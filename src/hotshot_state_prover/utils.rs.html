1use std::time::Duration;
2
3use alloy::providers::Provider;
4use anyhow::Result;
5
6#[derive(Debug, Clone, Copy)]
7pub struct ChainIdRetry {
8    pub request_timeout: Duration,
9    pub global_timeout: Duration,
10    pub retry_delay: Duration,
11}
12
13impl Default for ChainIdRetry {
14    fn default() -> Self {
15        Self {
16            request_timeout: Duration::from_secs(2),
17            global_timeout: Duration::from_secs(30),
18            retry_delay: Duration::from_secs(1),
19        }
20    }
21}
22
23impl ChainIdRetry {
24    /// Attempts to get the chain ID from the provider, retrying on failure.
25    ///
26    /// Combined with our `SwitchingTransport` this can be used to query
27    /// the chain ID from multiple providers.
28    ///
29    /// Each individual request is wrapped with a timeout to handle hanging requests.
30    /// The entire retry loop is wrapped with the global timeout to ensure we never
31    /// exceed the specified duration.
32    ///
33    /// Returns an error if unable to get chain ID after the global timeout period.
34    pub async fn get_chain_id(self, provider: &impl Provider) -> Result<u64> {
35        tokio::time::timeout(self.global_timeout, async {
36            loop {
37                match tokio::time::timeout(self.request_timeout, provider.get_chain_id()).await {
38                    Ok(Ok(id)) => return Ok(id),
39                    Ok(Err(err)) => {
40                        tracing::warn!("Failed to get chain ID, retrying: {err}");
41                    },
42                    Err(_) => {
43                        tracing::warn!(
44                            "Request timed out after {:?}, retrying",
45                            self.request_timeout
46                        );
47                    },
48                }
49                tokio::time::sleep(self.retry_delay).await;
50            }
51        })
52        .await
53        .map_err(|_| anyhow::anyhow!("Failed to get chain ID after {:?}", self.global_timeout))?
54    }
55}

58mod tests {
59    use alloy::{node_bindings::Anvil, providers::ProviderBuilder, rpc::client::RpcClient};
60    use espresso_types::{v0_1::SwitchingTransport, L1ClientOptions};
61    use rstest::rstest;

63    use super::*;
64
65    #[tokio::test]
66    async fn test_get_chain_id_switches_to_good_url() {
67        let anvil = Anvil::new().spawn();
68
69        let transport = SwitchingTransport::new(
70            L1ClientOptions::default(),
71            vec!["http://localhost:1".parse().unwrap(), anvil.endpoint_url()],
72        )
73        .expect("failed to create switching transport");
74        let rpc_client = RpcClient::new(transport, false);
75        let provider = ProviderBuilder::new().connect_client(rpc_client);
76
77        let chain_id = ChainIdRetry::default()
78            .get_chain_id(&provider)
79            .await
80            .unwrap();
81        assert_eq!(chain_id, anvil.chain_id());
82    }

84    #[rstest]
85    #[case::only_bad_urls(vec!["http://localhost:1", "http://localhost:2"])]
86    // A non-routable IP address to simulate a hanging request:
87    #[case::hanging_requests(vec!["http://10.255.255.1:1"])]
88    #[tokio::test]
89    async fn test_get_chain_id_failure(#[case] urls: Vec<&str>) {
90        let parsed_urls: Vec<_> = urls.iter().map(|u| u.parse().unwrap()).collect();
91        let transport = SwitchingTransport::new(L1ClientOptions::default(), parsed_urls)
92            .expect("failed to create switching transport");
93        let rpc_client = RpcClient::new(transport, false);
94        let provider = ProviderBuilder::new().connect_client(rpc_client);
95
96        let result = ChainIdRetry {
97            global_timeout: Duration::from_secs(1),
98            ..Default::default()
99        }
100        .get_chain_id(&provider)
101        .await;
102
103        assert!(result
104            .unwrap_err()
105            .to_string()
106            .contains("Failed to get chain ID"));
107    }
