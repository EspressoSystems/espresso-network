34/// Configuration/Parameters used for hotshot state prover
35#[derive(Debug, Clone)]
36pub struct StateProverConfig {
37    /// Url of the state relay server (a CDN that sequencers push their Schnorr signatures to)
38    pub relay_server: Url,
39    /// Interval between light client state update
40    pub update_interval: Duration,
41    /// Interval between retries if a state update fails
42    pub retry_interval: Duration,
43    /// RPC client to the L1 (or L2) provider
44    pub l1_rpc_client: RpcClient,
45    /// Address of LightClient proxy contract
46    pub light_client_address: Address,
47    /// Transaction signing key for Ethereum or any other layer 2
48    pub signer: LocalSigner<SigningKey>,
49    /// URL of a node that is currently providing the HotShot config.
50    /// This is used to initialize the stake table.
51    pub sequencer_url: Url,
52    /// If daemon and provided, the service will run a basic HTTP server on the given port.
53    ///
54    /// The server provides healthcheck and version endpoints.
55    pub port: Option<u16>,
56    /// Stake table capacity for the prover circuit.
57    pub stake_table_capacity: usize,
58    /// Epoch length in number of Hotshot blocks.
59    pub blocks_per_epoch: u64,
60    /// The epoch start block.
61    pub epoch_start_block: u64,
62    /// Maximum number of retires for one-shot prover
63    pub max_retries: u64,
64    /// optional gas price cap **in wei** to prevent prover sending updates during jammed base layer
65    pub max_gas_price: Option<u128>,
66}

68#[derive(Debug, Clone)]
69pub struct ProverServiceState {
70    /// The configuration of the prover service
71    pub config: StateProverConfig,
72    /// The current epoch number of the stake table
73    pub epoch: Option<<SeqTypes as NodeType>::Epoch>,
74    /// The stake table
75    pub stake_table: HSStakeTable<SeqTypes>,
76    /// The current stake table state
77    pub st_state: StakeTableState,
78}
79
80impl ProverServiceState {
81    pub async fn new_genesis(config: StateProverConfig) -> Result<Self> {
82        let stake_table = fetch_stake_table_from_sequencer(&config.sequencer_url, None)
83            .await
84            .with_context(|| "Failed to initialize stake table")?;
85        let st_state = stake_table
86            .commitment(config.stake_table_capacity)
87            .with_context(|| "Failed to compute stake table commitment")?;
88        Ok(Self {
89            config,
90            epoch: None,
91            stake_table,
92            st_state,
93        })
94    }
95
96    pub async fn sync_with_epoch(
97        &mut self,
98        epoch: Option<<SeqTypes as NodeType>::Epoch>,
99    ) -> Result<()> {
100        if epoch != self.epoch {
101            self.stake_table = fetch_stake_table_from_sequencer(&self.config.sequencer_url, epoch)
102                .await
103                .with_context(|| format!("Failed to update stake table for epoch: {epoch:?}"))?;
104            self.st_state = self
105                .stake_table
106                .commitment(self.config.stake_table_capacity)
107                .with_context(|| "Failed to compute stake table commitment")?;
108            self.epoch = epoch;
109        }
110        Ok(())
111    }
112}

114impl StateProverConfig {
115    pub async fn validate_light_client_contract(&self) -> Result<(), ProverError> {
116        let provider = ProviderBuilder::new().on_client(self.l1_rpc_client.clone());
117
118        if let Err(e) = is_proxy_contract(&provider, self.light_client_address).await {
119            Err(ProverError::ContractError(anyhow::anyhow!(
120                "Light Client contract's address {:?} is not a proxy: {e}",
121                self.light_client_address,
122            )))
123        } else {
124            Ok(())
125        }
126    }
127}

129#[derive(Debug, Display)]
130pub enum ProverError {
131    /// Invalid light client state or signatures: {0}
132    InvalidState(String),
133    /// Error when communicating with the smart contract: {0}
134    ContractError(anyhow::Error),
135    /// Error when communicating with the state relay server: {0}
136    RelayServerError(ServerError),
137    /// Error when communicating with the sequencer. Url: {0}, Error: {1}
138    SequencerCommunicationError(Url, ServerError),
139    /// Internal error when generating the SNARK proof: {0}
140    PlonkError(PlonkError),
141    /// Internal error: {0}
142    Internal(anyhow::Error),
143    /// Gas price too high: current {0} gwei, max allowed: {1} gwei
144    GasPriceTooHigh(String, String),
145    /// Epoch has already started on block {0}, please upgrade the contract to V2.
146    EpochAlreadyStarted(u64),
147}
148
149impl From<PlonkError> for ProverError {
150    fn from(err: PlonkError) -> Self {
151        Self::PlonkError(err)
152    }
153}
