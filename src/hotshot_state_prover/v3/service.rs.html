51pub fn load_proving_key(stake_table_capacity: usize) -> ProvingKey {
52    let srs = {
53        let num_gates = super::circuit::build_for_preprocessing::<
54            CircuitField,
55            ark_ed_on_bn254::EdwardsConfig,
56        >(stake_table_capacity)
57        .unwrap()
58        .0
59        .num_gates();
60
61        tracing::info!("Loading SRS from Aztec's ceremony...");
62        let srs_timer = Instant::now();
63        let srs = ark_srs::kzg10::aztec20::setup(num_gates + 2)
64            .expect("Error loading proving key: Aztec SRS fail to load.");
65        let srs_elapsed = Instant::now().signed_duration_since(srs_timer);
66        tracing::info!("Done in {srs_elapsed:.3}");
67
68        // convert to Jellyfish type
69        // TODO: (alex) use constructor instead https://github.com/EspressoSystems/jellyfish/issues/440
70        UnivariateUniversalParams {
71            powers_of_g: srs.powers_of_g,
72            h: srs.h,
73            beta_h: srs.beta_h,
74            powers_of_h: vec![srs.h, srs.beta_h],
75        }
76    };
77
78    tracing::info!("Generating proving key and verification key.");
79    let key_gen_timer = Instant::now();
80    let (pk, _) = super::snark::preprocess(&srs, stake_table_capacity)
81        .expect("Error loading proving key: failed to preprocess state prover circuit.");
82    let key_gen_elapsed = Instant::now().signed_duration_since(key_gen_timer);
83    tracing::info!("Done in {key_gen_elapsed:.3}");
84    pk
85}

100/// Read the following info from the LightClient contract storage on chain
101/// - latest finalized light client state
102/// - stake table commitment used in currently active epoch
103///
104/// Returned types are of Rust struct defined in `hotshot-types`.
105pub async fn read_contract_state(
106    provider: impl Provider,
107    address: Address,
108) -> Result<(LightClientState, StakeTableState), ProverError> {
109    let contract = LightClientV2::new(address, &provider);
110    let state: LightClientStateSol = match contract.finalizedState().call().await {
111        Ok(s) => s.into(),
112        Err(e) => {
113            tracing::error!("unable to read finalized_state from contract: {}", e);
114            return Err(ProverError::ContractError(e.into()));
115        },
116    };
117    let st_state: StakeTableStateSol = match contract.votingStakeTableState().call().await {
118        Ok(s) => s.into(),
119        Err(e) => {
120            tracing::error!(
121                "unable to read genesis_stake_table_state from contract: {}",
122                e
123            );
124            return Err(ProverError::ContractError(e.into()));
125        },
126    };
127
128    Ok((state.into(), st_state.into()))
129}

141async fn fetch_epoch_state_from_sequencer(
142    sequencer_url: &Url,
143    epoch: u64,
144) -> Result<LightClientStateUpdateCertificate<SeqTypes>, ProverError> {
145    let state_cert =
146        surf_disco::Client::<tide_disco::error::ServerError, StaticVersion<0, 1>>::new(
147            sequencer_url.clone(),
148        )
149        .get::<StateCertQueryData<SeqTypes>>(&format!("availability/state-cert/{epoch}"))
150        .send()
151        .await
152        .map_err(|err| {
153            ProverError::SequencerCommunicationError(
154                sequencer_url
155                    .join(&format!("availability/state-cert/{epoch}"))
156                    .unwrap(),
157                err,
158            )
159        })?;
160    Ok(state_cert.0)
161}

163async fn generate_proof(
164    state: &mut ProverServiceState,
165    light_client_state: LightClientState,
166    current_stake_table_state: StakeTableState,
167    next_stake_table_state: StakeTableState,
168    signature_map: HashMap<StateVerKey, StateSignature>,
169    proving_key: &ProvingKey,
170) -> Result<(Proof, PublicInput), ProverError> {
171    // Stake table update is already handled in the epoch catchup
172    let entries = state
173        .stake_table
174        .iter()
175        .map(|entry| {
176            (
177                entry.state_ver_key.clone(),
178                entry.stake_table_entry.stake_amount,
179            )
180        })
181        .collect::<Vec<_>>();
182    let mut signer_bit_vec = vec![false; entries.len()];
183    let mut signatures = vec![Default::default(); entries.len()];
184    let mut accumulated_weight = U256::ZERO;
185    entries.iter().enumerate().for_each(|(i, (key, stake))| {
186        if let Some(sig) = signature_map.get(key) {
187            // Check if the signature is valid
188            if <StateVerKey as LCV2StateSignatureKey>::verify_state_sig(
189                key,
190                sig,
191                &light_client_state,
192                &next_stake_table_state,
193            ) {
194                signer_bit_vec[i] = true;
195                signatures[i] = sig.clone();
196                accumulated_weight += *stake;
197            } else {
198                tracing::warn!("Invalid signature from key: {}", key);
199            }
200        }
201    });
202    tracing::debug!(
203        "Collected signatures with accumulated weight: {}",
204        accumulated_weight
205    );
206
207    if accumulated_weight < field_to_u256(current_stake_table_state.threshold) {
208        return Err(ProverError::InvalidState(
209            "The signers' total weight doesn't reach the threshold.".to_string(),
210        ));
211    }
212
213    tracing::info!("Collected latest state and signatures. Start generating SNARK proof.");
214    let proof_gen_start = Instant::now();
215    let proving_key_clone = proving_key.clone();
216    let stake_table_capacity = state.config.stake_table_capacity;
217    let auth_root = FixedBytes::from([0; 32]); // TODO(Chengyu): replace with actual auth_root
218    let signed_state_digest =
219        derive_signed_state_digest(&light_client_state, &next_stake_table_state, &auth_root);
220    let (proof, public_input) = spawn_blocking(move || {
221        super::snark::generate_state_update_proof(
222            &mut ark_std::rand::thread_rng(),
223            &proving_key_clone,
224            entries,
225            signer_bit_vec,
226            signatures,
227            &current_stake_table_state,
228            stake_table_capacity,
229            &signed_state_digest,
230        )
231    })
232    .await
233    .with_context(|| "Failed to join the proof generation task")
234    .map_err(ProverError::Internal)??;
235
236    let proof_gen_elapsed = Instant::now().signed_duration_since(proof_gen_start);
237    tracing::info!("Proof generation completed. Elapsed: {proof_gen_elapsed:.3}");
238
239    Ok((proof, public_input))
240}

242/// This function will fetch the cross epoch state update information from the sequencer query node
243/// and update the light client state in the contract to the `target_epoch`.
244/// In the end, both the locally stored stake table and the contract light client state will correspond
245/// to the `target_epoch`.
246/// It returns the final stake table state at the target epoch.
247async fn advance_epoch(
248    state: &mut ProverServiceState,
249    provider: impl Provider,
250    light_client_address: Address,
251    mut cur_st_state: StakeTableState,
252    proving_key: &ProvingKey,
253    contract_epoch: Option<<SeqTypes as NodeType>::Epoch>,
254    target_epoch: Option<<SeqTypes as NodeType>::Epoch>,
255) -> Result<StakeTableState, ProverError> {
256    let Some(target_epoch) = target_epoch else {
257        return Err(ProverError::Internal(anyhow!(
258            "Epoch related function called without a target epoch"
259        )));
260    };
261    // First sync the local stake table if necessary.
262    if state.epoch != contract_epoch {
263        state
264            .sync_with_epoch(contract_epoch)
265            .await
266            .with_context(|| format!("Failed to sync with epoch {contract_epoch:?}"))
267            .map_err(ProverError::Internal)?;
268    }
269    let base_epoch = contract_epoch
270        .map(|en| en.u64())
271        .unwrap_or(0)
272        .max(epoch_from_block_number(
273            state.config.epoch_start_block,
274            state.config.blocks_per_epoch,
275        ));
276    let target_epoch = target_epoch.u64();
277    for epoch in base_epoch..target_epoch {
278        tracing::info!("Performing epoch root state update for epoch {epoch}...");
279        let state_cert =
280            fetch_epoch_state_from_sequencer(&state.config.sequencer_url, epoch).await?;
281        let signature_map = state_cert
282            .signatures
283            .into_iter()
284            .collect::<HashMap<StateVerKey, StateSignature>>();
285
286        let (proof, public_input) = generate_proof(
287            state,
288            state_cert.light_client_state,
289            cur_st_state,
290            state_cert.next_stake_table_state,
291            signature_map,
292            proving_key,
293        )
294        .await?;
295
296        submit_state_and_proof(&provider, light_client_address, proof, public_input).await?;
297        tracing::info!("Epoch root state update successfully for epoch {epoch}.");
298
299        state
300            .sync_with_epoch(Some(EpochNumber::new(epoch + 1)))
301            .await
302            .with_context(|| format!("Failed to sync with epoch {}", epoch + 1))
303            .map_err(ProverError::Internal)?;
304        cur_st_state = state_cert.next_stake_table_state;
305    }
306    Ok(cur_st_state)
307}

309/// Sync the light client state from the relay server and submit the proof to the L1 LightClient contract
310pub async fn sync_state<ApiVer: StaticVersionType>(
311    state: &mut ProverServiceState,
312    proving_key: &ProvingKey,
313    relay_server_client: &Client<ServerError, ApiVer>,
314) -> Result<(), ProverError> {
315    let light_client_address = state.config.light_client_address;
316    let wallet = EthereumWallet::from(state.config.signer.clone());
317    let provider = ProviderBuilder::new()
318        .wallet(wallet)
319        .on_client(state.config.l1_rpc_client.clone());
320
321    // only sync light client state when gas price is sane
322    if let Some(max_gas_price) = state.config.max_gas_price {
323        let cur_gas_price = provider
324            .get_gas_price()
325            .await
326            .with_context(|| "Error checking gas price")
327            .map_err(ProverError::ContractError)?;
328        if cur_gas_price > max_gas_price {
329            let cur_gwei =
330                format_units(cur_gas_price, "gwei").map_err(|e| ProverError::Internal(e.into()))?;
331            let max_gwei =
332                format_units(max_gas_price, "gwei").map_err(|e| ProverError::Internal(e.into()))?;
333            return Err(ProverError::GasPriceTooHigh(cur_gwei, max_gwei));
334        }
335    }
336
337    let blocks_per_epoch = state.config.blocks_per_epoch;
338    let epoch_start_block = state.config.epoch_start_block;
339
340    let (contract_state, mut contract_st_state) =
341        read_contract_state(&provider, light_client_address).await?;
342    tracing::info!(
343        "Current HotShot block height on contract: {}",
344        contract_state.block_height
345    );
346
347    let bundle = fetch_latest_state(relay_server_client).await?;
348    tracing::debug!("Bundle accumulated weight: {}", bundle.accumulated_weight);
349    tracing::info!("Latest HotShot block height: {}", bundle.state.block_height);
350
351    if contract_state.block_height >= bundle.state.block_height {
352        tracing::info!("No update needed.");
353        return Ok(());
354    }
355    tracing::debug!("Old light client state: {contract_state}");
356    tracing::debug!("New light client state: {}", bundle.state);
357
358    tracing::debug!("Contract stake table state: {contract_st_state}");
359    tracing::debug!("Bundle stake table state: {}", bundle.next_stake);
360
361    let contract_state_epoch_enabled = contract_state.block_height >= epoch_start_block;
362    let epoch_enabled = bundle.state.block_height >= epoch_start_block;
363
364    if !epoch_enabled {
365        // If epoch hasn't been enabled, directly update the contract.
366        let (proof, public_input) = generate_proof(
367            state,
368            bundle.state,
369            contract_st_state,
370            contract_st_state,
371            bundle.signatures,
372            proving_key,
373        )
374        .await?;
375
376        submit_state_and_proof(&provider, light_client_address, proof, public_input).await?;
377
378        tracing::info!("Successfully synced light client state.");
379    } else {
380        // After the epoch is enabled
381        let contract_epoch = option_epoch_from_block_number::<SeqTypes>(
382            contract_state_epoch_enabled,
383            contract_state.block_height,
384            blocks_per_epoch,
385        );
386        // If the last contract update was on an epoch root, it's already on the next epoch.
387        let contract_epoch = if contract_state_epoch_enabled
388            && is_epoch_root(contract_state.block_height, blocks_per_epoch)
389        {
390            contract_epoch.map(|en| en + 1)
391        } else {
392            contract_epoch
393        };
394
395        let bundle_epoch = option_epoch_from_block_number::<SeqTypes>(
396            epoch_enabled,
397            bundle.state.block_height,
398            blocks_per_epoch,
399        );
400        let bundle_next_epoch = bundle_epoch.map(|en| en + 1);
401
402        // Update the local stake table if necessary
403        if contract_epoch != state.epoch {
404            state
405                .sync_with_epoch(contract_epoch)
406                .await
407                .map_err(ProverError::Internal)?;
408        }
409
410        // A catchup is needed if the contract epoch is behind.
411        if bundle_epoch > state.epoch {
412            tracing::info!(
413                "Catching up from epoch {contract_epoch:?} to epoch {bundle_epoch:?}..."
414            );
415            contract_st_state = advance_epoch(
416                state,
417                &provider,
418                light_client_address,
419                contract_st_state,
420                proving_key,
421                contract_epoch,
422                bundle_epoch,
423            )
424            .await?;
425        }
426
427        // Now that the contract epoch should be equal to the bundle epoch.
428
429        if is_ge_epoch_root(bundle.state.block_height as u64, blocks_per_epoch) {
430            // If we reached the epoch root, proceed to the next epoch directly
431            // In theory this should never happen because the node won't sign them.
432            tracing::info!("Epoch reaching an end, proceed to the next epoch...");
433            advance_epoch(
434                state,
435                &provider,
436                light_client_address,
437                contract_st_state,
438                proving_key,
439                bundle_epoch,
440                bundle_next_epoch,
441            )
442            .await?;
443        } else {
444            // Otherwise process the bundle update information as usual
445            let (proof, public_input) = generate_proof(
446                state,
447                bundle.state,
448                contract_st_state,
449                contract_st_state,
450                bundle.signatures,
451                proving_key,
452            )
453            .await?;
454
455            submit_state_and_proof(&provider, light_client_address, proof, public_input).await?;
456
457            tracing::info!("Successfully synced light client state.");
458        }
459    }
460    Ok(())
461}

463fn start_http_server<ApiVer: StaticVersionType + 'static>(
464    port: u16,
465    light_client_address: Address,
466    bind_version: ApiVer,
467) -> io::Result<()> {
468    let mut app = tide_disco::App::<_, ServerError>::with_state(());
469    let toml = toml::from_str::<toml::value::Value>(include_str!("../../api/prover-service.toml"))
470        .map_err(io::Error::other)?;
471
472    let mut api = Api::<_, ServerError, ApiVer>::new(toml).map_err(io::Error::other)?;
473
474    api.get("getlightclientcontract", move |_, _| {
475        async move { Ok(light_client_address) }.boxed()
476    })
477    .map_err(io::Error::other)?;
478    app.register_module("api", api).map_err(io::Error::other)?;
479
480    spawn(app.serve(format!("0.0.0.0:{port}"), bind_version));
481    Ok(())
482}

484/// Run prover in daemon mode
485pub async fn run_prover_service<ApiVer: StaticVersionType + 'static>(
486    config: StateProverConfig,
487    bind_version: ApiVer,
488) -> Result<()> {
489    let mut state = ProverServiceState::new_genesis(config).await?;
490
491    let stake_table_capacity = state.config.stake_table_capacity;
492    tracing::info!("Stake table capacity: {}", stake_table_capacity);
493
494    tracing::info!(
495        "Light client address: {:?}",
496        state.config.light_client_address
497    );
498
499    let relay_server_client = Arc::new(Client::<ServerError, ApiVer>::new(
500        state.config.relay_server.clone(),
501    ));
502
503    // Start the HTTP server to get a functioning healthcheck before any heavy computations.
504    if let Some(port) = state.config.port {
505        if let Err(err) = start_http_server(port, state.config.light_client_address, bind_version) {
506            tracing::error!("Error starting http server: {}", err);
507        }
508    }
509
510    let proving_key =
511        spawn_blocking(move || Arc::new(load_proving_key(state.config.stake_table_capacity)))
512            .await?;
513
514    let update_interval = state.config.update_interval;
515    let retry_interval = state.config.retry_interval;
516    loop {
517        if let Err(err) = sync_state(&mut state, &proving_key, &relay_server_client).await {
518            tracing::error!(
519                "Cannot sync the light client state, will retry in {:.1}s: {}",
520                retry_interval.as_secs_f32(),
521                err
522            );
523            sleep(retry_interval).await;
524        } else {
525            tracing::info!("Sleeping for {:.1}s", update_interval.as_secs_f32());
526            sleep(update_interval).await;
527        }
528    }
529}

531/// Run light client state prover once
532pub async fn run_prover_once<ApiVer: StaticVersionType>(
533    config: StateProverConfig,
534    _: ApiVer,
535) -> Result<()> {
536    let mut state = ProverServiceState::new_genesis(config).await?;
537
538    let stake_table_capacity = state.config.stake_table_capacity;
539    let proving_key =
540        spawn_blocking(move || Arc::new(load_proving_key(stake_table_capacity))).await?;
541    let relay_server_client = Client::<ServerError, ApiVer>::new(state.config.relay_server.clone());
542
543    for _ in 0..state.config.max_retries {
544        match sync_state(&mut state, &proving_key, &relay_server_client).await {
545            Ok(_) => return Ok(()),
546            Err(err) => {
547                tracing::error!(
548                    "Cannot sync the light client state, will retry in {:.1}s: {}",
549                    state.config.retry_interval.as_secs_f32(),
550                    err
551                );
552                sleep(state.config.retry_interval).await;
553            },
554        }
555    }
556    Err(anyhow::anyhow!("State update failed"))
557}
