23/// Different ways to ask the database for a leaf.
24#[derive(Clone, Copy, Debug, Display, From)]
25pub enum LeafRequest {
26    /// Ask for a leaf with a given ID.
27    #[display("leaf {_0}")]
28    Leaf(LeafId<SeqTypes>),
29
30    /// Ask for the leaf containing a header with a given ID.
31    #[display("header {_0}")]
32    Header(BlockId<SeqTypes>),
33}

35/// Maximum number of retries for a failed write before propagating the error.
36const WRITE_RETRY_MAX: u32 = 5;
37
38/// Backoff for retrying failed writes. Staggered so concurrent writers don't lock-step.
39const WRITE_BACKOFF: BackoffParams = BackoffParams::new(
40    Duration::from_millis(50),
41    Duration::from_millis(1_000),
42    2,
43    Ratio {
44        numerator: 5,
45        denominator: 10,
46    },
47);

49/// Client-side database for a [`LightClient`].
50pub trait Storage: Sized + Send + Sync + 'static {
51    /// Create a default, empty instance of the state.
52    ///
53    /// This is an async, fallible version of [`Default::default`]. If `Self: Default`, this is
54    /// equivalent to `ready(Ok(<Self as Default>::default()))`.
55    fn default() -> impl Send + Future<Output = Result<Self>>;
56
57    /// Get the number of blocks known to be in the chain.
58    ///
59    /// This is equivalent to one more than the block number of the latest known block.
60    ///
61    /// Because the database is not constantly being updated, this may be an underestimate of the
62    /// true number of blocks that exist.
63    fn block_height(&self) -> impl Send + Future<Output = Result<u64>>;
64
65    /// Get the earliest available leaf which is later than or equal to the requested leaf.
66    ///
67    /// This will either be the leaf requested, or can be used as the known-finalized endpoint in a
68    /// leaf chain proving that requested leaf is finalized (after the requested leaf is fetched
69    /// from elsewhere).
70    ///
71    /// If there is no known leaf later than the requested leaf, the result is [`None`].
72    fn leaf_upper_bound(
73        &self,
74        leaf: impl Into<LeafRequest> + Send,
75    ) -> impl Send + Future<Output = Result<Option<LeafQueryData<SeqTypes>>>>;
76
77    /// Get all leaves in the range [start, end)
78    fn get_leaves_in_range(
79        &self,
80        start: u32,
81        end: u32,
82    ) -> impl Send + Future<Output = Result<Vec<LeafQueryData<SeqTypes>>>>;
83
84    /// Add a leaf to the cache.
85    ///
86    /// This may result in an older leaf being removed.
87    fn insert_leaf(&self, leaf: LeafQueryData<SeqTypes>)
88    -> impl Send + Future<Output = Result<()>>;
89
90    /// Get the stake table for the latest epoch which is not later than `epoch`.
91    ///
92    /// If such a stake table is available in the database, returns the loaded epoch number, the
93    /// stake table state, the protocol version of the epoch root header under whose rules that
94    /// stake table's active set was selected, and the protocol version of the epoch root header
95    /// in the next epoch (used to seed iter 1 of a catchup that resumes from this row).
96    fn stake_table_lower_bound(
97        &self,
98        epoch: EpochNumber,
99    ) -> impl Send + Future<Output = Result<Option<(EpochNumber, StakeTableState, Version, Version)>>>;
100
101    /// Add a stake table to the cache.
102    ///
103    /// `epoch_root_protocol_version` is the protocol version of the epoch root header in epoch
104    /// `e-2` (the snapshot point), used so that future cache hits can apply the same active-set
105    /// selection rules without re-fetching the root.
106    ///
107    /// `next_epoch_root_protocol_version` is the protocol version of the epoch root header in
108    /// epoch `e-1` (the snapshot point for `e+1`), used so that a future catchup resuming from
109    /// this row can seed iter 1's filter version without re-fetching that root.
110    ///
111    /// This may result in an older stake table being removed.
112    fn insert_stake_table(
113        &self,
114        epoch: EpochNumber,
115        stake_table: &StakeTableState,
116        epoch_root_protocol_version: Version,
117        next_epoch_root_protocol_version: Version,
118    ) -> impl Send + Future<Output = Result<()>>;
119}

180impl LightClientSqliteOptions {
181    /// Create or connect to a database with the given options.
182    pub async fn connect(self) -> Result<SqliteStorage> {
183        let (path, _tmp) = match self.lc_path {
184            Some(p) => {
185                if let Some(parent) = p.parent().filter(|d| !d.as_os_str().is_empty()) {
186                    std::fs::create_dir_all(parent)
187                        .with_context(|| format!("creating parent directory {parent:?}"))?;
188                }
189                (p, None)
190            },
191            None => {
192                let mut builder = Builder::new();
193                builder.prefix("espresso-lc-");
194                #[cfg(unix)]
195                builder.permissions(Permissions::from_mode(0o700));
196                let dir = builder.tempdir().context(
197                    "creating temporary directory for light client database; set \
198                     LIGHT_CLIENT_DB_PATH to use a persistent location",
199                )?;
200                let path = dir.path().join("lc.db");
201                (path, Some(Arc::new(dir)))
202            },
203        };
204
205        let opt = hotshot_query_service::sqlite_options::sqlite_options().filename(&path);
206        let pool = SqlitePoolOptions::default()
207            .max_connections(self.num_connections)
208            .connect_with(opt)
209            .await?;
210        sqlx::migrate!("./migrations").run(&pool).await?;
211
212        Ok(SqliteStorage {
213            pool,
214            num_leaves: self.num_leaves,
215            num_stake_tables: self.num_stake_tables,
216            _tmp,
217        })
218    }
219}

235    async fn block_height(&self) -> Result<u64> {
236        let mut tx = self.pool.begin().await?;
237        let (height,) = query_as("SELECT COALESCE(max(height) + 1, 0) FROM leaf")
238            .fetch_one(tx.as_mut())
239            .await?;
240        Ok(height)
241    }

243    async fn leaf_upper_bound(
244        &self,
245        id: impl Into<LeafRequest> + Send,
246    ) -> Result<Option<LeafQueryData<SeqTypes>>> {
247        let mut tx = self.pool.begin().await?;
248
249        let mut q = QueryBuilder::new("SELECT height, data FROM leaf WHERE ");
250        match id.into() {
251            LeafRequest::Leaf(LeafId::Number(n)) | LeafRequest::Header(BlockId::Number(n)) => {
252                q.push("height >= ")
253                    .push_bind(n as i64)
254                    .push("ORDER BY HEIGHT");
255            },
256            LeafRequest::Leaf(LeafId::Hash(h)) => {
257                q.push("hash = ").push_bind(h.to_string());
258            },
259            LeafRequest::Header(BlockId::Hash(h)) => {
260                q.push("block_hash = ").push_bind(h.to_string());
261            },
262            LeafRequest::Header(BlockId::PayloadHash(h)) => {
263                q.push("payload_hash = ")
264                    .push_bind(h.to_string())
265                    .push("ORDER BY height");
266            },
267        }
268        q.push(" LIMIT 1");
269
270        let Some((height, data)) = q
271            .build_query_as::<(i64, _)>()
272            .fetch_optional(tx.as_mut())
273            .await?
274        else {
275            return Ok(None);
276        };
277        let leaf = serde_json::from_value(data)?;
278
279        // Mark this leaf as recently used.
280        let (id,): (i32,) = query_as("SELECT max(id) + 1 FROM leaf")
281            .fetch_one(tx.as_mut())
282            .await?;
283        query("UPDATE leaf SET id = $1 WHERE height = $2")
284            .bind(id)
285            .bind(height)
286            .execute(tx.as_mut())
287            .await?;
288        tx.commit().await?;
289
290        Ok(Some(leaf))
291    }

293    async fn get_leaves_in_range(
294        &self,
295        start_height: u32,
296        end_height: u32,
297    ) -> Result<Vec<LeafQueryData<SeqTypes>>> {
298        let mut tx = self.pool.begin().await?;
299
300        let leaves = query_as::<_, (i64, serde_json::Value)>(
301            "SELECT height, data FROM leaf WHERE height >= $1 AND height < $2 ORDER BY height",
302        )
303        .bind(start_height as i64)
304        .bind(end_height as i64)
305        .fetch_all(tx.as_mut())
306        .await?
307        .into_iter()
308        .map(|(_height, data)| serde_json::from_value(data))
309        .collect::<Result<Vec<_>, _>>()?;
310
311        tx.commit().await?;
312
313        Ok(leaves)
314    }

316    async fn insert_leaf(&self, leaf: LeafQueryData<SeqTypes>) -> Result<()> {
317        let height = leaf.height() as i64;
318        let hash = leaf.hash().to_string();
319        let block_hash = leaf.block_hash().to_string();
320        let payload_hash = leaf.payload_hash().to_string();
321        let data = serde_json::to_value(&leaf)?;
322
323        WRITE_BACKOFF
324            .retry_if(
325                WRITE_RETRY_MAX,
326                |_| true,
327                || async {
328                    let mut tx = self.pool.begin().await?;
329
330                    tracing::debug!(height, hash, "inserting leaf");
331                    let (id,): (i32,) = query_as(
332                        "INSERT INTO leaf (height, hash, block_hash, payload_hash, data) VALUES \
333                         ($1, $2, $3, $4, $5)
334                    ON CONFLICT (height) DO UPDATE SET id = excluded.id
335                    RETURNING id",
336                    )
337                    .bind(height)
338                    .bind(&hash)
339                    .bind(&block_hash)
340                    .bind(&payload_hash)
341                    .bind(&data)
342                    .fetch_one(tx.as_mut())
343                    .await
344                    .context("inserting new leaf")?;
345                    tracing::debug!(height, hash, id, "inserted leaf");
346
347                    // Delete the oldest leaves as necessary until the number of leaves stored does
348                    // not exceed `num_leaves`.
349                    let (num_leaves,): (u32,) = query_as("SELECT count(*) FROM leaf")
350                        .fetch_one(tx.as_mut())
351                        .await
352                        .context("counting leaves")?;
353                    let to_delete = num_leaves.saturating_sub(self.num_leaves);
354                    if to_delete > 0 {
355                        let (id_to_delete,): (i64,) =
356                            query_as("SELECT id FROM leaf ORDER BY id LIMIT 1 OFFSET $1")
357                                .bind(to_delete - 1)
358                                .fetch_one(tx.as_mut())
359                                .await
360                                .context("finding timestamp for GC")?;
361                        tracing::info!(id_to_delete, "garbage collecting {to_delete} leaves");
362                        let res = query("DELETE FROM leaf WHERE id <= $1")
363                            .bind(id_to_delete)
364                            .execute(tx.as_mut())
365                            .await
366                            .context("deleting old leaves")?;
367                        tracing::info!("deleted {} leaves", res.rows_affected());
368                    }
369
370                    tx.commit().await?;
371                    Ok(())
372                },
373            )
374            .await
375    }

377    async fn stake_table_lower_bound(
378        &self,
379        epoch: EpochNumber,
380    ) -> Result<Option<(EpochNumber, StakeTableState, Version, Version)>> {
381        let mut tx = self.pool.begin().await?;
382
383        let Some((epoch, epoch_root_protocol_version, next_epoch_root_protocol_version)) =
384            query_as::<_, (i64, String, String)>(
385                "SELECT epoch, epoch_root_protocol_version, next_epoch_root_protocol_version FROM \
386                 stake_table_epoch WHERE epoch <= $1 ORDER BY epoch DESC LIMIT 1",
387            )
388            .bind(*epoch as i64)
389            .fetch_optional(tx.as_mut())
390            .await
391            .context("loading epoch lower bound")?
392        else {
393            return Ok(None);
394        };
395        let epoch_root_protocol_version = versions::parse_version(&epoch_root_protocol_version)
396            .with_context(|| {
397                format!(
398                    "parsing stored epoch root protocol version {epoch_root_protocol_version:?} \
399                     for epoch {epoch}"
400                )
401            })?;
402        let next_epoch_root_protocol_version =
403            versions::parse_version(&next_epoch_root_protocol_version).with_context(|| {
404                format!(
405                    "parsing stored next epoch root protocol version \
406                     {next_epoch_root_protocol_version:?} for epoch {epoch}"
407                )
408            })?;
409
410        let validators = query_as::<_, (Value,)>(
411            "SELECT data FROM stake_table_validator WHERE epoch = $1 ORDER BY idx",
412        )
413        .bind(epoch)
414        .fetch(tx.as_mut())
415        .map_err(anyhow::Error::new)
416        .and_then(|(json,)| async move {
417            let validator: RegisteredValidator<PubKey> = serde_json::from_value(json)?;
418            Ok((validator.account, validator))
419        })
420        .try_collect()
421        .await
422        .context(format!("loading stake table for epoch {epoch}"))?;
423
424        let validator_exits =
425            query_as::<_, (String,)>("SELECT address FROM stake_table_exit WHERE epoch <= $1")
426                .bind(epoch)
427                .fetch(tx.as_mut())
428                .map_err(anyhow::Error::new)
429                .and_then(|(s,)| async move { Ok(Address::from_str(&s)?) })
430                .try_collect()
431                .await
432                .context(format!("loading validator exits for epoch {epoch}"))?;
433
434        let used_bls_keys =
435            query_as::<_, (String,)>("SELECT key FROM stake_table_bls_key WHERE epoch <= $1")
436                .bind(epoch)
437                .fetch(tx.as_mut())
438                .map_err(anyhow::Error::new)
439                .and_then(|(s,)| async move { Ok(PubKey::from_str(&s)?) })
440                .try_collect()
441                .await
442                .context(format!("loading BLS keys for epoch {epoch}"))?;
443
444        let used_schnorr_keys =
445            query_as::<_, (String,)>("SELECT key FROM stake_table_schnorr_key WHERE epoch <= $1")
446                .bind(epoch)
447                .fetch(tx.as_mut())
448                .map_err(anyhow::Error::new)
449                .and_then(|(s,)| async move { Ok(StateVerKey::from_str(&s)?) })
450                .try_collect()
451                .await
452                .context(format!("loading Schnorr keys for epoch {epoch}"))?;
453
454        let used_x25519_keys =
455            query_as::<_, (String,)>("SELECT key FROM stake_table_x25519_key WHERE epoch <= $1")
456                .bind(epoch)
457                .fetch(tx.as_mut())
458                .map_err(anyhow::Error::new)
459                .and_then(|(s,)| async move { Ok(x25519::PublicKey::from_str(&s)?) })
460                .try_collect()
461                .await
462                .context(format!("loading x25519 keys for epoch {epoch}"))?;
463
464        Ok(Some((
465            EpochNumber::new(epoch as u64),
466            StakeTableState::new(
467                validators,
468                validator_exits,
469                used_bls_keys,
470                used_schnorr_keys,
471                used_x25519_keys,
472            ),
473            epoch_root_protocol_version,
474            next_epoch_root_protocol_version,
475        )))
476    }

478    async fn insert_stake_table(
479        &self,
480        epoch: EpochNumber,
481        stake_table: &StakeTableState,
482        epoch_root_protocol_version: Version,
483        next_epoch_root_protocol_version: Version,
484    ) -> Result<()> {
485        let epoch = i64::try_from(*epoch).context("epoch overflow")?;
486        let epoch_root_protocol_version_str = epoch_root_protocol_version.to_string();
487        let next_epoch_root_protocol_version_str = next_epoch_root_protocol_version.to_string();
488        let validators = stake_table
489            .validators()
490            .values()
491            .cloned()
492            .map(serde_json::to_value)
493            .collect::<Result<Vec<_>, _>>()?;
494
495        WRITE_BACKOFF
496            .retry_if(
497                WRITE_RETRY_MAX,
498                |_| true,
499                || async {
500                    let mut tx = self.pool.begin().await?;
501
502                    // Record that the stake table for this epoch is available, along with the
503                    // versions of the epoch root headers in epochs `e-2` and `e-1` (snapshot points
504                    // for `e` and `e+1`).
505                    query(
506                        "INSERT INTO stake_table_epoch (epoch, epoch_root_protocol_version, \
507                         next_epoch_root_protocol_version) VALUES ($1, $2, $3)",
508                    )
509                    .bind(epoch)
510                    .bind(&epoch_root_protocol_version_str)
511                    .bind(&next_epoch_root_protocol_version_str)
512                    .execute(tx.as_mut())
513                    .await
514                    .context(format!(
515                        "recording stake table availability for epoch {epoch}"
516                    ))?;
517
518                    QueryBuilder::new("INSERT INTO stake_table_validator (epoch, idx, data) ")
519                        .push_values(validators.iter().enumerate(), |mut q, (i, data)| {
520                            q.push_bind(epoch).push_bind(i as i64).push_bind(data);
521                        })
522                        .build()
523                        .execute(tx.as_mut())
524                        .await
525                        .context(format!("inserting validators for epoch {epoch}"))?;
526
527                    // Insert only newly used BLS keys.
528                    QueryBuilder::new("INSERT INTO stake_table_bls_key (epoch, key) ")
529                    .push_values(stake_table.used_bls_keys(), |mut q, key| {
530                        q.push_bind(epoch).push_bind(key.to_string());
531                    })
532                    // If we insert keys out of order, make sure `epoch` reflects the earliest time
533                    // when this key was added to the state.
534                    .push(" ON CONFLICT (key) DO UPDATE SET epoch = min(epoch, excluded.epoch)")
535                    .build()
536                    .execute(tx.as_mut())
537                    .await
538                    .context(format!("inserting newly used BLS keys for epoch {epoch}"))?;
539
540                    // Insert only newly used Schnorr keys.
541                    QueryBuilder::new("INSERT INTO stake_table_schnorr_key (epoch, key) ")
542                    .push_values(stake_table.used_schnorr_keys(), |mut q, key| {
543                        q.push_bind(epoch).push_bind(key.to_string());
544                    })
545                    // If we insert keys out of order, make sure `epoch` reflects the earliest time
546                    // when this key was added to the state.
547                    .push(" ON CONFLICT (key) DO UPDATE SET epoch = min(epoch, excluded.epoch)")
548                    .build()
549                    .execute(tx.as_mut())
550                    .await
551                    .context(format!(
552                        "inserting newly used Schnorr keys for epoch {epoch}"
553                    ))?;
554
555                    // Insert only newly used x25519 keys.
556                    if !stake_table.used_x25519_keys().is_empty() {
557                        QueryBuilder::new("INSERT INTO stake_table_x25519_key (epoch, key) ")
558                        .push_values(stake_table.used_x25519_keys(), |mut q, key| {
559                            q.push_bind(epoch).push_bind(key.to_string());
560                        })
561                        // If we insert keys out of order, make sure `epoch` reflects the earliest
562                        // time when this key was added to the state.
563                        .push(" ON CONFLICT (key) DO UPDATE SET epoch = min(epoch, excluded.epoch)")
564                        .build()
565                        .execute(tx.as_mut())
566                        .await
567                        .context(format!(
568                            "inserting newly used x25519 keys for epoch {epoch}"
569                        ))?;
570                    }
571
572                    // Insert only the new validator exits.
573                    if !stake_table.validator_exits().is_empty() {
574                        QueryBuilder::new("INSERT INTO stake_table_exit (epoch, address) ")
575                        .push_values(stake_table.validator_exits(), |mut q, address| {
576                            q.push_bind(epoch).push_bind(address.to_string());
577                        })
578                        // If we insert exits out of order, make sure `epoch` reflects the earliest
579                        // time when this exit was added to the state.
580                        .push(
581                            " ON CONFLICT (address) DO UPDATE SET epoch = min(epoch, \
582                             excluded.epoch)",
583                        )
584                        .build()
585                        .execute(tx.as_mut())
586                        .await
587                        .context(format!("inserting new validator exits for epoch {epoch}"))?;
588                    }
589
590                    // Delete the second oldest stake table if necessary to ensure the number of stake
591                    // tables stored does not exceed `num_stake_tables`.
592                    let (num_stake_tables,): (u32,) =
593                        query_as("SELECT count(*) FROM stake_table_epoch")
594                            .fetch_one(tx.as_mut())
595                            .await
596                            .context("counting stake tables")?;
597                    if num_stake_tables > self.num_stake_tables {
598                        // We always delete the _second oldest_ stake table. We want to keep the oldest
599                        // around because it is the hardest to catch up for if we need it again (we
600                        // would have to go all the way back to genesis). The second oldest is the
601                        // least likely to be used again after the oldest, while still being easy to
602                        // replay if we do need it (because we can just replay from the cached oldest).
603                        let (epoch_to_delete,): (i64,) = query_as(
604                            "SELECT epoch FROM stake_table_epoch ORDER BY epoch LIMIT 1 OFFSET 1",
605                        )
606                        .fetch_one(tx.as_mut())
607                        .await
608                        .context("find second oldest epoch")?;
609                        tracing::info!(epoch_to_delete, "garbage collecting stake table");
610
611                        // Delete from the main epoch table. The corresponding rows from
612                        // `stake_table_validator` will be deleted automatically by cascading. The
613                        // corresponding rows in the BLS keys, Schnorr keys, and validator exits tables
614                        // cannot be deleted, because those tables are cumulative over later epochs.
615                        query("DELETE FROM stake_table_epoch WHERE epoch = $1")
616                            .bind(epoch_to_delete)
617                            .execute(tx.as_mut())
618                            .await
619                            .context("garbage collecting stake table")?;
620                    }
621
622                    tx.commit().await?;
623                    Ok(())
624                },
625            )
626            .await
627    }

1160    /// Make a stake table state with all fields populated.
1161    fn random_stake_table() -> StakeTableState {
1162        let validator = random_validator();
1163        let candidate: RegisteredValidator<PubKey> = validator.clone().into();
1164        let x25519_key =
1165            x25519::PublicKey::try_from(rand::random::<[u8; 32]>().as_slice()).unwrap();
1166        let candidate_bls = candidate
1167            .stake_table_key
1168            .expect("random_validator returns authenticated validator");
1169        StakeTableState::new(
1170            [(candidate.account, candidate.clone())]
1171                .into_iter()
1172                .collect(),
1173            [Address::random()].into_iter().collect(),
1174            [candidate_bls].into_iter().collect(),
1175            [candidate
1176                .state_ver_key
1177                .expect("random_validator has valid schnorr key")]
1178            .into_iter()
1179            .collect(),
1180            [x25519_key].into_iter().collect(),
1181        )
1182    }

1184    /// Create a new stake table state which is a possible successor to the given state.
1185    fn chain_stake_table(state: &StakeTableState) -> StakeTableState {
1186        let new_validator = random_validator();
1187        let new_candidate: RegisteredValidator<PubKey> = new_validator.clone().into();
1188        let new_candidate_bls = new_candidate
1189            .stake_table_key
1190            .expect("random_validator returns authenticated validator");
1191        let new_exit = Address::random();
1192        let new_x25519 =
1193            x25519::PublicKey::try_from(rand::random::<[u8; 32]>().as_slice()).unwrap();
1194        StakeTableState::new(
1195            state
1196                .validators()
1197                .values()
1198                .chain([&new_candidate])
1199                .map(|v| (v.account, v.clone()))
1200                .collect(),
1201            state
1202                .validator_exits()
1203                .iter()
1204                .chain([&new_exit])
1205                .cloned()
1206                .collect(),
1207            state
1208                .used_bls_keys()
1209                .iter()
1210                .chain([&new_candidate_bls])
1211                .cloned()
1212                .collect(),
1213            state
1214                .used_schnorr_keys()
1215                .iter()
1216                .chain([new_candidate
1217                    .state_ver_key
1218                    .as_ref()
1219                    .expect("random_validator has valid schnorr key")])
1220                .cloned()
1221                .collect(),
1222            state
1223                .used_x25519_keys()
1224                .iter()
1225                .chain([&new_x25519])
1226                .cloned()
1227                .collect(),
1228        )
1229    }
