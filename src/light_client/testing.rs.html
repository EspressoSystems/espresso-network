70/// Extract a chain of QCs from a chain of leaves.
71///
72/// The resulting QC chain will be one shorter than the leaf chain, and will justify the finality of
73/// the leaf _preceding_ this leaf chain, since we extract QCs from the justifying QC of each leaf.
74pub fn qc_chain_from_leaf_chain<'a>(
75    leaves: impl IntoIterator<Item = &'a LeafQueryData<SeqTypes>>,
76) -> Vec<Certificate> {
77    leaves
78        .into_iter()
79        .map(|leaf| Certificate::for_parent(leaf.leaf()))

100/// Construct a customized leaf chain for the given height range.
101///
102/// The chain will upgrade from `V::Base` to `V::Upgrade` at height `upgrade_height`.
103pub async fn custom_leaf_chain_with_upgrade<V: Versions>(
104    range: impl IntoIterator<Item = u64>,
105    upgrade_height: u64,
106    map: impl Fn(&mut QuorumProposal2<SeqTypes>),
107) -> Vec<LeafQueryData<SeqTypes>> {
108    let upgrade_leaf: Leaf2 = Leaf2::genesis::<V>(
109        &Default::default(),
110        &NodeState::mock()
111            .with_genesis_version(V::Upgrade::version())
112            .with_current_version(V::Upgrade::version()),
113    )
114    .await;
115    let upgrade_data = UpgradeProposalData {
116        old_version: V::Base::version(),
117        new_version: V::Upgrade::version(),
118        new_version_hash: Default::default(),
119        old_version_last_view: ViewNumber::new(upgrade_height - 1),
120        new_version_first_view: ViewNumber::new(upgrade_height),
121        decide_by: ViewNumber::new(upgrade_height),
122    };
123    let upgrade_commit = upgrade_data.commit();
124    let upgrade_cert = UpgradeCertificate::new(
125        upgrade_data,
126        upgrade_commit,
127        ViewNumber::new(upgrade_height),
128        Default::default(),
129        Default::default(),
130    );
131
132    custom_leaf_chain::<V>(range, |proposal| {
133        let height = proposal.block_header.height();
134        if height < upgrade_height {
135            // All views leading up to the upgrade get a certificate indicating the coming upgrade.
136            proposal.upgrade_certificate = Some(upgrade_cert.clone());
137        } else {
138            // After the upgrade takes effect we stop attaching the upgrade certificate, and we use
139            // the upgraded header version.
140            proposal.upgrade_certificate = None;
141            proposal.block_header = upgrade_leaf.block_header().clone();
142            *proposal.block_header.height_mut() = height;
143        }
144        map(proposal);
145    })
146    .await
147}

149/// Construct a customized leaf chain for the given height range.
150pub async fn custom_leaf_chain<V: Versions>(
151    range: impl IntoIterator<Item = u64>,
152    map: impl Fn(&mut QuorumProposal2<SeqTypes>),
153) -> Vec<LeafQueryData<SeqTypes>> {
154    let node_state = NodeState::mock()
155        .with_genesis_version(V::Base::version())
156        .with_current_version(V::Base::version());
157    let genesis_leaf: Leaf2 = Leaf2::genesis::<V>(&Default::default(), &node_state).await;
158    tracing::info!(?genesis_leaf, "leaf chain");
159
160    let mut qc = QuorumCertificate2::genesis::<V>(&Default::default(), &node_state).await;
161    let mut quorum_proposal = QuorumProposalWrapper::<SeqTypes> {
162        proposal: QuorumProposal2::<SeqTypes> {
163            epoch: None,
164            block_header: genesis_leaf.block_header().clone(),
165            view_number: genesis_leaf.view_number(),
166            justify_qc: qc.clone(),
167            upgrade_certificate: None,
168            view_change_evidence: None,
169            next_drb_result: None,
170            next_epoch_justify_qc: None,
171            state_cert: None,
172        },
173    };
174
175    let mut block_merkle_tree = BlockMerkleTree::new(BLOCK_MERKLE_TREE_HEIGHT);
176    let mut leaves = vec![];
177    for height in range {
178        *quorum_proposal.proposal.block_header.height_mut() = height;
179        *quorum_proposal
180            .proposal
181            .block_header
182            .block_merkle_tree_root_mut() = block_merkle_tree.commitment();
183        quorum_proposal.proposal.view_number = ViewNumber::new(height);
184        map(&mut quorum_proposal.proposal);
185        let leaf = Leaf2::from_quorum_proposal(&quorum_proposal);
186
187        qc.view_number = ViewNumber::new(height);
188        qc.data.leaf_commit = Committable::commit(&leaf);
189        if leaf.block_header().version() >= EpochVersion::version() {
190            qc.data.block_number = Some(height);
191        }
192
193        block_merkle_tree
194            .push(leaf.block_header().commit())
195            .unwrap();
196        leaves.push(LeafQueryData::new(leaf, qc.clone()).unwrap());
197        quorum_proposal.proposal.justify_qc = qc.clone();
198    }
199
200    leaves
201}

252/// A quorum which verifies that calls to `verify` use the correct version, but does not check
253/// signatures.
254#[derive(Clone, Debug, Default)]
255pub struct VersionCheckQuorum {
256    leaves: HashMap<Commitment<Leaf2>, Leaf2>,
257}
258
259impl VersionCheckQuorum {
260    pub fn new(leaves: impl IntoIterator<Item = Leaf2>) -> Self {
261        Self {
262            leaves: leaves
263                .into_iter()
264                .map(|leaf| (leaf.commit(), leaf))
265                .collect(),
266        }
267    }
268}
269
270impl Quorum for VersionCheckQuorum {
271    async fn verify_static<V: StaticVersionType + 'static>(
272        &self,
273        cert: &Certificate,
274    ) -> anyhow::Result<()> {
275        let leaf = self
276            .leaves
277            .get(&cert.leaf_commit())
278            .context(format!("unknown leaf {}", cert.leaf_commit()))?;
279        ensure!(
280            leaf.block_header().version() == V::version(),
281            "version mismatch: leaf has version {}, but verifier is using version {}",
282            leaf.block_header().version(),
283            V::version()
284        );
285        Ok(())
286    }
287}

352    fn quorum_for_epoch(&mut self, epoch: u64) -> &[(PrivKey, Validator<PubKey>)] {
353        // For testing purposes, we will say that one new node joins the quorum each epoch. The
354        // static stake table used before the first epoch with dynamic stake is the same as the
355        // stake table used in that epoch.
356        let quorum_size = max(epoch, self.first_epoch_with_dynamic_stake_table) as usize;
357
358        while self.quorum.len() < quorum_size {
359            let (stake_table_key, priv_key) =
360                PubKey::generated_from_seed_indexed(Default::default(), self.quorum.len() as u64);
361            let (state_ver_key, _) = SchnorrPubKey::generated_from_seed_indexed(
362                Default::default(),
363                self.quorum.len() as u64,
364            );
365            let stake = U256::from(self.quorum.len() + 1) * U256::from(1_000_000_000u128);
366            let validator = Validator {
367                account: Address::random(),
368                stake_table_key,
369                state_ver_key,
370                stake,
371                commission: 1,
372                delegators: [(Address::random(), stake)].into_iter().collect(),
373            };
374            self.quorum.push((priv_key, validator));
375        }
376
377        &self.quorum[..quorum_size]
378    }

380    fn stake_table_hash(&mut self, epoch: u64) -> StakeTableHash {
381        let quorum = self.quorum_for_epoch(epoch);
382        let mut validators = ValidatorMap::default();
383        let mut used_bls_keys = HashSet::default();
384        let mut used_schnorr_keys = HashSet::default();
385        for (_, validator) in quorum {
386            validators.insert(validator.account, validator.clone());
387            used_bls_keys.insert(validator.stake_table_key);
388            used_schnorr_keys.insert(validator.state_ver_key.clone());
389        }
390
391        let state = StakeTableState::new(
392            validators,
393            Default::default(),
394            used_bls_keys,
395            used_schnorr_keys,
396        );
397        state.commit()
398    }

400    async fn leaf(&mut self, height: usize, epoch_height: u64) -> LeafQueryData<SeqTypes> {
401        let epoch = epoch_from_block_number(height as u64, epoch_height);
402        let (quorum, stake_table): (Vec<_>, Vec<_>) =
403            self.quorum_for_epoch(epoch).iter().cloned().unzip();
404        let mut stake_entries = vec![];
405        let mut total_stake = U256::ZERO;
406        for validator in &stake_table {
407            stake_entries.push(StakeTableEntry {
408                stake_key: validator.stake_table_key,
409                stake_amount: validator.stake,
410            });
411            total_stake += validator.stake;
412        }
413
414        let pp = PubKey::public_parameter(&stake_entries, supermajority_threshold(total_stake));
415
416        for i in self.leaves.len()..=height {
417            let epoch = EpochNumber::new(epoch_from_block_number(i as u64, epoch_height));
418            let view_number = ViewNumber::new(i as u64);
419
420            let version = DrbAndHeaderUpgradeVersion::version();
421            let node_state = NodeState::mock_v3().with_genesis_version(version);
422            let (justify_qc, mt) = if i == 0 {
423                (
424                    QuorumCertificate2::genesis::<
425                        SequencerVersions<DrbAndHeaderUpgradeVersion, DrbAndHeaderUpgradeVersion>,
426                    >(&Default::default(), &node_state)
427                    .await,
428                    SHA3MerkleTree::new(BLOCK_MERKLE_TREE_HEIGHT),
429                )
430            } else {
431                let parent = &self.leaves[i - 1];
432                let mut mt = self.merkle_trees[i - 1].clone();
433                mt.push(parent.block_hash()).unwrap();
434                (parent.qc().clone(), mt)
435            };
436
437            let transactions = vec![Transaction::random(&mut rand::thread_rng())];
438            let (payload, ns_table) =
439                Payload::from_transactions_sync(transactions, node_state.chain_config).unwrap();
440            let payload_comm =
441                vid_commitment::<
442                    SequencerVersions<DrbAndHeaderUpgradeVersion, DrbAndHeaderUpgradeVersion>,
443                >(&payload.encode(), &ns_table.encode(), quorum.len(), version);
444
445            let mut block_header = Leaf2::genesis::<
446                SequencerVersions<DrbAndHeaderUpgradeVersion, DrbAndHeaderUpgradeVersion>,
447            >(&Default::default(), &node_state)
448            .await
449            .block_header()
450            .clone();
451            *block_header.height_mut() = i as u64;
452            *block_header.block_merkle_tree_root_mut() = mt.commitment();
453            *block_header.payload_commitment_mut() = payload_comm;
454            *block_header.ns_table_mut() = ns_table;
455            if *epoch + 1 >= self.first_epoch_with_dynamic_stake_table
456                && is_ge_epoch_root(block_header.height(), epoch_height)
457            {
458                assert!(block_header.set_next_stake_table_hash(self.stake_table_hash(*epoch + 1)));
459            }
460            let quorum_proposal = QuorumProposalWrapper::<SeqTypes> {
461                proposal: QuorumProposal2::<SeqTypes> {
462                    epoch: Some(epoch),
463                    block_header,
464                    view_number,
465                    justify_qc,
466                    upgrade_certificate: None,
467                    view_change_evidence: None,
468                    next_drb_result: None,
469                    next_epoch_justify_qc: None,
470                    state_cert: None,
471                },
472            };
473            let leaf = Leaf2::from_quorum_proposal(&quorum_proposal);
474            let quorum_data = QuorumData2 {
475                leaf_commit: leaf.commit(),
476                epoch: Some(epoch),
477                block_number: Some(i as u64),
478            };
479            let quorum_data_comm = VersionedVoteData::new_infallible(
480                quorum_data.clone(),
481                view_number,
482                &UpgradeLock::<
483                    SeqTypes,
484                    SequencerVersions<DrbAndHeaderUpgradeVersion, DrbAndHeaderUpgradeVersion>,
485                >::new(),
486            )
487            .await
488            .commit();
489            let signatures = quorum
490                .iter()
491                .map(|key| PubKey::sign(key, quorum_data_comm.as_ref()).unwrap())
492                .collect::<Vec<_>>();
493            let assembled = PubKey::assemble(
494                &pp,
495                &std::iter::repeat_n(true, quorum.len()).collect::<BitVec>(),
496                &signatures,
497            );
498            let qc = QuorumCertificate2::create_signed_certificate(
499                quorum_data_comm,
500                quorum_data,
501                assembled,
502                view_number,
503            );
504            let leaf = LeafQueryData::new(leaf, qc).unwrap();
505            self.leaf_hashes.insert(leaf.hash(), i);
506            self.block_hashes.insert(leaf.block_hash(), i);
507            self.payload_hashes.entry(leaf.payload_hash()).or_insert(i);
508            self.leaves.push(leaf);
509            self.payloads.push(payload);
510            self.merkle_trees.push(mt);
511        }
512
513        self.leaves[height].clone()
514    }

516    fn leaf_height(&self, req: LeafRequest) -> Result<usize> {
517        match req {
518            LeafRequest::Leaf(LeafId::Number(h)) | LeafRequest::Header(BlockId::Number(h)) => Ok(h),
519            LeafRequest::Leaf(LeafId::Hash(h)) => self
520                .leaf_hashes
521                .get(&h)
522                .copied()
523                .context(format!("missing leaf {h}")),
524            LeafRequest::Header(BlockId::Hash(h)) => self
525                .block_hashes
526                .get(&h)
527                .copied()
528                .context(format!("missing block {h}")),
529            LeafRequest::Header(BlockId::PayloadHash(h)) => self
530                .payload_hashes
531                .get(&h)
532                .copied()
533                .context(format!("missing payload {h}")),
534        }
535    }

652    async fn leaf_proof(
653        &self,
654        id: impl Into<LeafRequest> + Send,
655        finalized: Option<u64>,
656    ) -> Result<LeafProof> {
657        let mut inner = self.inner.lock().await;
658
659        let mut height = inner.leaf_height(id.into())?;
660        ensure!(
661            !inner.missing_leaves.contains(&height),
662            "missing leaf {height}"
663        );
664        if inner.invalid_proofs.contains(&height) {
665            tracing::info!(height, "return mock incorrect proof");
666            return Ok(LeafProof::default());
667        }
668        if let Some(sub) = inner.swapped_leaves.get(&height) {
669            tracing::info!(height, sub, "return wrong leaf");
670            height = *sub;
671        };
672
673        let leaf = inner.leaf(height, self.epoch_height).await;
674
675        let mut proof = LeafProof::default();
676        proof.push(leaf);
677        if let Some(finalized) = finalized {
678            ensure!(
679                finalized > (height as u64),
680                "assumed finalized leaf must be after requested leaf"
681            );
682            if finalized <= (height as u64) + 2 {
683                tracing::info!(
684                    height,
685                    finalized,
686                    "path to finalized is shorter than path to QC-chain, using finalized"
687                );
688                return Ok(proof);
689            }
690        }
691
692        proof.push(inner.leaf(height + 1, self.epoch_height).await);
693        assert!(proof.push(inner.leaf(height + 2, self.epoch_height).await));
694
695        Ok(proof)
696    }

698    async fn header_proof(&self, root: u64, id: BlockId<SeqTypes>) -> Result<HeaderProof> {
699        let mut inner = self.inner.lock().await;
700
701        let root = root as usize;
702        let mut height = inner.leaf_height(id.into())?;
703        ensure!(
704            !inner.missing_leaves.contains(&height),
705            "missing leaf {height}"
706        );
707        if inner.invalid_proofs.contains(&height) {
708            tracing::info!(root, height, "return mock invalid proof");
709            let leaf = inner.leaf(height, self.epoch_height).await;
710            // Construct a proof using a Merkle tree of the wrong height.
711            let mt = BlockMerkleTree::from_elems(
712                Some(BLOCK_MERKLE_TREE_HEIGHT + 1),
713                [leaf.block_hash()],
714            )
715            .unwrap();
716            let proof = mt.lookup(0).expect_ok().unwrap().1;
717            return Ok(HeaderProof::new(leaf.header().clone(), proof));
718        }
719        if let Some(sub) = inner.swapped_leaves.get(&height) {
720            tracing::info!(height, sub, "return wrong leaf");
721            height = *sub;
722        };
723
724        ensure!(height < root);
725
726        let mt = &inner.merkle_trees[root];
727        tracing::info!(height, root = %mt.commitment(), "get proof from Merkle tree");
728        let proof = mt.lookup(height as u64).expect_ok().unwrap().1;
729        let header = inner.leaf(height, self.epoch_height).await.header().clone();
730        Ok(HeaderProof::new(header, proof))
731    }

747    async fn stake_table_events(&self, epoch: EpochNumber) -> Result<Vec<StakeTableEvent>> {
748        let mut inner = self.inner.lock().await;
749
750        ensure!(
751            !inner.missing_quorums.contains(&*epoch),
752            "missing quorum for epoch {epoch}"
753        );
754
755        if inner.invalid_quorums.contains(&*epoch) {
756            // Return random events to create an invalid stake table.
757            let mut events = vec![];
758            let mut seed = [0; 32];
759            rand::thread_rng().fill_bytes(&mut seed);
760            register_validator_events(&mut events, &random_validator());
761            return Ok(events);
762        }
763
764        let mut events = vec![];
765        if *epoch == inner.first_epoch_with_dynamic_stake_table {
766            // Generate events such that the first dynamic stake table is the same as the static
767            // stake table we started with.
768            for (_, validator) in inner.quorum_for_epoch(*epoch) {
769                register_validator_events(&mut events, validator);
770            }
771        } else if *epoch > inner.first_epoch_with_dynamic_stake_table {
772            // For each subsequent epoch, just generate one event for the new validator which joined
773            // in that epoch.
774            let (_, validator) = &inner.quorum_for_epoch(*epoch)[(*epoch as usize) - 1];
775            register_validator_events(&mut events, validator);
776        }
777        Ok(events)
778    }

802    async fn namespace_proof(
803        &self,
804        height: u64,
805        mut namespace: NamespaceId,
806    ) -> Result<NamespaceProof> {
807        let mut inner = self.inner.lock().await;
808
809        let vid_common = inner.vid_common(height, self.epoch_height);
810        let height = height as usize;
811        ensure!(height < inner.payloads.len());
812
813        let (payload, ns_table) = if inner.invalid_payloads.contains(&height) {
814            // To mock an invalid proof, we use a different payload which doesn't match the actual
815            // payload for the requested block.
816            tracing::info!(height, "return mock incorrect namespace proof");
817            let mut payload = vec![0; 32];
818            rand::thread_rng().fill_bytes(&mut payload);
819            let tx = Transaction::new(namespace, payload);
820            let node_state = NodeState::mock_v3();
821            Payload::from_transactions_sync([tx], node_state.chain_config).unwrap()
822        } else {
823            (
824                inner.payloads[height].clone(),
825                inner.leaves[height].header().ns_table().clone(),
826            )
827        };
828
829        if inner.invalid_proofs.contains(&height) {
830            // If we are returning a mock invalid proof, return a proof for the wrong namespace so
831            // that verification will fail.
832            namespace = NamespaceId::from(u64::from(namespace) + 1);
833        }
834
835        let Some(ns_index) = ns_table.find_ns_id(&namespace) else {
836            return Ok(NamespaceProof::not_present());
837        };
838        let proof = NsProof::new(&payload, &ns_index, &vid_common)
839            .context("failed to construct NsProof")?;
840        Ok(NamespaceProof::new(proof, vid_common))
841    }

857fn register_validator_events(events: &mut Vec<StakeTableEvent>, validator: &Validator<PubKey>) {
858    events.push(StakeTableEvent::Register(ValidatorRegistered {
859        account: validator.account,
860        blsVk: validator.stake_table_key.into(),
861        schnorrVk: validator.state_ver_key.clone().into(),
862        commission: validator.commission,
863    }));
864    for (&delegator, &amount) in &validator.delegators {
865        events.push(StakeTableEvent::Delegate(Delegated {
866            delegator,
867            validator: validator.account,
868            amount,
869        }));
870    }
871}

873pub fn random_validator() -> Validator<PubKey> {
874    let account = Address::random();
875    let mut seed = [0; 32];
876    rand::thread_rng().fill_bytes(&mut seed);
877    let stake = U256::from(rand::thread_rng().next_u64());
878    Validator {
879        account,
880        stake_table_key: PubKey::generated_from_seed_indexed(seed, 0).0,
881        state_ver_key: SchnorrPubKey::generated_from_seed_indexed(seed, 0).0,
882        stake,
883        commission: 1,
884        delegators: [(Address::random(), stake)].into_iter().collect(),
885    }
