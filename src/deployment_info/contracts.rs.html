73impl AccessControlRole {

77            AccessControlRole::Pauser => alloy::primitives::keccak256("PAUSER_ROLE"),
78        }

155pub(crate) struct DeploymentQuerier<'a, P: Provider> {
156    provider: &'a P,
157    known: KnownAddresses,
158    block_id: BlockId,
159}
160
161impl<'a, P: Provider> DeploymentQuerier<'a, P> {
162    pub(crate) fn new(provider: &'a P, known: KnownAddresses, block_number: u64) -> Self {
163        Self {
164            provider,
165            known,
166            block_id: BlockId::number(block_number),
167        }
168    }
169
170    /// Finds which known address holds the given role. Errors if the holder is not
171    /// in `self.known` -- this validates that all role holders are tracked in the .env config.
172    async fn find_role_holder(
173        &self,
174        contract_addr: Address,
175        role: AccessControlRole,
176    ) -> Result<Address> {
177        let contract = IAccessControl::new(contract_addr, self.provider);
178        let role_hash = role.hash();
179        let mut holders = Vec::new();
180        for addr in self.known.keys() {
181            let has_role = contract
182                .hasRole(role_hash, *addr)
183                .block(self.block_id)
184                .call()
185                .await?;
186            if has_role {
187                holders.push(*addr);
188            }
189        }
190        match holders.len() {
191            0 => bail!(
192                "No known address holds {role} at {contract_addr}. The .env config may be missing \
193                 a multisig or other contract."
194            ),
195            1 => Ok(holders[0]),
196            _ => bail!(
197                "Multiple known addresses hold {role} at {contract_addr}: {holders:?}. This is \
198                 unexpected."
199            ),
200        }
201    }
202
203    async fn get_version(&self, addr: Address) -> Result<String> {
204        let v = IVersioned::new(addr, self.provider)
205            .getVersion()
206            .block(self.block_id)
207            .call()
208            .await?;
209        Ok(format!("{}.{}.{}", v._0, v._1, v._2))
210    }
211
212    fn resolve_role_holder(&self, addr: Address) -> Result<RoleHolder> {
213        let name = self.known.resolve(addr)?;
214        Ok(RoleHolder {
215            address: addr,
216            name,
217        })
218    }
219
220    pub(crate) async fn query_ownable(
221        &self,
222        addr: Address,
223        contract_type: ContractType,
224    ) -> Result<OwnableDeployment> {
225        tracing::info!("querying {contract_type} at {addr}");
226
227        let owner_addr = IOwnable::new(addr, self.provider)
228            .owner()
229            .block(self.block_id)
230            .call()
231            .await?;
232        let version = self.get_version(addr).await?;
233
234        let owner = self
235            .resolve_role_holder(owner_addr)
236            .context(format!("owner of {contract_type}"))?;
237
238        tracing::info!("  owner={} version={version}", owner.name);
239
240        Ok(OwnableDeployment::Deployed {
241            address: addr,
242            owner_address: owner.address,
243            owner_name: owner.name,
244            version,
245        })
246    }
247
248    pub(crate) async fn query_access_control(
249        &self,
250        addr: Address,
251        contract_type: ContractType,
252    ) -> Result<AccessControlDeployment> {
253        tracing::info!("querying {contract_type} at {addr}");
254
255        let admin_addr = self
256            .find_role_holder(addr, AccessControlRole::DefaultAdmin)
257            .await
258            .context(format!("default admin of {contract_type}"))?;
259        let version = self.get_version(addr).await?;
260
261        let admin = self
262            .resolve_role_holder(admin_addr)
263            .context(format!("default admin of {contract_type}"))?;
264
265        let pauser_addr = match contract_type {
266            ContractType::StakeTable | ContractType::RewardClaim => self
267                .find_role_holder(addr, AccessControlRole::Pauser)
268                .await
269                .context(format!("pauser of {contract_type}"))?,
270            other => bail!("{other} is not an AccessControl contract"),
271        };
272        let pauser = self
273            .resolve_role_holder(pauser_addr)
274            .context(format!("pauser of {contract_type}"))?;
275
276        tracing::info!(
277            "  default_admin={} version={version} pauser={}",
278            admin.name,
279            pauser.name
280        );
281
282        Ok(AccessControlDeployment::Deployed {
283            address: addr,
284            default_admin_address: admin.address,
285            default_admin_name: admin.name,
286            version,
287            pauser_address: pauser.address,
288            pauser_name: pauser.name,
289        })
290    }
291
292    async fn query_multisig(&self, name: &str, addr: Address) -> Result<MultisigDeployment> {
293        let contract = ISafe::new(addr, self.provider);
294        let version = contract
295            .VERSION()
296            .block(self.block_id)
297            .call()
298            .await
299            .with_context(|| format!("Failed to get VERSION for multisig '{name}' at {addr}"))?;
300        let owners = contract
301            .getOwners()
302            .block(self.block_id)
303            .call()
304            .await
305            .with_context(|| format!("Failed to get owners for multisig '{name}' at {addr}"))?;
306        let threshold: u64 = contract
307            .getThreshold()
308            .block(self.block_id)
309            .call()
310            .await
311            .with_context(|| format!("Failed to get threshold for multisig '{name}' at {addr}"))?
312            .try_into()
313            .context("threshold exceeds u64")?;
314        Ok(MultisigDeployment {
315            address: addr,
316            version,
317            owners,
318            threshold,
319        })
320    }
321
322    pub(crate) async fn query_timelock(&self, addr: Address) -> Result<TimelockDeployment> {
323        let min_delay_secs: u64 = ITimelock::new(addr, self.provider)
324            .getMinDelay()
325            .block(self.block_id)
326            .call()
327            .await?
328            .try_into()
329            .context("min_delay exceeds u64")?;
330        let min_delay = Duration::from_secs(min_delay_secs);
331
332        Ok(TimelockDeployment::Deployed {
333            address: addr,
334            min_delay,
335        })
336    }
337}

395impl CollectedDeployment {
396    pub(crate) async fn collect(
397        rpc_url: Url,
398        network: Network,
399        addresses: DeploymentAddresses,
400    ) -> Result<Self> {
401        let provider = alloy::providers::ProviderBuilder::new().connect_http(rpc_url);
402
403        let block_number = provider
404            .get_block_number()
405            .await
406            .context("Failed to get block number")?;
407        let block = provider
408            .get_block_by_number(block_number.into())
409            .await
410            .context("Failed to get block")?
411            .context("Block not found")?;
412        let block_timestamp = block.header.timestamp;
413
414        let known = KnownAddresses::from_deployment(&addresses);
415        let querier = DeploymentQuerier::new(&provider, known, block_number);
416
417        let mut multisigs = BTreeMap::new();
418        for (name, addr) in &addresses.multisigs {
419            let deployment = querier.query_multisig(name, *addr).await?;
420            multisigs.insert(name.clone(), deployment);
421        }
422
423        let ops_timelock = match addresses.ops_timelock {
424            Some(addr) => querier
425                .query_timelock(addr)
426                .await
427                .with_context(|| format!("Failed to query OpsTimelock at {addr}"))?,
428            None => TimelockDeployment::NotYetDeployed,
429        };
430
431        let safe_exit_timelock = match addresses.safe_exit_timelock {
432            Some(addr) => querier
433                .query_timelock(addr)
434                .await
435                .with_context(|| format!("Failed to query SafeExitTimelock at {addr}"))?,
436            None => TimelockDeployment::NotYetDeployed,
437        };
438
439        let esp_token = match addresses.esp_token {
440            Some(addr) => querier.query_ownable(addr, ContractType::EspToken).await?,
441            None => OwnableDeployment::NotYetDeployed,
442        };
443        let fee_contract = match addresses.fee_contract {
444            Some(addr) => {
445                querier
446                    .query_ownable(addr, ContractType::FeeContract)
447                    .await?
448            },
449            None => OwnableDeployment::NotYetDeployed,
450        };
451        let light_client = match addresses.light_client {
452            Some(addr) => {
453                querier
454                    .query_ownable(addr, ContractType::LightClient)
455                    .await?
456            },
457            None => OwnableDeployment::NotYetDeployed,
458        };
459        let reward_claim = match addresses.reward_claim {
460            Some(addr) => {
461                querier
462                    .query_access_control(addr, ContractType::RewardClaim)
463                    .await?
464            },
465            None => AccessControlDeployment::NotYetDeployed,
466        };
467        let stake_table = match addresses.stake_table {
468            Some(addr) => {
469                querier
470                    .query_access_control(addr, ContractType::StakeTable)
471                    .await?
472            },
473            None => AccessControlDeployment::NotYetDeployed,
474        };
475
476        Ok(CollectedDeployment {
477            info: DeploymentInfo {
478                network,
479                multisigs,
480                ops_timelock,
481                safe_exit_timelock,
482                stake_table,
483                esp_token,
484                light_client,
485                fee_contract,
486                reward_claim,
487            },
488            block_number,
489            block_timestamp,
490        })
491    }
