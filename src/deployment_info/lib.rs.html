15#[derive(Debug, Clone, Copy, PartialEq, ValueEnum, Serialize, Deserialize)]
16#[serde(rename_all = "kebab-case")]
17pub(crate) enum Network {
18    Decaf,
19    Hoodi,
20    Mainnet,
21}
22
23impl Network {
24    fn default_rpc_url(&self) -> Url {
25        match self {
26            Network::Decaf => "https://ethereum-sepolia-rpc.publicnode.com",
27            Network::Hoodi => "https://ethereum-hoodi-rpc.publicnode.com",
28            Network::Mainnet => "https://ethereum-rpc.publicnode.com",
29        }
30        .parse()
31        .expect("hardcoded URL is valid")
32    }
33
34    pub(crate) fn etherscan_base_url(&self) -> &'static str {
35        match self {
36            Network::Decaf => "https://sepolia.etherscan.io",
37            Network::Hoodi => "https://hoodi.etherscan.io",
38            Network::Mainnet => "https://etherscan.io",
39        }
40    }
41
42    pub(crate) fn display_order(&self) -> u8 {
43        match self {
44            Network::Mainnet => 0,
45            Network::Decaf => 1,
46            Network::Hoodi => 2,
47        }
48    }
49}

61#[derive(Debug, Parser)]
62#[clap(
63    name = "deployment-info",
64    about = "Collect and output deployment information for Espresso Network contracts"
65)]
66struct Args {
67    #[clap(
68        long,
69        env = "ESPRESSO_SEQUENCER_L1_PROVIDER",
70        help = "RPC URL for L1 provider. Defaults to publicnode when --network is specified."
71    )]
72    rpc_url: Option<Url>,
73
74    #[clap(
75        long,
76        value_enum,
77        help = "Known network. If not specified, all networks are processed."
78    )]
79    network: Option<Network>,
80
81    #[clap(long, help = "Path to input .env file. Only valid with --network.")]
82    env_file: Option<PathBuf>,
83
84    #[clap(long, help = "Output file path. Only valid with --network.")]
85    output: Option<PathBuf>,
86
87    #[clap(
88        long,
89        help = "Print to stdout instead of writing to a file. Only valid with --network."
90    )]
91    stdout: bool,
92
93    #[clap(long, help = "Write files even if deployment info is unchanged.")]
94    force: bool,
95}

97pub(crate) fn get_crate_dir() -> PathBuf {
98    PathBuf::from(env!("CARGO_MANIFEST_DIR"))
99}

101pub async fn run() -> Result<()> {
102    tracing_subscriber::fmt()
103        .with_env_filter(
104            tracing_subscriber::EnvFilter::try_from_default_env()
105                .unwrap_or_else(|_| tracing_subscriber::EnvFilter::new("info")),
106        )
107        .init();
108
109    let args = Args::parse();
110    let crate_dir = get_crate_dir();
111
112    let update_readme = if let Some(network) = args.network {
113        let env_file = match args.env_file {
114            Some(path) => path,
115            None => crate_dir.join(format!("addresses/{}.env", network)),
116        };
117
118        let addresses = DeploymentAddresses::from_env_file(&env_file)
119            .context("Failed to load addresses from env file")?;
120
121        let rpc_url = match args.rpc_url {
122            Some(url) => url,
123            None => network.default_rpc_url(),
124        };
125
126        tracing::info!("Collecting deployment info for network: {network}");
127
128        let collected = CollectedDeployment::collect(rpc_url, network, addresses)
129            .await
130            .context("Failed to collect deployment info")?;
131
132        let has_custom_output = args.output.is_some();
133
134        if args.stdout {
135            print!("{}", collected.to_toml_string()?);
136        } else {
137            let output_path = match args.output {
138                Some(path) => path,
139                None => crate_dir.join(format!("deployments/{}.toml", network)),
140            };
141
142            collected
143                .write_toml(&output_path, args.force)
144                .context("Failed to write deployment info to file")?;
145        }
146
147        !args.stdout && !has_custom_output
148    } else {
149        if args.env_file.is_some() || args.output.is_some() || args.stdout {
150            bail!("--env-file, --output, and --stdout are only valid with --network");
151        }
152
153        for network in Network::value_variants() {
154            let env_file = crate_dir.join(format!("addresses/{}.env", network));
155
156            let addresses = DeploymentAddresses::from_env_file(&env_file)
157                .context("Failed to load addresses from env file")?;
158
159            let rpc_url = match &args.rpc_url {
160                Some(url) => url.clone(),
161                None => network.default_rpc_url(),
162            };
163
164            tracing::info!("Collecting deployment info for network: {network}");
165
166            let collected = CollectedDeployment::collect(rpc_url, *network, addresses)
167                .await
168                .context("Failed to collect deployment info")?;
169
170            let output_path = crate_dir.join(format!("deployments/{}.toml", network));
171            collected
172                .write_toml(&output_path, args.force)
173                .context("Failed to write deployment info to file")?;
174        }
175        true
176    };
177
178    if update_readme {
179        update_readme_from_deployment_files()?;
180    } else {
181        tracing::info!("Skipping README update");
182    }
183
184    Ok(())
185}

285            .deployer(provider.clone())

289            .genesis_st_state(genesis_stake)
290            .blocks_per_epoch(100)

296            .initial_token_supply(U256::from(3590000000u64))
