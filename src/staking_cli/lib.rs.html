166#[derive(Debug, Error)]
167pub(crate) enum SignerConfigError {
168    #[error(
169        "Multiple signers provided: {provided}. Only one of --mnemonic, --private-key, or \
170         --ledger can be specified."
171    )]
172    MultipleSigners { provided: String },
173
174    #[error("--account-index is required when using --{signer}")]
175    MissingAccountIndex { signer: &'static str },
176
177    #[error("Either --mnemonic, --private-key, or --ledger flag must be provided")]
178    NoSigner,
179}

181#[derive(Clone, Debug)]
182pub(crate) enum ValidSignerConfig {
183    Mnemonic {
184        mnemonic: String,
185        account_index: u32,
186    },
187    PrivateKey {
188        private_key: String,
189    },
190    Ledger {
191        account_index: usize,
192    },
193}
194
195impl TryFrom<SignerConfig> for ValidSignerConfig {
196    type Error = SignerConfigError;
197
198    fn try_from(config: SignerConfig) -> std::result::Result<Self, SignerConfigError> {
199        let signers: Vec<&str> = [
200            config.ledger.then_some("--ledger"),
201            config.private_key.as_ref().map(|_| "--private-key"),
202            config.mnemonic.as_ref().map(|_| "--mnemonic"),
203        ]
204        .into_iter()
205        .flatten()
206        .collect();
207
208        if signers.len() > 1 {
209            return Err(SignerConfigError::MultipleSigners {
210                provided: signers.join(", "),
211            });
212        }
213
214        if config.ledger {
215            let account_index = config
216                .account_index
217                .ok_or(SignerConfigError::MissingAccountIndex { signer: "ledger" })?;
218            Ok(ValidSignerConfig::Ledger {
219                account_index: account_index as usize,
220            })
221        } else if let Some(private_key) = config.private_key {
222            Ok(ValidSignerConfig::PrivateKey { private_key })
223        } else if let Some(mnemonic) = config.mnemonic {
224            let account_index = config
225                .account_index
226                .ok_or(SignerConfigError::MissingAccountIndex { signer: "mnemonic" })?;
227            Ok(ValidSignerConfig::Mnemonic {
228                mnemonic,
229                account_index,
230            })
231        } else {
232            Err(SignerConfigError::NoSigner)
233        }
234    }
235}

237impl ValidSignerConfig {
238    pub async fn wallet(&self) -> Result<EthereumWallet> {
239        match self {
240            ValidSignerConfig::Mnemonic {
241                mnemonic,
242                account_index,
243            } => {
244                let signer = MnemonicBuilder::<English>::default()
245                    .phrase(mnemonic)
246                    .index(*account_index)?
247                    .build()?;
248                Ok(EthereumWallet::from(signer))
249            },
250            ValidSignerConfig::PrivateKey { private_key } => {
251                let signer: PrivateKeySigner = private_key.parse()?;
252                Ok(EthereumWallet::from(signer))
253            },
254            ValidSignerConfig::Ledger { account_index } => {
255                let signer = connect_ledger(*account_index).await?;
256                Ok(EthereumWallet::from(signer))
257            },
258        }
259    }
260}

283impl Config {
284    pub fn apply_env_var_overrides(self) -> Result<Self> {
285        let mut config = self.clone();
286        if self.stake_table_address == Address::ZERO {
287            let stake_table_env_var = "ESPRESSO_STAKE_TABLE_PROXY_ADDRESS";
288            if let Ok(stake_table_address) = std::env::var(stake_table_env_var) {
289                config.stake_table_address = stake_table_address.parse()?;
290                tracing::info!(
291                    "Using stake table address from env {stake_table_env_var}: \
292                     {stake_table_address}",
293                );
294            }
295        }
296        Ok(config)
297    }
298}
