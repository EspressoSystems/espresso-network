55impl Args {
56    fn config_path(&self) -> PathBuf {
57        // If the user provided a config path, use it.
58        self.config_path.clone().unwrap_or_else(|| {
59            // Otherwise create a config.toml in a platform specific config directory.
60            //
61            // (empty) qualifier, espresso organization, and application name
62            // see more <https://docs.rs/directories/5.0.1/directories/struct.ProjectDirs.html#method.from>
63            let project_dir =
64                directories::ProjectDirs::from("", "espresso", "espresso-staking-cli");
65            let basename = "config.toml";
66            if let Some(project_dir) = project_dir {
67                project_dir.config_dir().to_path_buf().join(basename)
68            } else {
69                // In the unlikely case that we can't find the config directory,
70                // create the config file in the current directory and issue a
71                // warning.
72                tracing::warn!("Unable to find config directory, using current directory");
73                basename.into()
74            }
75        })
76    }
77
78    fn config_dir(&self) -> PathBuf {
79        if let Some(path) = self.config_path().parent() {
80            path.to_path_buf()
81        } else {
82            // Try to use the current directory
83            PathBuf::from(".")
84        }
85    }
86}

106// Events containing custom structs do not get the Debug derive, due to a bug in
107// foundry. We instead format those types nicely with tagged base64.
108fn decode_and_display_logs(logs: &[Log]) {
109    for log in logs {
110        if let Ok(decoded) = StakeTableV2Events::decode_log(log.as_ref()) {
111            match &decoded.data {
112                StakeTableV2Events::ValidatorRegistered(e) => output_success(format!(
113                    "event: ValidatorRegistered {{ account: {}, blsVk: {}, schnorrVk: {}, \
114                     commission: {} }}",
115                    e.account,
116                    BLSPubKey::from(e.blsVk),
117                    StateVerKey::from(e.schnorrVk),
118                    e.commission
119                )),
120                StakeTableV2Events::ValidatorRegisteredV2(e) => output_success(format!(
121                    "event: ValidatorRegisteredV2 {{ account: {}, blsVK: {}, schnorrVK: {}, \
122                     commission: {}, metadataUri: {} }}",
123                    e.account,
124                    BLSPubKey::from(e.blsVK),
125                    StateVerKey::from(e.schnorrVK),
126                    e.commission,
127                    e.metadataUri
128                )),
129                StakeTableV2Events::Delegated(e) => output_success(format!("event: {e:?}")),
130                StakeTableV2Events::Undelegated(e) => output_success(format!("event: {e:?}")),
131                StakeTableV2Events::UndelegatedV2(e) => output_success(format!("event: {e:?}")),
132                StakeTableV2Events::ValidatorExit(e) => output_success(format!("event: {e:?}")),
133                StakeTableV2Events::ValidatorExitV2(e) => output_success(format!("event: {e:?}")),
134                StakeTableV2Events::ConsensusKeysUpdated(e) => output_success(format!(
135                    "event: ConsensusKeysUpdated {{ account: {}, blsVK: {}, schnorrVK: {} }}",
136                    e.account,
137                    BLSPubKey::from(e.blsVK),
138                    StateVerKey::from(e.schnorrVK)
139                )),
140                StakeTableV2Events::ConsensusKeysUpdatedV2(e) => output_success(format!(
141                    "event: ConsensusKeysUpdatedV2 {{ account: {}, blsVK: {}, schnorrVK: {} }}",
142                    e.account,
143                    BLSPubKey::from(e.blsVK),
144                    StateVerKey::from(e.schnorrVK)
145                )),
146                StakeTableV2Events::CommissionUpdated(e) => output_success(format!("event: {e:?}")),
147                StakeTableV2Events::MetadataUriUpdated(e) => output_success(format!(
148                    "event: MetadataUriUpdated {{ validator: {}, metadataUri: {} }}",
149                    e.validator, e.metadataUri
150                )),
151                StakeTableV2Events::Withdrawal(e) => output_success(format!("event: {e:?}")),
152                StakeTableV2Events::WithdrawalClaimed(e) => output_success(format!("event: {e:?}")),
153                StakeTableV2Events::ValidatorExitClaimed(e) => {
154                    output_success(format!("event: {e:?}"))
155                },
156
157                _ => {},
158            }
159        } else if let Ok(decoded) = EspTokenEvents::decode_log(log.as_ref()) {
160            match &decoded.data {
161                EspTokenEvents::Transfer(e) => output_success(format!("event: {e:?}")),
162                EspTokenEvents::Approval(e) => output_success(format!("event: {e:?}")),
163                _ => {},
164            }
165        } else if let Ok(decoded) = RewardClaimEvents::decode_log(log.as_ref()) {
166            if let RewardClaimEvents::RewardsClaimed(e) = &decoded.data {
167                output_success(format!("event: {e:?}"));
168            }
169        }
170    }
171}

173fn resolve_node_signatures(
174    signature_args: &crate::signature::NodeSignatureArgs,
175    export_calldata: bool,
176    wallet: Option<&EthereumWallet>,
177    sender_address: Option<Address>,
178) -> Result<NodeSignatures> {
179    if export_calldata {
180        let input = NodeSignatureInput::try_from((signature_args.clone(), sender_address))?;
181        NodeSignatures::try_from(input)
182    } else {
183        let wallet = wallet.ok_or_else(|| anyhow::anyhow!("Signer configuration required"))?;
184        let address = NetworkWallet::<Ethereum>::default_signer_address(wallet);
185        let input = NodeSignatureInput::try_from((signature_args.clone(), Some(address)))?;
186        NodeSignatures::try_from((input, wallet))
187    }
188}

190pub async fn run() -> Result<()> {
191    let mut cli = Args::parse();
192
193    // initialize the logging ASAP so we don't accidentally hide any messages.
194    cli.config.logging.clone().unwrap_or_default().init();
195
196    let config_path = cli.config_path();
197    // Get config file
198    let config = if let Ok(f) = std::fs::read_to_string(&config_path) {
199        // parse toml
200        match toml::from_str::<Config>(&f) {
201            Ok(config) => config.merge(&mut cli.config),
202            Err(err) => {
203                // This is a user error print the hopefully helpful error
204                // message without backtrace and exit.
205                exit_err(
206                    format!("Error in configuration file at {}", config_path.display()),
207                    err,
208                );
209            },
210        }
211    } else {
212        // If there is no config file return only config parsed from clap
213        Config::from(&mut cli.config)
214    };
215
216    if config.token_address.is_some() {
217        tracing::warn!("The `--token_address` argument is no longer necessary , and ignored");
218    };
219
220    // Run the init command first because config values required by other
221    // commands are not present.
222    match config.commands {
223        Commands::Init {
224            mnemonic,
225            private_key,
226            account_index,
227            ledger,
228            network,
229        } => {
230            let config_template = match network {
231                crate::Network::Mainnet => include_str!("../config.mainnet.toml"),
232                crate::Network::Decaf => include_str!("../config.decaf.toml"),
233                crate::Network::Local => include_str!("../config.demo-native.toml"),
234            };
235            let mut config = toml::from_str::<Config>(config_template)?;
236            config.signer.mnemonic = mnemonic;
237            config.signer.private_key = private_key;
238            config.signer.account_index = Some(account_index);
239            config.signer.ledger = ledger;
240
241            // Create directory where config file will be saved
242            std::fs::create_dir_all(cli.config_dir()).unwrap_or_else(|err| {
243                exit_err("failed to create config directory", err);
244            });
245
246            // Save the config file
247            std::fs::write(&config_path, toml::to_string(&config)?)
248                .unwrap_or_else(|err| exit_err("failed to write config file", err));
249
250            println!("New config file saved to {}", config_path.display());
251            return Ok(());
252        },
253        Commands::Purge { force } => {
254            // Check if the file exists
255            if !config_path.exists() {
256                println!("Config file not found at {}", config_path.display());
257                return Ok(());
258            }
259            if !force {
260                // Get a confirmation from the user before removing the config file.
261                println!(
262                    "Are you sure you want to remove the config file at {}? [y/N]",
263                    config_path.display()
264                );
265                let mut input = String::new();
266                std::io::stdin().read_line(&mut input).unwrap();
267                if !input.trim().to_lowercase().starts_with('y') {
268                    println!("Aborted");
269                    return Ok(());
270                }
271            }
272            // Remove the config file
273            std::fs::remove_file(&config_path).unwrap_or_else(|err| {
274                exit_err("failed to remove config file", err);
275            });
276
277            println!("Config file removed from {}", config_path.display());
278            return Ok(());
279        },
280        Commands::Config => {
281            if !config_path.exists() {
282                println!("No config file found at {}", config_path.display());
283                println!("Run `staking-cli init --network <network>` to create one.");
284                return Ok(());
285            }
286            println!("Config file at {}\n", config_path.display());
287            let mut config = config;
288            config.signer.mnemonic = config.signer.mnemonic.map(|_| "***".to_string());
289            config.signer.private_key = config.signer.private_key.map(|_| "***".to_string());
290            println!("{}", toml::to_string_pretty(&config)?);
291            return Ok(());
292        },
293        Commands::Version => {
294            println!("staking-cli version: {}", env!("CARGO_PKG_VERSION"));
295            println!("{}", git_version::git_version!(prefix = "git rev: "));
296            println!("OS: {}", System::long_os_version().unwrap_or_default());
297            println!("Arch: {}", System::cpu_arch());
298            return Ok(());
299        },
300        Commands::ExportNodeSignatures {
301            address,
302            consensus_private_key,
303            state_private_key,
304            output_args,
305        } => {
306            let destination = NodeSignatureDestination::try_from(output_args)?;
307
308            let payload = NodeSignatures::create(
309                address,
310                &consensus_private_key.into(),
311                &StateKeyPair::from_sign_key(state_private_key),
312            );
313
314            payload.handle_output(destination)?;
315            return Ok(());
316        },
317        Commands::PreviewMetadata { metadata_uri } => {
318            let url = url::Url::parse(&metadata_uri)
319                .with_context(|| format!("Invalid URL: {metadata_uri}"))?;
320            let metadata = fetch_metadata(&url).await?;
321            output_success(serde_json::to_string_pretty(&metadata)?);
322            return Ok(());
323        },
324        _ => {}, // Other commands handled after shared setup.
325    }
326
327    // When the staking CLI is used for our testnet, the env var names are different.
328    let config = config.apply_env_var_overrides()?;
329
330    // Commands that don't need a signer
331    if let Commands::StakeTable {
332        l1_block_number,
333        compact,
334    } = config.commands
335    {
336        let provider = ProviderBuilder::new().connect_http(config.rpc_url.clone());
337        let query_block = l1_block_number.unwrap_or(BlockId::latest());
338        let l1_block = provider.get_block(query_block).await?.unwrap_or_else(|| {
339            exit_err("Failed to get block {query_block}", "Block not found");
340        });
341        let l1_block_resolved = l1_block.header.number;
342        tracing::info!("Getting stake table info at block {l1_block_resolved}");
343        let stake_table = stake_table_info(
344            config.rpc_url.clone(),
345            config.stake_table_address,
346            l1_block_resolved,
347        )
348        .await?;
349        display_stake_table(stake_table, compact)?;
350        return Ok(());
351    }
352
353    // Clap serde will put default value if they aren't set. We check some
354    // common configuration mistakes.
355    if config.stake_table_address == Address::ZERO {
356        exit("Stake table address is not set use --stake-table-address or STAKE_TABLE_ADDRESS")
357    };
358
359    let stake_table_addr = config.stake_table_address;
360
361    // For export_calldata mode, we may not need a signer for most commands.
362    // We create the provider without a wallet first for token address fetching
363    // and contract version detection.
364    let readonly_provider = ProviderBuilder::new().connect_http(config.rpc_url.clone());
365
366    // Check if we need token address for this command
367    let token_addr = if config.commands.needs_token_address() {
368        fetch_token_address(config.rpc_url.clone(), stake_table_addr).await?
369    } else {
370        Address::ZERO
371    };
372
373    let wallet =
374        if let Ok(signer_config) = TryInto::<ValidSignerConfig>::try_into(config.signer.clone()) {
375            signer_config.wallet().await.ok()
376        } else {
377            None
378        };
379
380    // Commands that just read from chain
381    if let Commands::Account = config.commands {
382        let account = NetworkWallet::<Ethereum>::default_signer_address(
383            wallet.as_ref().context("Signer configuration required")?,
384        );
385        println!("{account}");
386        return Ok(());
387    }
388
389    if let Commands::TokenBalance { address } = config.commands {
390        let address = address
391            .or_from_wallet(wallet.as_ref())
392            .context("Address required - provide --address or configure a signer")?;
393        let token = EspToken::new(token_addr, &readonly_provider);
394        let balance = format_esp(token.balanceOf(address).call().await?);
395        output_success(format!("Token balance for {address}: {balance}"));
396        return Ok(());
397    }
398
399    if let Commands::TokenAllowance { owner } = config.commands {
400        let owner = owner
401            .or_from_wallet(wallet.as_ref())
402            .context("Owner address required - provide --owner or configure a signer")?;
403        let token = EspToken::new(token_addr, &readonly_provider);
404        let allowance = format_esp(
405            token
406                .allowance(owner, config.stake_table_address)
407                .call()
408                .await?,
409        );
410        output_success(format!(
411            "Stake table token allowance for {owner}: {allowance}"
412        ));
413        return Ok(());
414    }
415
416    if let Commands::UnclaimedRewards { address } = config.commands {
417        let address = address
418            .or_from_wallet(wallet.as_ref())
419            .context("Address required - provide --address or configure a signer")?;
420        let espresso_url = config.espresso_url.ok_or_else(|| {
421            anyhow::anyhow!("espresso_url not set, use --espresso-url or ESPRESSO_URL")
422        })?;
423        let unclaimed = crate::claim::unclaimed_rewards(
424            &readonly_provider,
425            stake_table_addr,
426            espresso_url,
427            address,
428        )
429        .await
430        .unwrap_or_else(|err| {
431            exit_err("Failed to check unclaimed rewards", err);
432        });
433        println!("{}", format_esp(unclaimed));
434        return Ok(());
435    }
436
437    if let Commands::StakeForDemo {
438        num_validators,
439        num_delegators_per_validator,
440        delegation_config,
441    } = config.commands
442    {
443        tracing::info!(
444            "Staking for demo with {num_validators} validators and config {delegation_config}"
445        );
446        stake_for_demo(
447            &config,
448            num_validators,
449            num_delegators_per_validator,
450            delegation_config,
451        )
452        .await
453        .unwrap();
454        return Ok(());
455    }
456
457    // Build Transaction for state-changing commands
458    let tx: Transaction = match &config.commands {
459        Commands::RegisterValidator {
460            signature_args,
461            commission,
462            metadata_uri_args,
463        } => {
464            let version = fetch_stake_table_version(&readonly_provider, stake_table_addr).await?;
465            if config.export_calldata && matches!(version, StakeTableContractVersion::V1) {
466                anyhow::bail!(
467                    "Calldata export is not supported for V1 stake table contracts. V1 is \
468                     deprecated."
469                );
470            }
471            let payload = resolve_node_signatures(
472                signature_args,
473                config.export_calldata,
474                wallet.as_ref(),
475                config.sender_address,
476            )?;
477            let metadata_uri: MetadataUri = metadata_uri_args.clone().try_into()?;
478
479            // Validate metadata URI if present and validation not skipped
480            if let Some(url) = metadata_uri.url() {
481                if !config.skip_metadata_validation {
482                    validate_metadata_uri(url, &payload.bls_vk)
483                        .await
484                        .context("use --skip-metadata-validation to skip")?;
485                }
486            }
487
488            Transaction::RegisterValidator {
489                stake_table: stake_table_addr,
490                commission: *commission,
491                metadata_uri,
492                payload,
493                version,
494            }
495        },
496        Commands::UpdateConsensusKeys { signature_args } => {
497            let version = fetch_stake_table_version(&readonly_provider, stake_table_addr).await?;
498            if config.export_calldata && matches!(version, StakeTableContractVersion::V1) {
499                anyhow::bail!(
500                    "Calldata export is not supported for V1 stake table contracts. V1 is \
501                     deprecated."
502                );
503            }
504            if let Some(w) = wallet.as_ref() {
505                let addr = NetworkWallet::<Ethereum>::default_signer_address(w);
506                tracing::info!("Updating validator {} with new keys", addr);
507            }
508            let payload = resolve_node_signatures(
509                signature_args,
510                config.export_calldata,
511                wallet.as_ref(),
512                config.sender_address,
513            )?;
514            Transaction::UpdateConsensusKeys {
515                stake_table: stake_table_addr,
516                payload,
517                version,
518            }
519        },
520        Commands::DeregisterValidator {} => Transaction::DeregisterValidator {
521            stake_table: stake_table_addr,
522        },
523        Commands::UpdateCommission { new_commission } => Transaction::UpdateCommission {
524            stake_table: stake_table_addr,
525            new_commission: *new_commission,
526        },
527        Commands::UpdateMetadataUri {
528            metadata_uri_args,
529            consensus_public_key,
530        } => {
531            let metadata_uri: MetadataUri = metadata_uri_args.clone().try_into()?;
532
533            // Validate metadata URI if present and validation not skipped
534            if let Some(url) = metadata_uri.url() {
535                if !config.skip_metadata_validation {
536                    let bls_vk = consensus_public_key.ok_or_else(|| {
537                        anyhow::anyhow!(
538                            "--consensus-public-key is required for metadata validation (use \
539                             --skip-metadata-validation to skip)"
540                        )
541                    })?;
542                    validate_metadata_uri(url, &bls_vk)
543                        .await
544                        .context("use --skip-metadata-validation to skip")?;
545                }
546            }
547
548            Transaction::UpdateMetadataUri {
549                stake_table: stake_table_addr,
550                metadata_uri,
551            }
552        },
553        Commands::Approve { amount } => Transaction::Approve {
554            token: token_addr,
555            spender: stake_table_addr,
556            amount: *amount,
557        },
558        Commands::Delegate {
559            validator_address,
560            amount,
561        } => Transaction::Delegate {
562            stake_table: stake_table_addr,
563            validator: *validator_address,
564            amount: *amount,
565        },
566        Commands::Undelegate {
567            validator_address,
568            amount,
569        } => Transaction::Undelegate {
570            stake_table: stake_table_addr,
571            validator: *validator_address,
572            amount: *amount,
573        },
574        Commands::ClaimWithdrawal { validator_address } => Transaction::ClaimWithdrawal {
575            stake_table: stake_table_addr,
576            validator: *validator_address,
577        },
578        Commands::ClaimValidatorExit { validator_address } => Transaction::ClaimValidatorExit {
579            stake_table: stake_table_addr,
580            validator: *validator_address,
581        },
582        Commands::ClaimRewards {} => {
583            let espresso_url = config.espresso_url.clone().ok_or_else(|| {
584                anyhow::anyhow!("espresso_url not set, use --espresso-url or ESPRESSO_URL")
585            })?;
586            let claimer_address = if config.export_calldata {
587                config.sender_address.ok_or_else(|| {
588                    anyhow::anyhow!(
589                        "claim-rewards with --export-calldata requires --sender-address"
590                    )
591                })?
592            } else {
593                NetworkWallet::<Ethereum>::default_signer_address(
594                    wallet.as_ref().context("Signer configuration required")?,
595                )
596            };
597            fetch_claim_rewards_inputs(
598                &readonly_provider,
599                stake_table_addr,
600                &espresso_url,
601                claimer_address,
602            )
603            .await?
604            .ok_or_else(|| anyhow::anyhow!("No reward claim data found for address"))?
605        },
606        Commands::Transfer { amount, to } => Transaction::Transfer {
607            token: token_addr,
608            to: *to,
609            amount: *amount,
610        },
611        Commands::Version
612        | Commands::Config
613        | Commands::Init { .. }
614        | Commands::Purge { .. }
615        | Commands::StakeTable { .. }
616        | Commands::Account
617        | Commands::UnclaimedRewards { .. }
618        | Commands::TokenBalance { .. }
619        | Commands::TokenAllowance { .. }
620        | Commands::ExportNodeSignatures { .. }
621        | Commands::PreviewMetadata { .. }
622        | Commands::StakeForDemo { .. } => {
623            unreachable!("Non-state-change commands are handled earlier in the function")
624        },
625    };
626
627    // Validate even for export mode to fail early if the transaction would fail on-chain.
628    tx.validate_delegate_amount(&readonly_provider).await?;
629
630    // Single code path for both export and execute modes
631    if config.export_calldata {
632        if config.skip_simulation {
633            output_warn("Skipping calldata validation (--skip-simulation)");
634        } else {
635            let sender = config.sender_address.ok_or_else(|| {
636                anyhow::anyhow!(
637                    "--sender-address is required for calldata simulation (use --skip-simulation \
638                     to skip)"
639                )
640            })?;
641            tx.simulate(&readonly_provider, sender).await?;
642        }
643        let (to, data) = tx.calldata();
644        return output_calldata(&CalldataInfo::new(to, data), &config.output);
645    }
646
647    // For execution, we need the wallet
648    let wallet = wallet.ok_or_else(|| {
649        anyhow::anyhow!("Signer configuration required for transaction execution")
650    })?;
651    let account = NetworkWallet::<Ethereum>::default_signer_address(&wallet);
652
653    // Check that our Ethereum balance isn't zero before proceeding.
654    let balance = readonly_provider.get_balance(account).await?;
655    if balance.is_zero() {
656        exit(format!(
657            "zero Ethereum balance for account {account}, please fund account"
658        ));
659    }
660
661    // Create provider with wallet for signing
662    let provider = ProviderBuilder::new()
663        .wallet(wallet)
664        .connect_http(config.rpc_url.clone());
665
666    // Execute the state change
667    let pending_tx = tx
668        .send(&provider)
669        .await
670        .unwrap_or_else(|err| exit_err("Error", err));
671
672    match pending_tx.get_receipt().await {
673        Ok(receipt) => {
674            output_success(format!(
675                "Success! transaction hash: {}",
676                receipt.transaction_hash
677            ));
678            decode_and_display_logs(receipt.inner.logs());
679            Ok(())
680        },
681        Err(err) => exit_err("Failed", err),
682    }
683}
