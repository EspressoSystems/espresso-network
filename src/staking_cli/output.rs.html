14pub fn output_success(msg: impl AsRef<str>) {
15    if std::env::var("RUST_LOG_FORMAT") == Ok("json".to_string()) {

39pub(crate) fn output_calldata(
40    info: &CalldataInfo,
41    output: &OutputArgs,
42    chain_id: u64,
43) -> Result<()> {
44    let fmt = output.format.unwrap_or(SerializationFormat::Safe);
45    match fmt {
46        SerializationFormat::Safe => {
47            output_safe_tx_builder(info, output.output.as_deref(), chain_id)?;
48        },
49        SerializationFormat::Json | SerializationFormat::Toml => {
50            // CalldataInfo derives Serialize with function_info skipped,
51            // producing the legacy {to, data, value} format.
52            let text = match fmt {
53                SerializationFormat::Toml => toml::to_string_pretty(info)?,
54                _ => serde_json::to_string_pretty(info)?,
55            };
56            if let Some(path) = &output.output {
57                std::fs::write(path, &text)?;
58                output_success(format!("Calldata written to {}", path.display()));
59            } else {
60                output_success(&text);
61            }
62        },
63    }
64
65    Ok(())
66}
