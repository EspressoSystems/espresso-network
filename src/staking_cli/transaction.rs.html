1//! State-changing operations for staking.
2//!
3//! The [`Transaction`] enum provides a single dispatch point for all state-changing operations.
4//! Each variant uses the same [`Transaction::calldata`] method for both execute and export modes,
5//! ensuring identical calldata generation.
6

36#[derive(Clone)]
37pub enum Transaction {
38    Approve {
39        token: Address,
40        spender: Address,
41        amount: U256,
42    },
43    Delegate {
44        stake_table: Address,
45        validator: Address,
46        amount: U256,
47    },
48    Undelegate {
49        stake_table: Address,
50        validator: Address,
51        amount: U256,
52    },
53    ClaimWithdrawal {
54        stake_table: Address,
55        validator: Address,
56    },
57    ClaimValidatorExit {
58        stake_table: Address,
59        validator: Address,
60    },
61    ClaimRewards {
62        reward_claim: Address,
63        lifetime_rewards: U256,
64        auth_data: Bytes,
65    },
66    RegisterValidator {
67        stake_table: Address,
68        commission: Commission,
69        metadata_uri: MetadataUri,
70        payload: NodeSignatures,
71        version: StakeTableContractVersion,
72    },
73    UpdateConsensusKeys {
74        stake_table: Address,
75        payload: NodeSignatures,
76        version: StakeTableContractVersion,
77    },
78    DeregisterValidator {
79        stake_table: Address,
80    },
81    UpdateCommission {
82        stake_table: Address,
83        new_commission: Commission,
84    },
85    UpdateMetadataUri {
86        stake_table: Address,
87        metadata_uri: MetadataUri,
88    },
89    Transfer {
90        token: Address,
91        to: Address,
92        amount: U256,
93    },
94}

96impl Transaction {
97    /// Returns the contract address and encoded calldata for this state change.
98    pub fn calldata(self) -> (Address, Bytes) {
99        match self {
100            Self::Approve {
101                token,
102                spender,
103                amount,
104            } => (
105                token,
106                approveCall {
107                    spender,
108                    value: amount,
109                }
110                .abi_encode()
111                .into(),
112            ),
113            Self::Delegate {
114                stake_table,
115                validator,
116                amount,
117            } => (
118                stake_table,
119                delegateCall { validator, amount }.abi_encode().into(),
120            ),
121            Self::Undelegate {
122                stake_table,
123                validator,
124                amount,
125            } => (
126                stake_table,
127                undelegateCall { validator, amount }.abi_encode().into(),
128            ),
129            Self::ClaimWithdrawal {
130                stake_table,
131                validator,
132            } => (
133                stake_table,
134                claimWithdrawalCall { validator }.abi_encode().into(),
135            ),
136            Self::ClaimValidatorExit {
137                stake_table,
138                validator,
139            } => (
140                stake_table,
141                claimValidatorExitCall { validator }.abi_encode().into(),
142            ),
143            Self::ClaimRewards {
144                reward_claim,
145                lifetime_rewards,
146                auth_data,
147            } => (
148                reward_claim,
149                claimRewardsCall {
150                    lifetimeRewards: lifetime_rewards,
151                    authData: auth_data,
152                }
153                .abi_encode()
154                .into(),
155            ),
156            Self::RegisterValidator {
157                stake_table,
158                commission,
159                metadata_uri,
160                payload,
161                version,
162            } => match version {
163                StakeTableContractVersion::V1 => (
164                    stake_table,
165                    registerValidatorCall::from((
166                        G2PointSol::from(payload.bls_vk),
167                        EdOnBN254PointSol::from(payload.schnorr_vk),
168                        G1PointSol::from(payload.bls_signature).into(),
169                        commission.to_evm(),
170                    ))
171                    .abi_encode()
172                    .into(),
173                ),
174                StakeTableContractVersion::V2 => (
175                    stake_table,
176                    registerValidatorV2Call::from((
177                        G2PointSol::from(payload.bls_vk),
178                        EdOnBN254PointSol::from(payload.schnorr_vk),
179                        G1PointSol::from(payload.bls_signature).into(),
180                        StateSignatureSol::from(payload.schnorr_signature).into(),
181                        commission.to_evm(),
182                        metadata_uri.to_string(),
183                    ))
184                    .abi_encode()
185                    .into(),
186                ),
187            },
188            Self::UpdateConsensusKeys {
189                stake_table,
190                payload,
191                version,
192            } => match version {
193                StakeTableContractVersion::V1 => (
194                    stake_table,
195                    updateConsensusKeysCall::from((
196                        G2PointSol::from(payload.bls_vk),
197                        EdOnBN254PointSol::from(payload.schnorr_vk),
198                        G1PointSol::from(payload.bls_signature).into(),
199                    ))
200                    .abi_encode()
201                    .into(),
202                ),
203                StakeTableContractVersion::V2 => (
204                    stake_table,
205                    updateConsensusKeysV2Call::from((
206                        G2PointSol::from(payload.bls_vk),
207                        EdOnBN254PointSol::from(payload.schnorr_vk),
208                        G1PointSol::from(payload.bls_signature).into(),
209                        StateSignatureSol::from(payload.schnorr_signature).into(),
210                    ))
211                    .abi_encode()
212                    .into(),
213                ),
214            },
215            Self::DeregisterValidator { stake_table } => {
216                (stake_table, deregisterValidatorCall {}.abi_encode().into())
217            },
218            Self::UpdateCommission {
219                stake_table,
220                new_commission,
221            } => (
222                stake_table,
223                updateCommissionCall {
224                    newCommission: new_commission.to_evm(),
225                }
226                .abi_encode()
227                .into(),
228            ),
229            Self::UpdateMetadataUri {
230                stake_table,
231                metadata_uri,
232            } => (
233                stake_table,
234                updateMetadataUriCall {
235                    metadataUri: metadata_uri.to_string(),
236                }
237                .abi_encode()
238                .into(),
239            ),
240            Self::Transfer { token, to, amount } => (
241                token,
242                transferCall { to, value: amount }.abi_encode().into(),
243            ),
244        }
245    }
246
247    fn to_transaction_request(&self) -> TransactionRequest {
248        let (to, data) = self.clone().calldata();
249        TransactionRequest::default()
250            .to(to)
251            .input(TransactionInput::new(data))
252    }

254    /// Validates the delegate amount against the minimum required by the contract.
255    ///
256    /// Currently only validates `Delegate` transactions; other variants pass through.
257    pub async fn validate_delegate_amount(&self, provider: &impl Provider) -> Result<()> {
258        if let Self::Delegate {
259            stake_table,
260            amount,
261            ..
262        } = self
263        {
264            use hotshot_contract_adapter::sol_types::StakeTableV2;
265            let st = StakeTableV2::new(*stake_table, provider);
266            let version: StakeTableContractVersion = st.getVersion().call().await?.try_into()?;
267            if let StakeTableContractVersion::V2 = version {
268                let min_amount = st.minDelegateAmount().call().await?;
269                if amount < &min_amount {
270                    bail!(
271                        "delegation amount {} is below minimum of {}",
272                        format_esp(*amount),
273                        format_esp(min_amount)
274                    );
275                }
276            }
277        }
278        Ok(())
279    }

281    fn decode_revert<T>(&self, result: impl DecodeRevert<T>) -> Result<T> {
282        match self {
283            Self::Approve { .. } | Self::Transfer { .. } => {
284                result.maybe_decode_revert::<EspTokenErrors>()
285            },
286            Self::ClaimRewards { .. } => result.maybe_decode_revert::<RewardClaimErrors>(),
287            Self::Delegate { .. }
288            | Self::Undelegate { .. }
289            | Self::ClaimWithdrawal { .. }
290            | Self::ClaimValidatorExit { .. }
291            | Self::RegisterValidator { .. }
292            | Self::UpdateConsensusKeys { .. }
293            | Self::DeregisterValidator { .. }
294            | Self::UpdateCommission { .. }
295            | Self::UpdateMetadataUri { .. } => result.maybe_decode_revert::<StakeTableV2Errors>(),
296        }
297    }

299    pub async fn simulate(&self, provider: &impl Provider, from: Address) -> Result<()> {
300        let tx = self.to_transaction_request().from(from);
301        let result = provider.call(tx).await;
302        self.decode_revert(result)
303            .context("Transaction simulation failed")?;
304        Ok(())
305    }

307    fn log_intent(&self) {
308        match self {
309            Self::Approve {
310                spender, amount, ..
311            } => {
312                tracing::info!("approve {} for {}", format_esp(*amount), spender);
313            },
314            Self::Delegate {
315                validator, amount, ..
316            } => {
317                tracing::info!("delegate {} to {}", format_esp(*amount), validator);
318            },
319            Self::Undelegate {
320                validator, amount, ..
321            } => {
322                tracing::info!("undelegate {} from {}", format_esp(*amount), validator);
323            },
324            Self::ClaimWithdrawal { validator, .. } => {
325                tracing::info!("claiming withdrawal for {}", validator);
326            },
327            Self::ClaimValidatorExit { validator, .. } => {
328                tracing::info!("claiming validator exit for {}", validator);
329            },
330            Self::ClaimRewards { reward_claim, .. } => {
331                tracing::info!("claiming rewards from {}", reward_claim);
332            },
333            Self::RegisterValidator {
334                payload,
335                commission,
336                ..
337            } => {
338                tracing::info!(
339                    "register validator {} with commission {}",
340                    payload.address,
341                    commission
342                );
343            },
344            Self::UpdateConsensusKeys { payload, .. } => {
345                tracing::info!("updating consensus keys for {}", payload.address);
346            },
347            Self::DeregisterValidator { .. } => {
348                tracing::info!("deregistering validator");
349            },
350            Self::UpdateCommission { new_commission, .. } => {
351                tracing::info!("updating commission to {}", new_commission);
352            },
353            Self::UpdateMetadataUri { .. } => {
354                tracing::info!("updating metadata URI");
355            },
356            Self::Transfer { to, amount, .. } => {
357                tracing::info!("transferring {} to {}", format_esp(*amount), to);
358            },
359        }
360    }
361
362    pub async fn send(
363        &self,
364        provider: impl Provider,
365    ) -> Result<PendingTransactionBuilder<Ethereum>> {
366        self.log_intent();
367        let tx = self.to_transaction_request();
368        let pending = provider.send_transaction(tx).await;
369        self.decode_revert(pending)
370    }
