1//! State-changing operations for staking.
2//!
3//! The [`Transaction`] enum provides a single dispatch point for all state-changing operations.
4//! Each variant uses the same [`Transaction::calldata`] method for both execute and export modes,
5//! ensuring identical calldata generation.

97impl Transaction {
98    /// Returns the contract address, encoded calldata, and optional function info for this state
99    /// change. Function info is `None` for calls with struct arguments that cannot be represented
100    /// as simple string values for Safe TX Builder.
101    pub fn calldata(self) -> (Address, Bytes, Option<FunctionInfo>) {
102        match self {
103            Self::Approve {
104                token,
105                spender,
106                amount,
107            } => (
108                token,
109                approveCall {
110                    spender,
111                    value: amount,
112                }
113                .abi_encode()
114                .into(),
115                Some(FunctionInfo {
116                    signature: "approve(address spender, uint256 value)".to_string(),
117                    args: vec![spender.to_string(), amount.to_string()],
118                }),
119            ),
120            Self::Delegate {
121                stake_table,
122                validator,
123                amount,
124            } => (
125                stake_table,
126                delegateCall { validator, amount }.abi_encode().into(),
127                Some(FunctionInfo {
128                    signature: "delegate(address validator, uint256 amount)".to_string(),
129                    args: vec![validator.to_string(), amount.to_string()],
130                }),
131            ),
132            Self::Undelegate {
133                stake_table,
134                validator,
135                amount,
136            } => (
137                stake_table,
138                undelegateCall { validator, amount }.abi_encode().into(),
139                Some(FunctionInfo {
140                    signature: "undelegate(address validator, uint256 amount)".to_string(),
141                    args: vec![validator.to_string(), amount.to_string()],
142                }),
143            ),
144            Self::ClaimWithdrawal {
145                stake_table,
146                validator,
147            } => (
148                stake_table,
149                claimWithdrawalCall { validator }.abi_encode().into(),
150                Some(FunctionInfo {
151                    signature: "claimWithdrawal(address validator)".to_string(),
152                    args: vec![validator.to_string()],
153                }),
154            ),
155            Self::ClaimValidatorExit {
156                stake_table,
157                validator,
158            } => (
159                stake_table,
160                claimValidatorExitCall { validator }.abi_encode().into(),
161                Some(FunctionInfo {
162                    signature: "claimValidatorExit(address validator)".to_string(),
163                    args: vec![validator.to_string()],
164                }),
165            ),
166            Self::ClaimRewards {
167                reward_claim,
168                lifetime_rewards,
169                auth_data,
170            } => (
171                reward_claim,
172                claimRewardsCall {
173                    lifetimeRewards: lifetime_rewards,
174                    authData: auth_data.clone(),
175                }
176                .abi_encode()
177                .into(),
178                Some(FunctionInfo {
179                    signature: "claimRewards(uint256 lifetimeRewards, bytes authData)".to_string(),
180                    args: vec![lifetime_rewards.to_string(), auth_data.to_string()],
181                }),
182            ),
183            // RegisterValidator and UpdateConsensusKeys use complex struct args (BLS/Schnorr keys)
184            // that cannot be represented as simple strings for Safe TX Builder.
185            Self::RegisterValidator {
186                stake_table,
187                commission,
188                metadata_uri,
189                payload,
190                version,
191            } => match version {
192                StakeTableContractVersion::V1 => (
193                    stake_table,
194                    registerValidatorCall::from((
195                        G2PointSol::from(payload.bls_vk),
196                        EdOnBN254PointSol::from(payload.schnorr_vk),
197                        G1PointSol::from(payload.bls_signature).into(),
198                        commission.to_evm(),
199                    ))
200                    .abi_encode()
201                    .into(),
202                    None,
203                ),
204                StakeTableContractVersion::V2 => (
205                    stake_table,
206                    registerValidatorV2Call::from((
207                        G2PointSol::from(payload.bls_vk),
208                        EdOnBN254PointSol::from(payload.schnorr_vk),
209                        G1PointSol::from(payload.bls_signature).into(),
210                        StateSignatureSol::from(payload.schnorr_signature).into(),
211                        commission.to_evm(),
212                        metadata_uri.to_string(),
213                    ))
214                    .abi_encode()
215                    .into(),
216                    None,
217                ),
218            },
219            Self::UpdateConsensusKeys {
220                stake_table,
221                payload,
222                version,
223            } => match version {
224                StakeTableContractVersion::V1 => (
225                    stake_table,
226                    updateConsensusKeysCall::from((
227                        G2PointSol::from(payload.bls_vk),
228                        EdOnBN254PointSol::from(payload.schnorr_vk),
229                        G1PointSol::from(payload.bls_signature).into(),
230                    ))
231                    .abi_encode()
232                    .into(),
233                    None,
234                ),
235                StakeTableContractVersion::V2 => (
236                    stake_table,
237                    updateConsensusKeysV2Call::from((
238                        G2PointSol::from(payload.bls_vk),
239                        EdOnBN254PointSol::from(payload.schnorr_vk),
240                        G1PointSol::from(payload.bls_signature).into(),
241                        StateSignatureSol::from(payload.schnorr_signature).into(),
242                    ))
243                    .abi_encode()
244                    .into(),
245                    None,
246                ),
247            },
248            Self::DeregisterValidator { stake_table } => (
249                stake_table,
250                deregisterValidatorCall {}.abi_encode().into(),
251                Some(FunctionInfo {
252                    signature: deregisterValidatorCall::SIGNATURE.to_string(),
253                    args: vec![],
254                }),
255            ),
256            Self::UpdateCommission {
257                stake_table,
258                new_commission,
259            } => (
260                stake_table,
261                updateCommissionCall {
262                    newCommission: new_commission.to_evm(),
263                }
264                .abi_encode()
265                .into(),
266                Some(FunctionInfo {
267                    signature: "updateCommission(uint16 newCommission)".to_string(),
268                    args: vec![new_commission.to_evm().to_string()],
269                }),
270            ),
271            Self::UpdateMetadataUri {
272                stake_table,
273                metadata_uri,
274            } => (
275                stake_table,
276                updateMetadataUriCall {
277                    metadataUri: metadata_uri.to_string(),
278                }
279                .abi_encode()
280                .into(),
281                Some(FunctionInfo {
282                    signature: "updateMetadataUri(string metadataUri)".to_string(),
283                    args: vec![metadata_uri.to_string()],
284                }),
285            ),
286            Self::Transfer { token, to, amount } => (
287                token,
288                transferCall { to, value: amount }.abi_encode().into(),
289                Some(FunctionInfo {
290                    signature: "transfer(address to, uint256 value)".to_string(),
291                    args: vec![to.to_string(), amount.to_string()],
292                }),
293            ),
294        }
295    }

297    pub fn description(&self) -> String {
298        match self {
299            Self::Approve {
300                spender, amount, ..
301            } => format!("Approve {} ESP for {}", format_esp(*amount), spender),
302            Self::Delegate {
303                validator, amount, ..
304            } => format!(
305                "Delegate {} ESP to validator {}",
306                format_esp(*amount),
307                validator
308            ),
309            Self::Undelegate {
310                validator, amount, ..
311            } => format!(
312                "Undelegate {} ESP from validator {}",
313                format_esp(*amount),
314                validator
315            ),
316            Self::ClaimWithdrawal { validator, .. } => {
317                format!("Claim withdrawal for validator {}", validator)
318            },
319            Self::ClaimValidatorExit { validator, .. } => {
320                format!("Claim validator exit for {}", validator)
321            },
322            Self::ClaimRewards { reward_claim, .. } => {
323                format!("Claim rewards from {}", reward_claim)
324            },
325            Self::RegisterValidator {
326                payload,
327                commission,
328                ..
329            } => format!(
330                "Register validator {} with {} commission",
331                payload.address, commission
332            ),
333            Self::UpdateConsensusKeys { payload, .. } => {
334                format!("Update consensus keys for {}", payload.address)
335            },
336            Self::DeregisterValidator { .. } => "Deregister validator".to_string(),
337            Self::UpdateCommission { new_commission, .. } => {
338                format!("Update commission to {}", new_commission)
339            },
340            Self::UpdateMetadataUri { .. } => "Update metadata URI".to_string(),
341            Self::Transfer { to, amount, .. } => {
342                format!("Transfer {} ESP to {}", format_esp(*amount), to)
343            },
344        }
345    }

347    fn to_transaction_request(&self) -> TransactionRequest {
348        let (to, data, _) = self.clone().calldata();
349        TransactionRequest::default()
350            .to(to)
351            .input(TransactionInput::new(data))
352    }

354    /// Validates the delegate amount against the minimum required by the contract.
355    ///
356    /// Currently only validates `Delegate` transactions; other variants pass through.
357    pub async fn validate_delegate_amount(&self, provider: &impl Provider) -> Result<()> {
358        if let Self::Delegate {
359            stake_table,
360            amount,
361            ..
362        } = self
363        {
364            use hotshot_contract_adapter::sol_types::StakeTableV2;
365            let st = StakeTableV2::new(*stake_table, provider);
366            let version: StakeTableContractVersion = st.getVersion().call().await?.try_into()?;
367            if let StakeTableContractVersion::V2 = version {
368                let min_amount = st.minDelegateAmount().call().await?;
369                if amount < &min_amount {
370                    bail!(
371                        "delegation amount {} is below minimum of {}",
372                        format_esp(*amount),
373                        format_esp(min_amount)
374                    );
375                }
376            }
377        }
378        Ok(())
379    }
380
381    fn decode_revert<T>(&self, result: impl DecodeRevert<T>) -> Result<T> {
382        match self {
383            Self::Approve { .. } | Self::Transfer { .. } => {
384                result.maybe_decode_revert::<EspTokenErrors>()
385            },
386            Self::ClaimRewards { .. } => result.maybe_decode_revert::<RewardClaimErrors>(),
387            Self::Delegate { .. }
388            | Self::Undelegate { .. }
389            | Self::ClaimWithdrawal { .. }
390            | Self::ClaimValidatorExit { .. }
391            | Self::RegisterValidator { .. }
392            | Self::UpdateConsensusKeys { .. }
393            | Self::DeregisterValidator { .. }
394            | Self::UpdateCommission { .. }
395            | Self::UpdateMetadataUri { .. } => result.maybe_decode_revert::<StakeTableV2Errors>(),
396        }
397    }
398
399    pub async fn simulate(&self, provider: &impl Provider, from: Address) -> Result<()> {
400        let tx = self.to_transaction_request().from(from);
401        let result = provider.call(tx).await;
402        self.decode_revert(result)
403            .context("Transaction simulation failed")?;
404        Ok(())
405    }
406
407    fn log_intent(&self) {
408        tracing::info!("{}", self.description());
409    }
410
411    pub async fn send(
412        &self,
413        provider: impl Provider,
414    ) -> Result<PendingTransactionBuilder<Ethereum>> {
415        self.log_intent();
416        let tx = self.to_transaction_request();
417        let pending = provider.send_transaction(tx).await;
418        self.decode_revert(pending)
419    }

428    /// Verify hand-written named signatures produce the same 4-byte selector as alloy's
429    /// `SIGNATURE` const. Catches typos or drift if the Solidity ABI changes.
430    #[test]
431    fn named_signatures_match_selectors() {
432        let cases: &[(&str, [u8; 4])] = &[
433            (
434                "approve(address spender, uint256 value)",
435                approveCall::SELECTOR,
436            ),
437            (
438                "delegate(address validator, uint256 amount)",
439                delegateCall::SELECTOR,
440            ),
441            (
442                "undelegate(address validator, uint256 amount)",
443                undelegateCall::SELECTOR,
444            ),
445            (
446                "claimWithdrawal(address validator)",
447                claimWithdrawalCall::SELECTOR,
448            ),
449            (
450                "claimValidatorExit(address validator)",
451                claimValidatorExitCall::SELECTOR,
452            ),
453            (
454                "claimRewards(uint256 lifetimeRewards, bytes authData)",
455                claimRewardsCall::SELECTOR,
456            ),
457            (
458                "updateCommission(uint16 newCommission)",
459                updateCommissionCall::SELECTOR,
460            ),
461            (
462                "updateMetadataUri(string metadataUri)",
463                updateMetadataUriCall::SELECTOR,
464            ),
465            (
466                "transfer(address to, uint256 value)",
467                transferCall::SELECTOR,
468            ),
469        ];
470
471        for (named_sig, expected_selector) in cases {
472            let func = Function::parse(named_sig)
473                .unwrap_or_else(|e| panic!("failed to parse '{named_sig}': {e}"));
474            assert_eq!(
475                func.selector().as_slice(),
476                expected_selector,
477                "selector mismatch for '{named_sig}'"
478            );
479        }
480    }
