1//! Per-view vote collection and certificate formation.
2//!
3//! Votes for views arrive concurrently, and each view is tallied on its own
4//! until it crosses a threshold and forms a certificate. [`VoteCollector`]
5//! owns the machinery common to every kind of vote, inspecting each vote only
6//! through the [`Ballot`] trait. It creates a task per view and routes votes
7//! to the appropriate task, drops duplicate and stale votes, buffers votes
8//! whose epoch is not yet resolved, and GCs decided views. How a view's task
9//! actually combines votes into an output is delegated to a pluggable [`Tally`]
10//! strategy.

37/// Information about a vote.
38///
39/// Used by [`VoteCollector`] to handle incoming votes, i.e. reject
40/// duplicate votes by the same signer, or stale votes for GCed views,
41/// before even tallying the vote.
42pub trait Ballot {
43    type Signer;
44
45    fn view(&self) -> ViewNumber;
46    fn epoch(&self) -> Option<EpochNumber>;
47    fn signer(&self) -> Self::Signer;
48}

50/// How to count votes to form a certificate.
51pub trait Tally<T: NodeType> {
52    type Vote: Send + 'static;
53    type Output: Send + 'static;
54
55    fn tally(
56        r: Receiver<Self::Vote>,
57        m: EpochMembership<T>,
58        l: UpgradeLock<T>,
59    ) -> Option<Self::Output>;
60}

97/// Accumulates votes into a single [`Certificate`] via a [`CheckedAccumulator`].
98#[allow(clippy::type_complexity)]
99pub struct SimpleTally<T, V, C>(PhantomData<fn() -> (T, V, C)>);
100
101impl<T, V, C> Tally<T> for SimpleTally<T, V, C>
102where
103    T: NodeType,
104    V: Vote<T> + Send + 'static,
105    C: Certificate<T, V::Commitment, Voteable = V::Commitment> + Send + 'static,
106{
107    type Vote = V;
108    type Output = C;
109
110    fn tally(r: Receiver<V>, m: EpochMembership<T>, l: UpgradeLock<T>) -> Option<Self::Output> {
111        let mut a = CheckedAccumulator::new(m, l);
112        while let Ok(v) = r.recv() {
113            if let Some(c) = a.add(v) {
114                return Some(c);
115            }
116        }
117        None
118    }
119}

127/// Accumulates epoch-root [`Vote1`]s into the (quorum, state) certificate pair.
128pub struct EpochRootTally<T>(PhantomData<fn() -> T>);
129
130impl<T: NodeType> Tally<T> for EpochRootTally<T> {
131    type Vote = Vote1<T>;
132    type Output = EpochRootCerts<T>;
133
134    fn tally(
135        rx: mpsc::Receiver<Vote1<T>>,
136        mem: EpochMembership<T>,
137        lock: UpgradeLock<T>,
138    ) -> Option<Self::Output> {
139        let mut quorum_accu = CheckedAccumulator::<T, QuorumVote2<T>, QuorumCertificate2<T>>::new(
140            mem.clone(),
141            lock.clone(),
142        );
143
144        let mut state_accu = LightClientStateUpdateVoteAccumulator::<T> {
145            vote_outcomes: HashMap::new(),
146            upgrade_lock: lock,
147        };
148
149        let mut quorum_cert = None;
150        let mut state_cert = None;
151
152        while let Ok(vote1) = rx.recv() {
153            let Some(state_vote) = vote1.state_vote else {
154                error!(view = %vote1.vote.view_number(), "epoch-root vote1 without state vote");
155                continue;
156            };
157            let bls_key = vote1.vote.signing_key();
158
159            if quorum_cert.is_none() {
160                quorum_cert = quorum_accu.add(vote1.vote);
161            }
162
163            // Unlike quorum votes, state votes are fully checked, including
164            // their signatures, by the accumulator, so the certificate does
165            // not need to be validated again.
166            if state_cert.is_none() {
167                state_cert = state_accu.accumulate(&bls_key, &state_vote, &mem);
168            }
169
170            if let (Some(q), Some(s)) = (&quorum_cert, &state_cert) {
171                info!(view = %q.view_number(), epoch = %s.epoch, "epoch-root certificates formed");
172                return Some((q.clone(), s.clone()));
173            }
174        }
175
176        None
177    }
178}

180/// Collects votes per view and forms certificate(s) using the strategy `S`.
181pub struct VoteCollector<T: NodeType, S: Tally<T>> {
182    /// Tasks collecting votes and forming certificates.
183    accumulators: JoinMap<ViewNumber, Option<S::Output>>,
184
185    /// Where callers submit their votes.
186    ballot_boxes: BTreeMap<ViewNumber, mpsc::Sender<S::Vote>>,
187
188    /// Votes for epochs we have yet to resolve, deduplicated by signer.
189    pending: BTreeMap<ViewNumber, HashMap<T::SignatureKey, S::Vote>>,
190
191    /// Views that had a valid certificate already.
192    completed: BTreeSet<ViewNumber>,
193
194    /// The signers per view.
195    signers: BTreeMap<ViewNumber, HashSet<T::SignatureKey>>,
196
197    /// The GC threshold.
198    lower_bound: ViewNumber,
199
200    membership: EpochMembershipCoordinator<T>,
201
202    upgrade_lock: UpgradeLock<T>,
203}
204
205impl<T: NodeType, S: Tally<T>> VoteCollector<T, S>
206where
207    <S as Tally<T>>::Vote: Ballot<Signer = T::SignatureKey>,
208{
209    pub fn new(mc: EpochMembershipCoordinator<T>, lock: UpgradeLock<T>) -> Self {
210        Self {
211            accumulators: JoinMap::new(),
212            ballot_boxes: BTreeMap::new(),
213            pending: BTreeMap::new(),
214            signers: BTreeMap::new(),
215            completed: BTreeSet::new(),
216            membership: mc,
217            upgrade_lock: lock,
218            lower_bound: ViewNumber::genesis(),
219        }
220    }
221
222    pub async fn next(&mut self) -> Option<S::Output> {
223        loop {
224            match self.accumulators.join_next().await {
225                Some((view, Ok(Some(cert)))) => {
226                    self.ballot_boxes.remove(&view);
227                    if view >= self.lower_bound {
228                        self.completed.insert(view);
229                        return Some(cert);
230                    }
231                },
232                Some((_, Ok(None))) => {},
233                Some((view, Err(err))) => {
234                    if err.is_panic() {
235                        error!(%view, %err, "vote collection task panic");
236                    }
237                    self.ballot_boxes.remove(&view);
238                },
239                None => return None,
240            }
241        }
242    }
243
244    pub fn accumulate_vote(&mut self, vote: S::Vote) {
245        let view = vote.view();
246
247        if view < self.lower_bound || self.completed.contains(&view) {
248            return;
249        }
250
251        let Some(membership) = self.resolve_membership(&vote) else {
252            // A missing epoch can never resolve, so we only buffer
253            // votes whose epoch's stake table will eventually become
254            // available:
255            if vote.epoch().is_some() {
256                self.pending
257                    .entry(view)
258                    .or_default()
259                    .insert(vote.signer(), vote);
260            }
261            return;
262        };
263
264        // Check that we have not received a vote from this signer already.
265        if !self.signers.entry(view).or_default().insert(vote.signer()) {
266            return;
267        }
268
269        if let Some(tx) = self.ballot_boxes.get(&view) {
270            let _ = tx.send(vote);
271            return;
272        }
273
274        let (tx, rx) = mpsc::channel();
275
276        let _ = tx.send(vote);
277        self.ballot_boxes.insert(view, tx);
278
279        let lock = self.upgrade_lock.clone();
280        self.accumulators
281            .spawn_blocking(view, move || S::tally(rx, membership, lock));
282    }
283
284    pub fn retry_pending_votes(&mut self) {
285        for vote in mem::take(&mut self.pending)
286            .into_values()
287            .flat_map(|votes| votes.into_values())
288        {
289            self.accumulate_vote(vote)
290        }
291    }
292
293    pub fn gc(&mut self, view: ViewNumber) {
294        self.ballot_boxes = self.ballot_boxes.split_off(&view);
295        self.completed = self.completed.split_off(&view);
296        self.pending = self.pending.split_off(&view);
297        self.signers = self.signers.split_off(&view);
298        self.lower_bound = view;
299    }
300
301    fn resolve_membership(&mut self, vote: &S::Vote) -> Option<EpochMembership<T>> {
302        let epoch = vote.epoch()?;
303        self.membership.membership_for_epoch(Some(epoch)).ok()
304    }
305}

307impl<T, V, C> VoteCollector<T, SimpleTally<T, V, C>>
308where
309    T: NodeType,
310    V: Vote<T> + Send + 'static,
311    C: Certificate<T, V::Commitment, Voteable = V::Commitment> + Send + 'static,
312{
313    /// Compute the accumulated stake.
314    ///
315    /// This is the sum across unique signers we've routed to the accumulator
316    /// for `view` and the cert threshold in `epoch`. Looks up each signer's
317    /// stake on demand — only intended for rare paths like timeout
318    /// diagnostics. Returns `None` if no votes have been seen for `view` or
319    /// `epoch`'s stake table is unavailable.
320    pub fn stats(&self, view: ViewNumber, epoch: EpochNumber) -> Option<VoteStats> {
321        let signers = self.signers.get(&view)?;
322        if signers.is_empty() {
323            return None;
324        }
325        let membership = self.membership.membership_for_epoch(Some(epoch)).ok()?;
326        let threshold = C::threshold(&membership);
327        let mut stake = U256::ZERO;
328        for signer in signers {
329            if let Some(peer) = C::stake_table_entry(&membership, signer) {
330                stake += peer.stake_table_entry.stake();
331            }
332        }
333        Some(VoteStats { stake, threshold })
334    }
335}
