20pub struct VoteCollector<T: NodeType, V, C> {
21    accumulators: BTreeMap<ViewNumber, (mpsc::Sender<V>, AbortHandle)>,
22    completed_certificates: BTreeSet<ViewNumber>,
23    epoch_membership_coordinator: EpochMembershipCoordinator<T>,
24    membership_cache: BTreeMap<EpochNumber, EpochMembership<T>>,
25    upgrade_lock: UpgradeLock<T>,
26    tasks: JoinSet<C>,
27}
28
29impl<T, V, C> VoteCollector<T, V, C>
30where
31    T: NodeType,
32    V: Vote<T> + HasEpoch + Send + Sync + 'static,
33    C: Certificate<T, V::Commitment, Voteable = V::Commitment> + Send + Sync + 'static,
34{
35    #[instrument(level = "debug", skip_all)]
36    pub fn new(
37        epoch_membership_coordinator: EpochMembershipCoordinator<T>,
38        upgrade_lock: UpgradeLock<T>,
39    ) -> Self {
40        Self {
41            accumulators: BTreeMap::new(),
42            completed_certificates: BTreeSet::new(),
43            epoch_membership_coordinator,
44            membership_cache: BTreeMap::new(),
45            upgrade_lock,
46            tasks: JoinSet::new(),
47        }
48    }
49
50    pub async fn next(&mut self) -> Option<C> {
51        loop {
52            match self.tasks.join_next().await {
53                Some(Ok(cert)) => {
54                    if self.completed_certificates.contains(&cert.view_number()) {
55                        continue;
56                    }
57                    self.completed_certificates.insert(cert.view_number());
58                    return Some(cert);
59                },
60                Some(Err(e)) if e.is_cancelled() => {
61                    debug!("Vote collection task cancelled: {e}");
62                },
63                Some(Err(e)) => {
64                    warn!("Error in vote collection task: {e}");
65                },
66                None => return None,
67            }
68        }
69    }
70
71    pub async fn accumulate_vote(&mut self, vote: V) {
72        let view = vote.view_number();
73        if self.completed_certificates.contains(&view) {
74            return;
75        }
76        let Some(membership) = self.resolve_membership(&vote).await else {
77            return;
78        };
79        let (tx, _abort_handle) = self.accumulators.entry(view).or_insert_with(|| {
80            let (tx, rx) = mpsc::channel(100);
81            let accumulator = VoteAccumulator::new(self.upgrade_lock.clone());
82            let upgrade_lock = self.upgrade_lock.clone();
83            let abort_handle = self.tasks.spawn(Self::run_per_view(
84                view,
85                rx,
86                accumulator,
87                membership,
88                upgrade_lock,
89            ));
90            (tx, abort_handle)
91        });
92        let _ = tx.send(vote).await;
93    }
94
95    async fn resolve_membership(&mut self, vote: &V) -> Option<EpochMembership<T>> {
96        let epoch = vote.epoch()?;
97        if let Some(m) = self.membership_cache.get(&epoch) {
98            return Some(m.clone());
99        }
100        let m = self
101            .epoch_membership_coordinator
102            .membership_for_epoch(Some(epoch))
103            .await
104            .ok()?;
105        self.membership_cache.insert(epoch, m.clone());
106        Some(m)
107    }
108
109    #[instrument(level = "debug", skip_all)]
110    async fn run_per_view(
111        _view: ViewNumber,
112        mut rx: mpsc::Receiver<V>,
113        mut accumulator: VoteAccumulator<T, V, C>,
114        membership: EpochMembership<T>,
115        upgrade_lock: UpgradeLock<T>,
116    ) -> C {
117        let mut votes = Vec::new();
118
119        while let Some(vote) = rx.recv().await {
120            if let Some(cert) = accumulator.accumulate(&vote, membership.clone()).await {
121                let stake_table = C::stake_table(&membership).await;
122                let threshold = C::threshold(&membership).await;
123                match cert.is_valid_cert(
124                    &StakeTableEntries::<T>::from(stake_table).0,
125                    threshold,
126                    &upgrade_lock,
127                ) {
128                    Ok(()) => {
129                        return cert;
130                    },
131                    Err(e) => {
132                        warn!("Invalid certificate formed: {e}");
133                        votes.push(vote);
134                        // Recover the good votes, this takes a long time
135                        // TODO make this more efficient by parallelizing the validation
136                        votes.retain(|v: &V| {
137                            let vote_commitment = generate_vote_commitment(v, &upgrade_lock);
138
139                            vote_commitment.is_some_and(|commitment| {
140                                v.signing_key()
141                                    .validate(&v.signature(), commitment.as_ref())
142                            })
143                        });
144                        accumulator = VoteAccumulator::new(upgrade_lock.clone());
145                        for vote in &votes {
146                            // after recovering the good votes, try to accumulate them again, but this time
147                            // we know the cert if good if we can form it
148                            if let Some(cert) =
149                                accumulator.accumulate(vote, membership.clone()).await
150                            {
151                                return cert;
152                            }
153                        }
154                    },
155                }
156            } else {
157                votes.push(vote);
158            }
159        }
160        unreachable!()
161    }
162    pub fn gc(&mut self, view: ViewNumber, epoch: EpochNumber) {
163        let keep = self.accumulators.split_off(&view);
164        self.completed_certificates = self.completed_certificates.split_off(&view);
165        for (_, handle) in self.accumulators.values_mut() {
166            handle.abort();
167        }
168        self.accumulators = keep;
169        self.membership_cache = self.membership_cache.split_off(&epoch);
170    }
171}

173fn generate_vote_commitment<T: NodeType, V: Vote<T>>(
174    vote: &V,
175    upgrade_lock: &UpgradeLock<T>,
176) -> Option<Commitment<VersionedVoteData<T, V::Commitment>>> {
177    match VersionedVoteData::new(vote.date().clone(), vote.view_number(), upgrade_lock) {
178        Ok(data) => Some(data.commit()),
179        Err(e) => {
180            tracing::warn!("Failed to generate versioned vote data: {e}");
181            None
182        },
183    }
184}

298    /// Wait for exactly `expected` certificates, then abort the task.
299    async fn _collect_certs<T: std::fmt::Debug>(
300        cert_rx: &mut mpsc::Receiver<T>,
301        expected: usize,
302    ) -> Vec<T> {
303        let mut results = Vec::new();
304        for _ in 0..expected {
305            let cert = tokio::time::timeout(CERT_TIMEOUT, cert_rx.recv())
306                .await
307                .expect("Timed out waiting for certificate")
308                .expect("Cert channel closed unexpectedly");
309            results.push(cert);
310        }
311        results
312    }

314    /// Confirm no certificates are produced within the timeout, then abort the task.
315    async fn assert_no_certs<
316        V: Vote<TestTypes> + HasEpoch + Send + Sync + 'static,
317        C: Certificate<TestTypes, V::Commitment, Voteable = V::Commitment>
318            + Debug
319            + Send
320            + Sync
321            + 'static,
322    >(
323        task: &mut VoteCollector<TestTypes, V, C>,
324    ) {
325        let result = tokio::time::timeout(NO_CERT_TIMEOUT, task.next()).await;
326        match result {
327            Err(_) => { /* timeout — good, no cert produced */ },
328            Ok(None) => { /* good, no cert produced */ },
329            Ok(Some(cert)) => panic!("Expected no certificate but got one: {cert:?}"),
330        }
331    }

333    /// Verify that a certificate's data commitment matches `expected_data` and that
334    /// the aggregate signature is valid against the stake table.
335    async fn verify_cert<C, D>(cert: &C, expected_data: &D, membership: &EpochMembership<TestTypes>)
336    where
337        D: Committable,
338        C: Certificate<TestTypes, D, Voteable = D>,
339    {
340        // Data commitment must match the vote data that produced the cert.
341        assert_eq!(
342            cert.data().commit(),
343            expected_data.commit(),
344            "Certificate data commitment does not match expected vote data"
345        );
346
347        // Aggregate signature must be valid against the stake table.
348        let stake_table = C::stake_table(membership).await;
349        let stake_table_entries = StakeTableEntries::<TestTypes>::from(stake_table).0;
350        let threshold = C::threshold(membership).await;
351        cert.is_valid_cert(&stake_table_entries, threshold, &test_upgrade_lock())
352            .expect("Certificate signature validation failed");
353    }
