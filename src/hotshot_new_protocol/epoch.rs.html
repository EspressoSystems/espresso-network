17pub enum EpochRootResult {
18    DrbResult(EpochNumber, DrbResult),
19    EpochRootAdded(EpochNumber),

22/// Epoch + error for the Err path so retries can re-kick the task.
23pub struct EpochFailure {
24    pub epoch: EpochNumber,
25    pub error: EpochManagerError,
26}
27
28/// Output of a spawned DRB task.  The leading `EpochNumber` tags the task
29/// with the epoch it was working on so [`EpochManager::next`] can update
30/// dedup state correctly even on the Err path.
31type TaskOutput = (EpochNumber, Result<EpochRootResult, EpochManagerError>);
32
33/// Manager for Epoch specific rules and actions.
34///
35/// Delegates all catchup (stake-table walk-back, epoch root fetch, DRB
36/// compute) to [`EpochMembershipCoordinator::membership_for_epoch`].  This
37/// manager exists only to issue the request and dedup concurrent callers.
38pub struct EpochManager<T: NodeType> {
39    epoch_height: BlockNumber,
40    membership_coordinator: EpochMembershipCoordinator<T>,
41    tasks: JoinSet<TaskOutput>,
42    handles: BTreeMap<EpochNumber, Vec<AbortHandle>>,
43    /// Epochs for which a `request_drb_result` task is currently in flight.
44    /// Prevents duplicate fetch/compute tasks while the first is running.
45    pending_drb_requests: BTreeSet<EpochNumber>,
46    /// Epochs whose DRB has already been computed and added to membership.
47    /// Subsequent `request_drb_result` calls for these epochs are no-ops.
48    completed_drb_requests: BTreeSet<EpochNumber>,
49}
50
51impl<T: NodeType> EpochManager<T> {
52    pub fn new<B>(epoch_height: B, membership_coordinator: EpochMembershipCoordinator<T>) -> Self
53    where
54        B: Into<BlockNumber>,
55    {
56        Self {
57            epoch_height: epoch_height.into(),
58            membership_coordinator,
59            tasks: JoinSet::new(),
60            handles: BTreeMap::new(),
61            pending_drb_requests: BTreeSet::new(),
62            completed_drb_requests: BTreeSet::new(),
63        }
64    }
65
66    pub async fn next(&mut self) -> Option<Result<EpochRootResult, EpochFailure>> {
67        loop {
68            match self.tasks.join_next().await {
69                Some(Ok((epoch, result))) => {
70                    match result {
71                        Ok(root @ EpochRootResult::DrbResult(..)) => {
72                            self.pending_drb_requests.remove(&epoch);
73                            self.completed_drb_requests.insert(epoch);
74                            return Some(Ok(root));
75                        },
76                        Ok(root @ EpochRootResult::EpochRootAdded(_)) => {
77                            return Some(Ok(root));
78                        },
79                        Err(error) => {
80                            // Clear the guard so a subsequent call can retry.
81                            self.pending_drb_requests.remove(&epoch);
82                            return Some(Err(EpochFailure { epoch, error }));
83                        },
84                    }
85                },
86                Some(Err(err)) => {
87                    if !err.is_cancelled() {
88                        error!(%err, "epoch manager task panic")
89                    }
90                },
91                None => return None,
92            }
93        }
94    }
95
96    pub fn handle_leaf_decided(&mut self, leaf: Leaf2<T>) {
97        let block_number = leaf.block_header().block_number();
98
99        // At every epoch root, trigger DRB computation for the epoch that
100        // will use this root (epoch + 2).
101        if is_epoch_root(block_number, *self.epoch_height) {
102            let Some(epoch) = leaf.epoch(*self.epoch_height) else {
103                error!("Leaf has no epoch");
104                return;
105            };
106
107            // Push the epoch root into membership directly from the decided
108            // header
109            let target_epoch = epoch + 2;
110            let membership_coordinator = self.membership_coordinator.clone();
111            let header = leaf.block_header().clone();
112            let handles = self.handles.entry(target_epoch).or_default();
113            handles.push(self.tasks.spawn(async move {
114                let result = membership_coordinator
115                    .membership()
116                    .add_epoch_root(header)
117                    .await
118                    .map(|()| EpochRootResult::EpochRootAdded(target_epoch))
119                    .map_err(|e| EpochManagerError::EpochRoot(anyhow::anyhow!("{e}")));
120                (target_epoch, result)
121            }));
122
123            self.request_drb_result(target_epoch);
124        }
125
126        // If this is the transition block of an epoch feed the DRB result to the coordinator.
127        if is_transition_block(block_number, *self.epoch_height)
128            && let Some(epoch) = leaf.epoch(*self.epoch_height)
129            && let Some(drb) = leaf.next_drb_result
130        {
131            let target_epoch = epoch + 1;
132            if !self.completed_drb_requests.contains(&target_epoch) {
133                self.membership_coordinator.supply_drb(target_epoch, drb);
134                self.completed_drb_requests.insert(target_epoch);
135                self.pending_drb_requests.remove(&target_epoch);
136            }
137        }
138    }
139
140    pub fn gc(&mut self, epoch: EpochNumber) {
141        let mut tmp = self.handles.split_off(&epoch);
142        swap(&mut tmp, &mut self.handles);
143        for handle in tmp.into_values().flatten() {
144            handle.abort();
145        }
146        // Drop tracking entries for epochs we no longer care about.  Keeps
147        // `completed_drb_requests` bounded while the protocol runs.
148        self.pending_drb_requests = self.pending_drb_requests.split_off(&epoch);
149        self.completed_drb_requests = self.completed_drb_requests.split_off(&epoch);
150    }
151
152    pub fn request_drb_result(&mut self, epoch: EpochNumber) {
153        // Already computed — caller can read the DRB from membership.
154        if self.completed_drb_requests.contains(&epoch) {
155            return;
156        }
157        // In-flight task will deliver the result; avoid spawning a duplicate.
158        if self.pending_drb_requests.contains(&epoch) {
159            return;
160        }
161        self.pending_drb_requests.insert(epoch);
162        let membership_coordinator = self.membership_coordinator.clone();
163        let handles = self.handles.entry(epoch).or_default();
164
165        handles.push(self.tasks.spawn(async move {
166            let result = async {
167                // Kick the membership coordinator.  If the stake table is
168                // already ready, this returns it immediately; otherwise it
169                // spawns a catchup task and returns a "catchup in progress"
170                // error.  Either way, `wait_for_catchup` resolves once the
171                // stake table + DRB are both in place.
172                let membership = match membership_coordinator.membership_for_epoch(Some(epoch)) {
173                    Ok(m) => m,
174                    Err(_) => membership_coordinator
175                        .wait_for_catchup(epoch)
176                        .await
177                        .map_err(EpochManagerError::DrbLookup)?,
178                };
179                let drb = membership
180                    .get_epoch_drb()
181                    .await
182                    .map_err(EpochManagerError::DrbLookup)?;
183                Ok(EpochRootResult::DrbResult(epoch, drb))
184            }
185            .await;
186            (epoch, result)
187        }));
188    }
189}
