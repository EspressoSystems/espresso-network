15/// Verify that a leaf is finalized by a new-protocol Certificate2.
16///
17/// `cert2` directly commits the newest leaf in `leaf_chain`. By the indirect
18/// commit rule, every ancestor of that leaf is finalized as well. This verifier
19/// validates `cert2`, then walks backward through the certified leaf's parent
20/// links until it finds `expected_height`.
21///
22/// View numbers are allowed to skip after timeouts, so the input may contain
23/// leaves that are not on the certified ancestry path. Those leaves are ignored;
24/// every accepted step must match the current leaf's justify QC, parent
25/// commitment, and block height.
26pub async fn verify_new_protocol_leaf_chain<T: NodeType>(
27    mut leaf_chain: Vec<Leaf2<T>>,
28    coordinator: &EpochMembershipCoordinator<T>,
29    expected_height: u64,
30    upgrade_lock: &UpgradeLock<T>,
31    cert2: Certificate2<T>,
32) -> anyhow::Result<Leaf2<T>> {
33    leaf_chain.sort_by_key(|l| l.view_number());
34    leaf_chain.reverse();
35
36    ensure!(!leaf_chain.is_empty(), "empty leaf chain");
37    let newest = &leaf_chain[0];
38
39    ensure!(
40        cert2.view_number() > ViewNumber::genesis(),
41        "cert2 must not be the genesis view"
42    );
43    let epoch = EpochNumber::new(epoch_from_block_number(
44        cert2.data.block_number,
45        *coordinator.epoch_height(),
46    ));
47    ensure!(
48        cert2.data.epoch == epoch,
49        "cert2 epoch {} does not match epoch {epoch} derived from its block number {}",
50        cert2.data.epoch,
51        cert2.data.block_number
52    );
53
54    let membership = coordinator
55        .stake_table_for_epoch(Some(epoch))
56        .map_err(|err| anyhow!("no stake table available for epoch {epoch}: {err:?}"))?;
57    let entries = StakeTableEntries::<T>::from_iter(membership.stake_table()).0;
58    cert2.is_valid_cert(&entries, membership.success_threshold(), upgrade_lock)?;
59
60    ensure!(
61        cert2.data.leaf_commit == newest.commit(),
62        "cert2 does not match the newest leaf in the chain"
63    );
64    ensure!(
65        cert2.data.block_number == newest.height(),
66        "cert2 block number does not match the newest leaf"
67    );
68    ensure!(
69        cert2.view_number() == newest.view_number(),
70        "cert2 view does not match the newest leaf"
71    );
72
73    if newest.height() == expected_height {
74        return Ok(newest.clone());
75    }
76
77    let mut current = newest;
78    for leaf in leaf_chain[1..].iter() {
79        let justify_qc = current.justify_qc();
80        if justify_qc.view_number() != leaf.view_number()
81            || justify_qc.data().leaf_commit != leaf.commit()
82        {
83            tracing::warn!(
84                view = ?leaf.view_number(),
85                expected_view = ?justify_qc.view_number(),
86                "leaf is off the leafchain path; expected only after a view timeout"
87            );
88            continue;
89        }
90        ensure!(
91            current.parent_commitment() == leaf.commit(),
92            "current leaf parent commitment does not match parent leaf"
93        );
94        ensure!(
95            leaf.height().checked_add(1) == Some(current.height()),
96            "leaf heights do not chain"
97        );
98        let qc_epoch = EpochNumber::new(epoch_from_block_number(
99            leaf.height(),
100            *coordinator.epoch_height(),
101        ));
102        ensure!(
103            justify_qc.data().epoch == Some(qc_epoch),
104            "justify QC claims epoch {:?} but certifies the leaf at height {} in epoch {qc_epoch}",
105            justify_qc.data().epoch,
106            leaf.height()
107        );
108        if let Some(block_number) = justify_qc.data().block_number {
109            ensure!(
110                block_number == leaf.height(),
111                "justify QC claims block number {block_number} but certifies the leaf at height {}",
112                leaf.height()
113            );
114        }
115        let membership = coordinator
116            .stake_table_for_epoch(Some(qc_epoch))
117            .map_err(|err| anyhow!("no stake table available for epoch {qc_epoch}: {err:?}"))?;
118        let entries = StakeTableEntries::<T>::from_iter(membership.stake_table()).0;
119        justify_qc.is_valid_cert(&entries, membership.success_threshold(), upgrade_lock)?;
120        if leaf.height() == expected_height {
121            return Ok(leaf.clone());
122        }
123        current = leaf;
124    }
125
126    Err(anyhow!(
127        "expected height was not found in the cert2-finalized chain"
128    ))
129}

165        indexes
166            .map(|i| {

168                    BLSPubKey::generated_from_seed_indexed(Default::default(), i);

206    fn sign((keys, peers): &Quorum, msg: &[u8]) -> <BLSPubKey as SignatureKey>::QcType {
207        let entries: Vec<_> = peers.iter().map(|p| p.stake_table_entry.clone()).collect();
208        let total: U256 = entries.iter().map(|e| e.stake_amount).sum();
209        let pp = BLSPubKey::public_parameter(&entries, supermajority_threshold(total));
210        let sigs: Vec<_> = keys
211            .iter()
212            .map(|key| BLSPubKey::sign(key, msg).unwrap())
213            .collect();
214        BLSPubKey::assemble(
215            &pp,
216            &std::iter::repeat_n(true, keys.len()).collect::<BitVec>(),
217            &sigs,
218        )
219    }

221    fn signed_qc(
222        data: QuorumData2<TestTypes>,
223        view: u64,
224        quorum: &Quorum,
225        upgrade_lock: &UpgradeLock<TestTypes>,

234        )

281    /// A legacy 3-chain deciding the last block of `epoch` (epoch height 10), where consecutive
282    /// epochs have disjoint quorums. The QC on the first block of `epoch + 1` is signed by
283    /// `deciding_quorum`; every other QC is signed by `epoch_quorum`.
284    fn boundary_3_chain(
285        epoch: u64,
286        epoch_quorum: &Quorum,
287        deciding_quorum: &Quorum,
288        upgrade_lock: &UpgradeLock<TestTypes>,
289    ) -> Vec<Leaf2<TestTypes>> {
290        let last = epoch * EPOCH_HEIGHT;
291        let parent_qc = signed_qc(
292            QuorumData2 {
293                leaf_commit: Commitment::from_raw([9; 32]),
294                epoch: Some(EpochNumber::new(epoch)),
295                block_number: Some(last - 1),
296            },
297            last - 1,
298            epoch_quorum,
299            upgrade_lock,
300        );
301        let last_leaf = make_leaf(last, last, epoch, parent_qc, EPOCH_VERSION);
302        let last_qc = signed_qc(
303            QuorumData2 {
304                leaf_commit: Committable::commit(&last_leaf),
305                epoch: Some(EpochNumber::new(epoch)),
306                block_number: Some(last),
307            },
308            last,
309            epoch_quorum,
310            upgrade_lock,
311        );
312        let boundary_leaf = make_leaf(last + 1, last + 1, epoch + 1, last_qc, EPOCH_VERSION);
313        let boundary_qc = signed_qc(
314            QuorumData2 {
315                leaf_commit: Committable::commit(&boundary_leaf),
316                epoch: Some(EpochNumber::new(epoch + 1)),
317                block_number: Some(last + 1),
318            },
319            last + 1,
320            deciding_quorum,
321            upgrade_lock,
322        );
323        let deciding_leaf = make_leaf(last + 2, last + 2, epoch + 1, boundary_qc, EPOCH_VERSION);
324        vec![last_leaf, boundary_leaf, deciding_leaf]
325    }

327    /// A self-consistent 3-chain starting at `first_height`, entirely within `epoch`, with
328    /// every QC signed by `epoch_quorum`.
329    fn same_epoch_3_chain(
330        first_height: u64,
331        epoch: u64,
332        epoch_quorum: &Quorum,
333        upgrade_lock: &UpgradeLock<TestTypes>,
334    ) -> Vec<Leaf2<TestTypes>> {
335        let parent_qc = signed_qc(
336            QuorumData2 {
337                leaf_commit: Commitment::from_raw([9; 32]),
338                epoch: Some(EpochNumber::new(epoch)),
339                block_number: Some(first_height - 1),
340            },
341            first_height - 1,
342            epoch_quorum,
343            upgrade_lock,
344        );
345        let mut chain = vec![make_leaf(
346            first_height,
347            first_height,
348            epoch,
349            parent_qc,
350            EPOCH_VERSION,
351        )];
352        for height in first_height + 1..first_height + 3 {
353            let qc = signed_qc(
354                QuorumData2 {
355                    leaf_commit: Committable::commit(chain.last().unwrap()),
356                    epoch: Some(EpochNumber::new(epoch)),
357                    block_number: Some(height - 1),
358                },
359                height - 1,
360                epoch_quorum,
361                upgrade_lock,
362            );
363            chain.push(make_leaf(height, height, epoch, qc, EPOCH_VERSION));
364        }
365        chain
366    }

441    /// A new-protocol leaf range deciding the last block of `epoch`, finalized by a cert2 on
442    /// the first block of `epoch + 1`, signed by `cert2_quorum`.
443    fn boundary_cert2_chain(
444        epoch: u64,
445        epoch_quorum: &Quorum,
446        cert2_quorum: &Quorum,
447        upgrade_lock: &UpgradeLock<TestTypes>,
448    ) -> (Vec<Leaf2<TestTypes>>, Certificate2<TestTypes>) {
449        let last = epoch * EPOCH_HEIGHT;
450        let parent_qc = signed_qc(
451            QuorumData2 {
452                leaf_commit: Commitment::from_raw([9; 32]),
453                epoch: Some(EpochNumber::new(epoch)),
454                block_number: Some(last - 1),
455            },
456            last - 1,
457            epoch_quorum,
458            upgrade_lock,
459        );
460        let last_leaf = make_leaf(last, last, epoch, parent_qc, NEW_PROTOCOL_VERSION);
461        let last_qc = signed_qc(
462            QuorumData2 {
463                leaf_commit: Committable::commit(&last_leaf),
464                epoch: Some(EpochNumber::new(epoch)),
465                block_number: Some(last),
466            },
467            last,
468            epoch_quorum,
469            upgrade_lock,
470        );
471        let boundary_leaf = make_leaf(last + 1, last + 1, epoch + 1, last_qc, NEW_PROTOCOL_VERSION);
472        let cert2 = signed_cert2(
473            Vote2Data {
474                leaf_commit: Committable::commit(&boundary_leaf),
475                epoch: EpochNumber::new(epoch + 1),
476                block_number: last + 1,
477            },
478            last + 1,
479            cert2_quorum,
480            upgrade_lock,
481        );
482        (vec![last_leaf, boundary_leaf], cert2)
483    }
