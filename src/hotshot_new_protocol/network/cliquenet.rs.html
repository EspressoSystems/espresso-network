35impl<T: NodeType> Cliquenet<T> {
36    pub async fn create<A, P>(
37        name: &str,
38        signing_key: T::SignatureKey,
39        keypair: Keypair,
40        addr: A,
41        parties: P,
42        upgrade_lock: UpgradeLock<T>,
43        metrics: Box<dyn Metrics>,
44    ) -> Result<Self, NetworkError>
45    where
46        A: Into<cliquenet::NetAddr>,
47        P: IntoIterator<Item = (T::SignatureKey, PeerConnectInfo)>,
48    {
49        let parties: HashMap<T::SignatureKey, PeerConnectInfo> = parties.into_iter().collect();
50
51        let cfg = cliquenet::Config::builder()
52            .name(name)
53            .keypair(keypair.into())
54            .bind(addr.into())
55            .parties(
56                parties
57                    .values()
58                    .map(|info| (info.x25519_key.into(), info.p2p_addr.clone())),
59            )
60            .build();
61
62        Self::create_with_config(signing_key, upgrade_lock, cfg, parties, metrics).await
63    }
64
65    pub(crate) async fn create_with_config<P>(
66        signing_key: T::SignatureKey,
67        upgrade_lock: UpgradeLock<T>,
68        mut config: cliquenet::Config,
69        parties: P,
70        metrics: Box<dyn Metrics>,
71    ) -> Result<Self, NetworkError>
72    where
73        P: IntoIterator<Item = (T::SignatureKey, PeerConnectInfo)>,
74    {
75        let public_key = config.public_key();
76        assert!(config.metrics.is_none());
77        config.metrics = Some(Arc::new(CliquenetMetrics::new(metrics)));
78        let network = cliquenet::Network::create(config)
79            .await
80            .map_err(to_network_error)?;
81
82        let peers: HashMap<_, _> = parties.into_iter().collect();
83
84        info!(peers = %peers.len(), "cliquenet created");
85
86        Ok(Self {
87            my_keys: (signing_key, public_key),
88            inner: network,
89            peers,
90            epoch: EpochNumber::new(0),
91            upgrade_lock,
92        })
93    }
94}

99    fn unicast(
100        &mut self,
101        v: ViewNumber,
102        to: &T::SignatureKey,
103        m: &Message<T, Validated>,
104    ) -> Result<(), NetworkError> {
105        let target = if *to == self.my_keys.0 {
106            self.my_keys.1
107        } else if let Some(info) = self.peers.get(to) {
108            info.x25519_key.into()
109        } else {
110            error!(peer = %to, "unicast target not found");
111            return Ok(());
112        };
113        let bytes = self.serialize(m)?;
114        self.inner
115            .unicast(Slot::new(*v), target, bytes)
116            .map_err(to_network_error)?;
117        Ok(())
118    }
119
120    fn multicast(
121        &mut self,
122        v: ViewNumber,
123        to: Vec<&T::SignatureKey>,
124        m: &Message<T, Validated>,
125    ) -> Result<(), NetworkError> {
126        let bytes = self.serialize(m)?;
127        let mut targets = Vec::new();
128        for t in to {
129            if let Some(info) = self.peers.get(t) {
130                targets.push(info.x25519_key.into())
131            } else if *t == self.my_keys.0 {
132                targets.push(self.my_keys.1)
133            } else {
134                error!(peer = %t, "multicast target not found");
135            }
136        }
137        self.inner
138            .multicast(Slot::new(*v), targets, bytes)
139            .map_err(to_network_error)?;
140        Ok(())
141    }
142
143    fn broadcast(&mut self, v: ViewNumber, m: &Message<T, Validated>) -> Result<(), NetworkError> {
144        let bytes = self.serialize(m)?;
145        self.inner
146            .broadcast(Slot::new(*v), bytes)
147            .map_err(to_network_error)?;
148        Ok(())
149    }

171    fn add_peers(
172        &mut self,
173        r: PeerRole,
174        ps: Vec<(T::SignatureKey, Self::PeerData)>,
175    ) -> Result<(), NetworkError> {
176        let mut targets = Vec::new();
177        for (k, (x, a)) in ps {
178            self.peers.insert(
179                k,
180                PeerConnectInfo {
181                    x25519_key: x.into(),
182                    p2p_addr: a.clone(),
183                },
184            );
185            targets.push((x, a))
186        }
187        self.inner
188            .add_peers(map_peer_role(r), targets)
189            .map_err(to_network_error)?;
190        Ok(())
191    }
192
193    fn remove_peers(&mut self, ps: Vec<&T::SignatureKey>) -> Result<(), NetworkError> {
194        let mut targets = Vec::new();
195        for k in ps {
196            if let Some(info) = self.peers.remove(k) {
197                targets.push(info.x25519_key.into())
198            }
199        }
200        self.inner.remove_peers(targets).map_err(to_network_error)?;
201        Ok(())
202    }
203
204    fn assign_role(&mut self, r: PeerRole, ps: Vec<&T::SignatureKey>) -> Result<(), NetworkError> {
205        let mut targets = Vec::new();
206        for k in ps {
207            if let Some(info) = self.peers.get(k) {
208                targets.push(info.x25519_key.into())
209            }
210        }
211        self.inner
212            .assign_peers(map_peer_role(r), targets)
213            .map_err(to_network_error)?;
214        Ok(())
215    }

217    /// Update peers on every epoch change.
218    ///
219    /// For any given epoch `e` we collect the validators of `e`, `e-1` and
220    /// `e+1` from the stake tables and merge their connection information.
221    ///
222    /// We keep validators that were in `e-1` but not in `e` for one additional
223    /// epoch and eagerly connect to new validators of `e+1`.
224    async fn apply_epoch(
225        &mut self,
226        epoch: EpochNumber,
227        coord: &EpochMembershipCoordinator<T>,
228    ) -> Result<(), NetworkError> {
229        if epoch <= self.epoch {
230            info!(%epoch, ours = %self.epoch, "epoch already seen");
231            return Ok(());
232        }
233
234        // Validators of the new epoch.
235        let Some(curr_infos) = coord.epoch_peers(Some(epoch)) else {
236            error!(%epoch, "no stake table available");
237            return Ok(());
238        };
239
240        // Validators leaving are retained as peers for one additional epoch.
241        let prev_infos = if *epoch > 0 {
242            coord.epoch_peers(Some(epoch - 1)).unwrap_or_else(|| {
243                info!(%epoch, "previous epoch's stake table unavailable");
244                HashMap::new()
245            })
246        } else {
247            HashMap::new()
248        };
249
250        // Validators joining in the next epoch are connected to early.
251        let next_infos = coord.epoch_peers(Some(epoch + 1)).unwrap_or_else(|| {
252            info!(%epoch, "next epoch's stake table not available");
253            HashMap::new()
254        });
255
256        // Since connection information may be updated, we need to merge them,
257        // preferring the newest epoch's data, i.e. `next(curr(prev))`.
258        let mut merged_infos = prev_infos.clone();
259        for (k, v) in curr_infos.iter().chain(&next_infos) {
260            merged_infos.insert(k.clone(), v.clone());
261        }
262
263        let wanted: HashSet<T::SignatureKey> = curr_infos
264            .keys()
265            .chain(next_infos.keys())
266            .cloned()
267            .collect();
268
269        let retained: HashSet<T::SignatureKey> = curr_infos
270            .keys()
271            .chain(prev_infos.keys())
272            .cloned()
273            .collect();
274
275        let mut to_add: Vec<(T::SignatureKey, PeerConnectInfo)> = Vec::new();
276        let mut to_del: Vec<(T::SignatureKey, PeerConnectInfo)> = Vec::new();
277
278        for k in &wanted {
279            if let Some(Some(new_info)) = merged_infos.get(k) {
280                if Some(new_info) != self.peers.get(k) {
281                    info!(%epoch, peer = %k, "adding/updating network peer");
282                    to_add.push((k.clone(), new_info.clone()));
283                } else {
284                    info!(%epoch, peer = %k, "peer unchanged");
285                }
286            } else {
287                info!(%epoch, peer = %k, "ignoring peer without connection info");
288            }
289        }
290
291        // Remove peers that have left both the current and previous epochs.
292        for (k, info) in &self.peers {
293            if !(retained.contains(k) || wanted.contains(k)) {
294                info!(%epoch, peer = %k, "removing network peer");
295                to_del.push((k.clone(), info.clone()));
296            }
297        }
298
299        for (k, _) in &to_del {
300            self.peers.remove(k);
301        }
302        for (k, info) in &to_add {
303            self.peers.insert(k.clone(), info.clone());
304        }
305
306        let add_targets: Vec<(PublicKey, NetAddr)> = to_add
307            .iter()
308            .map(|(_, i)| (i.x25519_key.into(), i.p2p_addr.clone()))
309            .collect();
310        let del_targets: Vec<PublicKey> = to_del.iter().map(|(_, i)| i.x25519_key.into()).collect();
311
312        if let Err(err) = self.inner.add_peers(Role::Active, add_targets) {
313            error!(%epoch, %err, "network down; could not add peers to network");
314            return Err(to_network_error(err));
315        }
316
317        if let Err(err) = self.inner.remove_peers(del_targets) {
318            error!(%epoch, %err, "network down; could not remove peers from network");
319            return Err(to_network_error(err));
320        }
321
322        info!(%epoch, peers = %self.peers.len());
323
324        self.epoch = epoch;
325        Ok(())
326    }
327}

329impl<T: NodeType> Cliquenet<T> {
330    fn serialize(&self, m: &Message<T, Validated>) -> Result<Vec<u8>, NetworkError> {
331        if let MessageType::External(bytes) = &m.message_type {
332            return Ok(bytes.clone());
333        }
334        self.upgrade_lock
335            .serialize(m)
336            .map_err(|e| NetworkError::Io(format!("serialization error: {e}").into()))
337    }
338
339    fn deserialize(&self, bytes: &[u8]) -> Result<Message<T, Unchecked>, NetworkError> {
340        match self
341            .upgrade_lock
342            .deserialize::<Message<T, Unchecked>>(bytes)
343        {
344            Ok((m, v)) => {
345                if v == EXTERNAL_MESSAGE_VERSION && !m.is_external() {
346                    let e = "received a non-external message with version 0.0".to_string();
347                    return Err(NetworkError::Io(e.into()));
348                }
349                Ok(m)
350            },
351            Err(primary_err) => {
352                // Fallback: bytes may be a hotshot-types `Message<T>` carrying
353                // an `External` payload (this is how `Leaf2Fetcher` in the
354                // membership layer frames leaf-catchup requests/responses).
355                // If so, surface it as `MessageType::External` so the
356                // Coordinator can route it to the membership external
357                // channel just like a native new-protocol external message.
358                if let Ok((_v, hs_msg)) =
359                    versions::decode::<hotshot_types::message::Message<T>>(bytes)
360                    && let MessageKind::External(data) = hs_msg.kind
361                {
362                    return Ok(Message {
363                        sender: hs_msg.sender,
364                        message_type: MessageType::External(data),
365                    });
366                }
367                Err(NetworkError::Io(primary_err.to_string().into()))
368            },
369        }
370    }
371}

373fn map_peer_role(r: PeerRole) -> Role {
374    match r {
375        PeerRole::Active => Role::Active,
376        PeerRole::Passive => Role::Passive,
377    }

380fn to_network_error(e: CliquenetError) -> NetworkError {
381    match e {
382        e @ CliquenetError::Bind(..) => NetworkError::Critical(e.into()),
383        e @ CliquenetError::ChannelClosed => NetworkError::Critical(e.into()),
384        e @ CliquenetError::BudgetClosed => NetworkError::Critical(e.into()),
385        e => NetworkError::Io(e.into()),
386    }
387}

389struct CliquenetMetrics {
390    metrics: Box<dyn Metrics>,
391    gauges: RwLock<Gauges>,
392    counters: RwLock<Counters>,
393}
394
395#[derive(Default)]
396struct Gauges {
397    gauges: HashMap<PublicKey, HashMap<String, Box<dyn Gauge>>>,
398    family: HashMap<String, Box<dyn GaugeFamily>>,
399}
400
401#[derive(Default)]
402struct Counters {
403    counters: HashMap<PublicKey, HashMap<String, Box<dyn Counter>>>,
404    family: HashMap<String, Box<dyn CounterFamily>>,
405}
406
407impl CliquenetMetrics {
408    pub fn new(m: Box<dyn Metrics>) -> Self {
409        Self {
410            metrics: m.subgroup("cliquenet".to_string()),
411            gauges: RwLock::new(Gauges::default()),
412            counters: RwLock::new(Counters::default()),
413        }
414    }
415}
416
417// In here we lazily create counters and gauges based on their labels.
418// If not found, we create a family using the label, e.g. "connect_attempts",
419// indexed by the peer (key). Afterwards we create the actual counter or gauge,
420// and update its value. On the next call, the metric would be found and
421// updated right away.
422impl cliquenet::Metrics for CliquenetMetrics {
423    fn set(&self, key: &PublicKey, label: &str, val: usize) {
424        if let Some(g) = self
425            .gauges
426            .read()
427            .gauges
428            .get(key)
429            .and_then(|m| m.get(label))
430        {
431            return g.set(val);
432        }
433
434        let mut gauges = self.gauges.write();
435
436        // Check again, in case a concurrent write has created the gauge:
437        if let Some(g) = gauges.gauges.get(key).and_then(|m| m.get(label)) {
438            return g.set(val);
439        }
440
441        let g = gauges
442            .family
443            .entry(label.to_string())
444            .or_insert_with(|| {
445                self.metrics
446                    .gauge_family(label.to_string(), vec!["peer".to_string()])
447            })
448            .create(vec![key.to_string()]);
449
450        gauges
451            .gauges
452            .entry(*key)
453            .or_default()
454            .entry(label.to_string())
455            .or_insert(g)
456            .set(val)
457    }
458
459    fn add(&self, key: &PublicKey, label: &str, val: usize) {
460        if let Some(c) = self
461            .counters
462            .read()
463            .counters
464            .get(key)
465            .and_then(|m| m.get(label))
466        {
467            return c.add(val);
468        }
469
470        let mut counters = self.counters.write();
471
472        // Check again, in case a concurrent write has created the counter:
473        if let Some(c) = counters.counters.get(key).and_then(|m| m.get(label)) {
474            return c.add(val);
475        }
476
477        let c = counters
478            .family
479            .entry(label.to_string())
480            .or_insert_with(|| {
481                self.metrics
482                    .counter_family(label.to_string(), vec!["peer".to_string()])
483            })
484            .create(vec![key.to_string()]);
485
486        counters
487            .counters
488            .entry(*key)
489            .or_default()
490            .entry(label.to_string())
491            .or_insert(c)
492            .add(val)
493    }
494
495    fn del(&self, key: &PublicKey) {
496        let key_string = key.to_string();
497
498        {
499            let mut gauges = self.gauges.write();
500            for (label, _) in gauges.gauges.remove(key).into_iter().flatten() {
501                if let Some(f) = gauges.family.get(&label) {
502                    f.destroy(&[&key_string]);
503                }
504            }
505        }
506
507        {
508            let mut counters = self.counters.write();
509            for (label, _) in counters.counters.remove(key).into_iter().flatten() {
510                if let Some(f) = counters.family.get(&label) {
511                    f.destroy(&[&key_string]);
512                }
513            }
514        }
515    }
