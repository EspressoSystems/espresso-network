20const RETRY_DELAY: Duration = Duration::from_millis(300);

22/// New protocol storage extension for data that is not part of the legacy HotShot storage trait.
23#[async_trait]
24pub trait NewProtocolStorage<T: NodeType>: StorageTrait<T> {
25    async fn append_cert2(&self, view: ViewNumber, cert: Certificate2<T>) -> anyhow::Result<()>;
26}
27
28pub struct Storage<T: NodeType, S: NewProtocolStorage<T>> {
29    storage: S,
30    private_key: <T::SignatureKey as SignatureKey>::PrivateKey,
31    handles: BTreeMap<ViewNumber, Vec<JoinHandle<()>>>,
32}
33
34impl<T: NodeType, S: NewProtocolStorage<T>> Storage<T, S> {
35    pub fn new(storage: S, private_key: <T::SignatureKey as SignatureKey>::PrivateKey) -> Self {
36        Self {
37            storage,
38            private_key,
39            handles: BTreeMap::new(),
40        }
41    }
42
43    pub fn append_vid(&mut self, vid_share: VidDisperseShare2<T>) {
44        let view = vid_share.view_number;
45        let storage = self.storage.clone();
46        let private_key = self.private_key.clone();
47        let handle = spawn(async move {
48            let share: VidDisperseShare<T> = VidDisperseShare::V2(vid_share);
49            let Some(proposal) = share.to_proposal(&private_key) else {
50                error!("failed to sign VID share for storage");
51                return;
52            };
53            loop {
54                match storage.append_vid(&proposal).await {
55                    Ok(()) => return,
56                    Err(err) => {
57                        warn!(%err, "failed to append VID share, retrying");
58                        sleep(RETRY_DELAY).await;
59                    },
60                }
61            }
62        });
63        self.handles.entry(view).or_default().push(handle);
64    }
65
66    pub fn append_da(
67        &mut self,
68        view_number: ViewNumber,
69        epoch: EpochNumber,
70        block_payload: T::BlockPayload,
71        metadata: <T::BlockPayload as BlockPayload<T>>::Metadata,
72        vid_commit: VidCommitment,
73    ) {
74        let storage = self.storage.clone();
75        let private_key = self.private_key.clone();
76        let handle = spawn(async move {
77            let data = DaProposal2 {
78                encoded_transactions: block_payload.encode(),
79                metadata,
80                view_number,
81                epoch: Some(epoch),
82                epoch_transition_indicator: EpochTransitionIndicator::NotInTransition,
83            };
84            let Ok(signature) = T::SignatureKey::sign(&private_key, &[]) else {
85                error!("failed to sign DA proposal for storage");
86                return;
87            };
88            let proposal = SignedProposal {
89                data,
90                signature,
91                _pd: PhantomData,
92            };
93            loop {
94                match storage.append_da2(&proposal, vid_commit).await {
95                    Ok(()) => return,
96                    Err(err) => {
97                        warn!(%err, "failed to append DA proposal, retrying");
98                        sleep(RETRY_DELAY).await;
99                    },
100                }
101            }
102        });
103        self.handles.entry(view_number).or_default().push(handle);
104    }
105
106    pub fn append_cert2(&mut self, view: ViewNumber, cert2: Certificate2<T>) {
107        let storage = self.storage.clone();
108        let handle = spawn(async move {
109            loop {
110                match storage.append_cert2(view, cert2.clone()).await {
111                    Ok(()) => return,
112                    Err(err) => {
113                        warn!(%err, %view, "failed to append cert2, retrying");
114                        sleep(RETRY_DELAY).await;
115                    },
116                }
117            }
118        });
119        self.handles.entry(view).or_default().push(handle);
120    }
121
122    pub fn append_proposal(&mut self, proposal: Proposal<T>) {
123        let view = proposal.view_number;
124        let storage = self.storage.clone();
125        let private_key = self.private_key.clone();
126        let handle = spawn(async move {
127            let data = QuorumProposalWrapper {
128                proposal: QuorumProposal2 {
129                    block_header: proposal.block_header,
130                    view_number: proposal.view_number,
131                    epoch: Some(proposal.epoch),
132                    justify_qc: proposal.justify_qc,
133                    next_epoch_justify_qc: None,
134                    upgrade_certificate: proposal.upgrade_certificate,
135                    view_change_evidence: proposal
136                        .view_change_evidence
137                        .map(ViewChangeEvidence2::Timeout),
138                    next_drb_result: proposal.next_drb_result,
139                    state_cert: proposal.state_cert,
140                },
141            };
142            let Ok(signature) = T::SignatureKey::sign(&private_key, &[]) else {
143                error!("failed to sign quorum proposal for storage");
144                return;
145            };
146            let signed = SignedProposal {
147                data,
148                signature,
149                _pd: PhantomData,
150            };
151            loop {
152                match storage.append_proposal_wrapper(&signed).await {
153                    Ok(()) => return,
154                    Err(err) => {
155                        warn!(%err, "failed to append proposal, retrying");
156                        sleep(RETRY_DELAY).await;
157                    },
158                }
159            }
160        });
161        self.handles.entry(view).or_default().push(handle);
162    }
163
164    pub fn gc(&mut self, view_number: ViewNumber) {
165        let keep = self.handles.split_off(&view_number);
166        for handles in self.handles.values() {
167            for handle in handles {
168                handle.abort();
169            }
170        }
171        self.handles = keep;
172    }
173}
