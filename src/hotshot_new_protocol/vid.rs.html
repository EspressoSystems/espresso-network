42/// Why a reconstruction attempt failed.
43#[derive(Clone, Copy, Debug, Eq, PartialEq, thiserror::Error)]
44pub enum VidReconstructErrorKind {
45    /// Unverifiable shares were weeded out; reconstruction retries once the
46    /// remaining shares cover the recovery threshold again.
47    #[error("awaiting more shares after weeding out unverifiable ones")]
48    AwaitingShares,
49    /// Every share verified yet the payload still does not re-commit: the
50    /// disperser committed to a non-codeword, so no subset can ever recover it.
51    #[error("unrecoverable: verified shares cannot decode to a payload matching the commitment")]
52    Unrecoverable,
53}

75    matches!(

116    pub fn request_vid_disperse(&mut self, vid_disperse_request: VidDisperseRequest<T>) {
117        let key = (
118            vid_disperse_request.view,
119            vid_disperse_request.payload_commitment,
120        );
121        if self.calculations.contains_key(&key) {
122            return;
123        }
124        let handle = self.tasks.spawn(Self::handle_vid_disperse_request(
125            self.epoch_membership_coordinator.clone(),
126            self.network.clone(),
127            self.public_key.clone(),
128            self.private_key.clone(),
129            vid_disperse_request,
130        ));
131        self.calculations.insert(key, handle);
132    }
133
134    pub async fn next(&mut self) -> Option<Result<VidDisperseOutput, VidDisperseError>> {
135        loop {
136            match self.tasks.join_next().await {
137                Some(Ok(result)) => return Some(result),
138                Some(Err(err)) => {
139                    if err.is_panic() {
140                        error!(%err, "vid disperse task panic");
141                    }
142                    continue;
143                },
144                None => return None,
145            }
146        }
147    }
148
149    async fn handle_vid_disperse_request(
150        epoch_membership_coordinator: EpochMembershipCoordinator<T>,
151        network: Sender<T>,
152        public_key: T::SignatureKey,
153        private_key: <T::SignatureKey as SignatureKey>::PrivateKey,
154        vid_disperse_request: VidDisperseRequest<T>,
155    ) -> Result<VidDisperseOutput, VidDisperseError> {
156        let view = vid_disperse_request.view;
157        let (disperse, _) = VidDisperse2::calculate_vid_disperse(
158            &vid_disperse_request.block,
159            &epoch_membership_coordinator,
160            view,
161            Some(vid_disperse_request.epoch),
162            Some(vid_disperse_request.epoch),
163            &vid_disperse_request.metadata,
164        )
165        .await
166        .map_err(VidDisperseError::Vid)?;
167
168        let payload_commitment = disperse.payload_commitment;
169
170        let signature = T::SignatureKey::sign(&private_key, payload_commitment.as_ref())
171            .map_err(|err| VidDisperseError::Sign(err.into()))?;
172
173        for proposal in disperse.to_share_proposals(&signature) {
174            let recipient = proposal.data.recipient_key.clone();
175            let message = Message {
176                sender: public_key.clone(),
177                message_type: MessageType::Consensus(ConsensusMessage::VidShare(proposal)),
178            };
179            if let Err(err) = network.unicast(view, &recipient, &message) {
180                if err.is_critical() {
181                    return Err(err.into());
182                } else {
183                    warn!(%err, "network error while sending vid share")
184                }
185            }
186        }
187
188        Ok(VidDisperseOutput {
189            view,
190            payload_commitment,
191        })
192    }

194    pub fn gc(&mut self, view_number: ViewNumber) {
195        let keep = self
196            .calculations
197            .split_off(&(view_number, VidCommitment2::default()));
198        for handle in self.calculations.values_mut() {
199            handle.abort();
200        }
201        self.calculations = keep;
202    }

205pub(crate) struct VidShareAccumulator<T: NodeType> {
206    /// The payload commitment claimed by the view's validated proposal.
207    payload_commitment: VidCommitment2,
208    metadata: Metadata<T>,
209    epoch: EpochNumber,
210    /// The VID erasure parameters the committee fixes for this view, used to
211    /// reject shares carrying a forged `common.param` (see [`Self::accept`]).
212    /// `None` if the committee could not be resolved; the param check is then
213    /// skipped, matching the previously unchecked path.
214    expected_param: Option<AvidmGf2Param>,
215    /// Common data pinned by the first admitted share; hash-bound to
216    /// `payload_commitment` (see [`Self::accept`]).
217    common: Option<AvidmGf2Common>,
218    /// Admitted shares by voter; their shard ranges are pairwise disjoint
219    /// (see [`Self::accept`]).
220    shares: BTreeMap<T::SignatureKey, AvidmGf2Share>,
221    /// Every voter whose share was admitted, including weeded ones: a voter
222    /// whose share failed verification doesn't get a second submission.
223    seen_keys: HashSet<T::SignatureKey>,
224    /// Set when a fully verified share set failed to decode to a payload
225    /// matching the commitment: the disperser is provably faulty and further
226    /// attempts are pointless.
227    exhausted: bool,
228}
229
230impl<T: NodeType> VidShareAccumulator<T> {
231    fn new(
232        payload_commitment: VidCommitment2,
233        metadata: Metadata<T>,
234        epoch: EpochNumber,
235        expected_param: Option<AvidmGf2Param>,
236    ) -> Self {
237        Self {
238            payload_commitment,
239            metadata,
240            epoch,
241            expected_param,
242            common: None,
243            shares: BTreeMap::new(),
244            seen_keys: HashSet::new(),
245            exhausted: false,
246        }
247    }
248
249    /// Admit `share` from the authenticated `sender`, dropping it if it fails
250    /// any intake check. The non-overlapping path stays crypto-free; a
251    /// shard-range overlap is the only case that triggers per-share
252    /// verification, via [`Self::resolve_conflict`].
253    fn accept(&mut self, view: ViewNumber, sender: T::SignatureKey, share: VidDisperseShare2<T>) {
254        if self.exhausted {
255            return;
256        }
257        if share.payload_commitment != self.payload_commitment {
258            warn!(%view, ?sender, "VID share commitment differs from the proposal's");
259            return;
260        }
261        // The commitment binds a share's `ns_commits` but not its `param`, so a
262        // Byzantine voter can pair real `ns_commits` with a forged `param` (e.g.
263        // an inflated `recovery_threshold`). Pinning that common as the
264        // verification oracle would reject every honest share, so reject it now.
265        if let Some(expected) = &self.expected_param
266            && share.common.param != *expected
267        {
268            warn!(%view, ?sender, "VID share common param differs from the committee's");
269            return;
270        }
271        // The commitment hash-binds the common, so trust it as the verification
272        // oracle only after that check; later shares must carry the same common.
273        if let Some(common) = &self.common {
274            if share.common != *common {
275                warn!(%view, ?sender, "VID share common differs from the accumulator's");
276                return;
277            }
278        } else if AvidmGf2Scheme::is_consistent(&self.payload_commitment, &share.common) {
279            self.common = Some(share.common.clone());
280        } else {
281            warn!(%view, ?sender, "VID share common is inconsistent with its commitment");
282            return;
283        }
284        // A share whose namespaces disagree on the shard range is malformed.
285        let Some(range) = share.share.range() else {
286            warn!(%view, ?sender, "VID share has an inconsistent shard range");
287            return;
288        };
289        // An empty range contributes nothing; positions past the end of the
290        // encoded payload would inflate coverage without aiding decoding.
291        if range.is_empty() || range.end > share.common.param.total_weights {
292            warn!(%view, ?sender, ?range, "VID share has an empty or out-of-bounds shard range");
293            return;
294        }
295        if self.seen_keys.contains(&sender) {
296            return;
297        }
298        // Honest dispersal assigns disjoint ranges, so an overlap with an
299        // admitted share proves a squat; resolve it (needs verification) below.
300        let conflicts: Vec<T::SignatureKey> = self
301            .shares
302            .iter()
303            .filter(|(_, admitted)| {
304                admitted
305                    .range()
306                    .is_some_and(|covered| covered.start < range.end && range.start < covered.end)
307            })
308            .map(|(key, _)| key.clone())
309            .collect();
310        if conflicts.is_empty() {
311            self.seen_keys.insert(sender.clone());
312            self.shares.insert(sender, share.share);
313            return;
314        }
315        self.resolve_conflict(view, sender, share, conflicts);
316    }
317
318    /// Resolve a shard-range collision between the incoming `share` and the
319    /// already-admitted `conflicts`: verify each against the commitment-bound
320    /// common, evict those that fail, and admit the newcomer only if it
321    /// verifies and no surviving share still covers its range.
322    fn resolve_conflict(
323        &mut self,
324        view: ViewNumber,
325        sender: T::SignatureKey,
326        share: VidDisperseShare2<T>,
327        conflicts: Vec<T::SignatureKey>,
328    ) {
329        // A conflict implies a prior admission, which pinned the common.
330        let Some(common) = self.common.clone() else {
331            return;
332        };
333        // The sender has used its one slot regardless of the outcome.
334        self.seen_keys.insert(sender.clone());
335        let mut survivor = false;
336        for key in conflicts {
337            let verified = self
338                .shares
339                .get(&key)
340                .is_some_and(|admitted| share_verifies(&common, admitted));
341            if verified {
342                survivor = true;
343            } else {
344                warn!(%view, ?key, "evicting unverifiable VID share squatting a shard range");
345                self.shares.remove(&key);
346            }
347        }
348        // A verified share still covers the contested range: the newcomer would
349        // double-cover it, so drop it.
350        if survivor {
351            return;
352        }
353        if share_verifies(&common, &share.share) {
354            self.shares.insert(sender, share.share);
355        } else {
356            warn!(%view, ?sender, "dropping unverifiable VID share at intake conflict");
357        }
358    }
359
360    fn ranges(&self) -> impl Iterator<Item = &Range<usize>> {
361        // Admitted shares always have a consistent range (checked in `accept`).
362        self.shares.values().filter_map(AvidmGf2Share::range)
363    }
364
365    /// Number of shard positions covered by the admitted shares; exact
366    /// because their ranges are disjoint.
367    fn coverage(&self) -> usize {
368        self.ranges().map(ExactSizeIterator::len).sum()
369    }
370
371    fn has_enough_shares(&self) -> bool {
372        self.common
373            .as_ref()
374            .is_some_and(|common| self.coverage() >= common.param.recovery_threshold)
375    }
376}

378#[derive(Default)]
379pub struct VidReconstructor<T: NodeType> {
380    /// Shares that arrived before their view's proposal, one per voter:
381    /// admitted (or dropped) once the proposal pins the view's commitment.
382    pending: BTreeMap<ViewNumber, BTreeMap<T::SignatureKey, VidDisperseShare2<T>>>,
383    /// One accumulator per view, created when its validated proposal
384    /// arrives: reconstruction needs the proposal's metadata, so only the
385    /// proposal's commitment is worth accumulating.
386    accumulators: BTreeMap<ViewNumber, VidShareAccumulator<T>>,
387    reconstructed: BTreeSet<ViewNumber>,
388    tasks: JoinSet<ReconstructResult<T>>,
389    calculations: BTreeMap<ViewNumber, AbortHandle>,
390}
391
392impl<T: NodeType> VidReconstructor<T> {
393    pub fn new() -> Self {
394        Self {
395            pending: BTreeMap::new(),
396            accumulators: BTreeMap::new(),
397            reconstructed: BTreeSet::new(),
398            tasks: JoinSet::new(),
399            calculations: BTreeMap::new(),
400        }
401    }
402
403    /// Pin `view` to its validated proposal's payload commitment and
404    /// metadata, and admit any shares that arrived before the proposal.
405    pub(crate) fn handle_proposal(
406        &mut self,
407        view: ViewNumber,
408        payload_commitment: VidCommitment2,
409        metadata: Metadata<T>,
410        epoch: EpochNumber,
411        expected_param: Option<AvidmGf2Param>,
412    ) {
413        if self.reconstructed.contains(&view) {
414            return;
415        }
416        let accumulator = match self.accumulators.entry(view) {
417            Entry::Occupied(existing) => {
418                // The first proposal wins: an equivocating leader cannot re-pin
419                // the view to another commitment.
420                if existing.get().payload_commitment != payload_commitment {
421                    warn!(%view, "conflicting proposal for a view pinned to another commitment");
422                }
423                return;
424            },
425            Entry::Vacant(slot) => slot.insert(VidShareAccumulator::new(
426                payload_commitment,
427                metadata,
428                epoch,
429                expected_param,
430            )),
431        };
432        for (sender, share) in self.pending.remove(&view).into_iter().flatten() {
433            accumulator.accept(view, sender, share);
434        }
435        self.try_reconstruct(view);
436    }
437
438    pub(crate) fn handle_vid_share(
439        &mut self,
440        sender: T::SignatureKey,
441        share: VidDisperseShare2<T>,
442    ) {
443        let view = share.view_number;
444        // A share carries the voter it belongs to; only the authenticated
445        // sender may contribute its own. This cheap check bounds each node to
446        // one slot and guards the pre-proposal `pending` window, where a share
447        // cannot yet be verified against a commitment.
448        if share.recipient_key != sender {
449            warn!(%view, ?sender, "VID share recipient key does not match its sender");
450            return;
451        }
452        if self.reconstructed.contains(&view) {
453            return;
454        }
455        let Some(accumulator) = self.accumulators.get_mut(&view) else {
456            // No validated proposal yet: hold the voter's share until
457            // `handle_proposal` pins the view's commitment.
458            self.pending
459                .entry(view)
460                .or_default()
461                .entry(sender)
462                .or_insert(share);
463            return;
464        };
465        accumulator.accept(view, sender, share);
466        self.try_reconstruct(view);
467    }
468
469    pub async fn next(&mut self) -> Option<ReconstructResult<T>> {
470        loop {
471            match self.tasks.join_next().await {
472                Some(Ok(Ok(out))) => {
473                    self.calculations.remove(&out.view);
474                    self.accumulators.remove(&out.view);
475                    self.reconstructed.insert(out.view);
476                    return Some(Ok(out));
477                },
478                Some(Ok(Err(err))) => {
479                    self.calculations.remove(&err.view);
480                    self.handle_failed_attempt(&err);
481                    return Some(Err(err));
482                },
483                Some(Err(_)) => continue,
484                None => return None,
485            }
486        }
487    }
488
489    /// Apply the outcome of a failed attempt: weed the bad shares out of the
490    /// accumulator, then either mark the payload as unrecoverable or retry
491    /// (`try_reconstruct` re-checks coverage, which shares that arrived while
492    /// the attempt ran may already restore).
493    fn handle_failed_attempt(&mut self, err: &VidReconstructError<T::SignatureKey>) {
494        let Some(accumulator) = self.accumulators.get_mut(&err.view) else {
495            return;
496        };
497        // Views are pinned to one commitment for their lifetime, so a
498        // finished attempt always matches; guard anyway so a future re-pin
499        // policy can't weed the wrong accumulator.
500        if accumulator.payload_commitment != err.payload_commitment {
501            return;
502        }
503        for key in &err.bad_share_keys {
504            accumulator.shares.remove(key);
505        }
506        match err.kind {
507            VidReconstructErrorKind::Unrecoverable => accumulator.exhausted = true,
508            VidReconstructErrorKind::AwaitingShares => self.try_reconstruct(err.view),
509        }
510    }
511
512    fn try_reconstruct(&mut self, view: ViewNumber) {
513        if self.calculations.contains_key(&view) {
514            return;
515        }
516        let Some(accumulator) = self.accumulators.get(&view) else {
517            return;
518        };
519        if accumulator.exhausted || !accumulator.has_enough_shares() {
520            return;
521        }
522        // Enough shares implies an admitted share, which pinned the common.
523        let Some(common) = accumulator.common.clone() else {
524            return;
525        };
526        let payload_commitment = accumulator.payload_commitment;
527        let metadata = accumulator.metadata.clone();
528        let shares: Vec<(T::SignatureKey, AvidmGf2Share)> = accumulator
529            .shares
530            .iter()
531            .map(|(key, share)| (key.clone(), share.clone()))
532            .collect();
533        let epoch = accumulator.epoch;
534        let task = self.tasks.spawn_blocking(move || {
535            reconstruct::<T>(view, epoch, payload_commitment, common, shares, metadata)
536        });
537        self.calculations.insert(view, task);
538    }
539
540    pub fn gc(&mut self, view_number: ViewNumber) {
541        let keep = self.calculations.split_off(&view_number);
542        for handle in self.calculations.values_mut() {
543            handle.abort();
544        }
545        self.calculations = keep;
546        self.pending = self.pending.split_off(&view_number);
547        self.accumulators = self.accumulators.split_off(&view_number);
548        self.reconstructed = self.reconstructed.split_off(&view_number);
549    }
550
551    /// Stop tracking `view`.
552    ///
553    /// Either because its payload was reconstructed (or obtained elsewhere)
554    /// or because it timed out and will never be decided: record it so
555    /// `handle_vid_share` ignores later shares, drop its accumulator and
556    /// pending shares, and abort any in-flight reconstruction task.
557    pub fn retire_view(&mut self, view: ViewNumber) {
558        self.reconstructed.insert(view);
559        self.pending.remove(&view);
560        self.accumulators.remove(&view);
561        if let Some(handle) = self.calculations.remove(&view) {
562            handle.abort();
563        }
564    }
565}

567/// Decode the shares and accept the result only if it re-commits to
568/// `payload_commitment`. On failure, report the shares that fail
569/// verification against the commitment (each share is self-authenticating
570/// via its merkle proofs) so they can be weeded out. If every share
571/// verifies, the payload is unrecoverable: the shares cover the recovery
572/// threshold with disjoint ranges, so the disperser committed to a
573/// non-codeword and no share subset can ever succeed.
574fn reconstruct<T: NodeType>(
575    view: ViewNumber,
576    epoch: EpochNumber,
577    payload_commitment: VidCommitment2,
578    common: AvidmGf2Common,
579    shares: Vec<(T::SignatureKey, AvidmGf2Share)>,
580    metadata: Metadata<T>,
581) -> ReconstructResult<T> {
582    let (keys, shares): (Vec<_>, Vec<_>) = shares.into_iter().unzip();
583    if let Some(bytes) =
584        decode_and_recommit::<T>(view, &common, &shares, &payload_commitment, &metadata)
585    {
586        return Ok(output(view, epoch, payload_commitment, &bytes, metadata));
587    }
588    let bad_share_keys: Vec<_> = keys
589        .into_iter()
590        .zip(&shares)
591        .filter(|(_, share)| !share_verifies(&common, share))
592        .map(|(key, _)| key)
593        .collect();
594    let kind = if bad_share_keys.is_empty() {
595        warn!(
596            %view,
597            %payload_commitment,
598            "verified shares cannot decode to a payload matching the commitment"
599        );
600        VidReconstructErrorKind::Unrecoverable
601    } else {
602        warn!(
603            %view,
604            %payload_commitment,
605            ?bad_share_keys,
606            "weeded out VID shares that failed verification"
607        );
608        VidReconstructErrorKind::AwaitingShares
609    };
610    Err(VidReconstructError {
611        view,
612        payload_commitment,
613        kind,
614        bad_share_keys,
615    })
616}

618/// Decode `shares` and return the payload bytes only if they re-commit to
619/// `payload_commitment`. Recovery alone does not bind the decoded bytes
620/// to the commitment: a Byzantine disperser can commit to a non-codeword,
621/// and a bad share poisons the erasure decoding.
622fn decode_and_recommit<T: NodeType>(
623    view: ViewNumber,
624    common: &AvidmGf2Common,
625    shares: &[AvidmGf2Share],
626    payload_commitment: &VidCommitment2,
627    metadata: &Metadata<T>,
628) -> Option<Vec<u8>> {
629    let bytes = match AvidmGf2Scheme::recover(common, shares) {
630        Ok(bytes) => bytes,
631        Err(err) => {
632            warn!(%view, %err, "VID recovery failed");
633            return None;
634        },
635    };
636    let ns_table = parse_ns_table(bytes.len(), &metadata.encode());
637    match AvidmGf2Scheme::commit(&common.param, &bytes, ns_table) {
638        Ok((recomputed, _)) if recomputed == *payload_commitment => Some(bytes),
639        Ok((recomputed, _)) => {
640            warn!(
641                %view,
642                expected = %payload_commitment,
643                %recomputed,
644                "reconstructed payload does not match the payload commitment"
645            );
646            None
647        },
648        Err(err) => {
649            warn!(%view, %err, "failed to recommit reconstructed VID payload");
650            None
651        },
652    }
653}
654
655fn output<T: NodeType>(
656    view: ViewNumber,
657    epoch: EpochNumber,
658    payload_commitment: VidCommitment2,
659    bytes: &[u8],
660    metadata: Metadata<T>,
661) -> VidReconstructOutput<T> {
662    let payload = T::BlockPayload::from_bytes(bytes, &metadata);
663    let tx_commitments = payload.transaction_commitments(&metadata);
664    VidReconstructOutput {
665        view,
666        epoch,
667        payload_commitment,
668        payload,
669        metadata,
670        tx_commitments,
671    }
672}

675pub enum VidDisperseError {

677    Net(#[from] NetworkError),
678
679    #[error("vid error: {0}")]

683    Sign(#[source] Box<dyn std::error::Error + Send + Sync>),
684}

688        match self {
689            Self::Net(e) => e.is_critical(),

691        }
692    }
