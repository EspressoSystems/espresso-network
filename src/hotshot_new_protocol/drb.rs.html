1use std::collections::{BTreeMap, BTreeSet};
2
3use hotshot_types::{
4    data::EpochNumber,
5    drb::{DrbInput, DrbResult, compute_drb_result},
6    traits::storage::{LoadDrbProgressFn, StoreDrbProgressFn},
7};
8use tokio::task::{AbortHandle, JoinSet};
9use tracing::warn;
10
11pub struct DrbRequester {
12    calculations: BTreeMap<EpochNumber, AbortHandle>,
13    completed_epochs: BTreeSet<EpochNumber>,
14    store_drb_progress: StoreDrbProgressFn,
15    load_drb_progress: LoadDrbProgressFn,
16    tasks: JoinSet<(EpochNumber, DrbResult)>,
17}
18
19impl DrbRequester {
20    pub fn new(
21        store_drb_progress: StoreDrbProgressFn,
22        load_drb_progress: LoadDrbProgressFn,
23    ) -> Self {
24        Self {
25            calculations: BTreeMap::new(),
26            completed_epochs: BTreeSet::new(),
27            store_drb_progress,
28            load_drb_progress,
29            tasks: JoinSet::new(),
30        }
31    }
32
33    pub async fn next(&mut self) -> Option<(EpochNumber, DrbResult)> {
34        loop {
35            match self.tasks.join_next().await {
36                Some(Ok((epoch, result))) => {
37                    self.calculations.remove(&epoch);
38                    self.completed_epochs.insert(epoch);
39                    return Some((epoch, result));
40                },
41                Some(Err(e)) => {
42                    warn!("Error in drb request task: {e}");
43                    continue;
44                },
45                None => return None,
46            }
47        }
48    }
49
50    pub fn request_drb(&mut self, drb_input: DrbInput) {
51        let store_drb_progress = self.store_drb_progress.clone();
52        let load_drb_progress = self.load_drb_progress.clone();
53        let epoch = EpochNumber::new(drb_input.epoch);
54        let handle = self.tasks.spawn(async move {
55            let result = compute_drb_result(drb_input, store_drb_progress, load_drb_progress).await;
56            (epoch, result)
57        });
58        self.calculations.insert(epoch, handle);
59    }
60
61    pub fn gc(&mut self, epoch: EpochNumber) {
62        let keep = self.calculations.split_off(&epoch);
63        self.completed_epochs = self.completed_epochs.split_off(&epoch);
64        for handle in self.calculations.values() {
65            handle.abort();
66        }
67        self.calculations = keep;
68    }
69}
