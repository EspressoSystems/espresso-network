200    /// Bootstrap the coordinator so the view-1 leader can propose.
201    ///
202    /// Emits an initial `ViewChanged(1)` and, if this node is the view-1
203    /// leader, a `RequestBlockAndHeader` for view 1.  Call this after
204    /// `seed_genesis` on the inner `Consensus` instance.
205    pub async fn start(&mut self) {
206        let view = ViewNumber::new(1);
207        let epoch = EpochNumber::genesis();
208
209        self.outbox
210            .push_back(ConsensusOutput::ViewChanged(view, epoch));
211
212        if let Some(leader) = self.leader(view, epoch).await
213            && leader == self.public_key
214        {
215            let genesis_proposal = self
216                .consensus
217                .proposal_at(ViewNumber::genesis())
218                .expect("genesis proposal must be seeded before start()")
219                .clone();
220            self.outbox
221                .push_back(ConsensusOutput::RequestBlockAndHeader(
222                    BlockAndHeaderRequest {
223                        view,
224                        epoch,
225                        parent_proposal: genesis_proposal,
226                    },
227                ));
228        }
229    }

235    pub async fn next_consensus_input(&mut self) -> Result<ConsensusInput<T>, CoordinatorError> {
236        loop {
237            select! {
238                message = self.network.receive() => match message {
239                    Ok(m) => {
240                        if let Some(input) = self.on_network_message(m).await {
241                            return Ok(input)
242                        }
243                    }
244                    Err(e) => {
245                        return Err(CoordinatorError::from(e).context("network receive"))
246                    }
247                },
248                () = &mut self.timer => {
249                    let input = ConsensusInput::Timeout(self.timer.view(), self.timer.epoch());
250                    // Timer is only reset so we can resend the timeout vote
251                    // This isn't strictly necessary for the protocol, but it's a good idea to
252                    // resend the timeout vote to avoid a situation where the network is stuck
253                    // view because we fail to form a timeout certificate.
254                    self.timer.reset();
255                    return Ok(input)
256                }
257                Some(output) = self.state_manager.next() => {
258                    if let Some(input) = self.on_state_manager_output(output) {
259                        return Ok(input)
260                    }
261                }
262                Some(request) = self.client.next_request() => {
263                    if let Err(err) = self.on_client_request(request) {
264                        error!(%err, "error while handling client request");
265                    }
266                }
267                Some(tcert) = self.timeout_collector.next() => {
268                    return Ok(ConsensusInput::TimeoutCertificate(tcert))
269                }
270                Some(out) = self.timeout_one_honest_collector.next() => {
271                    let Some(epoch) = out.data.epoch else {
272                        let msg = format!("missing epoch in view {}", out.view_number());
273                        return Err(CoordinatorError::regular(msg).context("gc timeout one honest"))
274                    };
275                    return Ok(ConsensusInput::TimeoutOneHonest(out.view_number(), epoch))
276                }
277                Some(cert1) = self.vote1_collector.next() => {
278                    return Ok(ConsensusInput::Certificate1(cert1))
279                }
280                Some(cert2) = self.vote2_collector.next() => {
281                    return Ok(ConsensusInput::Certificate2(cert2))
282                }
283                Some((cert1, state_cert)) = self.epoch_root_collector.next() => {
284                    return Ok(ConsensusInput::EpochRootCertificates { cert1, state_cert })
285                }
286                Some(item) = self.share_validator.next() => match item {
287                    Ok(vid_share) => {
288                        let view = vid_share.view_number();
289                        let Some(validated) = self.cached_validated_proposals.remove(&view) else {
290                            // Wait for the proposal
291                            self.cached_vid_shares.insert(view, vid_share);
292                            continue;
293                        };
294                        if !check_payload_commitment(&validated.message.proposal, &vid_share) {
295                            continue;
296                        }
297                        return self.handle_proposal_and_vid_share(validated, vid_share)
298                    },
299                    Err(e) => {
300                        return Err(CoordinatorError::regular(e).context("vid share validation"))
301                    }
302                },
303                Some(item) = self.proposal_validator.next() => match item {
304                    Ok(validated) => {
305                        // Refresh the network's peer set when a proposal is validated
306                        // on_epoch_change should return immediately if the epoch is not new
307                        let epoch = validated.message.proposal.data.epoch;
308                        if let Err(err) = self
309                            .network
310                            .on_epoch_change(epoch, &self.membership_coordinator)
311                            .await
312                        {
313                            error!(%epoch, %err, "network on_epoch_change failed");
314                        }
315
316                        let view = validated.message.proposal.data.view_number();
317                        let Some(vid_share) = self.cached_vid_shares.remove(&view) else {
318                            // Wait for the vid share
319                            self.cached_validated_proposals.insert(view, validated);
320                            continue;
321                        };
322                        // Check for commitment correspondence
323                        if !check_payload_commitment(&validated.message.proposal, &vid_share) {
324                            continue;
325                        }
326                        return self.handle_proposal_and_vid_share(validated, vid_share)
327                    }
328                    Err(e) => {
329                        return Err(CoordinatorError::regular(e).context("proposal validation"))
330                    }
331                },
332                Some(cert) = self.checkpoint_collector.next() => {
333                    let Some(epoch) = cert.epoch() else {
334                        let msg = format!("missing epoch in view {}", cert.view_number());
335                        return Err(CoordinatorError::critical(msg).context("gc certificate"))
336                    };
337                    self.gc(cert.view_number(), epoch);
338                }
339                Some(item) = self.block_builder.next() => match item {
340                    Ok(block) => {
341                        self.state_manager.request_header(HeaderRequest::from(&block));
342                        let next_view = block.view + 1;
343                        let epoch = block.epoch;
344                        let manifest = block.manifest.clone();
345                        self.storage.append_da(
346                            block.view,
347                            block.epoch,
348                            block.payload.payload.clone(),
349                            block.payload.metadata.clone(),
350                            block.payload_commitment,
351                        );
352                        self.unicast_to_leader(
353                            next_view,
354                            epoch,
355                            BlockMessage::DedupManifest(manifest),
356                        )
357                        .await?;
358                        return Ok(block.into())
359                    }
360                    Err(err) => {
361                        return Err(CoordinatorError::regular(err).context("block building"))
362                    }
363                },
364                Some(item) = self.vid_disperser.next() => match item {
365                    Ok(out) => {
366                        return Ok(ConsensusInput::VidDisperseCreated(out.view, out.disperse))
367                    }
368                    Err(()) => {
369                        return Err(CoordinatorError::unspecified().context("vid disperse"))
370                    }
371                },
372                Some(item) = self.vid_reconstructor.next() => match item {
373                    Ok(out) => {
374                        self.block_builder.on_block_reconstructed(out.tx_commitments);
375                        self.storage.append_da(
376                            out.view,
377                            out.epoch,
378                            out.payload,
379                            out.metadata,
380                            VidCommitment::V2(out.payload_commitment),
381                        );
382                        return Ok(ConsensusInput::BlockReconstructed(out.view, out.payload_commitment))
383                    }
384                    Err(()) => {
385                        return Err(CoordinatorError::unspecified().context("vid reconstruction"))
386                    }
387                },
388                Some(result) = self.epoch_manager.next() => match result {
389                    Ok(EpochRootResult::DrbResult(epoch, drb_result)) => {
390                        // New epoch data available — retry votes that were
391                        // buffered because their membership wasn't ready.
392                        self.vote1_collector.retry_pending_votes().await;
393                        self.vote2_collector.retry_pending_votes().await;
394                        self.timeout_collector.retry_pending_votes().await;
395                        self.timeout_one_honest_collector.retry_pending_votes().await;
396                        return Ok(ConsensusInput::DrbResult(epoch, drb_result))
397                    }
398                    Err(failure) => {
399                        // Catchup/compute failed. The epoch manager clears
400                        // the pending guard; consensus's `maybe_propose`
401                        // will re-request the DRB when it next tries to
402                        // build a transition proposal and finds it missing.
403                        warn!(%failure.error, epoch = %failure.epoch, "DRB request failed");
404                        continue;
405                    }
406                },
407                else => {
408                    return Err(CoordinatorError::critical(ErrorSource::NoInput))
409                }
410            }
411        }
412    }

414    pub fn handle_proposal_and_vid_share(
415        &mut self,
416        validated: ValidatedProposal<T>,
417        vid_share: VidDisperseShare2<T>,
418    ) -> Result<ConsensusInput<T>, CoordinatorError> {
419        self.storage.append_vid(vid_share.clone());
420        self.storage
421            .append_proposal(validated.message.proposal.data.clone());
422
423        let m = validated
424            .message
425            .proposal
426            .data
427            .block_header
428            .metadata()
429            .clone();
430        self.vid_reconstructor
431            .handle_vid_share(vid_share.clone(), m);
432
433        // GC for the cache
434        let view = validated.message.proposal.data.view_number();
435        self.cached_vid_shares = self.cached_vid_shares.split_off(&(view + 1));
436        self.cached_validated_proposals = self.cached_validated_proposals.split_off(&(view + 1));
437
438        Ok(ConsensusInput::ProposalWithVidShare(
439            validated.sender,
440            validated.message,
441            vid_share,
442        ))
443    }

504    pub async fn on_network_message(
505        &mut self,
506        message: Message<T, Unchecked>,
507    ) -> Option<ConsensusInput<T>> {
508        match message.message_type {
509            MessageType::Consensus(msg) => match msg {
510                ConsensusMessage::Proposal(p) => {
511                    if self.consensus.wants_proposal_for_view(&p.view_number()) {
512                        self.proposal_validator.validate(p);
513                    }
514                    None
515                },
516                ConsensusMessage::VidShare(share) => {
517                    if self
518                        .consensus
519                        .wants_proposal_for_view(&share.data.view_number())
520                    {
521                        self.share_validator.validate(share);
522                    }
523                    None
524                },
525                ConsensusMessage::Vote1(vote1) => {
526                    let bn = vote1.vote.data.block_number.unwrap_or(0);
527                    let epoch_height = *self.consensus.epoch_height;
528                    if is_epoch_root(bn, epoch_height) {
529                        // An epoch-root Vote1 MUST carry a state_vote.
530                        // Reject otherwise.
531                        vote1.state_vote.as_ref()?;
532                        self.epoch_root_collector.accumulate(vote1.clone()).await;
533                    } else {
534                        self.vote1_collector
535                            .accumulate_vote(vote1.vote.clone())
536                            .await;
537                    }
538                    self.vid_reconstructor
539                        .handle_vid_share(vote1.vid_share, None);
540                    None
541                },
542                ConsensusMessage::Vote2(vote2) => {
543                    self.vote2_collector.accumulate_vote(vote2).await;
544                    None
545                },
546                ConsensusMessage::Certificate1(certificate1, _key) => {
547                    Some(ConsensusInput::Certificate1(certificate1))
548                },
549                ConsensusMessage::Certificate2(certificate2, _key) => {
550                    Some(ConsensusInput::Certificate2(certificate2))
551                },
552                ConsensusMessage::TimeoutVote(timeout_msg) => {
553                    self.timeout_collector
554                        .accumulate_vote(timeout_msg.vote.clone())
555                        .await;
556                    self.timeout_one_honest_collector
557                        .accumulate_vote(timeout_msg.vote)
558                        .await;
559                    None
560                },
561                ConsensusMessage::TimeoutCertificate(tc) => {
562                    Some(ConsensusInput::TimeoutCertificate(tc))
563                },
564                ConsensusMessage::EpochChange(epoch_change) => {
565                    Some(ConsensusInput::EpochChange(epoch_change))
566                },
567                ConsensusMessage::Checkpoint(checkpoint) => {
568                    self.checkpoint_collector.accumulate_vote(checkpoint).await;
569                    None
570                },
571            },
572            MessageType::Block(msg) => {
573                match msg {
574                    BlockMessage::Transactions(msg) => self.block_builder.on_transactions(msg),
575                    BlockMessage::DedupManifest(manifest) => {
576                        if let Some(view_leader) = self.leader(manifest.view, manifest.epoch).await
577                            && view_leader == message.sender
578                        {
579                            self.block_builder.on_dedup_manifest(manifest)
580                        }
581                    },
582                }
583                None
584            },
585            MessageType::ProposalFetch(ProposalFetchMessage::Request(request)) => {
586                if !request.validate_sender(&message.sender) {
587                    warn!(
588                        sender = %message.sender,
589                        view = %request.view_number(),
590                        "ignoring invalid proposal fetch request signature"
591                    );
592                    return None;
593                }
594                if let Some(proposal) = self
595                    .consensus
596                    .signed_proposal(&request.view_number())
597                    .cloned()
598                {
599                    let response = Message {
600                        sender: self.public_key.clone(),
601                        message_type: MessageType::ProposalFetch(ProposalFetchMessage::Response(
602                            Box::new(proposal),
603                        )),
604                    };
605
606                    if let Err(err) =
607                        self.network
608                            .unicast(request.view_number(), &message.sender, &response)
609                    {
610                        let err = CoordinatorError::from(err).context("proposal response");
611                        warn!(%err, "network error while sending proposal response");
612                    }
613                }
614                None
615            },
616            MessageType::ProposalFetch(ProposalFetchMessage::Response(proposal)) => {
617                self.pending_proposal_fetches.resolve(&proposal);
618                None
619            },
620            MessageType::External(data) => {
621                self.coordinator_outbox
622                    .push_back(CoordinatorOutput::ExternalMessageReceived {
623                        sender: message.sender,
624                        data,
625                    });
626                None
627            },
628        }
629    }

631    pub async fn process_consensus_output(
632        &mut self,
633        output: ConsensusOutput<T>,
634    ) -> Result<(), CoordinatorError> {
635        match output {
636            ConsensusOutput::RequestState(state_request) => {
637                self.state_manager.request_state(state_request);
638            },
639            ConsensusOutput::RequestVidDisperse {
640                view,
641                epoch,
642                payload,
643                metadata,
644            } => {
645                self.vid_disperser.request_vid_disperse(VidDisperseRequest {
646                    view,
647                    epoch,
648                    block: payload,
649                    metadata,
650                });
651            },
652            ConsensusOutput::RequestDrbResult(epoch) => {
653                self.epoch_manager.request_drb_result(epoch);
654            },
655            ConsensusOutput::SendCheckpointVote(checkpoint_vote) => {
656                let message = Message {
657                    sender: self.public_key.clone(),
658                    message_type: MessageType::Consensus(ConsensusMessage::Checkpoint(
659                        checkpoint_vote,
660                    )),
661                };
662                self.network
663                    .broadcast(message.view_number(), &message)
664                    .map_err(|e| CoordinatorError::from(e).context("broadcast checkpoint vote"))?
665            },
666            ConsensusOutput::LeafDecided { leaves, cert2, .. } => {
667                if let Some(cert2) = cert2 {
668                    self.storage.append_cert2(cert2.view_number, cert2.clone());
669                }
670                for leaf in leaves {
671                    self.epoch_manager.handle_leaf_decided(leaf);
672                }
673            },
674            ConsensusOutput::LockUpdated(_) => {}, // TODO
675            ConsensusOutput::RequestBlockAndHeader(request) => {
676                self.block_builder.request_block(request);
677            },
678            ConsensusOutput::SendProposal(proposal) => {
679                self.storage.append_proposal(proposal.data.clone());
680                // TODO: This may be done async in network so we do not spend
681                // too much time here in this loop.
682
683                let message = Message {
684                    sender: self.public_key.clone(),
685                    message_type: MessageType::Consensus(ConsensusMessage::Proposal(
686                        ProposalMessage::validated(proposal.clone()),
687                    )),
688                };
689                if let Err(err) = self.network.broadcast(message.view_number(), &message) {
690                    let err = CoordinatorError::from(err).context("proposal broadcast");
691                    if err.severity == Severity::Critical {
692                        return Err(err);
693                    } else {
694                        warn!(%err, "network error while broadcasting proposal")
695                    }
696                }
697            },
698            ConsensusOutput::SendVidShares(vid_shares) => {
699                for share in vid_shares {
700                    let recipient = share.data.recipient_key.clone();
701                    let message = Message {
702                        sender: self.public_key.clone(),
703                        message_type: MessageType::Consensus(ConsensusMessage::VidShare(share)),
704                    };
705                    if let Err(err) =
706                        self.network
707                            .unicast(message.view_number(), &recipient, &message)
708                    {
709                        let err = CoordinatorError::from(err).context("vid share unicast");
710                        if err.severity == Severity::Critical {
711                            return Err(err);
712                        } else {
713                            warn!(%err, "network error while sending vid share")
714                        }
715                    }
716                }
717            },
718            ConsensusOutput::SendTimeoutVote(vote, lock) => {
719                let message = Message {
720                    sender: self.public_key.clone(),
721                    message_type: MessageType::Consensus(ConsensusMessage::TimeoutVote(
722                        message::TimeoutVoteMessage { vote, lock },
723                    )),
724                };
725                self.network
726                    .broadcast(message.view_number(), &message)
727                    .map_err(|e| CoordinatorError::from(e).context("broadcast timeout vote"))?
728            },
729            ConsensusOutput::SendTimeoutCertificate(tc, view, epoch) => {
730                if let Some(leader) = self.leader(view, epoch).await {
731                    let message = Message {
732                        sender: self.public_key.clone(),
733                        message_type: MessageType::Consensus(ConsensusMessage::TimeoutCertificate(
734                            tc,
735                        )),
736                    };
737                    self.network
738                        .unicast(message.view_number(), &leader, &message)
739                        .map_err(|e| CoordinatorError::from(e).context("timeout certificate"))?;
740                }
741            },
742            ConsensusOutput::SendVote1(vote1) => {
743                let message = Message {
744                    sender: self.public_key.clone(),
745                    message_type: MessageType::Consensus(ConsensusMessage::Vote1(vote1)),
746                };
747                self.network
748                    .broadcast(message.view_number(), &message)
749                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote1"))?
750            },
751            ConsensusOutput::SendVote2(vote2) => {
752                let message = Message {
753                    sender: self.public_key.clone(),
754                    message_type: MessageType::Consensus(ConsensusMessage::Vote2(vote2)),
755                };
756                self.network
757                    .broadcast(message.view_number(), &message)
758                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote2"))?
759            },
760            ConsensusOutput::SendEpochChange(epoch_change) => {
761                let message = Message {
762                    sender: self.public_key.clone(),
763                    message_type: MessageType::Consensus(ConsensusMessage::EpochChange(
764                        epoch_change,
765                    )),
766                };
767                self.network
768                    .broadcast(message.view_number(), &message)
769                    .map_err(|e| CoordinatorError::from(e).context("broadcast epoch change"))?
770            },
771            ConsensusOutput::SendCertificate1(cert1) => {
772                let message = Message {
773                    sender: self.public_key.clone(),
774                    message_type: MessageType::Consensus(ConsensusMessage::Certificate1(
775                        cert1,
776                        self.public_key.clone(),
777                    )),
778                };
779                self.network
780                    .broadcast(message.view_number(), &message)
781                    .map_err(|e| CoordinatorError::from(e).context("broadcast certificate1"))?
782            },
783            ConsensusOutput::ProposalValidated { .. } => {},
784            ConsensusOutput::ViewChanged(view, epoch) => {
785                self.consensus.set_view(view, epoch);
786                self.timer.reset_with_epoch(view, epoch);
787                let txns = self.block_builder.on_view_changed(view, epoch);
788                if !txns.is_empty() {
789                    let next_view = view + 1;
790                    self.unicast_to_leader(
791                        next_view,
792                        epoch,
793                        BlockMessage::Transactions(TransactionMessage {
794                            view: next_view,
795                            transactions: txns,
796                        }),
797                    )
798                    .await
799                    .map_err(|e| e.context("unicast transactions"))?;
800                }
801
802                // Proactively fetch DRBs for the next epoch so
803                // late-starting nodes have them before they need to
804                // propose or verify certs in a new epoch. The dedup
805                // in request_drb_result makes repeated calls free.
806                let next_epoch = epoch + 1;
807                if next_epoch > EpochNumber::genesis() + 1 {
808                    self.epoch_manager.request_drb_result(next_epoch);
809                    self.epoch_manager.request_drb_result(next_epoch + 1);
810                }
811            },
812        }
813        Ok(())
814    }
815
816    async fn unicast_to_leader(
817        &mut self,
818        view: ViewNumber,
819        epoch: EpochNumber,
820        msg: BlockMessage<T>,
821    ) -> Result<(), CoordinatorError> {
822        let Some(leader) = self.leader(view, epoch).await else {
823            warn!(%view, %epoch, "failed to resolve leader for unicast");
824            return Ok(());
825        };
826        let message = Message {
827            sender: self.public_key.clone(),
828            message_type: MessageType::Block(msg),
829        };
830        self.network
831            .unicast(message.view_number(), &leader, &message)
832            .map_err(|e| CoordinatorError::from(e).context("leader unicast"))
833    }
834
835    async fn leader(&mut self, view: ViewNumber, epoch: EpochNumber) -> Option<T::SignatureKey> {
836        let membership = self
837            .membership_coordinator
838            .membership_for_epoch(Some(epoch))
839            .await
840            .ok()?;
841        membership.leader(view).await.ok()
842    }

844    fn on_client_request(&mut self, request: ClientRequest<T>) -> Result<(), CoordinatorError> {
845        match request {
846            ClientRequest::CurrentView(tx) => {
847                let _ = tx.send(self.consensus.current_view());
848            },
849            ClientRequest::CurrentEpoch(tx) => {
850                let _ = tx.send(self.consensus.current_epoch());
851            },
852            ClientRequest::DecidedLeaf(tx) => {
853                let _ = tx.send(self.consensus.last_decided_leaf().clone());
854            },
855            ClientRequest::DecidedState(tx) => {
856                let view = self.consensus.last_decided_leaf().view_number();
857                let _ = tx.send(self.state_manager.get_state(&view));
858            },
859            ClientRequest::UndecidedLeaves(tx) => {
860                let _ = tx.send(self.consensus.undecided_leaves().cloned().collect());
861            },
862            ClientRequest::GetState { view, respond } => {
863                let _ = respond.send(self.state_manager.get_state(&view));
864            },
865            ClientRequest::GetStateAndDelta { view, respond } => {
866                let _ = respond.send(self.state_manager.get_state_and_delta(&view));
867            },
868            ClientRequest::SubmitTransaction { tx, respond } => {
869                self.block_builder.on_submit_transaction(tx);
870                let _ = respond.send(());
871            },
872            ClientRequest::UpdateLeaf { update, respond } => {
873                self.state_manager.update_state(update);
874                let _ = respond.send(());
875            },
876            ClientRequest::RequestProposal {
877                view,
878                leaf_commitment,
879                respond,
880            } => {
881                if let Some(proposal) = self.consensus.signed_proposal(&view)
882                    && proposal_commitment(&proposal.data) == leaf_commitment
883                {
884                    let _ = respond.send(Ok(proposal.clone()));
885                    return Ok(());
886                }
887                if !self
888                    .pending_proposal_fetches
889                    .contains_request(view, leaf_commitment)
890                {
891                    let request =
892                        self.consensus
893                            .signed_proposal_fetch_request(view)
894                            .map_err(|err| {
895                                let err = format!("failed to sign proposal request: {err}");
896                                CoordinatorError::regular(err).context("sign proposal request")
897                            })?;
898
899                    let message = Message {
900                        sender: self.public_key.clone(),
901                        message_type: MessageType::ProposalFetch(ProposalFetchMessage::Request(
902                            request,
903                        )),
904                    };
905
906                    self.network
907                        .broadcast(message.view_number(), &message)
908                        .map_err(|err| {
909                            CoordinatorError::from(err).context("broadcast proposal request")
910                        })?;
911                }
912                self.pending_proposal_fetches
913                    .push(view, leaf_commitment, respond);
914            },
915            ClientRequest::SendExternalMessage {
916                view,
917                payload,
918                recipient,
919                respond,
920            } => {
921                let message = Message {
922                    sender: self.public_key.clone(),
923                    message_type: MessageType::External(payload),
924                };
925                let result = self
926                    .network
927                    .unicast(view, &recipient, &message)
928                    .map_err(|err| {
929                        CoordinatorError::from(err)
930                            .context("send external message")
931                            .into()
932                    });
933                let _ = respond.send(result);
934            },
935        }
936
937        Ok(())
938    }

961fn check_payload_commitment<T: NodeType>(
962    proposal: &SignedProposal<T, Proposal<T>>,
963    vid_share: &VidDisperseShare2<T>,
964) -> bool {
965    let VidCommitment::V2(commit) = proposal.data.block_header.payload_commitment() else {
966        warn!(
967            "unexpected payload commitment type in view {}, proposal discarded",
968            proposal.data.view_number
969        );
970        return false;
971    };
972    if commit != vid_share.payload_commitment {
973        warn!(
974            "payload commitment mismatch in view {}, discard the proposal",
975            proposal.data.view_number
976        );
977        return false;
978    }
979    true
980}
981
982type ProposalFetchResponseSender<T> =
983    oneshot::Sender<Result<SignedProposal<T, Proposal<T>>, QueryError>>;
984
985#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
986struct ProposalFetchKey<T: NodeType> {
987    view: ViewNumber,
988    leaf_commitment: Commitment<Leaf2<T>>,
989}
990
991impl<T: NodeType> ProposalFetchKey<T> {
992    fn new(view: ViewNumber, leaf_commitment: Commitment<Leaf2<T>>) -> Self {
993        Self {
994            view,
995            leaf_commitment,
996        }
997    }
998}
999
1000#[derive(Default)]
1001struct PendingProposalFetches<T: NodeType> {
1002    pending: HashMap<ProposalFetchKey<T>, Vec<ProposalFetchResponseSender<T>>>,
1003}
1004
1005impl<T: NodeType> PendingProposalFetches<T> {
1006    fn prune_closed(&mut self) {
1007        self.pending.retain(|_, responders| {
1008            responders.retain(|respond| !respond.is_closed());
1009            !responders.is_empty()
1010        });
1011    }
1012
1013    fn contains_request(
1014        &mut self,
1015        view: ViewNumber,
1016        leaf_commitment: Commitment<Leaf2<T>>,
1017    ) -> bool {
1018        self.prune_closed();
1019        self.pending
1020            .contains_key(&ProposalFetchKey::new(view, leaf_commitment))
1021    }
1022
1023    fn push(
1024        &mut self,
1025        view: ViewNumber,
1026        leaf_commitment: Commitment<Leaf2<T>>,
1027        respond: ProposalFetchResponseSender<T>,
1028    ) {
1029        self.pending
1030            .entry(ProposalFetchKey::new(view, leaf_commitment))
1031            .or_default()
1032            .push(respond);
1033    }
1034
1035    fn gc(&mut self, view: ViewNumber) {
1036        self.pending.retain(|key, responders| {
1037            responders.retain(|respond| !respond.is_closed());
1038            key.view >= view && !responders.is_empty()
1039        });
1040    }
1041
1042    fn resolve(&mut self, proposal: &SignedProposal<T, Proposal<T>>) {
1043        self.prune_closed();
1044        let view = proposal.data.view_number;
1045        let leaf_commitment = proposal_commitment(&proposal.data);
1046        let key = ProposalFetchKey::new(view, leaf_commitment);
1047
1048        if let Some(responders) = self.pending.remove(&key) {
1049            for respond in responders {
1050                let _ = respond.send(Ok(proposal.clone()));
1051            }
1052        }
1053    }
1054}
