42impl<T: NodeType, I: NodeImplementation<T>> Coordinator<T, I> {
43    pub async fn run(mut self) {
44        loop {
45            select! {
46                message = self.network.receive() => match message {
47                    Ok(m) => {
48                        self.on_message(m).await
49                    }
50                    Err(err) if is_critical(&err) => {
51                        error!(%err, "critical network error => exiting");
52                        break
53                    }
54                    Err(err) => {
55                        warn!(%err, "network error")
56                    }
57                },
58                Some(state_event) = self.state_manager.next() => {
59                    if let Ok(input) = ConsensusInput::try_from(state_event) {
60                        self.evaluate(input).await;
61                    }
62                }
63                Some(tcert) = self.timeout_collector.next() => {
64                    self.evaluate(ConsensusInput::TimeoutCertificate(tcert)).await;
65                }
66                Some(cert1) = self.vote1_collector.next() => {
67                    self.evaluate(ConsensusInput::Certificate1(cert1)).await;
68                }
69                Some(cert2) = self.vote2_collector.next() => {
70                    self.evaluate(ConsensusInput::Certificate2(cert2)).await;
71                }
72                Some(item) = self.vid_disperser.next() => match item {
73                    Ok((view, _, disperse)) => {
74                        self.evaluate(ConsensusInput::VidDisperseCreated(view, disperse)).await;
75                    }
76                    Err(err) => {
77                        warn!(?err, "vid disperser error")
78                    }
79                },
80                Some(item) = self.vid_reconstructor.next() => match item {
81                    Ok((view, commitment, _)) => {
82                        self.evaluate(ConsensusInput::BlockReconstructed(view, commitment)).await;
83                    }
84                    Err(err) => {
85                        warn!(?err, "vid reconstructor error")
86                    }
87                },
88                Some((_epoch, drb_result)) = self.drb_requester.next() => {
89                    todo!()
90                }
91                else => {
92                    error!("all coordinator inputs are closed => exiting");
93                    break
94                }
95            }
96        }
97    }

99    /// Process an incoming network message.
100    async fn on_message(&mut self, msg: Message<T>) {
101        match msg.message_type {
102            MessageType::Consensus(msg) => match msg {
103                ConsensusMessage::Proposal(proposal) => {
104                    self.vid_reconstructor.handle_vid_share(VidShareInput {
105                        share: proposal.vid_share.clone(),
106                        metadata: Some(proposal.proposal.data.block_header.metadata().clone()),
107                    });
108                    self.evaluate(ConsensusInput::Proposal(proposal)).await;
109                },
110                ConsensusMessage::Vote1(vote1) => {
111                    self.vote1_collector.accumulate_vote(vote1.vote).await;
112                    self.vid_reconstructor.handle_vid_share(VidShareInput {
113                        share: vote1.vid_share,
114                        metadata: None,
115                    });
116                },
117                ConsensusMessage::Vote2(vote2) => {
118                    self.vote2_collector.accumulate_vote(vote2).await;
119                },
120                ConsensusMessage::Certificate1(certificate1, _key) => {
121                    self.evaluate(ConsensusInput::Certificate1(certificate1))
122                        .await;
123                },
124                ConsensusMessage::Certificate2(certificate2, _key) => {
125                    self.evaluate(ConsensusInput::Certificate2(certificate2))
126                        .await;
127                },
128                ConsensusMessage::TimeoutVote(timeout_vote) => {
129                    self.timeout_collector.accumulate_vote(timeout_vote).await;
130                },
131                ConsensusMessage::Transactions(transactions, view) => {
132                    todo!()
133                },
134                ConsensusMessage::Checkpoint(view, epoch) => {
135                    todo!()
136                },
137            },
138            MessageType::ViewSync(_) => todo!(),
139            MessageType::External(_) => todo!(),
140        }
141    }

143    async fn evaluate(&mut self, input: ConsensusInput<T>) {
144        self.consensus.apply(input, &mut self.outbox).await;
145        while let Some(output) = self.outbox.pop_front() {
146            match output {
147                ConsensusOutput::Action(a) => self.handle_action(a).await,
148                ConsensusOutput::Event(e) => self.handle_event(e),
149            }
150        }
151    }
152
153    async fn handle_action(&mut self, action: Action<T>) {
154        match action {
155            Action::SendProposal(..) => {},
156            Action::SendVote1(..) => {},
157            Action::SendVote2(..) => {},
158            Action::RequestState(state_request) => {
159                self.state_manager.request_state(state_request);
160            },
161            Action::RequestBlockAndHeader(req) => {
162                // TODO: add a block builder, and use it to build the block,
163                // Then on block built, request the header
164                todo!()
165            },
166            Action::RequestVidDisperse(view, epoch, block, metadata) => {
167                self.vid_disperser.request_vid_disperse(VidDisperseRequest {
168                    view,
169                    epoch,
170                    block,
171                    metadata,
172                });
173            },
174            Action::RequestProposal(_view, _commitment) => {},
175            Action::RequestDRB(drb_input) => {
176                self.drb_requester.request_drb(drb_input);
177            },
178        }
179    }
