64    /// Convenience method to run the coordinator event loop.
65    ///
66    /// Combines `next_consensus_input`, `apply_consensus` and outbox processing.
67    pub async fn run(mut self) {
68        loop {
69            match self.next_consensus_input().await {
70                Ok(input) => self.apply_consensus(input).await,
71                Err(err) if err.severity == Severity::Critical => {
72                    error!(%err, "while awaiting next consensus input");
73                    return;
74                },
75                Err(err) => {
76                    warn!(%err, "while awaiting next consensus input");
77                },
78            }
79            while let Some(output) = self.outbox.pop_front() {
80                if let Err(err) = self.process_consensus_output(output).await {
81                    if err.severity == Severity::Critical {
82                        error!(%err, "while processing consensus output");
83                        return;
84                    } else {
85                        warn!(%err, "while processing consensus output");
86                    }
87                }
88            }
89        }
90    }

92    pub async fn next_consensus_input(&mut self) -> Result<ConsensusInput<T>, CoordinatorError> {
93        loop {
94            select! {
95                message = self.network.receive() => match message {
96                    Ok(m) => {
97                        if let Some(input) = self.on_network_message(m).await {
98                            return Ok(input)
99                        }
100                    }
101                    Err(e) => {
102                        return Err(CoordinatorError::from(e).context("network receive"))
103                    }
104                },
105                () = &mut self.timer => {
106                    let input = ConsensusInput::Timeout(self.timer.view());
107                    self.timer.reset();
108                    return Ok(input)
109                }
110                Some(output) = self.state_manager.next() => {
111                    if let Some(input) = self.on_state_manager_output(output).await {
112                        return Ok(input)
113                    }
114                }
115                Some(tcert) = self.timeout_collector.next() => {
116                    return Ok(ConsensusInput::TimeoutCertificate(tcert))
117                }
118                Some(out) = self.timeout_one_honest_collector.next() => {
119                    return Ok(ConsensusInput::TimeoutOneHonest(out.view_number(), out.data.epoch))
120                }
121                Some(cert1) = self.vote1_collector.next() => {
122                    return Ok(ConsensusInput::Certificate1(cert1))
123                }
124                Some(cert2) = self.vote2_collector.next() => {
125                    return Ok(ConsensusInput::Certificate2(cert2))
126                }
127                Some(item) = self.proposal_validator.next() => match item {
128                    Ok(p) => {
129                        let s = p.vid_share.clone();
130                        let m = p.proposal.data.block_header.metadata().clone();
131                        self.vid_reconstructor.handle_vid_share(s, m);
132                        return Ok(ConsensusInput::Proposal(p))
133                    }
134                    Err(e) => {
135                        return Err(CoordinatorError::regular(e).context("proposal validation"))
136                    }
137                },
138                Some(cert) = self.checkpoint_collector.next() => {
139                    let Some(epoch) = cert.epoch() else {
140                        let msg = format!("missing epoch in view {}", cert.view_number());
141                        return Err(CoordinatorError::critical(msg).context("gc certificate"))
142                    };
143                    self.gc(cert.view_number(), epoch);
144                }
145                Some(item) = self.block_builder.next() => match item {
146                    Ok(block) => {
147                        self.state_manager.request_header(HeaderRequest::from(&block));
148                        let next_view = block.view + 1;
149                        let epoch = block.epoch;
150                        let manifest = block.manifest.clone();
151                        self.unicast_to_leader(
152                            next_view,
153                            epoch,
154                            BlockMessage::DedupManifest(manifest),
155                        )
156                        .await?;
157                        return Ok(block.into())
158                    }
159                    Err(err) => {
160                        return Err(CoordinatorError::regular(err).context("block building"))
161                    }
162                },
163                Some(item) = self.vid_disperser.next() => match item {
164                    Ok(out) => {
165                        return Ok(ConsensusInput::VidDisperseCreated(out.view, out.disperse))
166                    }
167                    Err(()) => {
168                        return Err(CoordinatorError::unspecified().context("vid disperse"))
169                    }
170                },
171                Some(item) = self.vid_reconstructor.next() => match item {
172                    Ok(out) => {
173                        self.block_builder.on_block_reconstructed(out.tx_commitments);
174                        return Ok(ConsensusInput::BlockReconstructed(out.view, out.payload_commitment))
175                    }
176                    Err(()) => {
177                        return Err(CoordinatorError::unspecified().context("vid reconstruction"))
178                    }
179                },
180                Some(result) = self.epoch_manager.next() => match result {
181                    Ok(EpochRootResult::DrbResult(epoch, drb_result)) => {
182                        return Ok(ConsensusInput::DrbResult(epoch, drb_result))
183                    }
184                    Ok(EpochRootResult::RootAdded(_epoch)) => {}
185                    Err(err) => {
186                        return Err(CoordinatorError::regular(err))
187                    }
188                },
189                else => {
190                    return Err(CoordinatorError::critical(ErrorSource::NoInput))
191                }
192            }
193        }
194    }

208    pub async fn on_state_manager_output(
209        &mut self,
210        output: StateManagerOutput<T>,
211    ) -> Option<ConsensusInput<T>> {
212        match output {
213            StateManagerOutput::State {
214                response,
215                validated: true,
216            } => Some(ConsensusInput::StateValidated(response)),
217            StateManagerOutput::State {
218                response,
219                validated: false,
220            } => Some(ConsensusInput::StateValidationFailed(response)),
221            StateManagerOutput::Header {
222                response,
223                header: Some(hdr),
224            } => Some(ConsensusInput::HeaderCreated(response.view, hdr)),
225            StateManagerOutput::Header {
226                response: _,
227                header: None,
228            } => {
229                todo!()
230            },
231        }

234    pub async fn on_network_message(
235        &mut self,
236        message: Message<T, Unchecked>,
237    ) -> Option<ConsensusInput<T>> {
238        match message.message_type {
239            MessageType::Consensus(msg) => match msg {
240                ConsensusMessage::Proposal(p) => {
241                    if self.consensus.wants_proposal(&p) {
242                        self.proposal_validator.validate(p);
243                    }
244                    None
245                },
246                ConsensusMessage::Vote1(vote1) => {
247                    self.vote1_collector.accumulate_vote(vote1.vote).await;
248                    self.vid_reconstructor
249                        .handle_vid_share(vote1.vid_share, None);
250                    None
251                },
252                ConsensusMessage::Vote2(vote2) => {
253                    self.vote2_collector.accumulate_vote(vote2).await;
254                    None
255                },
256                ConsensusMessage::Certificate1(certificate1, _key) => {
257                    Some(ConsensusInput::Certificate1(certificate1))
258                },
259                ConsensusMessage::Certificate2(certificate2, _key) => {
260                    Some(ConsensusInput::Certificate2(certificate2))
261                },
262                ConsensusMessage::TimeoutVote(timeout_msg) => {
263                    self.timeout_collector
264                        .accumulate_vote(timeout_msg.vote.clone())
265                        .await;
266                    self.timeout_one_honest_collector
267                        .accumulate_vote(timeout_msg.vote)
268                        .await;
269                    None
270                },
271                ConsensusMessage::TimeoutCertificate(tc) => {
272                    Some(ConsensusInput::TimeoutCertificate(tc))
273                },
274                ConsensusMessage::EpochChange(epoch_change) => {
275                    Some(ConsensusInput::EpochChange(epoch_change))
276                },
277                ConsensusMessage::Checkpoint(checkpoint) => {
278                    self.checkpoint_collector.accumulate_vote(checkpoint).await;
279                    None
280                },
281            },
282            MessageType::Block(msg) => {
283                match msg {
284                    BlockMessage::Transactions(msg) => self.block_builder.on_transactions(msg),
285                    BlockMessage::DedupManifest(manifest) => {
286                        if let Some(view_leader) = self.leader(manifest.view, manifest.epoch).await
287                            && view_leader == message.sender
288                        {
289                            self.block_builder.on_dedup_manifest(manifest)
290                        }
291                    },
292                }
293                None
294            },
295            MessageType::ViewSync(_) => todo!(),
296            MessageType::External(_) => todo!(),
297        }
298    }

300    pub async fn process_consensus_output(
301        &mut self,
302        output: ConsensusOutput<T>,
303    ) -> Result<(), CoordinatorError> {
304        match output {
305            ConsensusOutput::RequestState(state_request) => {
306                self.state_manager.request_state(state_request);
307            },
308            ConsensusOutput::RequestVidDisperse {
309                view,
310                epoch,
311                payload,
312                metadata,
313            } => {
314                self.vid_disperser.request_vid_disperse(VidDisperseRequest {
315                    view,
316                    epoch,
317                    block: payload,
318                    metadata,
319                });
320            },
321            ConsensusOutput::RequestDrbResult(epoch) => {
322                self.epoch_manager.request_drb_result(epoch);
323            },
324            ConsensusOutput::SendCheckpointVote(checkpoint_vote) => {
325                let message = Message {
326                    sender: self.public_key.clone(),
327                    message_type: MessageType::Consensus(ConsensusMessage::Checkpoint(
328                        checkpoint_vote,
329                    )),
330                };
331                self.network
332                    .broadcast(message)
333                    .await
334                    .map_err(|e| CoordinatorError::from(e).context("broadcast checkpoint vote"))?
335            },
336            ConsensusOutput::LeafDecided(leaves) => {
337                for leaf in leaves {
338                    self.epoch_manager.handle_leaf_decided(leaf);
339                }
340            },
341            ConsensusOutput::LockUpdated(_) => {}, // TODO
342            ConsensusOutput::RequestBlockAndHeader(request) => {
343                self.block_builder.request_block(request);
344            },
345            ConsensusOutput::RequestProposal(..) => {}, // TODO
346            ConsensusOutput::SendProposal(proposal, vid_disperse) => {
347                // TODO: This may be done async in network so we do not spend
348                // too much time here in this loop.
349                for vid_share in vid_disperse.to_shares() {
350                    let recipient_key = vid_share.recipient_key.clone();
351                    let message = Message {
352                        sender: self.public_key.clone(),
353                        message_type: MessageType::Consensus(ConsensusMessage::Proposal(
354                            ProposalMessage::validated(proposal.clone(), vid_share),
355                        )),
356                    };
357                    if let Err(err) = self.network.unicast(recipient_key, message).await {
358                        let err = CoordinatorError::from(err).context("vid share unicast");
359                        if err.severity == Severity::Critical {
360                            return Err(err);
361                        } else {
362                            warn!(%err, "network error while sending vid share")
363                        }
364                    }
365                }
366            },
367            ConsensusOutput::SendTimeoutVote(vote, lock) => {
368                let message = Message {
369                    sender: self.public_key.clone(),
370                    message_type: MessageType::Consensus(ConsensusMessage::TimeoutVote(
371                        message::TimeoutVoteMessage { vote, lock },
372                    )),
373                };
374                self.network
375                    .broadcast(message)
376                    .await
377                    .map_err(|e| CoordinatorError::from(e).context("broadcast timeout vote"))?
378            },
379            ConsensusOutput::SendTimeoutCertificate(tc, view, epoch) => {
380                if let Some(leader) = self.leader(view, epoch).await {
381                    let message = Message {
382                        sender: self.public_key.clone(),
383                        message_type: MessageType::Consensus(ConsensusMessage::TimeoutCertificate(
384                            tc,
385                        )),
386                    };
387                    self.network
388                        .unicast(leader, message)
389                        .await
390                        .map_err(|e| CoordinatorError::from(e).context("timeout certificate"))?;
391                }
392            },
393            ConsensusOutput::SendVote1(vote1) => {
394                let message = Message {
395                    sender: self.public_key.clone(),
396                    message_type: MessageType::Consensus(ConsensusMessage::Vote1(vote1)),
397                };
398                self.network
399                    .broadcast(message)
400                    .await
401                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote1"))?
402            },
403            ConsensusOutput::SendVote2(vote2) => {
404                let message = Message {
405                    sender: self.public_key.clone(),
406                    message_type: MessageType::Consensus(ConsensusMessage::Vote2(vote2)),
407                };
408                self.network
409                    .broadcast(message)
410                    .await
411                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote2"))?
412            },
413            ConsensusOutput::SendEpochChange(epoch_change) => {
414                let message = Message {
415                    sender: self.public_key.clone(),
416                    message_type: MessageType::Consensus(ConsensusMessage::EpochChange(
417                        epoch_change,
418                    )),
419                };
420                self.network
421                    .broadcast(message)
422                    .await
423                    .map_err(|e| CoordinatorError::from(e).context("broadcast epoch change"))?
424            },
425            ConsensusOutput::SendCertificate1(cert1) => {
426                let message = Message {
427                    sender: self.public_key.clone(),
428                    message_type: MessageType::Consensus(ConsensusMessage::Certificate1(
429                        cert1,
430                        self.public_key.clone(),
431                    )),
432                };
433                self.network
434                    .broadcast(message)
435                    .await
436                    .map_err(|e| CoordinatorError::from(e).context("broadcast certificate1"))?
437            },
438            ConsensusOutput::ViewChanged(view, epoch) => {
439                self.timer.reset_with(view);
440                let txns = self.block_builder.on_view_changed(view, epoch);
441                if !txns.is_empty() {
442                    let next_view = view + 1;
443                    self.unicast_to_leader(
444                        next_view,
445                        epoch,
446                        BlockMessage::Transactions(TransactionMessage {
447                            view: next_view,
448                            transactions: txns,
449                        }),
450                    )
451                    .await
452                    .map_err(|e| e.context("unicast transactions"))?;
453                }
454            },
455        }
456        Ok(())
457    }

459    async fn unicast_to_leader(
460        &mut self,
461        view: ViewNumber,
462        epoch: EpochNumber,
463        msg: BlockMessage<T>,
464    ) -> Result<(), CoordinatorError> {
465        let Some(leader) = self.leader(view, epoch).await else {
466            warn!(%view, %epoch, "failed to resolve leader for unicast");
467            return Ok(());
468        };
469        let message = Message {
470            sender: self.public_key.clone(),
471            message_type: MessageType::Block(msg),
472        };
473        self.network
474            .unicast(leader, message)
475            .await
476            .map_err(|e| CoordinatorError::from(e).context("leader unicast"))
477    }

479    async fn leader(&self, view: ViewNumber, epoch: EpochNumber) -> Option<T::SignatureKey> {
480        let membership = self
481            .membership_coordinator
482            .membership_for_epoch(Some(epoch))
483            .await
484            .ok()?;
485        membership.leader(view).await.ok()
486    }

488    fn gc(&mut self, view: ViewNumber, epoch: EpochNumber) {
489        self.consensus.gc(view, epoch);
490        self.checkpoint_collector.gc(view, epoch);
491        self.network.gc(view, epoch);
492        self.state_manager.gc(view);
493        self.vid_disperser.gc(view);
494        self.vid_reconstructor.gc(view);
495        self.vote1_collector.gc(view, epoch);
496        self.vote2_collector.gc(view, epoch);
497        self.timeout_collector.gc(view, epoch);
498        self.timeout_one_honest_collector.gc(view, epoch);
499        self.epoch_manager.gc(epoch);
500        self.block_builder.gc(view);
501    }
