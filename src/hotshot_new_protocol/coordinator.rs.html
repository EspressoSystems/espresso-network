85    pub fn maker(
86        membership_coordinator: EpochMembershipCoordinator<T>,
87        network: N,
88        initializer: &HotShotInitializer<T>,
89        public_key: T::SignatureKey,
90        private_key: <T::SignatureKey as SignatureKey>::PrivateKey,
91        timeout_duration: Duration,
92    ) -> Self {
93        let consensus = Consensus::new(
94            membership_coordinator.clone(),
95            public_key.clone(),
96            private_key,
97            initializer.anchor_leaf.clone(),
98            initializer.epoch_height,
99        );
100        let state_manager = StateManager::new(Arc::new(initializer.instance_state.clone()));
101
102        let lock = upgrade_lock();
103        Self::builder()
104            .consensus(consensus)
105            .network(Network::new(
106                network,
107                membership_coordinator.clone(),
108                lock.clone(),
109            ))
110            .state_manager(state_manager)
111            .vid_disperser(VidDisperser::new(membership_coordinator.clone()))
112            .vid_reconstructor(VidReconstructor::new())
113            .vote1_collector(VoteCollector::new(
114                membership_coordinator.clone(),
115                lock.clone(),
116            ))
117            .vote2_collector(VoteCollector::new(
118                membership_coordinator.clone(),
119                lock.clone(),
120            ))
121            .timeout_collector(VoteCollector::new(
122                membership_coordinator.clone(),
123                lock.clone(),
124            ))
125            .timeout_one_honest_collector(VoteCollector::new(
126                membership_coordinator.clone(),
127                lock.clone(),
128            ))
129            .checkpoint_collector(VoteCollector::new(membership_coordinator.clone(), lock))
130            .epoch_manager(EpochManager::new(
131                initializer.epoch_height,
132                membership_coordinator.clone(),
133            ))
134            .block_builder(BlockBuilder::new(
135                Arc::new(initializer.instance_state.clone()),
136                membership_coordinator.clone(),
137                BlockBuilderConfig::default(),
138            ))
139            .proposal_validator(ProposalValidator::new(membership_coordinator.clone()))
140            .membership_coordinator(membership_coordinator)
141            .timer(Timer::new(
142                timeout_duration,
143                ViewNumber::genesis(),
144                EpochNumber::genesis(),
145            ))
146            .public_key(public_key)
147            .build()
148    }

154    /// Bootstrap the coordinator so the view-1 leader can propose.
155    ///
156    /// Emits an initial `ViewChanged(1)` and, if this node is the view-1
157    /// leader, a `RequestBlockAndHeader` for view 1.  Call this after
158    /// `seed_genesis` on the inner `Consensus` instance.
159    pub async fn start(&mut self) {
160        let view = ViewNumber::new(1);
161        let epoch = EpochNumber::genesis();
162
163        self.outbox
164            .push_back(ConsensusOutput::ViewChanged(view, epoch));
165
166        if let Some(leader) = self.leader(view, epoch).await
167            && leader == self.public_key
168        {
169            let genesis_proposal = self
170                .consensus
171                .proposal_at(ViewNumber::genesis())
172                .expect("genesis proposal must be seeded before start()")
173                .clone();
174            self.outbox
175                .push_back(ConsensusOutput::RequestBlockAndHeader(
176                    BlockAndHeaderRequest {
177                        view,
178                        epoch,
179                        parent_proposal: genesis_proposal,
180                    },
181                ));
182        }
183    }

185    fn handle_request(&mut self, request: ClientRequest<T>) {
186        match request {
187            ClientRequest::CurrentView(tx) => {
188                let _ = tx.send(self.consensus.current_view());
189            },
190            ClientRequest::CurrentEpoch(tx) => {
191                let _ = tx.send(self.consensus.current_epoch());
192            },
193            ClientRequest::DecidedLeaf(tx) => {
194                let _ = tx.send(self.consensus.last_decided_leaf().clone());
195            },
196            ClientRequest::DecidedState(tx) => {
197                let view = self.consensus.last_decided_leaf().view_number();
198                let _ = tx.send(self.state_manager.get_state(&view));
199            },
200            ClientRequest::UndecidedLeaves(tx) => {
201                let _ = tx.send(self.consensus.undecided_leaves().cloned().collect());
202            },
203            ClientRequest::GetState { view, respond } => {
204                let _ = respond.send(self.state_manager.get_state(&view));
205            },
206            ClientRequest::GetStateAndDelta { view, respond } => {
207                let _ = respond.send(self.state_manager.get_state_and_delta(&view));
208            },
209            ClientRequest::UpdateLeaf { update, respond } => {
210                self.state_manager.update_state(update);
211                let _ = respond.send(());
212            },
213        }
214    }

216    pub async fn next_consensus_input(&mut self) -> Result<ConsensusInput<T>, CoordinatorError> {
217        loop {
218            select! {
219                message = self.network.receive() => match message {
220                    Ok(m) => {
221                        if let Some(input) = self.on_network_message(m).await {
222                            return Ok(input)
223                        }
224                    }
225                    Err(e) => {
226                        return Err(CoordinatorError::from(e).context("network receive"))
227                    }
228                },
229                () = &mut self.timer => {
230                    let input = ConsensusInput::Timeout(self.timer.view(), self.timer.epoch());
231                    // Timer is only reset so we can resend the timeout vote
232                    // This isn't strictly necessary for the protocol, but it's a good idea to
233                    // resend the timeout vote to avoid a situation where the network is stuck
234                    // view because we fail to form a timeout certificate.
235                    self.timer.reset();
236                    return Ok(input)
237                }
238                Some(output) = self.state_manager.next() => {
239                    if let Some(input) = self.on_state_manager_output(output).await {
240                        return Ok(input)
241                    }
242                }
243                Some(request) = self.client.next_request() => {
244                    self.handle_request(request);
245                }
246                Some(tcert) = self.timeout_collector.next() => {
247                    return Ok(ConsensusInput::TimeoutCertificate(tcert))
248                }
249                Some(out) = self.timeout_one_honest_collector.next() => {
250                    let Some(epoch) = out.data.epoch else {
251                        let msg = format!("missing epoch in view {}", out.view_number());
252                        return Err(CoordinatorError::regular(msg).context("gc timeout one honest"))
253                    };
254                    return Ok(ConsensusInput::TimeoutOneHonest(out.view_number(), epoch))
255                }
256                Some(cert1) = self.vote1_collector.next() => {
257                    return Ok(ConsensusInput::Certificate1(cert1))
258                }
259                Some(cert2) = self.vote2_collector.next() => {
260                    return Ok(ConsensusInput::Certificate2(cert2))
261                }
262                Some(item) = self.proposal_validator.next() => match item {
263                    Ok(validated) => {
264                        let s = validated.message.vid_share.clone();
265                        let m = validated.message.proposal.data.block_header.metadata().clone();
266                        self.vid_reconstructor.handle_vid_share(s, m);
267                        return Ok(ConsensusInput::Proposal(validated.sender, validated.message))
268                    }
269                    Err(e) => {
270                        return Err(CoordinatorError::regular(e).context("proposal validation"))
271                    }
272                },
273                Some(cert) = self.checkpoint_collector.next() => {
274                    let Some(epoch) = cert.epoch() else {
275                        let msg = format!("missing epoch in view {}", cert.view_number());
276                        return Err(CoordinatorError::critical(msg).context("gc certificate"))
277                    };
278                    self.gc(cert.view_number(), epoch);
279                }
280                Some(item) = self.block_builder.next() => match item {
281                    Ok(block) => {
282                        self.state_manager.request_header(HeaderRequest::from(&block));
283                        let next_view = block.view + 1;
284                        let epoch = block.epoch;
285                        let manifest = block.manifest.clone();
286                        self.unicast_to_leader(
287                            next_view,
288                            epoch,
289                            BlockMessage::DedupManifest(manifest),
290                        )
291                        .await?;
292                        return Ok(block.into())
293                    }
294                    Err(err) => {
295                        return Err(CoordinatorError::regular(err).context("block building"))
296                    }
297                },
298                Some(item) = self.vid_disperser.next() => match item {
299                    Ok(out) => {
300                        return Ok(ConsensusInput::VidDisperseCreated(out.view, out.disperse))
301                    }
302                    Err(()) => {
303                        return Err(CoordinatorError::unspecified().context("vid disperse"))
304                    }
305                },
306                Some(item) = self.vid_reconstructor.next() => match item {
307                    Ok(out) => {
308                        self.block_builder.on_block_reconstructed(out.tx_commitments);
309                        return Ok(ConsensusInput::BlockReconstructed(out.view, out.payload_commitment))
310                    }
311                    Err(()) => {
312                        return Err(CoordinatorError::unspecified().context("vid reconstruction"))
313                    }
314                },
315                Some(result) = self.epoch_manager.next() => match result {
316                    Ok(EpochRootResult::DrbResult(epoch, drb_result)) => {
317                        return Ok(ConsensusInput::DrbResult(epoch, drb_result))
318                    }
319                    Ok(EpochRootResult::RootAdded(_epoch)) => {}
320                    Err(err) => {
321                        return Err(CoordinatorError::regular(err))
322                    }
323                },
324                else => {
325                    return Err(CoordinatorError::critical(ErrorSource::NoInput))
326                }
327            }
328        }
329    }

377    pub async fn on_network_message(
378        &mut self,
379        message: Message<T, Unchecked>,
380    ) -> Option<ConsensusInput<T>> {
381        match message.message_type {
382            MessageType::Consensus(msg) => match msg {
383                ConsensusMessage::Proposal(p) => {
384                    if self.consensus.wants_proposal(&p) {
385                        self.proposal_validator.validate(p);
386                    }
387                    None
388                },
389                ConsensusMessage::Vote1(vote1) => {
390                    self.vote1_collector.accumulate_vote(vote1.vote).await;
391                    self.vid_reconstructor
392                        .handle_vid_share(vote1.vid_share, None);
393                    None
394                },
395                ConsensusMessage::Vote2(vote2) => {
396                    self.vote2_collector.accumulate_vote(vote2).await;
397                    None
398                },
399                ConsensusMessage::Certificate1(certificate1, _key) => {
400                    Some(ConsensusInput::Certificate1(certificate1))
401                },
402                ConsensusMessage::Certificate2(certificate2, _key) => {
403                    Some(ConsensusInput::Certificate2(certificate2))
404                },
405                ConsensusMessage::TimeoutVote(timeout_msg) => {
406                    self.timeout_collector
407                        .accumulate_vote(timeout_msg.vote.clone())
408                        .await;
409                    self.timeout_one_honest_collector
410                        .accumulate_vote(timeout_msg.vote)
411                        .await;
412                    None
413                },
414                ConsensusMessage::TimeoutCertificate(tc) => {
415                    Some(ConsensusInput::TimeoutCertificate(tc))
416                },
417                ConsensusMessage::EpochChange(epoch_change) => {
418                    Some(ConsensusInput::EpochChange(epoch_change))
419                },
420                ConsensusMessage::Checkpoint(checkpoint) => {
421                    self.checkpoint_collector.accumulate_vote(checkpoint).await;
422                    None
423                },
424            },
425            MessageType::Block(msg) => {
426                match msg {
427                    BlockMessage::Transactions(msg) => self.block_builder.on_transactions(msg),
428                    BlockMessage::DedupManifest(manifest) => {
429                        if let Some(view_leader) = self.leader(manifest.view, manifest.epoch).await
430                            && view_leader == message.sender
431                        {
432                            self.block_builder.on_dedup_manifest(manifest)
433                        }
434                    },
435                }
436                None
437            },
438            MessageType::External(data) => {
439                self.coordinator_outbox
440                    .push_back(CoordinatorOutput::ExternalMessageReceived {
441                        sender: message.sender,
442                        data,
443                    });
444                None
445            },
446        }
447    }

449    pub async fn process_consensus_output(
450        &mut self,
451        output: ConsensusOutput<T>,
452    ) -> Result<(), CoordinatorError> {
453        match output {
454            ConsensusOutput::RequestState(state_request) => {
455                self.state_manager.request_state(state_request);
456            },
457            ConsensusOutput::RequestVidDisperse {
458                view,
459                epoch,
460                payload,
461                metadata,
462            } => {
463                self.vid_disperser.request_vid_disperse(VidDisperseRequest {
464                    view,
465                    epoch,
466                    block: payload,
467                    metadata,
468                });
469            },
470            ConsensusOutput::RequestDrbResult(epoch) => {
471                self.epoch_manager.request_drb_result(epoch);
472            },
473            ConsensusOutput::SendCheckpointVote(checkpoint_vote) => {
474                let message = Message {
475                    sender: self.public_key.clone(),
476                    message_type: MessageType::Consensus(ConsensusMessage::Checkpoint(
477                        checkpoint_vote,
478                    )),
479                };
480                self.network
481                    .broadcast(message)
482                    .await
483                    .map_err(|e| CoordinatorError::from(e).context("broadcast checkpoint vote"))?
484            },
485            ConsensusOutput::LeafDecided { leaves, .. } => {
486                for leaf in leaves {
487                    self.epoch_manager.handle_leaf_decided(leaf);
488                }
489            },
490            ConsensusOutput::LockUpdated(_) => {}, // TODO
491            ConsensusOutput::RequestBlockAndHeader(request) => {
492                self.block_builder.request_block(request);
493            },
494            ConsensusOutput::RequestProposal(..) => {}, // TODO
495            ConsensusOutput::SendProposal(proposal, vid_disperse) => {
496                // TODO: This may be done async in network so we do not spend
497                // too much time here in this loop.
498                for vid_share in vid_disperse.to_shares() {
499                    let recipient_key = vid_share.recipient_key.clone();
500                    let message = Message {
501                        sender: self.public_key.clone(),
502                        message_type: MessageType::Consensus(ConsensusMessage::Proposal(
503                            ProposalMessage::validated(proposal.clone(), vid_share),
504                        )),
505                    };
506                    if let Err(err) = self.network.unicast(recipient_key, message).await {
507                        let err = CoordinatorError::from(err).context("vid share unicast");
508                        if err.severity == Severity::Critical {
509                            return Err(err);
510                        } else {
511                            warn!(%err, "network error while sending vid share")
512                        }
513                    }
514                }
515            },
516            ConsensusOutput::SendTimeoutVote(vote, lock) => {
517                let message = Message {
518                    sender: self.public_key.clone(),
519                    message_type: MessageType::Consensus(ConsensusMessage::TimeoutVote(
520                        message::TimeoutVoteMessage { vote, lock },
521                    )),
522                };
523                self.network
524                    .broadcast(message)
525                    .await
526                    .map_err(|e| CoordinatorError::from(e).context("broadcast timeout vote"))?
527            },
528            ConsensusOutput::SendTimeoutCertificate(tc, view, epoch) => {
529                if let Some(leader) = self.leader(view, epoch).await {
530                    let message = Message {
531                        sender: self.public_key.clone(),
532                        message_type: MessageType::Consensus(ConsensusMessage::TimeoutCertificate(
533                            tc,
534                        )),
535                    };
536                    self.network
537                        .unicast(leader, message)
538                        .await
539                        .map_err(|e| CoordinatorError::from(e).context("timeout certificate"))?;
540                }
541            },
542            ConsensusOutput::SendVote1(vote1) => {
543                let message = Message {
544                    sender: self.public_key.clone(),
545                    message_type: MessageType::Consensus(ConsensusMessage::Vote1(vote1)),
546                };
547                self.network
548                    .broadcast(message)
549                    .await
550                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote1"))?
551            },
552            ConsensusOutput::SendVote2(vote2) => {
553                let message = Message {
554                    sender: self.public_key.clone(),
555                    message_type: MessageType::Consensus(ConsensusMessage::Vote2(vote2)),
556                };
557                self.network
558                    .broadcast(message)
559                    .await
560                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote2"))?
561            },
562            ConsensusOutput::SendEpochChange(epoch_change) => {
563                let message = Message {
564                    sender: self.public_key.clone(),
565                    message_type: MessageType::Consensus(ConsensusMessage::EpochChange(
566                        epoch_change,
567                    )),
568                };
569                self.network
570                    .broadcast(message)
571                    .await
572                    .map_err(|e| CoordinatorError::from(e).context("broadcast epoch change"))?
573            },
574            ConsensusOutput::SendCertificate1(cert1) => {
575                let message = Message {
576                    sender: self.public_key.clone(),
577                    message_type: MessageType::Consensus(ConsensusMessage::Certificate1(
578                        cert1,
579                        self.public_key.clone(),
580                    )),
581                };
582                self.network
583                    .broadcast(message)
584                    .await
585                    .map_err(|e| CoordinatorError::from(e).context("broadcast certificate1"))?
586            },
587            ConsensusOutput::ProposalValidated { .. } => {},
588            ConsensusOutput::ViewChanged(view, epoch) => {
589                self.consensus.set_view(view, epoch);
590                self.timer.reset_with_epoch(view, epoch);
591                let txns = self.block_builder.on_view_changed(view, epoch);
592                if !txns.is_empty() {
593                    let next_view = view + 1;
594                    self.unicast_to_leader(
595                        next_view,
596                        epoch,
597                        BlockMessage::Transactions(TransactionMessage {
598                            view: next_view,
599                            transactions: txns,
600                        }),
601                    )
602                    .await
603                    .map_err(|e| e.context("unicast transactions"))?;
604                }
605            },
606        }
607        Ok(())
608    }

610    async fn unicast_to_leader(
611        &mut self,
612        view: ViewNumber,
613        epoch: EpochNumber,
614        msg: BlockMessage<T>,
615    ) -> Result<(), CoordinatorError> {
616        let Some(leader) = self.leader(view, epoch).await else {
617            warn!(%view, %epoch, "failed to resolve leader for unicast");
618            return Ok(());
619        };
620        let message = Message {
621            sender: self.public_key.clone(),
622            message_type: MessageType::Block(msg),
623        };
624        self.network
625            .unicast(leader, message)
626            .await
627            .map_err(|e| CoordinatorError::from(e).context("leader unicast"))
628    }

630    async fn leader(&self, view: ViewNumber, epoch: EpochNumber) -> Option<T::SignatureKey> {
631        let membership = self
632            .membership_coordinator
633            .membership_for_epoch(Some(epoch))
634            .await
635            .ok()?;
636        membership.leader(view).await.ok()
637    }

639    fn gc(&mut self, view: ViewNumber, epoch: EpochNumber) {
640        self.consensus.gc(view, epoch);
641        self.checkpoint_collector.gc(view, epoch);
642        self.network.gc(view, epoch);
643        self.state_manager.gc(view);
644        self.vid_disperser.gc(view);
645        self.vid_reconstructor.gc(view);
646        self.vote1_collector.gc(view, epoch);
647        self.vote2_collector.gc(view, epoch);
648        self.timeout_collector.gc(view, epoch);
649        self.timeout_one_honest_collector.gc(view, epoch);
650        self.epoch_manager.gc(epoch);
651        self.block_builder.gc(view);
652    }
