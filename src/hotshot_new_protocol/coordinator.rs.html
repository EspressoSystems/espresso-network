112    #[builder(builder_type = CoordinatorMaker, finish_fn = make)]
113    #[allow(clippy::too_many_arguments)]
114    pub fn maker(
115        membership_coordinator: EpochMembershipCoordinator<T>,
116        network: N,
117        initializer: &HotShotInitializer<T>,
118        upgrade_lock: UpgradeLock<T>,
119        public_key: T::SignatureKey,
120        private_key: <T::SignatureKey as SignatureKey>::PrivateKey,
121        state_private_key: <T::StateSignatureKey as StateSignatureKey>::StatePrivateKey,
122        stake_table_capacity: usize,
123        timeout_duration: Duration,
124        storage: S,
125        garbage_collection_interval: u64,
126        metrics: &dyn Metrics,
127    ) -> Self {
128        let mut consensus = Consensus::new(
129            membership_coordinator.clone(),
130            public_key.clone(),
131            private_key.clone(),
132            state_private_key,
133            stake_table_capacity,
134            upgrade_lock.clone(),
135            initializer.anchor_leaf.clone(),
136            initializer.epoch_height,
137            garbage_collection_interval,
138        );
139
140        let genesis_cert1 = initializer.high_qc.clone();
141        let genesis_proposal = message::Proposal {
142            block_header: initializer.anchor_leaf.block_header().clone(),
143            view_number: ViewNumber::genesis(),
144            epoch: EpochNumber::genesis(),
145            justify_qc: genesis_cert1.clone(),
146            next_epoch_justify_qc: None,
147            upgrade_certificate: None,
148            view_change_evidence: None,
149            next_drb_result: None,
150            state_cert: None,
151        };
152        let mut state_manager = StateManager::new(
153            Arc::new(initializer.instance_state.clone()),
154            upgrade_lock.clone(),
155        );
156        state_manager.seed_state(
157            initializer.anchor_leaf.view_number(),
158            initializer.anchor_state.clone(),
159            initializer.anchor_leaf.clone(),
160        );
161        // The synthetic genesis proposal has a non-null justify_qc (the genesis
162        // cert1) so the leaf derived from it has a different commitment than
163        // the anchor leaf produced by `Leaf2::genesis`. `request_header` for
164        // view 1 looks up the parent state by the *proposal's* leaf
165        // commitment, so seed the same state under that commitment too.
166        state_manager.seed_state(
167            ViewNumber::genesis(),
168            initializer.anchor_state.clone(),
169            Leaf2::from(genesis_proposal.clone()),
170        );
171        consensus.seed_genesis(genesis_cert1, genesis_proposal);
172
173        let lock = upgrade_lock.clone();
174        Self::builder()
175            .consensus(consensus)
176            .network(network)
177            .state_manager(state_manager)
178            .vid_disperser(VidDisperser::new(membership_coordinator.clone()))
179            .vid_reconstructor(VidReconstructor::new())
180            .vote1_collector(VoteCollector::new(
181                membership_coordinator.clone(),
182                lock.clone(),
183            ))
184            .vote2_collector(VoteCollector::new(
185                membership_coordinator.clone(),
186                lock.clone(),
187            ))
188            .timeout_collector(VoteCollector::new(
189                membership_coordinator.clone(),
190                lock.clone(),
191            ))
192            .timeout_one_honest_collector(VoteCollector::new(
193                membership_coordinator.clone(),
194                lock.clone(),
195            ))
196            .checkpoint_collector(VoteCollector::new(
197                membership_coordinator.clone(),
198                lock.clone(),
199            ))
200            .epoch_root_collector(EpochRootVoteCollector::new(
201                membership_coordinator.clone(),
202                lock,
203            ))
204            .epoch_manager(EpochManager::new(
205                initializer.epoch_height,
206                membership_coordinator.clone(),
207            ))
208            .block_builder(BlockBuilder::new(
209                Arc::new(initializer.instance_state.clone()),
210                membership_coordinator.clone(),
211                BlockBuilderConfig::default(),
212                upgrade_lock.clone(),
213            ))
214            .proposal_validator(ProposalValidator::new(
215                membership_coordinator.clone(),
216                initializer.epoch_height,
217                upgrade_lock.clone(),
218            ))
219            .share_validator(VidShareValidator::new(
220                membership_coordinator.clone(),
221                initializer.epoch_height,
222                upgrade_lock,
223            ))
224            .storage(Storage::new(storage, private_key))
225            .membership_coordinator(membership_coordinator)
226            .timer(Timer::new(
227                timeout_duration,
228                ViewNumber::genesis(),
229                EpochNumber::genesis(),
230            ))
231            .public_key(public_key)
232            .maybe_metrics(
233                metrics
234                    .is_recording()
235                    .then(|| metrics::Metrics::new(metrics)),
236            )
237            .build()
238    }

240    /// Emit `ViewChanged(current_view + 1)` and, if leader, a
241    /// `RequestBlockAndHeader`.
242    pub fn start(&mut self) {
243        let cur_view = self.consensus.current_view();
244        let next_view = cur_view + 1;
245        let epoch = self
246            .consensus
247            .current_epoch()
248            .unwrap_or(EpochNumber::genesis());
249
250        if self.consensus.last_decided_leaf().view_number() == ViewNumber::genesis() {
251            // Genesis DA never flows through the normal block-builder path.
252            let genesis_leaf = self.consensus.last_decided_leaf().clone();
253            let (payload, metadata) = T::BlockPayload::empty();
254            self.storage.append_da(
255                ViewNumber::genesis(),
256                EpochNumber::genesis(),
257                payload,
258                metadata,
259                genesis_leaf.payload_commitment(),
260            );
261
262            // Emit `LeafDecided` for genesis so persistence sees the header.
263            self.outbox.push_back(ConsensusOutput::LeafDecided {
264                leaves: vec![genesis_leaf],
265                cert1: self
266                    .consensus
267                    .cert1_at(ViewNumber::genesis())
268                    .cloned()
269                    .expect("genesis cert1 must be seeded"),
270                cert2: None,
271                vid_shares: vec![None],
272            });
273        }
274
275        self.outbox
276            .push_back(ConsensusOutput::ViewChanged(next_view, epoch));
277
278        if let Some(leader) = self.leader(next_view, epoch)
279            && leader == self.public_key
280        {
281            let parent_proposal = self
282                .consensus
283                .proposal_at(cur_view)
284                .expect("parent proposal must be seeded before start()")
285                .clone();
286            self.outbox
287                .push_back(ConsensusOutput::RequestBlockAndHeader(
288                    BlockAndHeaderRequest {
289                        view: next_view,
290                        epoch,
291                        parent_proposal,
292                    },
293                ));
294        }
295    }

301    pub async fn next_consensus_input(&mut self) -> Result<ConsensusInput<T>, CoordinatorError> {
302        loop {
303            let next_input = self
304                .metrics
305                .as_ref()
306                .map(|m| Measurement::start(m.next_consensus_input.clone()));
307            select! {
308                message = self.network.receive() => match message {
309                    Ok(m) => {
310                        finish_measurement(next_input);
311                        if let Some(input) = self.on_network_message(m).await {
312                            return Ok(input)
313                        }
314                    }
315                    Err(e) => {
316                        finish_measurement(next_input);
317                        return Err(CoordinatorError::from(e).context("network receive"))
318                    }
319                },
320                () = &mut self.timer => {
321                    let view = self.timer.view();
322                    let epoch = self.timer.epoch();
323                    if let Some(stats) = self.vote1_collector.stats(view, epoch).await {
324                        warn!(
325                            %view, %epoch,
326                            stake = %stats.stake,
327                            threshold = %stats.threshold,
328                            "timeout: vote1 stake observed (deduped by signer)"
329                        );
330                    } else {
331                        warn!(%view, %epoch, "timeout: no vote1 received for this view");
332                    }
333                    let input = ConsensusInput::Timeout(view, epoch);
334                    finish_measurement(next_input);
335                    if let Some(m) = &mut self.metrics {
336                        m.timeouts.add(1)
337                    }
338                    return Ok(input)
339                }
340                Some(output) = self.state_manager.next() => {
341                    finish_measurement(next_input);
342                    if let Some(input) = self.on_state_manager_output(output) {
343                        return Ok(input)
344                    }
345                }
346                Some(request) = self.client.next_request() => {
347                    finish_measurement(next_input);
348                    if let Err(err) = self.on_client_request(request).await {
349                        error!(%err, "error while handling client request");
350                    }
351                }
352                Some(tcert) = self.timeout_collector.next() => {
353                    finish_measurement(next_input);
354                    return Ok(ConsensusInput::TimeoutCertificate(tcert))
355                }
356                Some(out) = self.timeout_one_honest_collector.next() => {
357                    finish_measurement(next_input);
358                    let Some(epoch) = out.data.epoch else {
359                        let msg = format!("missing epoch in view {}", out.view_number());
360                        return Err(CoordinatorError::regular(msg).context("gc timeout one honest"))
361                    };
362                    return Ok(ConsensusInput::TimeoutOneHonest(out.view_number(), epoch))
363                }
364                Some(cert1) = self.vote1_collector.next() => {
365                    finish_measurement(next_input);
366                    return Ok(ConsensusInput::Certificate1(cert1))
367                }
368                Some(cert2) = self.vote2_collector.next() => {
369                    finish_measurement(next_input);
370                    return Ok(ConsensusInput::Certificate2(cert2))
371                }
372                Some((cert1, state_cert)) = self.epoch_root_collector.next() => {
373                    finish_measurement(next_input);
374                    self.storage.append_state_cert(
375                        ViewNumber::new(state_cert.light_client_state.view_number),
376                        state_cert.clone(),
377                    );
378                    return Ok(ConsensusInput::EpochRootCertificates { cert1, state_cert })
379                }
380                Some(item) = self.share_validator.next() => match item {
381                    Ok(vid_share) => {
382                        finish_measurement(next_input);
383                        let view = vid_share.view_number();
384                        let Some(validated) = self.cached_validated_proposals.remove(&view) else {
385                            // Wait for the proposal
386                            self.cached_vid_shares.insert(view, vid_share);
387                            continue;
388                        };
389                        if !check_payload_commitment(&validated.message.proposal, &vid_share) {
390                            continue;
391                        }
392                        return self.on_proposal_and_vid_share(validated, vid_share)
393                    },
394                    Err(e) => {
395                        finish_measurement(next_input);
396                        return Err(CoordinatorError::regular(e).context("vid share validation"))
397                    }
398                },
399                Some(item) = self.proposal_validator.next() => match item {
400                    Ok(validated) => {
401                        finish_measurement(next_input);
402                        // Refresh the network's peer set when a proposal is validated.
403                        let epoch = validated.message.proposal.data.epoch;
404                        if let Err(err) = self
405                            .network
406                            .apply_epoch(epoch, &self.membership_coordinator)
407                        {
408                            error!(%epoch, %err, "network apply_epoch failed");
409                        }
410
411                        let view = validated.message.proposal.data.view_number();
412                        let Some(vid_share) = self.cached_vid_shares.remove(&view) else {
413                            // Wait for the vid share
414                            self.cached_validated_proposals.insert(view, validated);
415                            continue;
416                        };
417                        // Check for commitment correspondence
418                        if !check_payload_commitment(&validated.message.proposal, &vid_share) {
419                            continue;
420                        }
421                        return self.on_proposal_and_vid_share(validated, vid_share)
422                    }
423                    Err(e) => {
424                        finish_measurement(next_input);
425                        return Err(CoordinatorError::regular(e).context("proposal validation"))
426                    }
427                },
428                Some(cert) = self.checkpoint_collector.next() => {
429                    finish_measurement(next_input);
430                    let Some(epoch) = cert.epoch() else {
431                        let msg = format!("missing epoch in view {}", cert.view_number());
432                        return Err(CoordinatorError::critical(msg).context("gc certificate"))
433                    };
434                    self.gc(cert.view_number(), epoch);
435                }
436                Some(item) = self.block_builder.next() => match item {
437                    Ok(block) => {
438                        finish_measurement(next_input);
439                        self.state_manager.request_header(HeaderRequest::from(&block));
440                        let next_view = block.view + 1;
441                        let epoch = block.epoch;
442                        let manifest = block.manifest.clone();
443                        self.storage.append_da(
444                            block.view,
445                            block.epoch,
446                            block.payload.payload.clone(),
447                            block.payload.metadata.clone(),
448                            block.payload_commitment,
449                        );
450                        // We built this block; skip reconstructing it from our own loopback share.
451                        self.vid_reconstructor.mark_reconstructed(block.view);
452                        self.unicast_to_leader(
453                            next_view,
454                            epoch,
455                            BlockMessage::DedupManifest(manifest),
456                        )?;
457                        return Ok(block.into())
458                    }
459                    Err(err) => {
460                        finish_measurement(next_input);
461                        return Err(CoordinatorError::regular(err).context("block building"))
462                    }
463                },
464                Some(item) = self.vid_disperser.next() => match item {
465                    Ok(out) => {
466                        finish_measurement(next_input);
467                        return Ok(ConsensusInput::VidDisperseCreated(out.view, out.disperse))
468                    }
469                    Err(()) => {
470                        finish_measurement(next_input);
471                        return Err(CoordinatorError::unspecified().context("vid disperse"))
472                    }
473                },
474                Some(item) = self.vid_reconstructor.next() => match item {
475                    Ok(out) => {
476                        finish_measurement(next_input);
477                        self.block_builder.on_block_reconstructed(out.tx_commitments);
478                        self.storage.append_da(
479                            out.view,
480                            out.epoch,
481                            out.payload.clone(),
482                            out.metadata.clone(),
483                            VidCommitment::V2(out.payload_commitment),
484                        );
485                        if let Some(proposal) = self.consensus.proposal_at(out.view) {
486                            self.outbox.push_back(ConsensusOutput::BlockPayloadReconstructed {
487                                view: out.view,
488                                header: proposal.block_header.clone(),
489                                payload: out.payload,
490                            });
491                        }
492                        return Ok(ConsensusInput::BlockReconstructed(out.view, out.payload_commitment))
493                    }
494                    Err(()) => {
495                        finish_measurement(next_input);
496                        return Err(CoordinatorError::unspecified().context("vid reconstruction"))
497                    }
498                },
499                Some(result) = self.epoch_manager.next() => match result {
500                    Ok(EpochRootResult::DrbResult(epoch, drb_result)) => {
501                        finish_measurement(next_input);
502                        // New epoch data available — retry votes that were
503                        // buffered because their membership wasn't ready.
504                        self.vote1_collector.retry_pending_votes().await;
505                        self.vote2_collector.retry_pending_votes().await;
506                        self.timeout_collector.retry_pending_votes().await;
507                        self.timeout_one_honest_collector.retry_pending_votes().await;
508                        return Ok(ConsensusInput::DrbResult(epoch, drb_result))
509                    }
510                    Err(failure) => {
511                        finish_measurement(next_input);
512                        // Catchup/compute failed. The epoch manager clears
513                        // the pending guard; consensus's `maybe_propose`
514                        // will re-request the DRB when it next tries to
515                        // build a transition proposal and finds it missing.
516                        warn!(%failure.error, epoch = %failure.epoch, "DRB request failed");
517                        continue;
518                    }
519                },
520                else => {
521                    finish_measurement(next_input);
522                    return Err(CoordinatorError::critical(ErrorSource::NoInput))
523                }
524            }
525        }
526    }

536    pub fn process_consensus_output(
537        &mut self,
538        output: ConsensusOutput<T>,
539    ) -> Result<(), CoordinatorError> {
540        let node = self.node_id;
541        let _m = self
542            .metrics
543            .as_ref()
544            .map(|m| Measurement::start(m.process_consensus_output.clone()));
545        match output {
546            ConsensusOutput::RequestState(state_request) => {
547                debug!(
548                    %node,
549                    view = %state_request.view,
550                    epoch = %state_request.epoch,
551                    block = %state_request.block,
552                    "request state validation"
553                );
554                self.state_manager.request_state(state_request);
555            },
556            ConsensusOutput::RequestVidDisperse {
557                view,
558                epoch,
559                payload,
560                metadata,
561            } => {
562                debug!(%node, %view, %epoch, "request vid disperse");
563                self.vid_disperser.request_vid_disperse(VidDisperseRequest {
564                    view,
565                    epoch,
566                    block: payload,
567                    metadata,
568                });
569            },
570            ConsensusOutput::RequestDrbResult(epoch) => {
571                debug!(%node, %epoch, "request drb result");
572                self.epoch_manager.request_drb_result(epoch);
573            },
574            ConsensusOutput::SendCheckpointVote(checkpoint_vote) => {
575                let view = checkpoint_vote.view_number();
576                let epoch = checkpoint_vote.data.epoch;
577                debug!(%node, %view, %epoch, "send checkpoint vote");
578                let message = Message {
579                    sender: self.public_key.clone(),
580                    message_type: MessageType::Consensus(ConsensusMessage::Checkpoint(
581                        checkpoint_vote,
582                    )),
583                };
584                self.network
585                    .broadcast(message.view_number(), &message)
586                    .map_err(|e| CoordinatorError::from(e).context("broadcast checkpoint vote"))?
587            },
588            ConsensusOutput::LeafDecided {
589                leaves,
590                cert1,
591                cert2,
592                ..
593            } => {
594                info!(
595                    %node,
596                    view = %cert1.view_number(),
597                    epoch = ?cert1.epoch().map(|e| *e),
598                    leaves = leaves.len(),
599                    "leaves decided"
600                );
601                if let Some(cert2) = cert2 {
602                    self.storage.append_cert2(cert2.view_number, cert2.clone());
603                }
604                for leaf in leaves {
605                    self.epoch_manager.handle_leaf_decided(leaf);
606                }
607            },
608            ConsensusOutput::LockUpdated(cert) => {
609                debug!(
610                    %node,
611                    view = %cert.view_number(),
612                    epoch = ?cert.epoch().map(|e| *e),
613                    "lock updated"
614                );
615            },
616            ConsensusOutput::RequestBlockAndHeader(request) => {
617                debug!(
618                    %node,
619                    view = %request.view,
620                    epoch = %request.epoch,
621                    "request block and header"
622                );
623                self.block_builder.request_block(request);
624            },
625            ConsensusOutput::SendProposal(proposal) => {
626                let view = proposal.data.view_number;
627                let epoch = proposal.data.epoch;
628                let block = proposal.data.block_header.block_number();
629                info!(%node, %view, %epoch, %block, "send proposal");
630                self.storage.append_proposal(proposal.data.clone());
631                // TODO: This may be done async in network so we do not spend
632                // too much time here in this loop.
633
634                let message = Message {
635                    sender: self.public_key.clone(),
636                    message_type: MessageType::Consensus(ConsensusMessage::Proposal(
637                        ProposalMessage::validated(proposal.clone()),
638                    )),
639                };
640                if let Err(err) = self.network.broadcast(message.view_number(), &message) {
641                    let err = CoordinatorError::from(err).context("proposal broadcast");
642                    if err.severity == Severity::Critical {
643                        return Err(err);
644                    } else {
645                        warn!(%node, %err, "network error while broadcasting proposal")
646                    }
647                }
648            },
649            ConsensusOutput::SendVidShares(vid_shares) => {
650                debug!(%node, count = vid_shares.len(), "send vid shares");
651                for share in vid_shares {
652                    let recipient = share.data.recipient_key.clone();
653                    let message = Message {
654                        sender: self.public_key.clone(),
655                        message_type: MessageType::Consensus(ConsensusMessage::VidShare(share)),
656                    };
657                    if let Err(err) =
658                        self.network
659                            .unicast(message.view_number(), &recipient, &message)
660                    {
661                        let err = CoordinatorError::from(err).context("vid share unicast");
662                        if err.severity == Severity::Critical {
663                            return Err(err);
664                        } else {
665                            warn!(%node, %err, "network error while sending vid share")
666                        }
667                    }
668                }
669            },
670            ConsensusOutput::SendTimeoutVote(vote, lock) => {
671                let view = vote.view_number();
672                debug!(%node, %view, has_lock = lock.is_some(), "send timeout vote");
673                let message = Message {
674                    sender: self.public_key.clone(),
675                    message_type: MessageType::Consensus(ConsensusMessage::TimeoutVote(
676                        message::TimeoutVoteMessage { vote, lock },
677                    )),
678                };
679                self.network
680                    .broadcast(message.view_number(), &message)
681                    .map_err(|e| CoordinatorError::from(e).context("broadcast timeout vote"))?
682            },
683            ConsensusOutput::SendTimeoutCertificate(tc, view, epoch) => {
684                debug!(
685                    %node, %view, %epoch,
686                    cert_view = %tc.view_number(),
687                    "send timeout certificate"
688                );
689                if let Some(leader) = self.leader(view, epoch) {
690                    let message = Message {
691                        sender: self.public_key.clone(),
692                        message_type: MessageType::Consensus(ConsensusMessage::TimeoutCertificate(
693                            tc,
694                        )),
695                    };
696                    self.network
697                        .unicast(message.view_number(), &leader, &message)
698                        .map_err(|e| CoordinatorError::from(e).context("timeout certificate"))?;
699                }
700            },
701            ConsensusOutput::SendVote1(vote1) => {
702                let view = vote1.vote.view_number();
703                debug!(
704                    %node, %view,
705                    epoch_root = vote1.state_vote.is_some(),
706                    "send vote1"
707                );
708                let message = Message {
709                    sender: self.public_key.clone(),
710                    message_type: MessageType::Consensus(ConsensusMessage::Vote1(vote1)),
711                };
712                self.network
713                    .broadcast(message.view_number(), &message)
714                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote1"))?
715            },
716            ConsensusOutput::SendVote2(vote2) => {
717                debug!(%node, view = %vote2.view_number(), "send vote2");
718                let message = Message {
719                    sender: self.public_key.clone(),
720                    message_type: MessageType::Consensus(ConsensusMessage::Vote2(vote2)),
721                };
722                self.network
723                    .broadcast(message.view_number(), &message)
724                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote2"))?
725            },
726            ConsensusOutput::SendEpochChange(epoch_change) => {
727                info!(
728                    %node,
729                    view = %epoch_change.cert1.view_number(),
730                    epoch = ?epoch_change.cert1.epoch().map(|e| *e),
731                    "send epoch change"
732                );
733                let message = Message {
734                    sender: self.public_key.clone(),
735                    message_type: MessageType::Consensus(ConsensusMessage::EpochChange(
736                        epoch_change,
737                    )),
738                };
739                self.network
740                    .broadcast(message.view_number(), &message)
741                    .map_err(|e| CoordinatorError::from(e).context("broadcast epoch change"))?
742            },
743            ConsensusOutput::SendCertificate1(cert1) => {
744                debug!(
745                    %node,
746                    view = %cert1.view_number(),
747                    epoch = ?cert1.epoch().map(|e| *e),
748                    "send certificate1"
749                );
750                let message = Message {
751                    sender: self.public_key.clone(),
752                    message_type: MessageType::Consensus(ConsensusMessage::Certificate1(
753                        cert1,
754                        self.public_key.clone(),
755                    )),
756                };
757                self.network
758                    .broadcast(message.view_number(), &message)
759                    .map_err(|e| CoordinatorError::from(e).context("broadcast certificate1"))?
760            },
761            ConsensusOutput::ProposalValidated { proposal, sender } => {
762                debug!(
763                    %node,
764                    view = %proposal.data.view_number,
765                    sender = %KeyPrefix::from(&sender),
766                    "proposal validated"
767                );
768            },
769            ConsensusOutput::ViewChanged(view, epoch) => {
770                info!(%node, %view, %epoch, "view changed");
771                self.consensus.set_view(view, epoch);
772                self.timer.reset_with_epoch(view, epoch);
773                let txns = self.block_builder.on_view_changed(view, epoch);
774                if !txns.is_empty() {
775                    let next_view = view + 1;
776                    self.unicast_to_leader(
777                        next_view,
778                        epoch,
779                        BlockMessage::Transactions(TransactionMessage {
780                            view: next_view,
781                            transactions: txns,
782                        }),
783                    )
784                    .map_err(|e| e.context("unicast transactions"))?;
785                }
786
787                // Proactively fetch the DRB for the next epoch so
788                // late-starting nodes have it before they need it
789                let next_epoch = epoch + 1;
790                if next_epoch > EpochNumber::genesis() + 1 {
791                    self.epoch_manager.request_drb_result(next_epoch);
792                }
793            },
794            ConsensusOutput::BlockPayloadReconstructed { .. } => {},
795        }
796        Ok(())
797    }

831    pub(crate) async fn on_network_message(
832        &mut self,
833        message: Message<T, Unchecked>,
834    ) -> Option<ConsensusInput<T>> {
835        let sender = KeyPrefix::from(&message.sender);
836        let node = self.node_id;
837        let _m = self
838            .metrics
839            .as_ref()
840            .map(|m| Measurement::start(m.on_network_message.clone()));
841        match message.message_type {
842            MessageType::Consensus(msg) => match msg {
843                ConsensusMessage::Proposal(p) => {
844                    let view = p.view_number();
845                    let epoch = p.proposal.data.epoch;
846                    let block = p.proposal.data.block_header.block_number();
847                    debug!(%node, %sender, %view, %epoch, %block, "recv proposal");
848                    if self.consensus.wants_proposal_for_view(&view) {
849                        self.proposal_validator.validate(p);
850                    }
851                    None
852                },
853                ConsensusMessage::VidShare(share) => {
854                    let view = share.data.view_number();
855                    debug!(%node, %sender, %view, "recv vid share");
856                    if self.consensus.wants_proposal_for_view(&view) {
857                        self.share_validator.validate(share);
858                    }
859                    None
860                },
861                ConsensusMessage::Vote1(vote1) => {
862                    let view = vote1.vote.view_number();
863                    let bn = vote1.vote.data.block_number.unwrap_or(0);
864                    let epoch_height = *self.consensus.epoch_height;
865                    let is_epoch_root_vote = is_epoch_root(bn, epoch_height);
866                    debug!(
867                        %node, %sender, %view,
868                        epoch_root = is_epoch_root_vote,
869                        has_state_vote = vote1.state_vote.is_some(),
870                        "recv vote1"
871                    );
872                    if is_epoch_root_vote {
873                        // An epoch-root Vote1 MUST carry a state_vote.
874                        // Reject otherwise.
875                        vote1.state_vote.as_ref()?;
876                        self.epoch_root_collector.accumulate(vote1.clone()).await;
877                    } else {
878                        self.vote1_collector
879                            .accumulate_vote(vote1.vote.clone())
880                            .await;
881                    }
882                    self.vid_reconstructor
883                        .handle_vid_share(vote1.vid_share, None);
884                    None
885                },
886                ConsensusMessage::Vote2(vote2) => {
887                    let view = vote2.view_number();
888                    debug!(%node, %sender, %view, "recv vote2");
889                    self.vote2_collector.accumulate_vote(vote2).await;
890                    None
891                },
892                ConsensusMessage::Certificate1(certificate1, _key) => {
893                    debug!(
894                        %node, %sender,
895                        view = %certificate1.view_number(),
896                        epoch = ?certificate1.epoch().map(|e| *e),
897                        "recv certificate1"
898                    );
899                    Some(ConsensusInput::Certificate1(certificate1))
900                },
901                ConsensusMessage::Certificate2(certificate2, _key) => {
902                    debug!(
903                        %node, %sender,
904                        view = %certificate2.view_number(),
905                        epoch = ?certificate2.epoch().map(|e| *e),
906                        "recv certificate2"
907                    );
908                    Some(ConsensusInput::Certificate2(certificate2))
909                },
910                ConsensusMessage::TimeoutVote(timeout_msg) => {
911                    let view = timeout_msg.vote.view_number();
912                    debug!(
913                        %node, %sender, %view,
914                        has_lock = timeout_msg.lock.is_some(),
915                        "recv timeout vote"
916                    );
917                    self.timeout_collector
918                        .accumulate_vote(timeout_msg.vote.clone())
919                        .await;
920                    self.timeout_one_honest_collector
921                        .accumulate_vote(timeout_msg.vote)
922                        .await;
923                    None
924                },
925                ConsensusMessage::TimeoutCertificate(tc) => {
926                    debug!(
927                        %node, %sender,
928                        view = %tc.view_number(),
929                        epoch = ?tc.epoch().map(|e| *e),
930                        "recv timeout certificate"
931                    );
932                    Some(ConsensusInput::TimeoutCertificate(tc))
933                },
934                ConsensusMessage::EpochChange(epoch_change) => {
935                    debug!(
936                        %node, %sender,
937                        view = %epoch_change.cert1.view_number(),
938                        epoch = ?epoch_change.cert1.epoch().map(|e| *e),
939                        "recv epoch change"
940                    );
941                    Some(ConsensusInput::EpochChange(epoch_change))
942                },
943                ConsensusMessage::Checkpoint(checkpoint) => {
944                    debug!(
945                        %node, %sender,
946                        view = %checkpoint.view_number(),
947                        epoch = %checkpoint.data.epoch,
948                        "recv checkpoint vote"
949                    );
950                    self.checkpoint_collector.accumulate_vote(checkpoint).await;
951                    None
952                },
953            },
954            MessageType::Block(msg) => {
955                match msg {
956                    BlockMessage::Transactions(msg) => {
957                        debug!(
958                            %node, %sender,
959                            view = %msg.view,
960                            count = msg.transactions.len(),
961                            "recv transactions"
962                        );
963                        self.block_builder.on_transactions(msg)
964                    },
965                    BlockMessage::DedupManifest(manifest) => {
966                        debug!(
967                            %node, %sender,
968                            view = %manifest.view,
969                            epoch = %manifest.epoch,
970                            hashes = manifest.hashes.len(),
971                            "recv dedup manifest"
972                        );
973                        if let Some(view_leader) = self.leader(manifest.view, manifest.epoch)
974                            && view_leader == message.sender
975                        {
976                            self.block_builder.on_dedup_manifest(manifest)
977                        }
978                    },
979                }
980                None
981            },
982            MessageType::ProposalFetch(ProposalFetchMessage::Request(request)) => {
983                let view = request.view_number();
984                debug!(%node, %sender, %view, "recv proposal fetch request");
985                if !request.validate_sender(&message.sender) {
986                    warn!(
987                        %node,
988                        sender = %message.sender,
989                        %view,
990                        "ignoring invalid proposal fetch request signature"
991                    );
992                    return None;
993                }
994                if let Some(proposal) = self.consensus.signed_proposal(&view).cloned() {
995                    let response = Message {
996                        sender: self.public_key.clone(),
997                        message_type: MessageType::ProposalFetch(ProposalFetchMessage::Response(
998                            Box::new(proposal),
999                        )),
1000                    };
1001
1002                    if let Err(err) = self.network.unicast(view, &message.sender, &response) {
1003                        let err = CoordinatorError::from(err).context("proposal response");
1004                        warn!(%node, %err, "network error while sending proposal response");
1005                    }
1006                }
1007                None
1008            },
1009            MessageType::ProposalFetch(ProposalFetchMessage::Response(proposal)) => {
1010                debug!(
1011                    %node, %sender,
1012                    view = %proposal.data.view_number,
1013                    "recv proposal fetch response"
1014                );
1015                self.pending_proposal_fetches.resolve(&proposal);
1016                None
1017            },
1018            MessageType::External(data) => {
1019                debug!(%node, %sender, bytes = data.len(), "recv external message");
1020                self.coordinator_outbox
1021                    .push_back(CoordinatorOutput::ExternalMessageReceived {
1022                        sender: message.sender,
1023                        data,
1024                    });
1025                None
1026            },
1027        }
1028    }

1100    fn unicast_to_leader(
1101        &mut self,
1102        view: ViewNumber,
1103        epoch: EpochNumber,
1104        msg: BlockMessage<T>,
1105    ) -> Result<(), CoordinatorError> {
1106        let Some(leader) = self.leader(view, epoch) else {
1107            warn!(%view, %epoch, "failed to resolve leader for unicast");
1108            return Ok(());
1109        };
1110        let message = Message {
1111            sender: self.public_key.clone(),
1112            message_type: MessageType::Block(msg),
1113        };
1114        self.network
1115            .unicast(message.view_number(), &leader, &message)
1116            .map_err(|e| CoordinatorError::from(e).context("leader unicast"))
1117    }

1127    async fn on_client_request(
1128        &mut self,
1129        request: ClientRequest<T>,
1130    ) -> Result<(), CoordinatorError> {
1131        let _m = self
1132            .metrics
1133            .as_ref()
1134            .map(|m| Measurement::start(m.on_client_request.clone()));
1135        match request {
1136            ClientRequest::CurrentView(tx) => {
1137                let _ = tx.send(self.consensus.current_view());
1138            },
1139            ClientRequest::CurrentEpoch(tx) => {
1140                let _ = tx.send(self.consensus.current_epoch());
1141            },
1142            ClientRequest::DecidedLeaf(tx) => {
1143                let _ = tx.send(self.consensus.last_decided_leaf().clone());
1144            },
1145            ClientRequest::DecidedState(tx) => {
1146                let view = self.consensus.last_decided_leaf().view_number();
1147                let _ = tx.send(self.state(view).map(|s| s.state.clone()));
1148            },
1149            ClientRequest::UndecidedLeaves(tx) => {
1150                let _ = tx.send(self.consensus.undecided_leaves().cloned().collect());
1151            },
1152            ClientRequest::GetState { view, respond } => {
1153                let _ = respond.send(self.state(view).map(|s| s.state.clone()));
1154            },
1155            ClientRequest::GetStateAndDelta { view, respond } => {
1156                let _ = respond.send(match self.state(view) {
1157                    Some(s) => (Some(s.state.clone()), s.delta.clone()),
1158                    None => (None, None),
1159                });
1160            },
1161            ClientRequest::SubmitTransaction { tx, respond } => {
1162                self.block_builder.on_submit_transaction(tx);
1163                let _ = respond.send(());
1164            },
1165            ClientRequest::UpdateLeaf { update, respond } => {
1166                self.state_manager.update_state(update);
1167                let _ = respond.send(());
1168            },
1169            ClientRequest::RequestProposal {
1170                view,
1171                leaf_commitment,
1172                respond,
1173            } => {
1174                if let Some(proposal) = self.consensus.signed_proposal(&view)
1175                    && proposal_commitment(&proposal.data) == leaf_commitment
1176                {
1177                    let _ = respond.send(Ok(proposal.clone()));
1178                    return Ok(());
1179                }
1180                if !self
1181                    .pending_proposal_fetches
1182                    .contains_request(view, leaf_commitment)
1183                {
1184                    let request =
1185                        self.consensus
1186                            .signed_proposal_fetch_request(view)
1187                            .map_err(|err| {
1188                                let err = format!("failed to sign proposal request: {err}");
1189                                CoordinatorError::regular(err).context("sign proposal request")
1190                            })?;
1191
1192                    let message = Message {
1193                        sender: self.public_key.clone(),
1194                        message_type: MessageType::ProposalFetch(ProposalFetchMessage::Request(
1195                            request,
1196                        )),
1197                    };
1198
1199                    self.network
1200                        .broadcast(message.view_number(), &message)
1201                        .map_err(|err| {
1202                            CoordinatorError::from(err).context("broadcast proposal request")
1203                        })?;
1204                }
1205                self.pending_proposal_fetches
1206                    .push(view, leaf_commitment, respond);
1207            },
1208            ClientRequest::SendExternalMessage {
1209                view,
1210                payload,
1211                recipient,
1212                respond,
1213            } => {
1214                let message = Message {
1215                    sender: self.public_key.clone(),
1216                    message_type: MessageType::External(payload),
1217                };
1218                let result = self
1219                    .network
1220                    .unicast(view, &recipient, &message)
1221                    .map_err(|err| {
1222                        CoordinatorError::from(err)
1223                            .context("send external message")
1224                            .into()
1225                    });
1226                let _ = respond.send(result);
1227            },
1228            ClientRequest::SeedPreCutover { seed, respond } => {
1229                tracing::info!(
1230                    undecided = seed.undecided.len(),
1231                    anchor_view = *seed.decided_anchor.view_number(),
1232                    high_qc_view = seed.high_qc.as_ref().map(|qc| *qc.view_number()),
1233                    cutover_view = *seed.cutover_view,
1234                    states = seed.validated_states.len(),
1235                    "coordinator: applying legacy → new-protocol seed",
1236                );
1237
1238                // State manager is owned by the coordinator, so the
1239                // validated-state map must be applied here before the
1240                // seed is consumed by consensus.
1241                let anchor_view = seed.decided_anchor.view_number();
1242                if let Some(state) = seed.validated_states.get(&anchor_view).cloned() {
1243                    self.state_manager
1244                        .seed_state(anchor_view, state, seed.decided_anchor.clone());
1245                }
1246                for leaf in &seed.undecided {
1247                    let view = leaf.view_number();
1248                    if let Some(state) = seed.validated_states.get(&view).cloned() {
1249                        self.state_manager.seed_state(view, state, leaf.clone());
1250                    }
1251                }
1252
1253                let highest_seeded_leaf = seed.undecided.last().unwrap_or(&seed.decided_anchor);
1254                let cutover_epoch = EpochNumber::new(epoch_from_block_number(
1255                    highest_seeded_leaf.block_header().block_number(),
1256                    *self.consensus.epoch_height,
1257                ));
1258                let cutover_view = seed.cutover_view;
1259
1260                self.consensus.apply_pre_cutover_seed(seed);
1261
1262                // Refresh peers for the cutover epoch before kicking the
1263                // leader — the proposal-driven site can't fire yet.
1264                if let Err(err) = self
1265                    .network
1266                    .apply_epoch(cutover_epoch, &self.membership_coordinator)
1267                {
1268                    tracing::error!(
1269                        %cutover_epoch,
1270                        %err,
1271                        "network on_epoch_change failed during seed_pre_cutover",
1272                    );
1273                }
1274
1275                let cur_view = self.consensus.current_view();
1276                if self.consensus.timeout_cert_at(cur_view).is_some() {
1277                    self.resume_after_cutover_tc();
1278                } else if cur_view + 1 == cutover_view
1279                    && self.consensus.cert1_at(cur_view).is_some()
1280                    && self.consensus.proposal_at(cur_view).is_some()
1281                {
1282                    self.start();
1283                } else {
1284                    let epoch = self
1285                        .consensus
1286                        .current_epoch()
1287                        .unwrap_or(EpochNumber::genesis());
1288                    self.outbox
1289                        .push_back(ConsensusOutput::ViewChanged(cur_view, epoch));
1290                }
1291                while let Some(output) = self.outbox.pop_front() {
1292                    if let Err(err) = self.process_consensus_output(output) {
1293                        tracing::warn!(
1294                            %err,
1295                            "error processing post-seed bootstrap output"
1296                        );
1297                    }
1298                }
1299                let _ = respond.send(());
1300            },
1301            ClientRequest::SubmitTimeoutVote { vote, respond } => {
1302                self.timeout_collector.accumulate_vote(vote.clone()).await;
1303                self.timeout_one_honest_collector
1304                    .accumulate_vote(vote.clone())
1305                    .await;
1306                // Rebroadcast so peer coordinators can aggregate too.
1307                let message = Message {
1308                    sender: self.public_key.clone(),
1309                    message_type: MessageType::Consensus(ConsensusMessage::TimeoutVote(
1310                        message::TimeoutVoteMessage { vote, lock: None },
1311                    )),
1312                };
1313                if let Err(err) = self.network.broadcast(message.view_number(), &message) {
1314                    tracing::warn!(%err, "failed to rebroadcast bridged timeout vote");
1315                }
1316                let _ = respond.send(());
1317            },
1318            ClientRequest::BumpNetworkEpoch { epoch, respond } => {
1319                if let Err(err) = self
1320                    .network
1321                    .apply_epoch(epoch, &self.membership_coordinator)
1322                {
1323                    tracing::warn!(%epoch, %err, "network on_epoch_change failed");
1324                }
1325                let _ = respond.send(());
1326            },
1327        }
1328
1329        Ok(())
1330    }
1331
1332    /// Kick the leader after the seed lands when a forwarded TC2 had
1333    /// already advanced `current_view`. No-op unless leader and all
1334    /// prerequisites are present.
1335    fn resume_after_cutover_tc(&mut self) {
1336        let cur_view = self.consensus.current_view();
1337        if self.consensus.timeout_cert_at(cur_view).is_none() {
1338            return;
1339        }
1340        let epoch = self
1341            .consensus
1342            .current_epoch()
1343            .unwrap_or(EpochNumber::genesis());
1344        let Some(leader) = self.leader(cur_view, epoch) else {
1345            return;
1346        };
1347        if leader != self.public_key {
1348            return;
1349        }
1350        let Some(locked_view) = self.consensus.locked_view() else {
1351            return;
1352        };
1353        let Some(parent_proposal) = self.consensus.proposal_at(locked_view).cloned() else {
1354            return;
1355        };
1356        self.outbox
1357            .push_back(ConsensusOutput::RequestBlockAndHeader(
1358                BlockAndHeaderRequest {
1359                    view: cur_view,
1360                    epoch,
1361                    parent_proposal,
1362                },
1363            ));
1364    }
1365
1366    fn gc(&mut self, view: ViewNumber, epoch: EpochNumber) {
1367        info!(node = %self.node_id, %view, "garbage collecting");
1368        self.consensus.gc(view, epoch);
1369        self.checkpoint_collector.gc(view, epoch);
1370        let _ = self.network.gc(view); // TODO
1371        self.state_manager.gc(view);
1372        self.vid_disperser.gc(view);
1373        self.vid_reconstructor.gc(view);
1374        self.vote1_collector.gc(view, epoch);
1375        self.vote2_collector.gc(view, epoch);
1376        self.timeout_collector.gc(view, epoch);
1377        self.timeout_one_honest_collector.gc(view, epoch);
1378        self.epoch_root_collector.gc(view, epoch);
1379        self.epoch_manager.gc(epoch);
1380        self.block_builder.gc(view);
1381        self.pending_proposal_fetches.gc(view);
1382        self.storage.gc(view);
1383        self.cached_validated_proposals = self.cached_validated_proposals.split_off(&view);
1384        self.cached_vid_shares = self.cached_vid_shares.split_off(&view);
1385    }
1386}

1388fn check_payload_commitment<T: NodeType>(
1389    proposal: &SignedProposal<T, Proposal<T>>,
1390    vid_share: &VidDisperseShare2<T>,
1391) -> bool {
1392    let VidCommitment::V2(commit) = proposal.data.block_header.payload_commitment() else {
1393        warn!(
1394            "unexpected payload commitment type in view {}, proposal discarded",
1395            proposal.data.view_number
1396        );
1397        return false;
1398    };
1399    if commit != vid_share.payload_commitment {
1400        warn!(
1401            "payload commitment mismatch in view {}, discard the proposal",
1402            proposal.data.view_number
1403        );
1404        return false;
1405    }
1406    true
1407}
1408
1409type ProposalFetchResponseSender<T> =
1410    oneshot::Sender<Result<SignedProposal<T, Proposal<T>>, QueryError>>;
1411
1412#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
1413struct ProposalFetchKey<T: NodeType> {
1414    view: ViewNumber,
1415    leaf_commitment: Commitment<Leaf2<T>>,
1416}
1417
1418impl<T: NodeType> ProposalFetchKey<T> {
1419    fn new(view: ViewNumber, leaf_commitment: Commitment<Leaf2<T>>) -> Self {
1420        Self {
1421            view,
1422            leaf_commitment,
1423        }
1424    }
1425}
1426
1427#[derive(Default)]
1428struct PendingProposalFetches<T: NodeType> {
1429    pending: HashMap<ProposalFetchKey<T>, Vec<ProposalFetchResponseSender<T>>>,
1430}
1431
1432impl<T: NodeType> PendingProposalFetches<T> {
1433    fn prune_closed(&mut self) {
1434        self.pending.retain(|_, responders| {
1435            responders.retain(|respond| !respond.is_closed());
1436            !responders.is_empty()
1437        });
1438    }
1439
1440    fn contains_request(
1441        &mut self,
1442        view: ViewNumber,
1443        leaf_commitment: Commitment<Leaf2<T>>,
1444    ) -> bool {
1445        self.prune_closed();
1446        self.pending
1447            .contains_key(&ProposalFetchKey::new(view, leaf_commitment))
1448    }
1449
1450    fn push(
1451        &mut self,
1452        view: ViewNumber,
1453        leaf_commitment: Commitment<Leaf2<T>>,
1454        respond: ProposalFetchResponseSender<T>,
1455    ) {
1456        self.pending
1457            .entry(ProposalFetchKey::new(view, leaf_commitment))
1458            .or_default()
1459            .push(respond);
1460    }
1461
1462    #[allow(dead_code)]
1463    fn gc(&mut self, view: ViewNumber) {
1464        self.pending.retain(|key, responders| {
1465            responders.retain(|respond| !respond.is_closed());
1466            key.view >= view && !responders.is_empty()
1467        });
1468    }
1469
1470    fn resolve(&mut self, proposal: &SignedProposal<T, Proposal<T>>) {
1471        self.prune_closed();
1472        let view = proposal.data.view_number;
1473        let leaf_commitment = proposal_commitment(&proposal.data);
1474        let key = ProposalFetchKey::new(view, leaf_commitment);
1475
1476        if let Some(responders) = self.pending.remove(&key) {
1477            for respond in responders {
1478                let _ = respond.send(Ok(proposal.clone()));
1479            }
1480        }
1481    }
1482}
