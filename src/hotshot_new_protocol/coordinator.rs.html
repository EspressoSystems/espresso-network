110    pub fn maker(
111        membership_coordinator: EpochMembershipCoordinator<T>,
112        network: N,
113        initializer: &HotShotInitializer<T>,
114        upgrade_lock: UpgradeLock<T>,
115        public_key: T::SignatureKey,
116        private_key: <T::SignatureKey as SignatureKey>::PrivateKey,
117        state_private_key: <T::StateSignatureKey as StateSignatureKey>::StatePrivateKey,
118        stake_table_capacity: usize,
119        timeout_duration: Duration,
120        storage: S,
121    ) -> Self {
122        let mut consensus = Consensus::new(
123            membership_coordinator.clone(),
124            public_key.clone(),
125            private_key.clone(),
126            state_private_key,
127            stake_table_capacity,
128            upgrade_lock.clone(),
129            initializer.anchor_leaf.clone(),
130            initializer.epoch_height,
131        );
132
133        let genesis_cert1 = initializer.high_qc.clone();
134        let genesis_proposal = message::Proposal {
135            block_header: initializer.anchor_leaf.block_header().clone(),
136            view_number: ViewNumber::genesis(),
137            epoch: EpochNumber::genesis(),
138            justify_qc: genesis_cert1.clone(),
139            next_epoch_justify_qc: None,
140            upgrade_certificate: None,
141            view_change_evidence: None,
142            next_drb_result: None,
143            state_cert: None,
144        };
145        consensus.seed_genesis(genesis_cert1, genesis_proposal);
146
147        let mut state_manager = StateManager::new(
148            Arc::new(initializer.instance_state.clone()),
149            upgrade_lock.clone(),
150        );
151        state_manager.seed_state(
152            initializer.anchor_leaf.view_number(),
153            initializer.anchor_state.clone(),
154            initializer.anchor_leaf.clone(),
155        );
156
157        let lock = upgrade_lock.clone();
158        Self::builder()
159            .consensus(consensus)
160            .network(network)
161            .state_manager(state_manager)
162            .vid_disperser(VidDisperser::new(membership_coordinator.clone()))
163            .vid_reconstructor(VidReconstructor::new())
164            .vote1_collector(VoteCollector::new(
165                membership_coordinator.clone(),
166                lock.clone(),
167            ))
168            .vote2_collector(VoteCollector::new(
169                membership_coordinator.clone(),
170                lock.clone(),
171            ))
172            .timeout_collector(VoteCollector::new(
173                membership_coordinator.clone(),
174                lock.clone(),
175            ))
176            .timeout_one_honest_collector(VoteCollector::new(
177                membership_coordinator.clone(),
178                lock.clone(),
179            ))
180            .checkpoint_collector(VoteCollector::new(
181                membership_coordinator.clone(),
182                lock.clone(),
183            ))
184            .epoch_root_collector(EpochRootVoteCollector::new(
185                membership_coordinator.clone(),
186                lock,
187            ))
188            .epoch_manager(EpochManager::new(
189                initializer.epoch_height,
190                membership_coordinator.clone(),
191            ))
192            .block_builder(BlockBuilder::new(
193                Arc::new(initializer.instance_state.clone()),
194                membership_coordinator.clone(),
195                BlockBuilderConfig::default(),
196                upgrade_lock.clone(),
197            ))
198            .proposal_validator(ProposalValidator::new(
199                membership_coordinator.clone(),
200                initializer.epoch_height,
201                upgrade_lock.clone(),
202            ))
203            .share_validator(VidShareValidator::new(
204                membership_coordinator.clone(),
205                initializer.epoch_height,
206                upgrade_lock,
207            ))
208            .storage(Storage::new(storage, private_key))
209            .membership_coordinator(membership_coordinator)
210            .timer(Timer::new(
211                timeout_duration,
212                ViewNumber::genesis(),
213                EpochNumber::genesis(),
214            ))
215            .public_key(public_key)
216            .build()
217    }

219    /// Bootstrap the coordinator so the view-1 leader can propose.
220    pub fn start(&mut self) {
221        let view = ViewNumber::new(1);
222        let epoch = EpochNumber::genesis();
223
224        if self.consensus.last_decided_leaf().view_number() == ViewNumber::genesis() {
225            // Append the genesis DA proposal to storage.
226            //
227            // The genesis payload is always empty, but it never flows through the
228            // regular block-builder/VID path that would otherwise persist a DA
229            // proposal for view 0. Storage consumers downstream still expect one,
230            // so we synthesize and append it here.
231            let genesis_leaf = self.consensus.last_decided_leaf().clone();
232            let (payload, metadata) = T::BlockPayload::empty();
233            self.storage.append_da(
234                ViewNumber::genesis(),
235                EpochNumber::genesis(),
236                payload,
237                metadata,
238                genesis_leaf.payload_commitment(),
239            );
240
241            // Genesis is never decided through the normal consensus path, so
242            // downstream consumers (persistence, query service) would never see
243            // the genesis header. We emit a `LeafDecided` for it here so that
244            // application layer sees this event
245            self.outbox.push_back(ConsensusOutput::LeafDecided {
246                leaves: vec![genesis_leaf],
247                cert1: self
248                    .consensus
249                    .cert1_at(ViewNumber::genesis())
250                    .cloned()
251                    .expect("genesis cert1 must be seeded"),
252                cert2: None,
253                vid_shares: vec![None],
254            });
255        }
256
257        self.outbox
258            .push_back(ConsensusOutput::ViewChanged(view, epoch));
259
260        if let Some(leader) = self.leader(view, epoch)
261            && leader == self.public_key
262        {
263            let genesis_proposal = self
264                .consensus
265                .proposal_at(ViewNumber::genesis())
266                .expect("genesis proposal must be seeded before start()")
267                .clone();
268            self.outbox
269                .push_back(ConsensusOutput::RequestBlockAndHeader(
270                    BlockAndHeaderRequest {
271                        view,
272                        epoch,
273                        parent_proposal: genesis_proposal,
274                    },
275                ));
276        }
277    }

283    pub async fn next_consensus_input(&mut self) -> Result<ConsensusInput<T>, CoordinatorError> {
284        loop {
285            select! {
286                message = self.network.receive() => match message {
287                    Ok(m) => {
288                        if let Some(input) = self.on_network_message(m).await {
289                            return Ok(input)
290                        }
291                    }
292                    Err(e) => {
293                        return Err(CoordinatorError::from(e).context("network receive"))
294                    }
295                },
296                () = &mut self.timer => {
297                    let input = ConsensusInput::Timeout(self.timer.view(), self.timer.epoch());
298                    // Timer is only reset so we can resend the timeout vote
299                    // This isn't strictly necessary for the protocol, but it's a good idea to
300                    // resend the timeout vote to avoid a situation where the network is stuck
301                    // view because we fail to form a timeout certificate.
302                    self.timer.reset();
303                    return Ok(input)
304                }
305                Some(output) = self.state_manager.next() => {
306                    if let Some(input) = self.on_state_manager_output(output) {
307                        return Ok(input)
308                    }
309                }
310                Some(request) = self.client.next_request() => {
311                    if let Err(err) = self.on_client_request(request) {
312                        error!(%err, "error while handling client request");
313                    }
314                }
315                Some(tcert) = self.timeout_collector.next() => {
316                    return Ok(ConsensusInput::TimeoutCertificate(tcert))
317                }
318                Some(out) = self.timeout_one_honest_collector.next() => {
319                    let Some(epoch) = out.data.epoch else {
320                        let msg = format!("missing epoch in view {}", out.view_number());
321                        return Err(CoordinatorError::regular(msg).context("gc timeout one honest"))
322                    };
323                    return Ok(ConsensusInput::TimeoutOneHonest(out.view_number(), epoch))
324                }
325                Some(cert1) = self.vote1_collector.next() => {
326                    return Ok(ConsensusInput::Certificate1(cert1))
327                }
328                Some(cert2) = self.vote2_collector.next() => {
329                    return Ok(ConsensusInput::Certificate2(cert2))
330                }
331                Some((cert1, state_cert)) = self.epoch_root_collector.next() => {
332                    self.storage.append_state_cert(
333                        ViewNumber::new(state_cert.light_client_state.view_number),
334                        state_cert.clone(),
335                    );
336                    return Ok(ConsensusInput::EpochRootCertificates { cert1, state_cert })
337                }
338                Some(item) = self.share_validator.next() => match item {
339                    Ok(vid_share) => {
340                        let view = vid_share.view_number();
341                        let Some(validated) = self.cached_validated_proposals.remove(&view) else {
342                            // Wait for the proposal
343                            self.cached_vid_shares.insert(view, vid_share);
344                            continue;
345                        };
346                        if !check_payload_commitment(&validated.message.proposal, &vid_share) {
347                            continue;
348                        }
349                        return self.on_proposal_and_vid_share(validated, vid_share)
350                    },
351                    Err(e) => {
352                        return Err(CoordinatorError::regular(e).context("vid share validation"))
353                    }
354                },
355                Some(item) = self.proposal_validator.next() => match item {
356                    Ok(validated) => {
357                        // Refresh the network's peer set when a proposal is validated.
358                        let epoch = validated.message.proposal.data.epoch;
359                        if let Err(err) = self
360                            .network
361                            .apply_epoch(epoch, &self.membership_coordinator)
362                            .await
363                        {
364                            error!(%epoch, %err, "network apply_epoch failed");
365                        }
366
367                        let view = validated.message.proposal.data.view_number();
368                        let Some(vid_share) = self.cached_vid_shares.remove(&view) else {
369                            // Wait for the vid share
370                            self.cached_validated_proposals.insert(view, validated);
371                            continue;
372                        };
373                        // Check for commitment correspondence
374                        if !check_payload_commitment(&validated.message.proposal, &vid_share) {
375                            continue;
376                        }
377                        return self.on_proposal_and_vid_share(validated, vid_share)
378                    }
379                    Err(e) => {
380                        return Err(CoordinatorError::regular(e).context("proposal validation"))
381                    }
382                },
383                Some(cert) = self.checkpoint_collector.next() => {
384                    let Some(epoch) = cert.epoch() else {
385                        let msg = format!("missing epoch in view {}", cert.view_number());
386                        return Err(CoordinatorError::critical(msg).context("gc certificate"))
387                    };
388                    self.gc(cert.view_number(), epoch);
389                }
390                Some(item) = self.block_builder.next() => match item {
391                    Ok(block) => {
392                        self.state_manager.request_header(HeaderRequest::from(&block));
393                        let next_view = block.view + 1;
394                        let epoch = block.epoch;
395                        let manifest = block.manifest.clone();
396                        self.storage.append_da(
397                            block.view,
398                            block.epoch,
399                            block.payload.payload.clone(),
400                            block.payload.metadata.clone(),
401                            block.payload_commitment,
402                        );
403                        self.unicast_to_leader(
404                            next_view,
405                            epoch,
406                            BlockMessage::DedupManifest(manifest),
407                        )?;
408                        return Ok(block.into())
409                    }
410                    Err(err) => {
411                        return Err(CoordinatorError::regular(err).context("block building"))
412                    }
413                },
414                Some(item) = self.vid_disperser.next() => match item {
415                    Ok(out) => {
416                        return Ok(ConsensusInput::VidDisperseCreated(out.view, out.disperse))
417                    }
418                    Err(()) => {
419                        return Err(CoordinatorError::unspecified().context("vid disperse"))
420                    }
421                },
422                Some(item) = self.vid_reconstructor.next() => match item {
423                    Ok(out) => {
424                        self.block_builder.on_block_reconstructed(out.tx_commitments);
425                        self.storage.append_da(
426                            out.view,
427                            out.epoch,
428                            out.payload,
429                            out.metadata,
430                            VidCommitment::V2(out.payload_commitment),
431                        );
432                        return Ok(ConsensusInput::BlockReconstructed(out.view, out.payload_commitment))
433                    }
434                    Err(()) => {
435                        return Err(CoordinatorError::unspecified().context("vid reconstruction"))
436                    }
437                },
438                Some(result) = self.epoch_manager.next() => match result {
439                    Ok(EpochRootResult::DrbResult(epoch, drb_result)) => {
440                        // New epoch data available — retry votes that were
441                        // buffered because their membership wasn't ready.
442                        self.vote1_collector.retry_pending_votes().await;
443                        self.vote2_collector.retry_pending_votes().await;
444                        self.timeout_collector.retry_pending_votes().await;
445                        self.timeout_one_honest_collector.retry_pending_votes().await;
446                        return Ok(ConsensusInput::DrbResult(epoch, drb_result))
447                    }
448                    Err(failure) => {
449                        // Catchup/compute failed. The epoch manager clears
450                        // the pending guard; consensus's `maybe_propose`
451                        // will re-request the DRB when it next tries to
452                        // build a transition proposal and finds it missing.
453                        warn!(%failure.error, epoch = %failure.epoch, "DRB request failed");
454                        continue;
455                    }
456                },
457                else => {
458                    return Err(CoordinatorError::critical(ErrorSource::NoInput))
459                }
460            }
461        }
462    }

468    pub fn process_consensus_output(
469        &mut self,
470        output: ConsensusOutput<T>,
471    ) -> Result<(), CoordinatorError> {
472        match output {
473            ConsensusOutput::RequestState(state_request) => {
474                self.state_manager.request_state(state_request);
475            },
476            ConsensusOutput::RequestVidDisperse {
477                view,
478                epoch,
479                payload,
480                metadata,
481            } => {
482                self.vid_disperser.request_vid_disperse(VidDisperseRequest {
483                    view,
484                    epoch,
485                    block: payload,
486                    metadata,
487                });
488            },
489            ConsensusOutput::RequestDrbResult(epoch) => {
490                self.epoch_manager.request_drb_result(epoch);
491            },
492            ConsensusOutput::SendCheckpointVote(checkpoint_vote) => {
493                let message = Message {
494                    sender: self.public_key.clone(),
495                    message_type: MessageType::Consensus(ConsensusMessage::Checkpoint(
496                        checkpoint_vote,
497                    )),
498                };
499                self.network
500                    .broadcast(message.view_number(), &message)
501                    .map_err(|e| CoordinatorError::from(e).context("broadcast checkpoint vote"))?
502            },
503            ConsensusOutput::LeafDecided { leaves, cert2, .. } => {
504                if let Some(cert2) = cert2 {
505                    self.storage.append_cert2(cert2.view_number, cert2.clone());
506                }
507                for leaf in leaves {
508                    self.epoch_manager.handle_leaf_decided(leaf);
509                }
510            },
511            ConsensusOutput::LockUpdated(_) => {}, // TODO
512            ConsensusOutput::RequestBlockAndHeader(request) => {
513                self.block_builder.request_block(request);
514            },
515            ConsensusOutput::SendProposal(proposal) => {
516                self.storage.append_proposal(proposal.data.clone());
517                // TODO: This may be done async in network so we do not spend
518                // too much time here in this loop.
519
520                let message = Message {
521                    sender: self.public_key.clone(),
522                    message_type: MessageType::Consensus(ConsensusMessage::Proposal(
523                        ProposalMessage::validated(proposal.clone()),
524                    )),
525                };
526                if let Err(err) = self.network.broadcast(message.view_number(), &message) {
527                    let err = CoordinatorError::from(err).context("proposal broadcast");
528                    if err.severity == Severity::Critical {
529                        return Err(err);
530                    } else {
531                        warn!(%err, "network error while broadcasting proposal")
532                    }
533                }
534            },
535            ConsensusOutput::SendVidShares(vid_shares) => {
536                for share in vid_shares {
537                    let recipient = share.data.recipient_key.clone();
538                    let message = Message {
539                        sender: self.public_key.clone(),
540                        message_type: MessageType::Consensus(ConsensusMessage::VidShare(share)),
541                    };
542                    if let Err(err) =
543                        self.network
544                            .unicast(message.view_number(), &recipient, &message)
545                    {
546                        let err = CoordinatorError::from(err).context("vid share unicast");
547                        if err.severity == Severity::Critical {
548                            return Err(err);
549                        } else {
550                            warn!(%err, "network error while sending vid share")
551                        }
552                    }
553                }
554            },
555            ConsensusOutput::SendTimeoutVote(vote, lock) => {
556                let message = Message {
557                    sender: self.public_key.clone(),
558                    message_type: MessageType::Consensus(ConsensusMessage::TimeoutVote(
559                        message::TimeoutVoteMessage { vote, lock },
560                    )),
561                };
562                self.network
563                    .broadcast(message.view_number(), &message)
564                    .map_err(|e| CoordinatorError::from(e).context("broadcast timeout vote"))?
565            },
566            ConsensusOutput::SendTimeoutCertificate(tc, view, epoch) => {
567                if let Some(leader) = self.leader(view, epoch) {
568                    let message = Message {
569                        sender: self.public_key.clone(),
570                        message_type: MessageType::Consensus(ConsensusMessage::TimeoutCertificate(
571                            tc,
572                        )),
573                    };
574                    self.network
575                        .unicast(message.view_number(), &leader, &message)
576                        .map_err(|e| CoordinatorError::from(e).context("timeout certificate"))?;
577                }
578            },
579            ConsensusOutput::SendVote1(vote1) => {
580                let message = Message {
581                    sender: self.public_key.clone(),
582                    message_type: MessageType::Consensus(ConsensusMessage::Vote1(vote1)),
583                };
584                self.network
585                    .broadcast(message.view_number(), &message)
586                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote1"))?
587            },
588            ConsensusOutput::SendVote2(vote2) => {
589                let message = Message {
590                    sender: self.public_key.clone(),
591                    message_type: MessageType::Consensus(ConsensusMessage::Vote2(vote2)),
592                };
593                self.network
594                    .broadcast(message.view_number(), &message)
595                    .map_err(|e| CoordinatorError::from(e).context("broadcast vote2"))?
596            },
597            ConsensusOutput::SendEpochChange(epoch_change) => {
598                let message = Message {
599                    sender: self.public_key.clone(),
600                    message_type: MessageType::Consensus(ConsensusMessage::EpochChange(
601                        epoch_change,
602                    )),
603                };
604                self.network
605                    .broadcast(message.view_number(), &message)
606                    .map_err(|e| CoordinatorError::from(e).context("broadcast epoch change"))?
607            },
608            ConsensusOutput::SendCertificate1(cert1) => {
609                let message = Message {
610                    sender: self.public_key.clone(),
611                    message_type: MessageType::Consensus(ConsensusMessage::Certificate1(
612                        cert1,
613                        self.public_key.clone(),
614                    )),
615                };
616                self.network
617                    .broadcast(message.view_number(), &message)
618                    .map_err(|e| CoordinatorError::from(e).context("broadcast certificate1"))?
619            },
620            ConsensusOutput::ProposalValidated { .. } => {},
621            ConsensusOutput::ViewChanged(view, epoch) => {
622                self.consensus.set_view(view, epoch);
623                self.timer.reset_with_epoch(view, epoch);
624                let txns = self.block_builder.on_view_changed(view, epoch);
625                if !txns.is_empty() {
626                    let next_view = view + 1;
627                    self.unicast_to_leader(
628                        next_view,
629                        epoch,
630                        BlockMessage::Transactions(TransactionMessage {
631                            view: next_view,
632                            transactions: txns,
633                        }),
634                    )
635                    .map_err(|e| e.context("unicast transactions"))?;
636                }
637
638                // Proactively fetch DRBs for the next epoch so
639                // late-starting nodes have them before they need to
640                // propose or verify certs in a new epoch. The dedup
641                // in request_drb_result makes repeated calls free.
642                let next_epoch = epoch + 1;
643                if next_epoch > EpochNumber::genesis() + 1 {
644                    self.epoch_manager.request_drb_result(next_epoch);
645                    self.epoch_manager.request_drb_result(next_epoch + 1);
646                }
647            },
648        }
649        Ok(())
650    }

684    pub(crate) async fn on_network_message(
685        &mut self,
686        message: Message<T, Unchecked>,
687    ) -> Option<ConsensusInput<T>> {
688        match message.message_type {
689            MessageType::Consensus(msg) => match msg {
690                ConsensusMessage::Proposal(p) => {
691                    if self.consensus.wants_proposal_for_view(&p.view_number()) {
692                        self.proposal_validator.validate(p);
693                    }
694                    None
695                },
696                ConsensusMessage::VidShare(share) => {
697                    if self
698                        .consensus
699                        .wants_proposal_for_view(&share.data.view_number())
700                    {
701                        self.share_validator.validate(share);
702                    }
703                    None
704                },
705                ConsensusMessage::Vote1(vote1) => {
706                    let bn = vote1.vote.data.block_number.unwrap_or(0);
707                    let epoch_height = *self.consensus.epoch_height;
708                    if is_epoch_root(bn, epoch_height) {
709                        // An epoch-root Vote1 MUST carry a state_vote.
710                        // Reject otherwise.
711                        vote1.state_vote.as_ref()?;
712                        self.epoch_root_collector.accumulate(vote1.clone()).await;
713                    } else {
714                        self.vote1_collector
715                            .accumulate_vote(vote1.vote.clone())
716                            .await;
717                    }
718                    self.vid_reconstructor
719                        .handle_vid_share(vote1.vid_share, None);
720                    None
721                },
722                ConsensusMessage::Vote2(vote2) => {
723                    self.vote2_collector.accumulate_vote(vote2).await;
724                    None
725                },
726                ConsensusMessage::Certificate1(certificate1, _key) => {
727                    Some(ConsensusInput::Certificate1(certificate1))
728                },
729                ConsensusMessage::Certificate2(certificate2, _key) => {
730                    Some(ConsensusInput::Certificate2(certificate2))
731                },
732                ConsensusMessage::TimeoutVote(timeout_msg) => {
733                    self.timeout_collector
734                        .accumulate_vote(timeout_msg.vote.clone())
735                        .await;
736                    self.timeout_one_honest_collector
737                        .accumulate_vote(timeout_msg.vote)
738                        .await;
739                    None
740                },
741                ConsensusMessage::TimeoutCertificate(tc) => {
742                    Some(ConsensusInput::TimeoutCertificate(tc))
743                },
744                ConsensusMessage::EpochChange(epoch_change) => {
745                    Some(ConsensusInput::EpochChange(epoch_change))
746                },
747                ConsensusMessage::Checkpoint(checkpoint) => {
748                    self.checkpoint_collector.accumulate_vote(checkpoint).await;
749                    None
750                },
751            },
752            MessageType::Block(msg) => {
753                match msg {
754                    BlockMessage::Transactions(msg) => self.block_builder.on_transactions(msg),
755                    BlockMessage::DedupManifest(manifest) => {
756                        if let Some(view_leader) = self.leader(manifest.view, manifest.epoch)
757                            && view_leader == message.sender
758                        {
759                            self.block_builder.on_dedup_manifest(manifest)
760                        }
761                    },
762                }
763                None
764            },
765            MessageType::ProposalFetch(ProposalFetchMessage::Request(request)) => {
766                if !request.validate_sender(&message.sender) {
767                    warn!(
768                        sender = %message.sender,
769                        view = %request.view_number(),
770                        "ignoring invalid proposal fetch request signature"
771                    );
772                    return None;
773                }
774                if let Some(proposal) = self
775                    .consensus
776                    .signed_proposal(&request.view_number())
777                    .cloned()
778                {
779                    let response = Message {
780                        sender: self.public_key.clone(),
781                        message_type: MessageType::ProposalFetch(ProposalFetchMessage::Response(
782                            Box::new(proposal),
783                        )),
784                    };
785
786                    if let Err(err) =
787                        self.network
788                            .unicast(request.view_number(), &message.sender, &response)
789                    {
790                        let err = CoordinatorError::from(err).context("proposal response");
791                        warn!(%err, "network error while sending proposal response");
792                    }
793                }
794                None
795            },
796            MessageType::ProposalFetch(ProposalFetchMessage::Response(proposal)) => {
797                self.pending_proposal_fetches.resolve(&proposal);
798                None
799            },
800            MessageType::External(data) => {
801                self.coordinator_outbox
802                    .push_back(CoordinatorOutput::ExternalMessageReceived {
803                        sender: message.sender,
804                        data,
805                    });
806                None
807            },
808        }
809    }

838    fn on_proposal_and_vid_share(
839        &mut self,
840        validated: ValidatedProposal<T>,
841        vid_share: VidDisperseShare2<T>,
842    ) -> Result<ConsensusInput<T>, CoordinatorError> {
843        self.storage.append_vid(vid_share.clone());
844        self.storage
845            .append_proposal(validated.message.proposal.data.clone());
846
847        let m = validated
848            .message
849            .proposal
850            .data
851            .block_header
852            .metadata()
853            .clone();
854        self.vid_reconstructor
855            .handle_vid_share(vid_share.clone(), m);
856
857        // GC for the cache
858        let view = validated.message.proposal.data.view_number();
859        self.cached_vid_shares = self.cached_vid_shares.split_off(&(view + 1));
860        self.cached_validated_proposals = self.cached_validated_proposals.split_off(&(view + 1));
861
862        Ok(ConsensusInput::ProposalWithVidShare(
863            validated.sender,
864            validated.message,
865            vid_share,
866        ))
867    }

869    fn unicast_to_leader(
870        &mut self,
871        view: ViewNumber,
872        epoch: EpochNumber,
873        msg: BlockMessage<T>,
874    ) -> Result<(), CoordinatorError> {
875        let Some(leader) = self.leader(view, epoch) else {
876            warn!(%view, %epoch, "failed to resolve leader for unicast");
877            return Ok(());
878        };
879        let message = Message {
880            sender: self.public_key.clone(),
881            message_type: MessageType::Block(msg),
882        };
883        self.network
884            .unicast(message.view_number(), &leader, &message)
885            .map_err(|e| CoordinatorError::from(e).context("leader unicast"))
886    }

896    fn on_client_request(&mut self, request: ClientRequest<T>) -> Result<(), CoordinatorError> {
897        match request {
898            ClientRequest::CurrentView(tx) => {
899                let _ = tx.send(self.consensus.current_view());
900            },
901            ClientRequest::CurrentEpoch(tx) => {
902                let _ = tx.send(self.consensus.current_epoch());
903            },
904            ClientRequest::DecidedLeaf(tx) => {
905                let _ = tx.send(self.consensus.last_decided_leaf().clone());
906            },
907            ClientRequest::DecidedState(tx) => {
908                let view = self.consensus.last_decided_leaf().view_number();
909                let _ = tx.send(self.state(view).map(|s| s.state.clone()));
910            },
911            ClientRequest::UndecidedLeaves(tx) => {
912                let _ = tx.send(self.consensus.undecided_leaves().cloned().collect());
913            },
914            ClientRequest::GetState { view, respond } => {
915                let _ = respond.send(self.state(view).map(|s| s.state.clone()));
916            },
917            ClientRequest::GetStateAndDelta { view, respond } => {
918                let _ = respond.send(match self.state(view) {
919                    Some(s) => (Some(s.state.clone()), s.delta.clone()),
920                    None => (None, None),
921                });
922            },
923            ClientRequest::SubmitTransaction { tx, respond } => {
924                self.block_builder.on_submit_transaction(tx);
925                let _ = respond.send(());
926            },
927            ClientRequest::UpdateLeaf { update, respond } => {
928                self.state_manager.update_state(update);
929                let _ = respond.send(());
930            },
931            ClientRequest::RequestProposal {
932                view,
933                leaf_commitment,
934                respond,
935            } => {
936                if let Some(proposal) = self.consensus.signed_proposal(&view)
937                    && proposal_commitment(&proposal.data) == leaf_commitment
938                {
939                    let _ = respond.send(Ok(proposal.clone()));
940                    return Ok(());
941                }
942                if !self
943                    .pending_proposal_fetches
944                    .contains_request(view, leaf_commitment)
945                {
946                    let request =
947                        self.consensus
948                            .signed_proposal_fetch_request(view)
949                            .map_err(|err| {
950                                let err = format!("failed to sign proposal request: {err}");
951                                CoordinatorError::regular(err).context("sign proposal request")
952                            })?;
953
954                    let message = Message {
955                        sender: self.public_key.clone(),
956                        message_type: MessageType::ProposalFetch(ProposalFetchMessage::Request(
957                            request,
958                        )),
959                    };
960
961                    self.network
962                        .broadcast(message.view_number(), &message)
963                        .map_err(|err| {
964                            CoordinatorError::from(err).context("broadcast proposal request")
965                        })?;
966                }
967                self.pending_proposal_fetches
968                    .push(view, leaf_commitment, respond);
969            },
970            ClientRequest::SendExternalMessage {
971                view,
972                payload,
973                recipient,
974                respond,
975            } => {
976                let message = Message {
977                    sender: self.public_key.clone(),
978                    message_type: MessageType::External(payload),
979                };
980                let result = self
981                    .network
982                    .unicast(view, &recipient, &message)
983                    .map_err(|err| {
984                        CoordinatorError::from(err)
985                            .context("send external message")
986                            .into()
987                    });
988                let _ = respond.send(result);
989            },
990        }
991
992        Ok(())
993    }

1016fn check_payload_commitment<T: NodeType>(
1017    proposal: &SignedProposal<T, Proposal<T>>,
1018    vid_share: &VidDisperseShare2<T>,
1019) -> bool {
1020    let VidCommitment::V2(commit) = proposal.data.block_header.payload_commitment() else {
1021        warn!(
1022            "unexpected payload commitment type in view {}, proposal discarded",
1023            proposal.data.view_number
1024        );
1025        return false;
1026    };
1027    if commit != vid_share.payload_commitment {
1028        warn!(
1029            "payload commitment mismatch in view {}, discard the proposal",
1030            proposal.data.view_number
1031        );
1032        return false;
1033    }
1034    true
1035}
1036
1037type ProposalFetchResponseSender<T> =
1038    oneshot::Sender<Result<SignedProposal<T, Proposal<T>>, QueryError>>;
1039
1040#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
1041struct ProposalFetchKey<T: NodeType> {
1042    view: ViewNumber,
1043    leaf_commitment: Commitment<Leaf2<T>>,
1044}
1045
1046impl<T: NodeType> ProposalFetchKey<T> {
1047    fn new(view: ViewNumber, leaf_commitment: Commitment<Leaf2<T>>) -> Self {
1048        Self {
1049            view,
1050            leaf_commitment,
1051        }
1052    }
1053}
1054
1055#[derive(Default)]
1056struct PendingProposalFetches<T: NodeType> {
1057    pending: HashMap<ProposalFetchKey<T>, Vec<ProposalFetchResponseSender<T>>>,
1058}
1059
1060impl<T: NodeType> PendingProposalFetches<T> {
1061    fn prune_closed(&mut self) {
1062        self.pending.retain(|_, responders| {
1063            responders.retain(|respond| !respond.is_closed());
1064            !responders.is_empty()
1065        });
1066    }
1067
1068    fn contains_request(
1069        &mut self,
1070        view: ViewNumber,
1071        leaf_commitment: Commitment<Leaf2<T>>,
1072    ) -> bool {
1073        self.prune_closed();
1074        self.pending
1075            .contains_key(&ProposalFetchKey::new(view, leaf_commitment))
1076    }
1077
1078    fn push(
1079        &mut self,
1080        view: ViewNumber,
1081        leaf_commitment: Commitment<Leaf2<T>>,
1082        respond: ProposalFetchResponseSender<T>,
1083    ) {
1084        self.pending
1085            .entry(ProposalFetchKey::new(view, leaf_commitment))
1086            .or_default()
1087            .push(respond);
1088    }
1089
1090    #[allow(dead_code)]
1091    fn gc(&mut self, view: ViewNumber) {
1092        self.pending.retain(|key, responders| {
1093            responders.retain(|respond| !respond.is_closed());
1094            key.view >= view && !responders.is_empty()
1095        });
1096    }
1097
1098    fn resolve(&mut self, proposal: &SignedProposal<T, Proposal<T>>) {
1099        self.prune_closed();
1100        let view = proposal.data.view_number;
1101        let leaf_commitment = proposal_commitment(&proposal.data);
1102        let key = ProposalFetchKey::new(view, leaf_commitment);
1103
1104        if let Some(responders) = self.pending.remove(&key) {
1105            for respond in responders {
1106                let _ = respond.send(Ok(proposal.clone()));
1107            }
1108        }
1109    }
1110}
