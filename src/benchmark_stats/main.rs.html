43fn main() -> Result<(), Box<dyn std::error::Error>> {
44    let command = Command::parse();
45
46    match command.subcommand {
47        SubCommands::Replica { path, output } => {
48            let replica_view_stats = read_replica_view_stats(&path)?;
49            plot_replica_stats(
50                &replica_view_stats,
51                output
52                    .as_deref()
53                    .unwrap_or_else(|| Path::new("replica_stats.html")),
54            )?;
55            let stats = generate_replica_stats(&replica_view_stats);
56            print_replica_stats(&stats);
57        },
58        SubCommands::Leader { path, output } => {
59            let leader_view_stats = read_leader_view_stats(&path)?;
60            plot_and_print_leader_stats(
61                &leader_view_stats,
62                output
63                    .as_deref()
64                    .unwrap_or_else(|| Path::new("leader_stats.html")),
65            )?;
66        },
67    }
68
69    Ok(())
70}

72struct ReplicaStats {

81/// Read replica stats from CSV into a BTreeMap
82fn read_replica_view_stats(
83    path: &Path,
84) -> Result<BTreeMap<ViewNumber, ReplicaViewStats<SeqTypes>>, Box<dyn std::error::Error>> {
85    println!("\n**--- Replica Stats ---**");
86    let mut reader = csv::Reader::from_path(path)
87        .map_err(|e| format!("Failed to open replica stats CSV at {path:?}: {e}"))?;
88    let mut replica_view_stats = BTreeMap::new();
89
90    for result in reader.deserialize() {
91        let record: ReplicaViewStats<SeqTypes> = result?;
92        replica_view_stats.insert(record.view, record);
93    }
94
95    Ok(replica_view_stats)
96}

98/// Generate plots of replica stats
99fn plot_replica_stats(
100    replica_view_stats: &BTreeMap<ViewNumber, ReplicaViewStats<SeqTypes>>,
101    output_file: &Path,
102) -> Result<(), Box<dyn std::error::Error>> {
103    let mut x_views = Vec::new();
104    let mut y_timestamps_normal = Vec::new();
105    let mut hover_texts_normal = Vec::new();
106
107    let mut x_views_timeout = Vec::new();
108    let mut y_timestamps_timeout = Vec::new();
109    let mut hover_texts_timeout = Vec::new();
110
111    let mut first_event_counts = HashMap::new();
112    let mut views = Vec::new();
113    let mut proposal_times = Vec::new();
114    let mut vid_share_times = Vec::new();
115    let mut dac_times = Vec::new();
116
117    for (&view, record) in replica_view_stats {
118        // Skip views without a proposal receive event
119        let proposal_ts = match record.proposal_recv {
120            Some(t) => t,
121            None => continue,
122        };
123
124        // Collect all events with timestamps in milliseconds
125        let mut events_with_ts = Vec::new();
126        if let Some(ts) = record.proposal_recv {
127            events_with_ts.push(("proposal_recv", ts / 1_000_000));
128        }
129        if let Some(ts) = record.vote_send {
130            events_with_ts.push(("vote_send", ts / 1_000_000));
131        }
132        if let Some(ts) = record.timeout_vote_send {
133            events_with_ts.push(("timeout_vote_send", ts / 1_000_000));
134        }
135        if let Some(ts) = record.da_proposal_received {
136            events_with_ts.push(("da_proposal_received", ts / 1_000_000));
137        }
138        if let Some(ts) = record.da_proposal_validated {
139            events_with_ts.push(("da_proposal_validated", ts / 1_000_000));
140        }
141        if let Some(ts) = record.da_certificate_recv {
142            events_with_ts.push(("da_certificate_recv", ts / 1_000_000));
143        }
144        if let Some(ts) = record.proposal_prelim_validated {
145            events_with_ts.push(("proposal_prelim_validated", ts / 1_000_000));
146        }
147        if let Some(ts) = record.proposal_validated {
148            events_with_ts.push(("proposal_validated", ts / 1_000_000));
149        }
150        if let Some(ts) = record.timeout_triggered {
151            events_with_ts.push(("timeout_triggered", ts / 1_000_000));
152        }
153        if let Some(ts) = record.vid_share_validated {
154            events_with_ts.push(("vid_share_validated", ts / 1_000_000));
155        }
156        if let Some(ts) = record.vid_share_recv {
157            events_with_ts.push(("vid_share_recv", ts / 1_000_000));
158        }
159
160        // Count which event appeared first
161        if let Some((first_event, _)) = events_with_ts.clone().into_iter().min_by_key(|(_, ts)| *ts)
162        {
163            *first_event_counts.entry(first_event).or_insert(0) += 1;
164        }
165
166        events_with_ts.sort_by_key(|&(_, ts)| ts);
167        let ordered_events = events_with_ts
168            .iter()
169            .enumerate()
170            .map(|(i, (name, _))| format!("{}. {}", i + 1, name))
171            .collect::<Vec<_>>()
172            .join("<br>");
173        let hover = format!("View: {view}<br>Events:<br>{ordered_events}");
174
175        // Separate views where timeout was triggered vs normal
176        if record.timeout_triggered.is_some() {
177            x_views_timeout.push(view);
178            y_timestamps_timeout.push((proposal_ts as f64) / 1_000_000_000.0);
179            hover_texts_timeout.push(hover);
180        } else {
181            x_views.push(view);
182            y_timestamps_normal.push((proposal_ts as f64) / 1_000_000_000.0);
183            hover_texts_normal.push(hover);
184        }
185
186        views.push(view);
187        proposal_times.push(record.proposal_recv.map(|t| t as f64));
188        vid_share_times.push(record.vid_share_recv.map(|t| t as f64));
189        dac_times.push(record.da_certificate_recv.map(|t| t as f64));
190    }
191
192    // Aggregate first event stats for bar chart
193    let mut first_events: Vec<_> = first_event_counts.into_iter().collect();
194    first_events.sort_by(|a, b| b.1.cmp(&a.1));
195    let (bar_labels, bar_values): (Vec<_>, Vec<_>) = first_events
196        .into_iter()
197        .map(|(k, v)| (k.to_string(), v))
198        .unzip();
199
200    let trace_normal = Scatter::new(x_views, y_timestamps_normal)
201        .mode(Mode::Markers)
202        .hover_info(HoverInfo::Text)
203        .hover_text_array(hover_texts_normal)
204        .name("Proposal Received");
205
206    let trace_timeout = Scatter::new(x_views_timeout, y_timestamps_timeout)
207        .mode(Mode::Markers)
208        .hover_info(HoverInfo::Text)
209        .hover_text_array(hover_texts_timeout)
210        .marker(Marker::new().color("red").symbol(MarkerSymbol::Circle))
211        .name("Timeout Triggered");
212
213    let trace_bar = Bar::new(bar_labels, bar_values)
214        .name("First Event Frequency")
215        .x_axis("x2")
216        .y_axis("y2");
217
218    // scatter plots for Proposal / VID / DAC against on same graph with different traces
219    // against the view on x-axis
220    let trace_proposal_time = Scatter::new(views.clone(), proposal_times)
221        .mode(Mode::Markers)
222        .name("Proposal Received")
223        .x_axis("x3")
224        .y_axis("y3")
225        .marker(
226            Marker::new()
227                .symbol(MarkerSymbol::Circle)
228                .size(6)
229                .color("rgba(0,0,0,0)")
230                .line(Line::new().color("green").width(1.0)),
231        );
232
233    let trace_vid_share_time = Scatter::new(views.clone(), vid_share_times)
234        .mode(Mode::Markers)
235        .name("VID Share Received")
236        .x_axis("x3")
237        .y_axis("y3")
238        .marker(
239            Marker::new()
240                .symbol(MarkerSymbol::Square)
241                .size(8)
242                .color("rgba(0,0,0,0)")
243                .line(Line::new().color("orange").width(1.0)),
244        );
245
246    let trace_dac_time = Scatter::new(views.clone(), dac_times)
247        .mode(Mode::Markers)
248        .name("DAC Received")
249        .x_axis("x3")
250        .y_axis("y3")
251        .marker(
252            Marker::new()
253                .symbol(MarkerSymbol::Diamond)
254                .size(10)
255                .color("rgba(0,0,0,0)")
256                .line(Line::new().color("blue").width(1.0)),
257        );
258
259    let mut plot = Plot::new();
260    plot.add_trace(trace_normal.clone());
261    plot.add_trace(trace_timeout.clone());
262    plot.add_trace(trace_bar.clone());
263    plot.add_trace(trace_proposal_time.clone());
264    plot.add_trace(trace_vid_share_time.clone());
265    plot.add_trace(trace_dac_time.clone());
266
267    plot.set_layout(
268        Layout::new()
269            .title("Replica ReplicaStats")
270            .auto_size(true)
271            .grid(
272                LayoutGrid::new()
273                    .rows(3)
274                    .columns(1)
275                    .pattern(GridPattern::Independent),
276            )
277            .x_axis(Axis::new().title("View"))
278            .y_axis(Axis::new().title("Proposal Received Timestamp (s)"))
279            .x_axis2(Axis::new().title("Event").domain(&[0.0, 0.5]))
280            .y_axis2(Axis::new().title("First Event Count"))
281            .x_axis3(Axis::new().title("View"))
282            .y_axis3(Axis::new().title("DAC, VID, Proposal Timestamps (s)"))
283            .height(2500)
284            .margin(layout::Margin::new().left(130)),
285    );
286
287    plot.write_html(output_file);
288    println!("Plot saved to {output_file:?}");
289
290    Ok(())
291}

293/// Computes replica stats
294/// it generates the time difference for VID/DAC/Proposal
295/// from the view change event
296fn generate_replica_stats(
297    replica_view_stats: &BTreeMap<ViewNumber, ReplicaViewStats<SeqTypes>>,
298) -> ReplicaStats {
299    let mut vid_deltas_from_vc = Vec::new();
300    let mut dac_deltas_from_vc = Vec::new();
301    let mut proposal_deltas_from_vc = Vec::new();
302
303    for record in replica_view_stats.values() {
304        if let Some(vc) = record.view_change {
305            if let Some(vid) = record.vid_share_recv {
306                let delta_ms = (vid - vc) as f64 / 1_000_000.0;
307                vid_deltas_from_vc.push(delta_ms);
308            }
309
310            if let Some(dac) = record.da_certificate_recv {
311                let delta_ms = (dac - vc) as f64 / 1_000_000.0;
312                dac_deltas_from_vc.push(delta_ms);
313            }
314
315            if let Some(prop) = record.proposal_recv {
316                let delta_ms = (prop - vc) as f64 / 1_000_000.0;
317                proposal_deltas_from_vc.push(delta_ms);
318            }
319        }
320    }
321
322    ReplicaStats {
323        vid_deltas_from_vc,
324        dac_deltas_from_vc,
325        proposal_deltas_from_vc,
326    }
327}

329fn print_replica_stats(stats: &ReplicaStats) {
330    println!("Deltas calculated from view change time:");
331    print_delta_stats("DA Cert:", &stats.dac_deltas_from_vc);
332    print_delta_stats("VID Disperse:", &stats.vid_deltas_from_vc);
333    print_delta_stats("Proposal Received:", &stats.proposal_deltas_from_vc);
334}

352fn plot_and_print_leader_stats(
353    leader_view_stats: &BTreeMap<ViewNumber, LeaderViewStats<SeqTypes>>,
354    output_file: &Path,
355) -> Result<(), Box<dyn std::error::Error>> {
356    let mut views = Vec::new();
357    let mut vid_ts = Vec::new();
358    let mut dac_ts = Vec::new();
359    let mut qc_ts = Vec::new();
360
361    // Deltas relative to block built
362    let mut da_cert_deltas = Vec::new();
363    let mut vid_disperse_deltas = Vec::new();
364    // Deltas relative to previous proposal received
365    let mut block_built_prev_prop_deltas = Vec::new();
366
367    // For stats
368    let mut first_event_counts = HashMap::new();
369    let mut qc_vid_diffs = Vec::new();
370    let mut qc_dac_diffs = Vec::new();
371    let mut vid_dac_diffs = Vec::new();
372
373    for (&view, record) in leader_view_stats.iter() {
374        // Skip if either Block built, DA, VID, or QC is missing
375        let (block_built, dac, vid, qc) = match (
376            record.block_built,
377            record.da_cert_send,
378            record.vid_disperse_send,
379            record.qc_formed,
380        ) {
381            (Some(block_built), Some(dac), Some(vid), Some(qc)) => (block_built, dac, vid, qc),
382            _ => continue, // need all three
383        };
384
385        views.push(view);
386
387        // Determine first among QC / VID / DAC
388        let mut events = [("QC", qc), ("VID", vid), ("DAC", dac)];
389        events.sort_by_key(|&(_, ts)| ts);
390        let first = events[0].0;
391        *first_event_counts.entry(first).or_insert(0) += 1;
392
393        qc_vid_diffs.push(((qc as i64 - vid as i64) as f64) / 1_000_000.0);
394        qc_dac_diffs.push(((qc as i64 - dac as i64) as f64) / 1_000_000.0);
395        vid_dac_diffs.push(((vid as i64 - dac as i64) as f64) / 1_000_000.0);
396
397        // Difference relative to block built
398        da_cert_deltas.push((dac - block_built) as f64 / 1_000_000.0);
399        vid_disperse_deltas.push((vid - block_built) as f64 / 1_000_000.0);
400
401        // Difference between block built and previous proposal
402        if let Some(prev_prop) = record.prev_proposal_send {
403            block_built_prev_prop_deltas.push((block_built - prev_prop) as f64 / 1_000_000.0);
404        }
405
406        vid_ts.push(Some(vid as f64 / 1_000_000_000.0));
407        dac_ts.push(Some(dac as f64 / 1_000_000_000.0));
408        qc_ts.push(Some(qc as f64 / 1_000_000_000.0));
409    }
410
411    let trace_da_cert_deltas = Scatter::new(views.clone(), da_cert_deltas.clone())
412        .mode(Mode::Markers)
413        .name("DA Cert Δ (ms)")
414        .marker(
415            Marker::new()
416                .symbol(MarkerSymbol::Diamond)
417                .size(10)
418                .color("rgba(0,0,0,0)")
419                .line(Line::new().color("blue").width(1.0)),
420        );
421
422    let trace_vid_disperse_deltas = Scatter::new(views.clone(), vid_disperse_deltas.clone())
423        .mode(Mode::Markers)
424        .name("VID Disperse Δ (ms)")
425        .marker(
426            Marker::new()
427                .symbol(MarkerSymbol::Square)
428                .size(8)
429                .color("rgba(0,0,0,0)")
430                .line(Line::new().color("orange").width(1.0)),
431        );
432
433    let mut plot = Plot::new();
434
435    plot.add_trace(trace_da_cert_deltas);
436    plot.add_trace(trace_vid_disperse_deltas);
437
438    let trace_block_built_prev_prop =
439        Scatter::new(views.clone(), block_built_prev_prop_deltas.clone())
440            .mode(Mode::Markers)
441            .name("Block Built Δ from previous proposal (ms)")
442            .x_axis("x2")
443            .y_axis("y2")
444            .marker(Marker::new().symbol(MarkerSymbol::Circle));
445    plot.add_trace(trace_block_built_prev_prop);
446
447    let trace_vid = Scatter::new(views.clone(), vid_ts)
448        .mode(Mode::Markers)
449        .name("VID Timestamp")
450        .x_axis("x3")
451        .y_axis("y3")
452        .marker(
453            Marker::new()
454                .symbol(MarkerSymbol::Square)
455                .size(8)
456                .color("rgba(0,0,0,0)")
457                .line(Line::new().color("orange").width(1.0)),
458        );
459
460    let trace_dac = Scatter::new(views.clone(), dac_ts)
461        .mode(Mode::Markers)
462        .name("DAC Timestamp")
463        .x_axis("x3")
464        .y_axis("y3")
465        .marker(
466            Marker::new()
467                .symbol(MarkerSymbol::Diamond)
468                .size(10)
469                .color("rgba(0,0,0,0)")
470                .line(Line::new().color("blue").width(1.0)),

472
473    let trace_qc = Scatter::new(views.clone(), qc_ts)
474        .mode(Mode::Markers)
475        .name("QC Timestamp")
476        .x_axis("x3")
477        .y_axis("y3")
478        .marker(
479            Marker::new()
480                .symbol(MarkerSymbol::Circle)
481                .size(6)
482                .color("rgba(0,0,0,0)")
483                .line(Line::new().color("green").width(1.0)),

485
486    plot.add_trace(trace_vid);
487    plot.add_trace(trace_dac);
488    plot.add_trace(trace_qc);
489
490    plot.set_layout(
491        Layout::new()
492            .title("Leader Stats")
493            .grid(
494                LayoutGrid::new()
495                    .rows(3)
496                    .columns(1)
497                    .pattern(GridPattern::Independent),
498            )
499            .height(2000)
500            .x_axis(Axis::new().title("View"))
501            .y_axis(Axis::new().title("VID/DAC Δ from Block Built (ms)"))
502            .x_axis2(Axis::new().title("View"))
503            .y_axis2(Axis::new().title("Block built Δ from previous proposal (ms)"))
504            .x_axis3(Axis::new().title("View"))
505            .y_axis3(Axis::new().title("VID/DAC/QC Timestamps (ms)"))
506            .margin(layout::Margin::new().left(130)),
507    );
508
509    println!("\nOrdering of VID, DAC, QC):");
510    for (event, count) in &first_event_counts {
511        println!(" {event} was first in {count} views");
512    }
513
514    println!("\nDeltas calculated from block built:");
515    print_delta_stats("DA Cert:", &da_cert_deltas);
516    print_delta_stats("VID Disperse:", &vid_disperse_deltas);
517
518    println!("\nDeltas calculated from previous proposal:");
519    print_delta_stats("Block built:", &block_built_prev_prop_deltas);
520
521    plot.write_html(output_file);
522    println!("\n\nPlot saved to {output_file:?}");
523    Ok(())
524}

526fn print_delta_stats(label: &str, values: &[f64]) {
527    if values.is_empty() {
528        println!("\n--- {label} ---\nNo data available.");
529        return;
530    }
531
532    let min = values.iter().copied().fold(f64::INFINITY, f64::min);
533    let max = values.iter().copied().fold(f64::NEG_INFINITY, f64::max);
534    let sum: f64 = values.iter().sum();
535    let avg = sum / values.len() as f64;
536
537    println!("\n{label}");
538    println!("Count: {}", values.len());
539    println!("Min: {min:.2} ms");
540    println!("Max: {max:.2} ms");
541    println!("Avg: {avg:.2} ms");
