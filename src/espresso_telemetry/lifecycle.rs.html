1//! In-process OTel logs export plus a periodic Prometheus remote-write push for
2//! metrics. Both pipelines share one six-month BLS-BN254 JWT minted from the
3//! staking key. Tracing events go through an OTLP/HTTP batch exporter; an
4//! externally supplied `prometheus::Registry` is scraped on a tokio interval and
5//! POSTed to `/api/v1/write` as snappy-compressed protobuf.
6//!
7//! Failure modes:
8//! - Proxy down: `BatchLogProcessor` queue fills (~2k records) then drops;
9//!   metrics push `warn!`s and retries next tick. Neither uses disk nor blocks
10//!   consensus.
11//! - JWT misconfig: [`init`] returns `Err`; the caller continues without telemetry.
12//! - Token TTL expiry mid-process is not handled; the six-month TTL outlasts
13//!   typical restart cadence.
14

37const SERVICE_NAME: &str = "espresso-node";
38const SHUTDOWN_TIMEOUT: Duration = Duration::from_secs(10);

40/// Join `thread`, detaching it after [`SHUTDOWN_TIMEOUT`] so a wedged exporter
41/// can't hang shutdown on the main thread.
42fn join_bounded(thread: std::thread::JoinHandle<()>, what: &str) {
43    let (done_tx, done_rx) = std::sync::mpsc::sync_channel::<()>(0);
44    if let Err(e) = std::thread::Builder::new()
45        .name("espresso-telemetry-join".into())
46        .spawn(move || {
47            let _ = thread.join();
48            let _ = done_tx.send(());
49        })
50    {
51        tracing::warn!(error = %e, "telemetry: cannot spawn {what} join watcher");
52        return;
53    }
54    if done_rx.recv_timeout(SHUTDOWN_TIMEOUT).is_err() {
55        tracing::warn!(
56            timeout_secs = SHUTDOWN_TIMEOUT.as_secs(),
57            "telemetry: {what} shutdown timed out; detaching thread",
58        );
59    }
60}

62/// Global handoff for the prometheus `Registry` populated by HotShot.
63///
64/// The API setup builds the `Registry` after [`init`] and can't reach the
65/// telemetry wiring, so it deposits a clone here for the run path to read back.
66/// Single writer, single reader, ordered by node startup. Tests pass the
67/// registry directly into [`init`] instead.
68static REGISTRY: OnceLock<Arc<Registry>> = OnceLock::new();
69
70/// Deposit the `Registry`. Idempotent: subsequent calls are no-ops, since
71/// `OnceLock::set` returns `Err`.
72pub fn set_registry(registry: Arc<Registry>) {
73    let _ = REGISTRY.set(registry);
74}
75
76/// Read the registry deposited by [`set_registry`]. `None` if no API setup has
77/// run yet (tests, CLI tools without the HTTP module).
78pub fn registry() -> Option<Arc<Registry>> {
79    REGISTRY.get().cloned()
80}

82/// Operator-facing telemetry configuration.
83#[derive(Parser, Clone, Derivative)]
84#[derivative(Debug)]
85pub struct TelemetryOptions {
86    /// Enable the OTel logs pipeline.
87    #[clap(
88        long,
89        env = "ESPRESSO_NODE_TELEMETRY_LOGS_ENABLE",
90        default_value = "false"
91    )]
92    pub logs_enable: bool,
93
94    /// Enable the Prometheus remote-write metrics pipeline.
95    #[clap(
96        long,
97        env = "ESPRESSO_NODE_TELEMETRY_METRICS_ENABLE",
98        default_value = "false"
99    )]
100    pub metrics_enable: bool,
101
102    /// OTLP/HTTP base URL override. When unset, the caller selects the default
103    /// endpoint (the node picks it by chain ID).
104    #[clap(long, env = "ESPRESSO_NODE_TELEMETRY_ENDPOINT")]
105    pub endpoint: Option<Url>,
106
107    /// `EnvFilter` for the OTel log layer only; the local stderr layer is
108    /// unaffected. Default `warn`; per-target syntax works (e.g.
109    /// `warn,hotshot=info`).
110    #[clap(long, env = "ESPRESSO_NODE_TELEMETRY_LOG", default_value = "warn")]
111    pub log_filter: String,
112
113    /// Seconds between Prometheus remote-write pushes.
114    #[clap(
115        long,
116        env = "ESPRESSO_NODE_TELEMETRY_METRICS_INTERVAL",
117        default_value = "60"
118    )]
119    pub metrics_interval_secs: u64,
120}

134/// Handle to the metrics push thread. The thread owns a dedicated
135/// single-threaded runtime so a slow proxy can't starve a consensus worker.
136struct MetricsPushHandle {
137    shutdown: oneshot::Sender<()>,
138    thread: std::thread::JoinHandle<()>,
139}
140
141/// Owns the OTel logger provider and, optionally, the metrics push task, so
142/// both flush on graceful shutdown. Stashes the JWT and endpoint so the push
143/// task can be spawned later via [`TelemetryHandle::attach_metrics_push`] once
144/// the API setup has built the `Registry`.
145pub struct TelemetryHandle {
146    /// `None` when only metrics are enabled (logs pipeline disabled).
147    logger_provider: Option<SdkLoggerProvider>,
148    log_filter: String,
149    jwt: String,
150    endpoint: String,
151    metrics_interval: Duration,
152    metrics_push: Option<MetricsPushHandle>,
153    /// When false, `attach_metrics_push` is a no-op.
154    metrics_enabled: bool,
155    /// Labels stamped onto every pushed TimeSeries (e.g. `service`, `instance`),
156    /// mirroring the OTel resource attributes so the aggregator partitions logs
157    /// and metrics consistently.
158    metrics_external_labels: Vec<Label>,
159    /// Latch flipped on the first HTTP 429 from the metrics push, so the
160    /// operator-facing ERROR is logged once per process across push ticks. The
161    /// logs pipeline retries 429s via opentelemetry-otlp's built-in
162    /// `experimental-http-retry` and does not surface them here.
163    rate_limit_warned: Arc<AtomicBool>,
164    /// `log_filter` embedded verbatim in the rate-limit ERROR so operators see
165    /// their active filter.
166    telemetry_log_filter: Arc<String>,
167}
168
169impl std::fmt::Debug for TelemetryHandle {
170    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
171        f.debug_struct("TelemetryHandle")
172            .field("log_filter", &self.log_filter)
173            .field("metrics_push_active", &self.metrics_push.is_some())
174            .finish()

178impl TelemetryHandle {
179    /// Layer bridging tracing events into the OTLP exporter, filtered by
180    /// `log_filter`. `None` when the logs pipeline is disabled.
181    pub fn tracing_layer<S>(&self) -> Option<impl Layer<S> + Send + Sync + 'static>
182    where
183        S: Subscriber + for<'a> LookupSpan<'a>,
184    {
185        let provider = self.logger_provider.as_ref()?;
186        let bridge = OpenTelemetryTracingBridge::new(provider);
187        Some(bridge.with_filter(EnvFilter::new(self.log_filter.clone())))
188    }
189
190    /// Wrapper over [`attach_metrics_push_buffered`] that emits setup warnings
191    /// directly, for callers that already have a subscriber installed.
192    pub fn attach_metrics_push(&mut self, registry: Arc<Registry>) {
193        let mut deferred = Vec::new();
194        self.attach_metrics_push_buffered(registry, &mut deferred);
195        for w in deferred {
196            tracing::warn!("{w}");
197        }
198    }
199
200    /// Spawn the periodic metrics push on its own thread. Idempotent: no-op if
201    /// already attached or if metrics are disabled. Setup warnings buffer into
202    /// `deferred` so [`init`] can replay them after a subscriber is installed.
203    fn attach_metrics_push_buffered(
204        &mut self,
205        registry: Arc<Registry>,
206        deferred: &mut Vec<String>,
207    ) {
208        if !self.metrics_enabled {
209            return;
210        }
211        if self.metrics_push.is_some() {
212            return;
213        }
214        let push_endpoint: Url = match self.endpoint.parse() {
215            Ok(u) => u,
216            Err(e) => {
217                deferred.push(format!(
218                    "telemetry: cannot parse endpoint {endpoint:?} as URL ({e}); skipping metrics \
219                     push",
220                    endpoint = self.endpoint,
221                ));
222                return;
223            },
224        };
225        let (shutdown_tx, shutdown_rx) = oneshot::channel();
226        let jwt = self.jwt.clone();
227        let interval = self.metrics_interval;
228        let rate_limit_warned = self.rate_limit_warned.clone();
229        let telemetry_log_filter = self.telemetry_log_filter.clone();
230        let external_labels = self.metrics_external_labels.clone();
231        let thread = match std::thread::Builder::new()
232            .name("espresso-telemetry-metrics".into())
233            .spawn(move || {
234                let rt = match tokio::runtime::Builder::new_current_thread()
235                    .enable_all()
236                    .build()
237                {
238                    Ok(rt) => rt,
239                    Err(e) => {
240                        tracing::warn!(
241                            error = %e,
242                            "telemetry: cannot build dedicated runtime; metrics push disabled"
243                        );
244                        return;
245                    },
246                };
247                rt.block_on(push_task::run(
248                    registry,
249                    push_endpoint,
250                    jwt,
251                    interval,
252                    external_labels,
253                    rate_limit_warned,
254                    telemetry_log_filter,
255                    shutdown_rx,
256                ));
257            }) {
258            Ok(t) => t,
259            Err(e) => {
260                deferred.push(format!("telemetry: cannot spawn metrics push thread: {e}"));
261                return;
262            },
263        };
264        self.metrics_push = Some(MetricsPushHandle {
265            shutdown: shutdown_tx,
266            thread,
267        });
268    }
269
270    /// Flush the push thread, then shut down the OTel logger provider.
271    /// Best-effort; failures are logged, never bubbled. Both joins run through
272    /// [`join_bounded`] (the provider shutdown can deadlock on a current-thread
273    /// runtime).
274    pub fn shutdown(self) {
275        if let Some(MetricsPushHandle { shutdown, thread }) = self.metrics_push {
276            let _ = shutdown.send(());
277            join_bounded(thread, "metrics push");
278        }
279        if let Some(provider) = self.logger_provider {
280            match std::thread::Builder::new()
281                .name("espresso-telemetry-shutdown".into())
282                .spawn(move || {
283                    if let Err(e) = provider.shutdown() {
284                        tracing::warn!(error = %e, "telemetry: logger provider shutdown error");
285                    }
286                }) {
287                Ok(thread) => join_bounded(thread, "logger provider"),
288                Err(e) => tracing::warn!(error = %e, "telemetry: cannot spawn shutdown thread"),
289            }
290        }
291    }
292
293    /// True once the metrics push task is spawned.
294    #[doc(hidden)]
295    pub fn metrics_push_active(&self) -> bool {
296        self.metrics_push.is_some()
297    }
298
299    /// True when the metrics pipeline is enabled.
300    #[doc(hidden)]
301    pub fn metrics_enabled(&self) -> bool {
302        self.metrics_enabled
303    }
304}

306/// Initialize the OTel logger pipeline and, when `registry` is `Some`, spawn the
307/// metrics push immediately. Production passes `None` (the `Registry` isn't built
308/// yet) and later calls [`TelemetryHandle::attach_metrics_push`]; tests pass it
309/// directly.
310///
311/// `Ok((None, _))` when telemetry is disabled; `Err` on misconfig (bad endpoint,
312/// JWT mint failure). The returned warnings are produced before any subscriber is
313/// installed; callers MUST replay them via `tracing::warn!` afterward or they are
314/// lost.
315pub fn init(
316    opts: &TelemetryOptions,
317    staking_key: &SignKey,
318    node_name: Option<&str>,
319    company_name: Option<&str>,
320    endpoint: &Url,
321    registry: Option<Arc<Registry>>,
322) -> anyhow::Result<(Option<TelemetryHandle>, Vec<String>)> {
323    let logs_on = opts.logs_enable;
324    let metrics_on = opts.metrics_enable;
325
326    if !logs_on && !metrics_on {
327        return Ok((None, Vec::new()));
328    }
329
330    let mut deferred: Vec<String> = Vec::new();
331
332    if logs_on && let Err(e) = EnvFilter::try_new(&opts.log_filter) {
333        deferred.push(format!(
334            "telemetry: invalid log_filter {filter:?} ({e}); falling back to lossy parse, some \
335             directives may be ignored",
336            filter = opts.log_filter,
337        ));
338    }
339
340    let jwt = UnauthenticatedToken::generate_with(staking_key, node_name, company_name)
341        .context("mint telemetry JWT")?
342        .encode();
343
344    // Both signals share this base URL. Reject non-http(s) so the metrics push
345    // can't silently no-op later.
346    let scheme = endpoint.scheme();
347    if scheme != "http" && scheme != "https" {
348        anyhow::bail!(
349            "telemetry endpoint must use http or https scheme, got {scheme:?}: {endpoint}"
350        );
351    }
352    let endpoint = endpoint.as_str().to_owned();
353
354    let telemetry_log_filter: Arc<String> = Arc::new(opts.log_filter.clone());
355    let rate_limit_warned: Arc<AtomicBool> = Arc::new(AtomicBool::new(false));
356
357    let logger_provider = if logs_on {
358        Some(build_logger_provider(jwt.clone(), &endpoint, node_name)?)
359    } else {
360        None
361    };
362
363    let metrics_interval = Duration::from_secs(opts.metrics_interval_secs.max(1));
364    let mut metrics_external_labels = vec![Label {
365        name: "service".to_owned(),
366        value: SERVICE_NAME.to_owned(),
367    }];
368    if let Some(name) = node_name {
369        metrics_external_labels.push(Label {
370            name: "instance".to_owned(),
371            value: name.to_owned(),
372        });
373    }
374    let mut handle = TelemetryHandle {
375        logger_provider,
376        log_filter: opts.log_filter.clone(),
377        jwt,
378        endpoint,
379        metrics_interval,
380        metrics_push: None,
381        metrics_enabled: metrics_on,
382        metrics_external_labels,
383        rate_limit_warned,
384        telemetry_log_filter,
385    };
386
387    if let Some(registry) = registry {
388        handle.attach_metrics_push_buffered(registry, &mut deferred);
389    }
390
391    Ok((Some(handle), deferred))
392}

394fn build_logger_provider(
395    jwt: String,
396    endpoint: &str,
397    node_name: Option<&str>,
398) -> anyhow::Result<SdkLoggerProvider> {
399    let logs_endpoint = format!("{}/v1/logs", endpoint.trim_end_matches('/'));
400
401    let mut headers = HashMap::new();
402    headers.insert("authorization".to_string(), format!("Bearer {jwt}"));
403
404    // Logs are highly compressible; gzip keeps flush payloads small.
405    let exporter = LogExporter::builder()
406        .with_http()
407        .with_protocol(Protocol::HttpBinary)
408        .with_compression(Compression::Gzip)
409        .with_endpoint(logs_endpoint)
410        .with_headers(headers)
411        .build()
412        .context("build OTLP log exporter")?;
413
414    let mut resource = Resource::builder().with_service_name(SERVICE_NAME);
415    if let Some(name) = node_name {
416        // service.instance.id distinguishes individual operators in the aggregator.
417        resource = resource.with_attribute(KeyValue::new("service.instance.id", name.to_owned()));
418    }
419
420    Ok(SdkLoggerProvider::builder()
421        .with_resource(resource.build())
422        .with_batch_exporter(exporter)
423        .build())
