21/// Which directory to read PSI files from. Detected once at startup.
22#[derive(Clone, Copy, Debug, PartialEq, Eq)]
23enum PressureSource {
24    /// cgroup v2: `/sys/fs/cgroup/{cpu,memory,io}.pressure`
25    CgroupV2,
26    /// host-wide: `/proc/pressure/{cpu,memory,io}`
27    Host,
28    /// PSI unavailable on this kernel/container.
29    None,
30}
31
32impl PressureSource {
33    fn detect() -> Self {
34        let cgroup = Path::new(CGROUP_ROOT).join("cpu.pressure");
35        if Path::new(CGROUP_ROOT).join("cgroup.controllers").exists() && cgroup.exists() {
36            return Self::CgroupV2;
37        }
38        if Path::new(HOST_PRESSURE_DIR).join("cpu").exists() {
39            return Self::Host;
40        }
41        Self::None
42    }
43
44    fn path(self, resource: &str) -> Option<String> {
45        match self {
46            Self::CgroupV2 => Some(format!("{CGROUP_ROOT}/{resource}.pressure")),
47            Self::Host => Some(format!("{HOST_PRESSURE_DIR}/{resource}")),
48            Self::None => None,
49        }
50    }
51}

53/// Whether cgroup v2 cpu/memory accounting files are readable.
54fn detect_cgroup_v2() -> bool {
55    Path::new(CGROUP_ROOT).join("cpu.stat").exists()
56        && Path::new(CGROUP_ROOT).join("memory.current").exists()
57}

59/// Accumulates fractional units (µs or ticks) into a counter measured in whole seconds,
60/// preserving sub-second precision across many ticks.
61#[derive(Default)]
62struct SecondsAccumulator {
63    /// Last absolute reading from the kernel, in the source unit (µs or ticks).
64    last: Option<u64>,
65    /// Sub-second remainder carried between calls, in the source unit.
66    remainder: u64,
67}
68
69impl SecondsAccumulator {
70    /// Feed an absolute monotonic reading. Returns whole-seconds delta to add to the counter.
71    fn observe(&mut self, current: u64, units_per_second: u64) -> usize {
72        let Some(prev) = self.last.replace(current) else {
73            return 0;
74        };
75        let delta = current.saturating_sub(prev);
76        let total = self.remainder + delta;
77        let whole = total / units_per_second;
78        self.remainder = total % units_per_second;
79        whole as usize
80    }
81}

83/// Tracks the previous absolute value of a `u64` counter for delta-add against a `Counter`.
84#[derive(Default)]
85struct U64Delta {
86    last: Option<u64>,
87}
88
89impl U64Delta {
90    fn observe(&mut self, current: u64) -> usize {
91        let Some(prev) = self.last.replace(current) else {
92            return 0;
93        };
94        current.saturating_sub(prev) as usize
95    }
96}

154impl ProcessMetrics {
155    pub fn new(metrics: &(impl Metrics + ?Sized)) -> Self {
156        let bytes = || Some("bytes".into());
157        let seconds = || Some("seconds".into());
158
159        // `memory.max` is either a u64 or the literal "max" (unlimited). Only register the
160        // gauge when finite so operators don't see a perpetual 0 that looks like a 0-byte
161        // limit. Container memory limits don't change at runtime, so set it once here; the
162        // registry retains its own handle, so we don't keep the `Box` around.
163        if let Some(max_bytes) = read_cgroup_memory_max() {
164            metrics
165                .create_gauge("cgroup_memory_max_bytes".into(), bytes())
166                .set(max_bytes as usize);
167        }
168
169        Self {
170            resident_memory_bytes: metrics
171                .create_gauge("process_resident_memory_bytes".into(), bytes()),
172            virtual_memory_bytes: metrics
173                .create_gauge("process_virtual_memory_bytes".into(), bytes()),
174            open_fds: metrics.create_gauge("process_open_fds".into(), None),
175            threads: metrics.create_gauge("process_threads".into(), None),
176            uptime_seconds: metrics.create_gauge("process_uptime_seconds".into(), seconds()),
177
178            cpu_count: metrics.create_gauge("node_cpu_count".into(), None),
179            load1_milli: metrics.create_gauge("node_load1_milli".into(), None),
180            load5_milli: metrics.create_gauge("node_load5_milli".into(), None),
181            load15_milli: metrics.create_gauge("node_load15_milli".into(), None),
182
183            process_cpu_seconds_total: metrics
184                .create_counter("process_cpu_seconds_total".into(), seconds()),
185
186            pressure_cpu_some_total: metrics
187                .create_counter("node_pressure_cpu_waiting_seconds_total".into(), seconds()),
188            pressure_memory_some_total: metrics.create_counter(
189                "node_pressure_memory_waiting_seconds_total".into(),
190                seconds(),
191            ),
192            pressure_memory_full_total: metrics.create_counter(
193                "node_pressure_memory_stalled_seconds_total".into(),
194                seconds(),
195            ),
196            pressure_io_some_total: metrics
197                .create_counter("node_pressure_io_waiting_seconds_total".into(), seconds()),
198            pressure_io_full_total: metrics
199                .create_counter("node_pressure_io_stalled_seconds_total".into(), seconds()),
200
201            cgroup_cpu_periods_total: metrics
202                .create_counter("cgroup_cpu_periods_total".into(), None),
203            cgroup_cpu_throttled_periods_total: metrics
204                .create_counter("cgroup_cpu_throttled_periods_total".into(), None),
205            cgroup_cpu_throttled_seconds_total: metrics
206                .create_counter("cgroup_cpu_throttled_seconds_total".into(), seconds()),
207
208            cgroup_memory_current_bytes: metrics
209                .create_gauge("cgroup_memory_current_bytes".into(), bytes()),
210
211            process_read_bytes_total: metrics
212                .create_counter("process_read_bytes_total".into(), bytes()),
213            process_write_bytes_total: metrics
214                .create_counter("process_write_bytes_total".into(), bytes()),
215        }

218    pub async fn run(mut self) {
219        let pid = match sysinfo::get_current_pid() {
220            Ok(pid) => pid,
221            Err(err) => {
222                tracing::warn!(%err, "could not determine current pid; process metrics disabled");
223                return;
224            },
225        };
226
227        let env = Env {
228            pid,
229            start: Instant::now(),
230            pressure: PressureSource::detect(),
231            cgroup_v2: detect_cgroup_v2(),
232            ticks_per_second: procfs::ticks_per_second(),
233        };
234        tracing::info!(
235            pressure = ?env.pressure,
236            cgroup_v2 = env.cgroup_v2,
237            ticks_per_second = env.ticks_per_second,
238            "process metrics source detection complete"
239        );
240
241        // CPU count is process-invariant; set once and drop the periodic sample.
242        self.cpu_count.set(sysinfo::System::new().cpus().len());
243
244        let mut system = System::new();
245        let mut previous = Previous::default();
246        let mut ticker = interval(SAMPLE_INTERVAL);
247        loop {
248            ticker.tick().await;
249            self.sample(&mut system, env, &mut previous);
250        }
251    }
252
253    fn sample(&mut self, system: &mut System, env: Env, prev: &mut Previous) {
254        let Env {
255            pid,
256            start,
257            pressure,
258            cgroup_v2,
259            ticks_per_second,
260        } = env;
261        system.refresh_processes_specifics(
262            ProcessesToUpdate::Some(&[pid]),
263            true,
264            ProcessRefreshKind::nothing().with_memory(),
265        );
266        if let Some(process) = system.process(pid) {
267            self.resident_memory_bytes.set(process.memory() as usize);
268            self.virtual_memory_bytes
269                .set(process.virtual_memory() as usize);
270        }
271        self.open_fds.set(count_dir_entries("/proc/self/fd"));
272        self.threads.set(count_dir_entries("/proc/self/task"));
273        self.uptime_seconds
274            .set(Instant::now().duration_since(start).as_secs() as usize);
275
276        if let Some(load) = read_or_debug("loadavg", LoadAverage::current) {
277            self.load1_milli.set(milli(load.one));
278            self.load5_milli.set(milli(load.five));
279            self.load15_milli.set(milli(load.fifteen));
280        }
281
282        if let Some(p) = read_or_debug("process self", Process::myself) {
283            if let Some(stat) = read_or_debug("/proc/self/stat", || p.stat()) {
284                let total_ticks = stat.utime + stat.stime;
285                self.process_cpu_seconds_total
286                    .add(prev.cpu_ticks.observe(total_ticks, ticks_per_second));
287            }
288            if let Some(Io {
289                read_bytes,
290                write_bytes,
291                ..
292            }) = read_or_debug("/proc/self/io", || p.io())
293            {
294                self.process_read_bytes_total
295                    .add(prev.read_bytes.observe(read_bytes));
296                self.process_write_bytes_total
297                    .add(prev.write_bytes.observe(write_bytes));
298            }
299        }
300
301        self.sample_pressure(pressure, prev);
302
303        if cgroup_v2 {
304            self.sample_cgroup_cpu(prev);
305            self.sample_cgroup_memory();
306        }
307    }
308
309    fn sample_pressure(&self, pressure: PressureSource, prev: &mut Previous) {
310        if let Some(cpu_path) = pressure.path("cpu")
311            && let Some((some, _full)) = read_pressure(&cpu_path)
312        {
313            self.pressure_cpu_some_total
314                .add(prev.pressure_cpu_some.observe(some.total, 1_000_000));
315        }
316
317        if let Some(mem_path) = pressure.path("memory")
318            && let Some((some, full)) = read_pressure(&mem_path)
319        {
320            self.pressure_memory_some_total
321                .add(prev.pressure_memory_some.observe(some.total, 1_000_000));
322            self.pressure_memory_full_total
323                .add(prev.pressure_memory_full.observe(full.total, 1_000_000));
324        }
325
326        if let Some(io_path) = pressure.path("io")
327            && let Some((some, full)) = read_pressure(&io_path)
328        {
329            self.pressure_io_some_total
330                .add(prev.pressure_io_some.observe(some.total, 1_000_000));
331            self.pressure_io_full_total
332                .add(prev.pressure_io_full.observe(full.total, 1_000_000));
333        }
334    }
335
336    fn sample_cgroup_cpu(&self, prev: &mut Previous) {
337        let Some(stat) = read_cgroup_cpu_stat() else {
338            return;
339        };
340        self.cgroup_cpu_periods_total
341            .add(prev.cgroup_cpu_periods.observe(stat.nr_periods));
342        self.cgroup_cpu_throttled_periods_total
343            .add(prev.cgroup_cpu_throttled_periods.observe(stat.nr_throttled));
344        self.cgroup_cpu_throttled_seconds_total.add(
345            prev.cgroup_cpu_throttled_us
346                .observe(stat.throttled_usec, 1_000_000),
347        );
348    }
349
350    fn sample_cgroup_memory(&self) {
351        if let Some(bytes) = read_u64_file(&format!("{CGROUP_ROOT}/memory.current")) {
352            self.cgroup_memory_current_bytes.set(bytes as usize);
353        }
354        // `cgroup_memory_max_bytes` is set once at startup in `new()`.
355    }
356}

358fn milli(v: f32) -> usize {
359    (v * 1000.0).max(0.0) as usize

362fn count_dir_entries(path: &str) -> usize {
363    match fs::read_dir(path) {
364        Ok(d) => d.filter(Result::is_ok).count(),
365        Err(err) => {
366            tracing::debug!(%path, %err, "could not read directory for process metrics");
367            0
368        },
369    }
370}

372fn read_or_debug<T, E, F>(what: &str, f: F) -> Option<T>
373where
374    F: FnOnce() -> Result<T, E>,
375    E: std::fmt::Display,
376{
377    match f() {
378        Ok(v) => Some(v),
379        Err(err) => {
380            tracing::debug!(%what, %err, "process metrics read failed");
381            None
382        },
383    }
384}

386fn read_pressure(path: &str) -> Option<(PressureRecord, PressureRecord)> {
387    let file = match fs::File::open(path) {
388        Ok(f) => f,
389        Err(err) => {
390            tracing::debug!(%path, %err, "could not open pressure file");
391            return None;
392        },
393    };
394    match get_pressure(BufReader::new(file)) {
395        Ok(v) => Some(v),
396        Err(err) => {
397            tracing::debug!(%path, %err, "could not parse pressure file");
398            None
399        },
400    }
401}

404struct CpuStat {
405    nr_periods: u64,
406    nr_throttled: u64,
407    throttled_usec: u64,
408}
409
410fn read_cgroup_cpu_stat() -> Option<CpuStat> {
411    let path = format!("{CGROUP_ROOT}/cpu.stat");
412    let contents = read_string_file(&path)?;
413    let mut out = CpuStat::default();
414    let mut saw_any = false;
415    for line in contents.lines() {
416        let mut parts = line.split_whitespace();
417        let (Some(key), Some(value)) = (parts.next(), parts.next()) else {
418            continue;
419        };
420        let Ok(value) = value.parse::<u64>() else {
421            continue;
422        };
423        match key {
424            "nr_periods" => {
425                out.nr_periods = value;
426                saw_any = true;
427            },
428            "nr_throttled" => {
429                out.nr_throttled = value;
430                saw_any = true;
431            },
432            "throttled_usec" => {
433                out.throttled_usec = value;
434                saw_any = true;
435            },
436            _ => {},
437        }
438    }
439    saw_any.then_some(out)
440}

444        Ok(s) => Some(s),
445        Err(err) => {

452/// Read `cgroup_root/memory.max`. Returns `None` when the file is missing/unreadable or
453/// holds the literal `max` (unlimited).
454fn read_cgroup_memory_max() -> Option<u64> {
455    let raw = read_string_file(&format!("{CGROUP_ROOT}/memory.max"))?;
456    raw.trim().parse::<u64>().ok()
457}

474    #[test]
475    fn seconds_accumulator_first_sample_is_zero() {
476        let mut acc = SecondsAccumulator::default();
477        assert_eq!(acc.observe(123_456, 1_000_000), 0);

480    #[test]
481    fn seconds_accumulator_preserves_remainder() {
482        let mut acc = SecondsAccumulator::default();
483        // First call seeds the baseline.
484        assert_eq!(acc.observe(0, 1_000_000), 0);
485        // 0.5s delta — no whole second yet.
486        assert_eq!(acc.observe(500_000, 1_000_000), 0);
487        // Another 0.6s delta — one whole second, 0.1s remainder.
488        assert_eq!(acc.observe(1_100_000, 1_000_000), 1);
489        // Another 0.95s — total now 1.05s of remainder + delta → 1 sec.
490        assert_eq!(acc.observe(2_050_000, 1_000_000), 1);
491    }
492
493    #[test]
494    fn seconds_accumulator_handles_counter_reset() {
495        let mut acc = SecondsAccumulator::default();
496        acc.observe(10_000_000, 1_000_000);
497        // Apparent regression (e.g. proc remount or wraparound), saturate to 0.
498        assert_eq!(acc.observe(5_000_000, 1_000_000), 0);
499        // After saturating, `last` should equal the most recent reading; the next
500        // legitimate delta from there should still register.
501        assert_eq!(acc.observe(6_000_000, 1_000_000), 1);
502    }
503
504    #[test]
505    fn u64_delta_first_sample_is_zero() {
506        let mut d = U64Delta::default();
507        assert_eq!(d.observe(42), 0);
508        assert_eq!(d.observe(45), 3);
509        // Reset: don't emit a negative spike.
510        assert_eq!(d.observe(10), 0);
511    }
512
513    #[test]
514    fn milli_clamps_negative_to_zero() {
515        assert_eq!(milli(0.0), 0);
516        assert_eq!(milli(1.25), 1250);
517        assert_eq!(milli(-0.1), 0);
518    }
