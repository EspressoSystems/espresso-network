5/// Meta information appended at the end of a message.
6#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
7pub struct Trailer {
8    pub(crate) ty: TrailerType,
9    /// The slot the message corresponds to.
10    pub(crate) slot: Slot,
11    /// The message ID.
12    pub(crate) id: MsgId,
13}
14
15#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
16#[repr(u8)]
17pub enum TrailerType {
18    Std = 0,
19    NoAck = 1,
20    Unknown,
21}
22
23impl From<u8> for TrailerType {
24    fn from(value: u8) -> Self {
25        match value {
26            0 => Self::Std,
27            1 => Self::NoAck,
28            _ => Self::Unknown,
29        }
30    }
31}
32
33impl TrailerType {
34    pub fn is_ack(self) -> bool {
35        matches!(self, Self::Std | Self::Unknown)
36    }
37}
38
39// Besides the actual meta information, the last byte of the trailer encodes its
40// total length (including the length byte itself).
41impl Trailer {
42    pub const SIZE: usize = 18;
43    pub const MAX_SIZE: usize = u8::MAX as usize;
44
45    pub fn new(ty: TrailerType, slot: Slot, id: MsgId) -> Self {
46        Self { ty, slot, id }
47    }
48
49    pub fn from_bytes(bytes: &mut Bytes) -> Option<Self> {
50        let len = bytes.len();
51        let trailer_len: usize = (*bytes.last()?).into();
52        if trailer_len < Self::SIZE || len < trailer_len {
53            return None;
54        }
55        let ty = TrailerType::from(bytes[len - 2]);
56        let id = u64::from_be_bytes(bytes[len - 10..len - 2].try_into().ok()?);
57        let slot = u64::from_be_bytes(bytes[len - 18..len - 10].try_into().ok()?);
58        bytes.truncate(len - trailer_len);
59        Some(Self {
60            ty,
61            slot: Slot(slot),
62            id: MsgId(id),
63        })
64    }
65
66    pub fn to_bytes(self) -> [u8; Self::SIZE] {
67        let mut buf = [0; Self::SIZE];
68        buf[..8].copy_from_slice(&self.slot.0.to_be_bytes()[..]);
69        buf[8..16].copy_from_slice(&self.id.0.to_be_bytes()[..]);
70        buf[16] = self.ty as u8;
71        buf[17] = Self::SIZE as u8;
72        buf
73    }
74}
