43    pub async fn accept(conf: Arc<Config>, mut stream: TcpStream) -> Result<Self> {
44        let node = conf.keypair.public_key();
45
46        if let Err(err) = stream.set_nodelay(true) {
47            warn!(name = %conf.name, %node, %err, "failed to enable NO_DELAY option")
48        }
49
50        until(conf.handshake_timeout, async move {
51            let (version, prologue) = select_version(&conf, &mut stream, false).await?;
52
53            debug!(name = %conf.name, %node, %version, "negotiated version");
54
55            let noise_proto = conf
56                .noise_protocols
57                .get(&version)
58                .expect("selected version has noise config");
59
60            let hs = Builder::new(noise_proto.noise_params())
61                .local_private_key(&conf.keypair.secret_key().as_bytes())
62                .expect("valid private key")
63                .prologue(&prologue)
64                .expect("1st time we set the prologue")
65                .build_responder()
66                .expect("valid noise params yield valid handshake state");
67
68            let addr = stream.peer_addr()?;
69            let state = on_handshake(&mut stream, hs).await?;
70
71            if let Some(key) = remote_static_key(&state) {
72                Ok(Self {
73                    key,
74                    addr,
75                    stream,
76                    state,
77                })
78            } else {
79                warn!(name = %conf.name, %node, %addr, "invalid static key");
80                Err(NetworkError::InvalidHandshakeMessage)
81            }
82        })
83        .await
84    }

86    pub async fn connect(conf: Arc<Config>, peer: PublicKey, addr: NetAddr) -> Self {
87        let mut delays = once({
88            if conf.random_connect_delay {
89                Duration::from_millis(rand::rng().random_range(0..1000))
90            } else {
91                Duration::ZERO
92            }
93        })
94        .chain(
95            conf.retry_delays
96                .iter()
97                .map(|&d| Duration::from_secs(d.into())),
98        )
99        .chain(repeat(conf.max_retry_delay));
100
101        let addr = addr.to_string();
102        let node = conf.keypair.public_key();
103
104        let mut backoff = None;
105
106        loop {
107            if let Some(d) = backoff.take() {
108                sleep(d).await;
109            } else {
110                sleep(delays.next().expect("delays iterator is infinite")).await;
111            }
112
113            debug!(name = %conf.name, %node, %peer, %addr, "connecting");
114
115            match try_connect(&conf, &peer, &addr).await {
116                Ok(mut conn) => {
117                    let hello_exchange = until(conf.handshake_timeout, async {
118                        conn.send_hello(Hello::Ok).await?;
119                        conn.recv_hello().await
120                    });
121                    match hello_exchange.await {
122                        Ok(h) if h.is_ok() => break conn,
123                        Ok(h) => {
124                            warn!(
125                                name = %conf.name,
126                                %node,
127                                %peer,
128                                remote = %conn.key,
129                                %addr,
130                                "hello response was not ok"
131                            );
132                            backoff = h.backoff_duration();
133                        },
134                        Err(err) => {
135                            warn!(
136                                name = %conf.name,
137                                %node,
138                                %peer,
139                                remote = %conn.key,
140                                %addr,
141                                %err,
142                                "failed to exchange hello"
143                            );
144                        },
145                    }
146                },
147                Err(err) => {
148                    warn!(name = %conf.name, %node, %peer, %addr, %err, "connect/handshake error")
149                },
150            }
151        }
152    }

154    /// Send a `Hello` frame.
155    pub async fn send_hello(&mut self, h: Hello) -> Result<()> {
156        let mut b = [0u8; 64];
157        let n = self
158            .state
159            .write_message(h.to_bytes().as_ref(), &mut b[Header::SIZE..])?;
160        let h = Header::data(n as u16);
161        send_frame(&mut self.stream, h, &mut b[..Header::SIZE + n]).await?;
162        Ok(())
163    }
164
165    /// Read a `Hello` frame.
166    pub async fn recv_hello(&mut self) -> Result<Hello> {
167        let mut a = [0u8; 64];
168        let h = recv_frame(&mut self.stream, &mut a).await?;
169        let mut b = [0u8; 64];
170        let n = self.state.read_message(&a[..h.len().into()], &mut b)?;
171        let h = Hello::from_bytes(&b[..n]).ok_or(NetworkError::InvalidHello)?;
172        Ok(h)
173    }
174}

176async fn try_connect(conf: &Config, peer: &PublicKey, addr: &str) -> Result<Connection> {
177    let new_handshake_state = |prologue: &Prologue, params: NoiseParams| {
178        Builder::new(params)
179            .local_private_key(conf.keypair.secret_key().as_slice())
180            .expect("valid private key")
181            .remote_public_key(peer.as_slice())
182            .expect("valid remote pub key")
183            .prologue(prologue)
184            .expect("1st time we set the prologue")
185            .build_initiator()
186            .expect("valid noise params yield valid handshake state")
187    };
188
189    let mut stream = until(conf.connect_timeout, TcpStream::connect(addr)).await?;
190
191    let node = conf.keypair.public_key();
192    let addr = stream.peer_addr()?;
193
194    debug!(name = %conf.name, %node, %peer, %addr, "tcp connection established");
195
196    if let Err(err) = stream.set_nodelay(true) {
197        warn!(name = %conf.name, %node, %err, "failed to enable NO_DELAY option");
198    }
199
200    until(conf.handshake_timeout, async move {
201        let (version, prologue) = select_version(conf, &mut stream, true).await?;
202
203        debug!(name = %conf.name, %node, %peer, %addr, %version, "negotiated version");
204
205        let noise_proto = conf
206            .noise_protocols
207            .get(&version)
208            .expect("selected version has noise config");
209
210        let hshake = new_handshake_state(&prologue, noise_proto.noise_params());
211        let state = handshake(&mut stream, hshake).await?;
212        match remote_static_key(&state) {
213            Some(key) if key == *peer => Ok(Connection {
214                key,
215                addr,
216                stream,
217                state,
218            }),
219            Some(key) => {
220                warn!(name = %conf.name, %node, %peer, remote = %key, %addr, "static key mismatch");
221                Err(NetworkError::InvalidHandshakeMessage)
222            },
223            None => {
224                warn!(name = %conf.name, %node, %peer, %addr, "invalid static key");
225                Err(NetworkError::InvalidHandshakeMessage)
226            },
227        }
228    })
229    .await
230}

232fn remote_static_key(state: &TransportState) -> Option<PublicKey> {
233    let k = state.get_remote_static()?;
234    PublicKey::try_from(k).ok()
235}

237/// Select a version from the range that both sides support.
238///
239/// This will be the minimum of the max. supported ones from both sides.
240async fn select_version(
241    conf: &Config,
242    stream: &mut TcpStream,
243    is_initiator: bool,
244) -> Result<(Version, Prologue)> {
245    const INIT_PAYLOAD_LEN: usize = 4;
246
247    let our_min = conf
248        .noise_protocols
249        .first_key_value()
250        .map(|(k, _)| *k)
251        .expect("noise_configs is not empty");
252    let our_max = conf
253        .noise_protocols
254        .last_key_value()
255        .map(|(k, _)| *k)
256        .expect("noise_configs is not empty");
257
258    let mut send_buf = [0u8; INIT_PAYLOAD_LEN];
259    let mut recv_buf = [0u8; INIT_PAYLOAD_LEN];
260
261    send_buf[0..2].copy_from_slice(&u16::from(our_min).to_be_bytes());
262    send_buf[2..4].copy_from_slice(&u16::from(our_max).to_be_bytes());
263
264    let (mut r, mut w) = stream.split();
265    try_join!(w.write_all(&send_buf), r.read_exact(&mut recv_buf))?;
266
267    let their_min = Version::from(u16::from_be_bytes([recv_buf[0], recv_buf[1]]));
268    let their_max = Version::from(u16::from_be_bytes([recv_buf[2], recv_buf[3]]));
269
270    let selected = min(our_max, their_max);
271
272    if selected < their_min || selected < our_min {
273        return Err(NetworkError::IncompatibleVersions {
274            ours: (our_min, our_max),
275            theirs: (their_min, their_max),
276        });
277    }
278
279    // Construct the prologue so that both sides end up with the same value.
280    // We include the sent and received version ranges to ensure no one has
281    // tampered with those values as they were sent in plain text.
282    let mut prologue = Vec::new();
283    prologue.extend_from_slice(conf.name.as_bytes());
284    if is_initiator {
285        prologue.extend_from_slice(&send_buf);
286        prologue.extend_from_slice(&recv_buf);
287    } else {
288        prologue.extend_from_slice(&recv_buf);
289        prologue.extend_from_slice(&send_buf);
290    }
291
292    Ok((selected, prologue))
293}

295/// Perform a noise handshake as initiator with the remote party.
296async fn handshake(stream: &mut TcpStream, mut hs: HandshakeState) -> Result<TransportState> {
297    let mut a = [0u8; MAX_NOISE_HANDSHAKE_SIZE];
298    let n = hs.write_message(&[], &mut a[Header::SIZE..])?;
299    let h = Header::data(n as u16);
300    send_frame(stream, h, &mut a[..Header::SIZE + n]).await?;
301    let mut b = [0u8; MAX_NOISE_HANDSHAKE_SIZE];
302    let h = recv_frame(stream, &mut b).await?;
303    if !h.is_data() || h.is_partial() {
304        return Err(NetworkError::InvalidHandshakeMessage);
305    }
306    hs.read_message(&b[..h.len().into()], &mut a)?;
307    Ok(hs.into_transport_mode()?)
308}
309
310/// Perform a noise handshake as responder with a remote party.
311async fn on_handshake(stream: &mut TcpStream, mut hs: HandshakeState) -> Result<TransportState> {
312    let mut a = [0u8; MAX_NOISE_HANDSHAKE_SIZE];
313    let h = recv_frame(stream, &mut a).await?;
314    if !h.is_data() || h.is_partial() {
315        return Err(NetworkError::InvalidHandshakeMessage);
316    }
317    let mut b = [0u8; MAX_NOISE_HANDSHAKE_SIZE];
318    hs.read_message(&a[..h.len().into()], &mut b)?;
319    let n = hs.write_message(&[], &mut b[Header::SIZE..])?;
320    let h = Header::data(n as u16);
321    send_frame(stream, h, &mut b[..Header::SIZE + n]).await?;
322    Ok(hs.into_transport_mode()?)
323}

325/// Read a single frame (header + payload) from the remote.
326async fn recv_frame<R, const N: usize>(stream: &mut R, buf: &mut [u8; N]) -> io::Result<Header>
327where
328    R: AsyncReadExt + Unpin,
329{
330    let h = {
331        let n = stream.read_u32().await?;
332        Header::unvalidated(n)
333    };
334    let n = h.len().into();
335    if n > N {
336        return Err(io::ErrorKind::InvalidInput.into());
337    }
338    stream.read_exact(&mut buf[..n]).await?;
339    Ok(h)
340}
341
342/// Write a single frame (header + payload) to the remote.
343///
344/// The header is serialised into the first 4 bytes of `msg`. It is the
345/// caller's responsibility to ensure there is room at the beginning.
346async fn send_frame<W>(stream: &mut W, hdr: Header, msg: &mut [u8]) -> io::Result<()>
347where
348    W: AsyncWriteExt + Unpin,
349{
350    debug_assert!(msg.len() <= MAX_NOISE_MESSAGE_SIZE);
351    msg[..Header::SIZE].copy_from_slice(&hdr.to_bytes());
352    stream.write_all(msg).await?;
353    Ok(())
354}

383    async fn negotiate(
384        a: &Config,
385        b: &Config,
386    ) -> (Result<(Version, Prologue)>, Result<(Version, Prologue)>) {
387        let listener = TcpListener::bind("127.0.0.1:0").await.unwrap();
388        let port = listener.local_addr().unwrap().port();
389        tokio::join!(
390            async {
391                let mut s = TcpStream::connect(("127.0.0.1", port)).await.unwrap();
392                select_version(a, &mut s, true).await
393            },
394            async {
395                let (mut s, _) = listener.accept().await.unwrap();
396                select_version(b, &mut s, false).await
397            },
398        )
399    }
