51/// The states of a peer.
52///
53/// When a party is created there exists no peer yet. After a connection has
54/// been accepted or a connect attempt succeeded, a peer task is created and
55/// the state transitions to `Connected`. The cancellation token is used if
56/// the peer task should stop in order to replace the peer's connection. When
57/// cancelled the next connection is stored in `Replace` and once the peer
58/// task finishes and returns the peer, its connection is set and a new peer
59/// task (with the same peer object) is spawned and the state transitions to
60/// `Connected` again.
61///
62/// The `Reconnect` state is entered if the peer itself errors and its task
63/// finishes. We keep the peer here until we have a new connection to resume.
64/// Once a connect or accept task finishes we set the connection and spawn
65/// a new peer task again.
66enum PeerState {
67    /// Initial state.
68    None,
69    /// A peer task is running.
70    ///
71    /// The cancellation token can be used to interrupt the peer. The task
72    /// will end and the peer object is returned.
73    Connected(CancellationToken),
74    /// A peer has errored and wants a fresh connection.
75    ///
76    /// We store the peer here until a new connection is available.
77    Reconnect(Peer),
78    /// The server wants to replace a peer's connection.
79    ///
80    /// Only entered from `Connected` after the server has cancelled the
81    /// peer task. While waiting for the task to return the peer object
82    /// we park the connection here.
83    Replace(Connection),
84}

86impl Server {
87    pub(super) fn spawn(
88        conf: Arc<Config>,
89        listener: TcpListener,
90        role: Role,
91        tx: UnboundedSender<PeerMessage>,
92        rx: UnboundedReceiver<Command>,
93        sx: watch::Receiver<Slot>,
94        metrics: Arc<dyn Metrics>,
95    ) -> JoinHandle<()> {
96        let our_key = conf.keypair.public_key();
97        let parties = conf
98            .parties
99            .iter()
100            .filter(|&(k, _)| *k != our_key)
101            .map(|(k, a)| {
102                let p = Party::new(Role::Active, a.clone());
103                (*k, p)
104            })
105            .collect();
106
107        let this = Self {
108            key: our_key,
109            conf,
110            role,
111            ibound: tx,
112            obound: rx,
113            parties,
114            accept_tasks: JoinSet::new(),
115            connect_tasks: JoinMap::new(),
116            hello_tasks: JoinMap::new(),
117            peer_tasks: JoinMap::new(),
118            msgid: MsgId::new(0),
119            next_slot: sx,
120            lower_bound: Slot::MIN,
121            metrics,
122        };
123
124        spawn(this.run(listener))
125    }

127    async fn run(mut self, listener: TcpListener) {
128        // Connect to all peers.
129        for (k, a) in self
130            .parties
131            .iter()
132            .map(|(k, p)| (*k, p.addr.clone()))
133            .collect::<Vec<_>>()
134        {
135            self.spawn_connect(k, a)
136        }
137
138        loop {
139            select! {
140                x = listener.accept() => match x {
141                    Ok((stream, addr)) => {
142                        debug!(
143                            name = %self.conf.name,
144                            node = %self.key,
145                            %addr,
146                            "accepted new tcp connection"
147                        );
148                        self.spawn_accept(stream)
149                    }
150                    Err(err) => {
151                        warn!(
152                            name = %self.conf.name,
153                            node = %self.key,
154                            %err,
155                            "error accepting tcp connection"
156                        )
157                    }
158                },
159
160                Some(h) = self.accept_tasks.join_next() => match h {
161                    Ok(Ok(conn)) => {
162                        self.metrics.set(&self.key, ACCEPT_TASKS, self.accept_tasks.len());
163                        if conn.key == self.key {
164                            warn!(
165                                name = %self.conf.name,
166                                node = %self.key,
167                                peer = %conn.key,
168                                addr = %conn.addr,
169                                "rejecting connection with the same key"
170                            );
171                            self.spawn_hello(conn, Hello::BackOff(Duration::MAX));
172                            continue
173                        }
174                        let Some(party) = self.parties.get_mut(&conn.key) else {
175                            info!(
176                                name = %self.conf.name,
177                                node = %self.key,
178                                peer = %conn.key,
179                                addr = %conn.addr,
180                                "unknown party"
181                            );
182                            self.spawn_hello(conn, Hello::BackOff(self.conf.backoff_duration));
183                            continue
184                        };
185                        if party.ip_addr_mismatch(conn.addr.ip()) {
186                            warn!(
187                                name = %self.conf.name,
188                                node = %self.key,
189                                peer = %conn.key,
190                                addr = %conn.addr,
191                                "party has invalid ip addr"
192                            );
193                            self.spawn_hello(conn, Hello::BackOff(self.conf.backoff_duration));
194                            continue
195                        }
196                        self.spawn_hello(conn, Hello::Ok);
197                    }
198                    Ok(Err(err)) => {
199                        self.metrics.set(&self.key, ACCEPT_TASKS, self.accept_tasks.len());
200                        warn!(name = %self.conf.name, node = %self.key, %err, "handshake failed")
201                    }
202                    Err(err) => {
203                        self.metrics.set(&self.key, ACCEPT_TASKS, self.accept_tasks.len());
204                        if !err.is_cancelled() {
205                            error!(
206                                name = %self.conf.name,
207                                node = %self.key,
208                                %err,
209                                "handshake task panic"
210                            )
211                        }
212                    }
213                },
214
215                Some(r) = self.hello_tasks.join_next() => match r {
216                    (_, Ok(Ok((our_hello, conn, their_hello)))) => {
217                        self.metrics.set(&self.key, HELLO_TASKS, self.hello_tasks.len());
218                        if conn.key == self.key {
219                            // This case has been addressed already by rejecting the peer,
220                            // i.e. we told the peer to backoff forever.
221                            continue
222                        }
223                        let Some(party) = self.parties.get_mut(&conn.key) else {
224                            info!(
225                                name = %self.conf.name,
226                                node = %self.key,
227                                peer = %conn.key,
228                                addr = %conn.addr,
229                                "unknown party"
230                            );
231                            continue
232                        };
233                        if !(our_hello.is_ok() && their_hello.is_ok()) {
234                            warn!(
235                                name   = %self.conf.name,
236                                node   = %self.key,
237                                peer   = %conn.key,
238                                addr   = %conn.addr,
239                                ours   = ?our_hello,
240                                theirs = ?their_hello,
241                                "hello failed"
242                            );
243                            continue
244                        }
245                        match party.peer.take() {
246                            PeerState::None => {
247                                self.connect_tasks.abort(&conn.key);
248                                let key = conn.key;
249                                let peer = Peer::builder()
250                                    .config(self.conf.clone())
251                                    .budget(self.conf.peer_budget)
252                                    .next_slot(self.next_slot.clone())
253                                    .inbound(self.ibound.clone())
254                                    .messages(party.outbox.clone())
255                                    .connection(conn)
256                                    .metrics(self.metrics.clone())
257                                    .build();
258                                party.peer = PeerState::Connected(peer.cancel_token());
259                                self.spawn_peer(key, peer);
260                            }
261                            PeerState::Reconnect(mut peer) => {
262                                self.connect_tasks.abort(&conn.key);
263                                let key = conn.key;
264                                peer.set_connection(conn);
265                                party.peer = PeerState::Connected(peer.cancel_token());
266                                self.spawn_peer(key, peer);
267                            }
268                            PeerState::Connected(cancel) => {
269                                if conn.key > self.key {
270                                    info!(
271                                        name = %self.conf.name,
272                                        node = %self.key,
273                                        peer = %conn.key,
274                                        addr = %conn.addr,
275                                        "replacing connection with accepted one"
276                                    );
277                                    cancel.cancel();
278                                    party.peer = PeerState::Replace(conn);
279                                } else {
280                                    party.peer = PeerState::Connected(cancel);
281                                }
282                            }
283                            PeerState::Replace(_) => {
284                                party.peer = PeerState::Replace(conn);
285                            }
286                        }
287                    }
288                    (key, Ok(Err(err))) => {
289                        self.metrics.set(&self.key, HELLO_TASKS, self.hello_tasks.len());
290                        warn!(
291                            name = %self.conf.name,
292                            node = %self.key,
293                            peer = %key,
294                            %err,
295                            "hello task error"
296                        )
297                    }
298                    (key, Err(err)) => {
299                        self.metrics.set(&self.key, HELLO_TASKS, self.hello_tasks.len());
300                        if !err.is_cancelled() {
301                            error!(
302                                name = %self.conf.name,
303                                node = %self.key,
304                                peer = %key,
305                                %err,
306                                "hello task panic"
307                            )
308                        }
309                    }
310                },
311
312                Some(x) = self.connect_tasks.join_next() => match x {
313                    (_, Ok(conn)) => {
314                        self.metrics.set(&self.key, CONNECT_TASKS, self.connect_tasks.len());
315                        let Some(party) = self.parties.get_mut(&conn.key) else {
316                            debug!(
317                                name = %self.conf.name,
318                                node = %self.key,
319                                peer = %conn.key,
320                                addr = %conn.addr,
321                                "party has been removed"
322                            );
323                            continue
324                        };
325                        match party.peer.take() {
326                            PeerState::None => {
327                                let key = conn.key;
328                                let peer = Peer::builder()
329                                    .config(self.conf.clone())
330                                    .budget(self.conf.peer_budget)
331                                    .next_slot(self.next_slot.clone())
332                                    .inbound(self.ibound.clone())
333                                    .messages(party.outbox.clone())
334                                    .connection(conn)
335                                    .metrics(self.metrics.clone())
336                                    .build();
337                                party.peer = PeerState::Connected(peer.cancel_token());
338                                self.spawn_peer(key, peer);
339                            }
340                            PeerState::Reconnect(mut peer) => {
341                                let key = conn.key;
342                                peer.set_connection(conn);
343                                party.peer = PeerState::Connected(peer.cancel_token());
344                                self.spawn_peer(key, peer);
345                            }
346                            PeerState::Connected(cancel) => {
347                                if conn.key < self.key {
348                                    info!(
349                                        name = %self.conf.name,
350                                        node = %self.key,
351                                        peer = %conn.key,
352                                        addr = %conn.addr,
353                                        "replacing connection with outgoing one"
354                                    );
355                                    cancel.cancel();
356                                    party.peer = PeerState::Replace(conn);
357                                } else {
358                                    party.peer = PeerState::Connected(cancel);
359                                }
360                            }
361                            PeerState::Replace(_) => {
362                                party.peer = PeerState::Replace(conn);
363                            }
364                        }
365                    }
366                    (key, Err(err)) => {
367                        self.metrics.set(&self.key, CONNECT_TASKS, self.connect_tasks.len());
368                        if !err.is_cancelled() {
369                            error!(
370                                name = %self.conf.name,
371                                node = %self.key,
372                                peer = %key,
373                                %err,
374                                "connect task panic"
375                            )
376                        }
377                    }
378                },
379
380                Some(p) = self.peer_tasks.join_next() => match p {
381                    (key, Ok(mut peer)) => {
382                        self.metrics.set(&self.key, PEER_TASKS, self.peer_tasks.len());
383                        if self.ibound.is_closed() {
384                            return
385                        }
386                        let Some(party) = self.parties.get_mut(peer.public_key()) else {
387                            debug!(
388                                name = %self.conf.name,
389                                node = %self.key,
390                                peer = %peer.public_key(),
391                                addr = %peer.socket_addr(),
392                                "party has been removed"
393                            );
394                            continue
395                        };
396                        if let PeerState::Replace(conn) = party.peer.take() {
397                            let key = conn.key;
398                            peer.set_connection(conn);
399                            party.peer = PeerState::Connected(peer.cancel_token());
400                            self.spawn_peer(key, peer);
401                        } else {
402                            let addr = party.addr.clone();
403                            party.peer = PeerState::Reconnect(peer);
404                            self.spawn_connect(key, addr);
405                        }
406                    }
407                    (key, Err(err)) => {
408                        self.metrics.set(&self.key, PEER_TASKS, self.peer_tasks.len());
409                        if !err.is_cancelled() {
410                            error!(
411                                name = %self.conf.name,
412                                node = %self.key,
413                                peer = %key,
414                                %err,
415                                "peer task panic"
416                            );
417                            if self.ibound.is_closed() {
418                                return
419                            }
420                            if let Some(party) = self.parties.get_mut(&key) {
421                                let addr = party.addr.clone();
422                                party.peer = PeerState::None;
423                                self.spawn_connect(key, addr);
424                            }
425                        }
426                    }
427                },
428
429                r = self.next_slot.changed() => {
430                    if r.is_err() {
431                        return
432                    }
433                    let s = *self.next_slot.borrow_and_update();
434                    debug_assert!(s > self.lower_bound); // ensured by controller
435                    self.lower_bound = s;
436                    self.metrics.set(&self.key, LOWER_BOUND, u64::from(s) as usize);
437                    for party in self.parties.values() {
438                        party.outbox.gc(s)
439                    }
440                }
441
442                cmd = self.obound.recv() => {
443                    self.metrics.set(&self.key, CHANNEL_SIZE, self.obound.len());
444                    match cmd {
445                        Some(Command::Peer(PeerCommand::Add(role, parties))) => {
446                            for (k, a) in parties {
447                                if k == self.key {
448                                    self.role = role;
449                                    continue
450                                }
451                                if let Some(p) = self.parties.get_mut(&k) {
452                                    if p.addr == a {
453                                        p.role = role;
454                                    } else {
455                                        info!(
456                                            name = %self.conf.name,
457                                            node = %self.key,
458                                            peer = %k,
459                                            addr = %a,
460                                            "updating party address"
461                                        );
462                                        p.addr = a.clone();
463                                        p.role = role;
464                                        self.connect_tasks.abort(&k);
465                                        if let PeerState::Connected(cancel) = &p.peer {
466                                            cancel.cancel()
467                                        } else {
468                                            self.spawn_connect(k, a)
469                                        }
470                                    }
471                                    continue
472                                }
473                                info!(
474                                    name = %self.conf.name,
475                                    node = %self.key,
476                                    peer = %k,
477                                    addr = %a,
478                                    "adding new peer"
479                                );
480                                self.parties.insert(k, Party::new(role, a.clone()));
481                                self.spawn_connect(k, a)
482                            }
483                        }
484                        Some(Command::Peer(PeerCommand::Remove(peers))) => {
485                            for k in &peers {
486                                if *k == self.key {
487                                    info!(
488                                        name = %self.conf.name,
489                                        node = %self.key,
490                                        "removing self sets role to passive"
491                                    );
492                                    self.role = Role::Passive;
493                                    continue
494                                }
495                                info!(
496                                    name = %self.conf.name,
497                                    node = %self.key,
498                                    peer = %k,
499                                    "removing peer"
500                                );
501                                self.parties.remove(k);
502                                self.connect_tasks.abort(k);
503                                self.peer_tasks.abort(k);
504                            }
505                        }
506                        Some(Command::Peer(PeerCommand::Assign(role, peers))) => {
507                            for k in &peers {
508                                if *k == self.key {
509                                    self.role = role;
510                                    continue
511                                }
512                                if let Some(p) = self.parties.get_mut(k) {
513                                    info!(
514                                        name = %self.conf.name,
515                                        node = %self.key,
516                                        peer = %k,
517                                        %role,
518                                        "assigning role to peer"
519                                    );
520                                    p.role = role
521                                } else {
522                                    warn!(
523                                        name = %self.conf.name,
524                                        node = %self.key,
525                                        peer = %k,
526                                        role = %role,
527                                        "peer to assign role to not found"
528                                    );
529                                }
530                            }
531                        }
532                        Some(Command::Send(cmd)) => match cmd.action {
533                            SendAction::Unicast(to, m) => {
534                                if cmd.slot < self.lower_bound {
535                                    continue
536                                }
537
538                                if to == self.key {
539                                    trace!(name = %self.conf.name, node = %self.key, "sending message");
540                                    if let Err(err) = self.ibound.send((self.key, m.into(), None)) {
541                                        warn!(
542                                            name = %self.conf.name,
543                                            node = %self.key,
544                                            err  = %err,
545                                            "channel closed"
546                                        );
547                                        return
548                                    }
549                                    trace!(name = %self.conf.name, node = %self.key, "message delivered");
550                                    continue
551                                }
552
553                                let msgid = self.next_msgid();
554                                let bytes = append_trailer(cmd.retry, cmd.slot, msgid, m);
555
556                                if let Some(party) = self.parties.get(&to) {
557                                    party.outbox.enqueue(cmd.slot, msgid, (cmd.retry, bytes));
558                                } else {
559                                    warn!(
560                                        name = %self.conf.name,
561                                        node = %self.key,
562                                        peer = %to,
563                                        "unicast target not found"
564                                    );
565                                }
566                            }
567                            SendAction::Multicast(parties, m) => {
568                                if cmd.slot < self.lower_bound {
569                                    continue
570                                }
571
572                                let msgid = self.next_msgid();
573                                let bytes = append_trailer(cmd.retry, cmd.slot, msgid, m);
574
575                                if parties.contains(&self.key) {
576                                    let bytes = remove_trailer(bytes.clone());
577                                    trace!(name = %self.conf.name, node = %self.key, "sending message");
578                                    if let Err(err) = self.ibound.send((self.key, bytes, None)) {
579                                        warn!(
580                                            name = %self.conf.name,
581                                            node = %self.key,
582                                            err  = %err,
583                                            "channel closed"
584                                        );
585                                        return
586                                    }
587                                    trace!(name = %self.conf.name, node = %self.key, "message delivered");
588                                }
589
590                                for (to, party) in &self.parties {
591                                    if !parties.contains(to) {
592                                        continue
593                                    }
594                                    trace!(name = %self.conf.name, node = %self.key, %to, "sending message");
595                                    party.outbox.enqueue(cmd.slot, msgid, (cmd.retry, bytes.clone()));
596                                }
597                            }
598                            SendAction::Broadcast(m) => {
599                                if cmd.slot < self.lower_bound {
600                                    continue
601                                }
602
603                                let msgid = self.next_msgid();
604                                let bytes = append_trailer(cmd.retry, cmd.slot, msgid, m);
605
606                                if self.role.is_active() {
607                                    let bytes = remove_trailer(bytes.clone());
608                                    trace!(name = %self.conf.name, node = %self.key, "sending message");
609                                    if let Err(err) = self.ibound.send((self.key, bytes, None)) {
610                                        warn!(
611                                            name = %self.conf.name,
612                                            node = %self.key,
613                                            err  = %err,
614                                            "channel closed"
615                                        );
616                                        return
617                                    }
618                                    trace!(name = %self.conf.name, node = %self.key, "message delivered");
619                                }
620                                for (key, party) in &self.parties {
621                                    if party.role.is_active() {
622                                        trace!(
623                                            name  = %self.conf.name,
624                                            node  = %self.key,
625                                            to    = %key,
626                                            "sending message"
627                                        );
628                                        party.outbox.enqueue(cmd.slot, msgid, (cmd.retry, bytes.clone()));
629                                    }
630                                }
631                            }
632                        }
633                        Some(Command::Shutdown(tx)) => {
634                            debug!(name = %self.conf.name, node = %self.key, "shutting down");
635                            let _ = tx.send(());
636                            return
637                        }
638                        None => return
639                    }
640                }
641            }
642        }
643    }

645    fn spawn_connect(&mut self, key: PublicKey, addr: NetAddr) {
646        if self.key == key {
647            return;
648        }
649        debug!(
650            name = %self.conf.name,
651            node = %self.key,
652            peer = %key,
653            addr = %addr,
654            "spawning connect task"
655        );
656        let conn = Connection::connect(self.conf.clone(), key, addr);
657        self.connect_tasks.spawn(key, conn);
658        self.metrics.add(&key, CONNECT_ATTEMPTS, 1);
659        self.metrics
660            .set(&self.key, CONNECT_TASKS, self.connect_tasks.len());
661    }
662
663    fn spawn_accept(&mut self, stream: TcpStream) {
664        debug!(name = %self.conf.name, node = %self.key, "spawning accept task");
665        let conn = Connection::accept(self.conf.clone(), stream);
666        self.accept_tasks.spawn(conn);
667        self.metrics
668            .set(&self.key, ACCEPT_TASKS, self.accept_tasks.len());
669    }

671    fn spawn_hello(&mut self, mut conn: Connection, ours: Hello) {
672        debug!(
673            name = %self.conf.name,
674            node = %self.key,
675            peer = %conn.key,
676            addr = %conn.addr,
677            "spawning hello task"
678        );
679
680        self.metrics.add(&conn.key, HELLOS, 1);
681
682        self.hello_tasks.abort(&conn.key);
683        self.hello_tasks.spawn(
684            conn.key,
685            until(self.conf.handshake_timeout, async move {
686                let theirs = conn.recv_hello().await?;
687                conn.send_hello(ours.clone()).await?;
688                Ok::<_, NetworkError>((ours, conn, theirs))
689            }),
690        );
691
692        self.metrics
693            .set(&self.key, HELLO_TASKS, self.hello_tasks.len());
694    }

696    fn spawn_peer(&mut self, key: PublicKey, mut peer: Peer) {
697        debug!(
698            name = %self.conf.name,
699            node = %self.key,
700            peer = %peer.public_key(),
701            addr = %peer.socket_addr(),
702            "spawning peer task"
703        );
704        let node = self.key;
705        let name = self.conf.name.clone();
706        let metrics = self.metrics.clone();
707        self.peer_tasks.spawn(key, async move {
708            let Err(err) = peer.start().await;
709            if !matches!(err, NetworkError::PeerInterrupt) {
710                warn!(
711                    %name,
712                    %node,
713                    peer = %peer.public_key(),
714                    addr = %peer.socket_addr(),
715                    %err,
716                    "peer failure"
717                );
718                metrics.add(peer.public_key(), ERRORS, 1)
719            }
720            peer
721        });
722        self.metrics
723            .set(&self.key, PEER_TASKS, self.peer_tasks.len());
724    }
725
726    fn next_msgid(&mut self) -> MsgId {
727        let current = self.msgid;
728        self.msgid = MsgId::new(self.msgid.0.wrapping_add(1));
729        current
730    }
731}

735        Self {
736            addr: a,

744        let NetAddr::Inet(ip, _) = &self.addr else {
745            return false;
746        };
747        *ip != addr
748    }

757fn append_trailer(pol: RetryPolicy, slot: Slot, id: MsgId, bytes: Vec<u8>) -> Bytes {
758    let t = match pol {
759        RetryPolicy::Default => Trailer::Std { slot, id },
760        RetryPolicy::NoRetry => Trailer::NoAck { slot },
761    };
762    let mut msg = BytesMut::from(Bytes::from(bytes));
763    msg.extend_from_slice(t.to_bytes().as_ref());
764    msg.freeze()
765}
766
767fn remove_trailer(mut bytes: Bytes) -> Bytes {
768    let _t = Trailer::from_bytes(&mut bytes);
769    debug_assert!(_t.is_some());
770    bytes
771}
