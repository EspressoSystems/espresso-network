36/// A peer sends and receives messages over a connection with a remote.
37///
38/// Peers are initialised with a [`Connection`], which can be replaced
39/// later, should it be necessary.
40///
41/// Messages sent are expected to be acknowledged, i.e. the remote needs to
42/// send back an ACK frame. Otherwise the message is resent after some time.
43pub struct Peer {
44    /// Network configuration.
45    conf: Arc<Config>,
46
47    /// A budget limits how many message a peer can deliver to the application.
48    budget: Budget,
49
50    /// The current TCP+Noise connection.
51    conn: Connection,
52
53    /// Messages the application wants to be sent to the remote.
54    msgs: Queue<(RetryPolicy, Bytes)>,
55
56    /// Messages waiting to be retried if no ACK has been received.
57    retry: DelayQueue,
58
59    /// Receive notifications about changes to the GC threshold
60    next_slot: watch::Receiver<Slot>,
61
62    /// Our current GC threshold.
63    lower_bound: Slot,
64
65    /// The channel over which to deliver inbound messages to the application.
66    tx: UnboundedSender<PeerMessage>,
67
68    /// A healthcheck countdown.
69    ///
70    /// When dropped to 0 the connection should be replaced.
71    countdown: Countdown,
72
73    /// Token to interrupt processing.
74    cancel: CancellationToken,
75
76    /// The true max. message size.
77    ///
78    /// It accounts for the additional `Trailer` bytes.
79    max_message_size: usize,
80}
81
82/// A budget limits how many messages a peer can delivery to the application.
83pub struct Budget(Arc<Semaphore>);
84
85impl Budget {
86    pub fn new(amount: NonZeroUsize) -> Self {
87        Self(Arc::new(Semaphore::new(amount.get())))
88    }
89}

139    /// Start I/O with a connected peer.
140    ///
141    /// This method continues until an error occurs, after which callers may
142    /// want to reconnect and resume peer operation with a new `Connection`.
143    pub async fn start(&mut self) -> Result<Empty> {
144        /// Messages are broken into frames and each frame has a header.
145        ///
146        /// The `ReadState` tracks what we have received from the remote.
147        enum ReadState<'a> {
148            Header {
149                off: usize,
150                buf: [u8; Header::SIZE],
151            },
152            Frame {
153                hdr: Header,
154                off: usize,
155                buf: &'a mut Vec<u8>,
156            },
157        }
158
159        /// This state tracks what we have sent to the remote. We currently
160        /// support either data or ACK frames. The offset and length values
161        /// are relative to the write buffer (see below).
162        enum WriteState {
163            /// No write operation is in progress.
164            Idle,
165            /// An ACK frame is sent.
166            Ack { off: usize, len: usize },
167            /// A data frame is sent.
168            Data { off: usize, len: usize },
169        }
170
171        impl WriteState {
172            fn is_idle(&self) -> bool {
173                matches!(self, Self::Idle)
174            }
175
176            /// Encrypt a single data frame with Noise.
177            fn data_frame(
178                data: &[u8],
179                is_partial: bool,
180                state: &mut TransportState,
181                buf: &mut NoiseBuf,
182            ) -> Result<Self> {
183                let n = state.write_message(data, &mut buf[Header::SIZE..])?;
184                let h = if is_partial {
185                    Header::data(n as u16).partial()
186                } else {
187                    Header::data(n as u16)
188                };
189                buf[..Header::SIZE].copy_from_slice(&h.to_bytes());
190                Ok(Self::Data {
191                    off: 0,
192                    len: n + Header::SIZE,
193                })
194            }
195
196            /// Encrypt a single ACK frame with Noise.
197            fn ack_frame(a: Ack, state: &mut TransportState, buf: &mut NoiseBuf) -> Result<Self> {
198                let n = state.write_message(&a.0, &mut buf[Header::SIZE..])?;
199                let h = Header::ack(n as u16);
200                buf[..Header::SIZE].copy_from_slice(&h.to_bytes());
201                Ok(Self::Ack {
202                    off: 0,
203                    len: n + Header::SIZE,
204                })
205            }
206        }
207
208        // Early check if we already got cancelled which can happen with many
209        // simultaneous connects where we need to drop connections.
210        if self.cancel.is_cancelled() {
211            return Err(NetworkError::PeerInterrupt);
212        }
213
214        // Noise packages are limited to 64KiB.
215        //
216        // We retain one such buffer for reading and one for writing.
217        let mut rbuf = NoiseBuf::new([0; _]);
218        let mut wbuf = NoiseBuf::new([0; _]);
219
220        // A frame buffer for reading before it is decrypted.
221        let mut fbuf = Vec::new();
222
223        // An incoming message that is assembled from its frames.
224        let mut ibound_msg = BytesMut::new();
225
226        // An outgoing message. The integer tracks the chunk size as we need
227        // to break the message into frames that fit into a noise package.
228        let mut obound_msg: Option<(RetryPolicy, Bytes, usize)> = None;
229
230        // Pending outbound ACK messages. This is appended when we received a
231        // message and picked up and interleaved when sending frames or else
232        // at the start of the loop (see below).
233        let mut obound_acks: VecDeque<Ack> = VecDeque::new();
234
235        // Track write and read states:
236        let mut wstate = WriteState::Idle;
237        let mut rstate = ReadState::Header {
238            off: 0,
239            buf: [0; _],
240        };
241
242        // Each read needs a permit taken from our budget in order to deliver
243        // to the application.
244        let mut read_permit: Option<OwnedSemaphorePermit> = None;
245
246        // Measure time.
247        //
248        // Used to trigger resends of messages that have not been ACKed yet.
249        let mut clock = interval(Duration::from_secs(1));
250        clock.set_missed_tick_behavior(MissedTickBehavior::Skip);
251
252        // Ensure any previously fired countdown is reset.
253        self.countdown.stop();
254
255        loop {
256            trace!(
257                name     = %self.conf.name,
258                peer     = %self.conn.key,
259                addr     = %self.conn.addr,
260                writing  = %!wstate.is_idle(),
261                acks     = %obound_acks.len(),
262                retries  = %self.retry.len(),
263                can_read = %read_permit.is_some(),
264                "entering event loop"
265            );
266
267            select! {
268                // Wait for the next message if no write is in progress.
269                //
270                // We limit writing if we expect too many ACKs that the remote
271                // has not sent yet. This is to prevent an attack were a
272                // malicious peer never sends ACKs, which would cause our
273                // delay queue to grow unbounded.
274                m = self.msgs.dequeue(), if wstate.is_idle() && self.retry.len() < self.conf.peer_budget.get() => {
275                    trace!(name = %self.conf.name, peer = %self.conn.key, "next outbound message");
276                    let (slot, id, (policy, bytes)) = m;
277                    if policy.is_retry() {
278                        self.retry.add(slot, id, bytes.clone());
279                    }
280                    let chunk = min(bytes.len(), MAX_PAYLOAD_SIZE);
281                    wstate = WriteState::data_frame(
282                        &bytes[..chunk],
283                        chunk < bytes.len(),
284                        &mut self.conn.state,
285                        &mut wbuf
286                    )?;
287                    obound_msg = Some((policy, bytes, chunk))
288                }
289
290                // Pick up an ACK and send it if possible.
291                () = ready(()), if wstate.is_idle() && !obound_acks.is_empty() => {
292                    trace!(name = %self.conf.name, peer = %self.conn.key, "next outbound ack");
293                    let ack = obound_acks.pop_front().expect("obound_acks is not empty");
294                    wstate = WriteState::ack_frame(ack, &mut self.conn.state, &mut wbuf)?
295                }
296
297                // If requested, interrupt all I/O processing.
298                //
299                // Once a peer has been interrupted, its connection needs to be
300                // replaced before calling start again.
301                _ = self.cancel.cancelled() => {
302                    trace!(name = %self.conf.name, peer = %self.conn.key, "interrupt");
303                    return Err(NetworkError::PeerInterrupt)
304                }
305
306                // Wait for a slot change, signaling GC.
307                r = self.next_slot.changed() => {
308                    trace!(name = %self.conf.name, peer = %self.conn.key, "gc");
309                    if r.is_err() {
310                        return Err(NetworkError::ChannelClosed)
311                    }
312                    let s = *self.next_slot.borrow_and_update();
313                    debug_assert!(s > self.lower_bound);
314                    self.lower_bound = s;
315                    self.retry.gc(s);
316                }
317
318                // Check if there are messages that should be re-sent:
319                t = clock.tick(), if !self.retry.is_empty() => {
320                    trace!(name = %self.conf.name, peer = %self.conn.key, "retry check");
321                    self.retry.check(t)
322                }
323
324                // If messages should be re-sent and we can do so, send it:
325                //
326                // This is separate from the retry check above because the check
327                // scans multiple items and here we just take the next one that
328                // is ready and go with it.
329                () = ready(()), if self.retry.is_ready() && wstate.is_idle() => {
330                    trace!(name = %self.conf.name, peer = %self.conn.key, "resending message");
331                    let Some(bytes) = self.retry.next() else {
332                        continue
333                    };
334                    let chunk = min(bytes.len(), MAX_PAYLOAD_SIZE);
335                    wstate = WriteState::data_frame(
336                        &bytes[..chunk],
337                        chunk < bytes.len(),
338                        &mut self.conn.state,
339                        &mut wbuf
340                    )?;
341                    obound_msg = Some((RetryPolicy::Default, bytes, chunk))
342                }
343
344                // Continue an ongoing write operation.
345                r = self.conn.stream.writable(), if !wstate.is_idle() => {
346                    trace!(name = %self.conf.name, peer = %self.conn.key, "continue writing");
347                    if let Err(e) = r {
348                        return Err(e.into())
349                    }
350                    match &mut wstate {
351                        WriteState::Ack { off, len } => {
352                            match self.conn.stream.try_write(&wbuf[*off..*len]) {
353                                Ok(n) => {
354                                    *off += n;
355                                    if *off < *len {
356                                        continue
357                                    }
358                                    if let Some((policy, bytes, chunk)) = &mut obound_msg {
359                                        if *chunk < bytes.len() {
360                                            let end = min(*chunk + MAX_PAYLOAD_SIZE, bytes.len());
361                                            wstate = WriteState::data_frame(
362                                                &bytes[*chunk..end],
363                                                end < bytes.len(),
364                                                &mut self.conn.state,
365                                                &mut wbuf
366                                            )?;
367                                            *chunk = end;
368                                        } else {
369                                            if policy.is_retry() {
370                                                self.countdown.start(self.conf.receive_timeout)
371                                            }
372                                            obound_msg = None;
373                                            wstate = WriteState::Idle;
374                                        }
375                                    } else {
376                                        wstate = WriteState::Idle;
377                                    }
378                                }
379                                Err(e) => if e.kind() != io::ErrorKind::WouldBlock {
380                                    return Err(e.into())
381                                }
382                            }
383                        }
384                        WriteState::Data { off, len } => {
385                            match self.conn.stream.try_write(&wbuf[*off..*len]) {
386                                Ok(n) => {
387                                    *off += n;
388                                    if *off < *len {
389                                        continue
390                                    }
391                                    if let Some(ack) = obound_acks.pop_front() {
392                                        wstate = WriteState::ack_frame(ack, &mut self.conn.state, &mut wbuf)?
393                                    } else if let Some((policy, bytes, chunk)) = &mut obound_msg {
394                                        if *chunk < bytes.len() {
395                                            let end = min(*chunk + MAX_PAYLOAD_SIZE, bytes.len());
396                                            wstate = WriteState::data_frame(
397                                                &bytes[*chunk..end],
398                                                end < bytes.len(),
399                                                &mut self.conn.state,
400                                                &mut wbuf
401                                            )?;
402                                            *chunk = end;
403                                        } else {
404                                            if policy.is_retry() {
405                                                self.countdown.start(self.conf.receive_timeout)
406                                            }
407                                            obound_msg = None;
408                                            wstate = WriteState::Idle;
409                                        }
410                                    } else {
411                                        wstate = WriteState::Idle;
412                                    }
413                                }
414                                Err(e) => if e.kind() != io::ErrorKind::WouldBlock {
415                                    return Err(e.into())
416                                }
417                            }
418                        },
419                        WriteState::Idle => { /* unreachable!() */ }
420                    }
421                }
422
423                // Wait for the healthcheck countdown to finish.
424                //
425                // The countdown is started after writing a message and reset
426                // when we received a frame.
427                () = &mut self.countdown => {
428                    trace!(name = %self.conf.name, peer = %self.conn.key, "read timeout");
429                    return Err(NetworkError::Timeout)
430                }
431
432                // Await the next read permit.
433                //
434                // If our budget is used up, we need to wait for capacity to become
435                // available before we can continue to read from the socket (and
436                // eventually deliver the message to the application).
437                p = self.budget.0.clone().acquire_owned(), if read_permit.is_none() => {
438                    trace!(name = %self.conf.name, peer = %self.conn.key, "next read permit");
439                    read_permit = Some(p.map_err(|_| NetworkError::BudgetClosed)?);
440                }
441
442                // Continue reading from the socket if possible.
443                //
444                // NB that we require the ACKs that we have appended before to
445                // picked up. This should be very fast as writing interleaves
446                // ACKs in between frames. We do this to exercise backpressure
447                // because if the remote does not or can not read what we write
448                // but keeps sending us data we would accumulate ACKs without
449                // bound.
450                r = self.conn.stream.readable(), if read_permit.is_some() => {
451                    trace!(name = %self.conf.name, peer = %self.conn.key, "continue reading");
452                    if let Err(e) = r {
453                        return Err(e.into())
454                    }
455                    if obound_acks.len() > self.conf.peer_budget.get() {
456                        return Err(NetworkError::TooManyPendingAcks(self.conn.key))
457                    }
458                    match &mut rstate {
459                        ReadState::Header { off, buf } => {
460                            match self.conn.stream.try_read(&mut buf[*off..]) {
461                                Ok(0) => {
462                                    let e = io::ErrorKind::UnexpectedEof.into();
463                                    return Err(NetworkError::Io(e))
464                                }
465                                Ok(n) => {
466                                    self.countdown.stop();
467                                    *off += n;
468                                    if *off < buf.len() {
469                                        continue
470                                    }
471                                    let hdr = Header::unvalidated(*buf);
472                                    fbuf.resize(hdr.len().into(), 0);
473                                    rstate = ReadState::Frame { hdr, off: 0, buf: &mut fbuf }
474                                }
475                                Err(e) => if e.kind() != io::ErrorKind::WouldBlock {
476                                    return Err(e.into())
477                                }
478                            }
479                        }
480                        ReadState::Frame { hdr, off, buf } => {
481                            match self.conn.stream.try_read(&mut buf[*off..]) {
482                                Ok(0) => {
483                                    let e = io::ErrorKind::UnexpectedEof.into();
484                                    return Err(NetworkError::Io(e))
485                                }
486                                Ok(n) => {
487                                    self.countdown.stop();
488                                    *off += n;
489                                    if *off < buf.len() {
490                                        continue
491                                    }
492                                    match hdr.frame_type() {
493                                        Ok(FrameType::Data) => {
494                                            let n = self.conn.state.read_message(buf, &mut *rbuf)?;
495                                            ibound_msg.extend_from_slice(&rbuf[..n]);
496                                            if ibound_msg.len() > self.max_message_size {
497                                                return Err(NetworkError::MessageTooLarge)
498                                            }
499                                            if !hdr.is_partial() { // message complete
500                                                let mut msg = mem::take(&mut ibound_msg).freeze();
501                                                let Some(t) = Trailer::from_bytes(&mut msg) else {
502                                                    warn!(
503                                                        name = %self.conf.name,
504                                                        node = %self.conf.keypair.public_key(),
505                                                        peer = %self.conn.key,
506                                                        addr = %self.conn.addr,
507                                                        "invalid trailer"
508                                                    );
509                                                    return Err(NetworkError::InvalidTrailer);
510                                                };
511                                                if t.ty.is_ack() {
512                                                    obound_acks.push_back(Ack::from((t.slot, t.id)));
513                                                }
514                                                if t.slot >= self.lower_bound {
515                                                    let p = read_permit.take();
516                                                    debug_assert!(p.is_some());
517                                                    if self.tx.send((self.conn.key, msg, p)).is_err() {
518                                                        return Err(NetworkError::ChannelClosed)
519                                                    }
520                                                    trace!(
521                                                        name = %self.conf.name,
522                                                        node = %self.conf.keypair.public_key(),
523                                                        peer = %self.conn.key,
524                                                        addr = %self.conn.addr,
525                                                        "message delivered"
526                                                    );
527                                                }
528                                            }
529                                            rstate = ReadState::Header { off: 0, buf: [0; _] };
530                                        }
531                                        Ok(FrameType::Ack) if hdr.is_partial() => {
532                                            return Err(NetworkError::InvalidAck)
533                                        }
534                                        Ok(FrameType::Ack) => {
535                                            let n = self.conn.state.read_message(buf, &mut *rbuf)?;
536                                            let Ok(a) = Ack::try_from(&rbuf[..n]) else {
537                                                return Err(NetworkError::InvalidAck)
538                                            };
539                                            let (s, i) = a.into();
540                                            self.retry.del(s, i);
541                                            rstate = ReadState::Header { off: 0, buf: [0; _] };
542                                        }
543                                        Err(t) => {
544                                            return Err(NetworkError::UnknownFrameType(t))
545                                        }
546                                    }
547                                }
548                                Err(e) => if e.kind() != io::ErrorKind::WouldBlock {
549                                    return Err(e.into())
550                                }
551                            }
552                        }
553                    }
554                }
555            }
556        }
557    }
