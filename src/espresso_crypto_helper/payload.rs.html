18#[repr(C)]
19pub struct DecodingResult {
20    // The operation succeeded
21    pub success: bool,
22    // The error message if the operation failed, otherwise null
23    pub error: *mut c_char,
24    // Pointer to the array of transactions (null on error)
25    pub transactions: *mut Transaction,
26    // Number of transactions in the array
27    pub transactions_len: usize,
28    // Capacity of the transactions array (for freeing)
29    pub transactions_cap: usize,
30}
31
32impl DecodingResult {
33    fn err(msg: &str) -> DecodingResult {
34        let ptr = CString::new(msg)
35            .unwrap_or(c"<invalid error string>".to_owned())
36            .into_raw();
37        DecodingResult {
38            success: false,
39            error: ptr,
40            transactions: std::ptr::null_mut(),
41            transactions_len: 0,
42            transactions_cap: 0,
43        }
44    }
45
46    fn success(transactions: Vec<Transaction>) -> DecodingResult {
47        let len = transactions.len();
48        let cap = transactions.capacity();
49        let ptr = transactions.leak().as_mut_ptr();
50        DecodingResult {
51            success: true,
52            error: std::ptr::null_mut(),
53            transactions: ptr,
54            transactions_len: len,
55            transactions_cap: cap,
56        }
57    }
58}

60#[no_mangle]
61/// Decode a payload from payload bytes and namespace table bytes.
62///
63/// # Safety
64///
65/// payload_ptr and ns_table_ptr must be valid pointers to initialized slices,
66/// valid for the duration of the call. It is okay to pass null pointers if the length is zero.
67pub unsafe extern "C" fn decode_payload(
68    mut payload_ptr: *const u8,
69    payload_len: usize,
70    mut ns_table_ptr: *const u8,
71    ns_table_len: usize,
72) -> DecodingResult {
73    // Pointers must be non-null even for zero-length slices
74    if payload_len == 0 {
75        payload_ptr = std::ptr::dangling();
76    }
77    if ns_table_len == 0 {
78        ns_table_ptr = std::ptr::dangling();
79    }
80
81    // Pointers must be non-null
82    if payload_ptr.is_null() {
83        return DecodingResult::err("Invalid payload pointer or length");
84    }
85    if ns_table_ptr.is_null() {
86        return DecodingResult::err("Invalid ns_table pointer or length");
87    }
88
89    // Safety:
90    //
91    // We have ensured that the pointers are non-null, even for zero-length slices
92    // It's on caller to pass initialized slices and not do anything weird with them
93    // in another thread
94    let payload_bytes = unsafe { std::slice::from_raw_parts(payload_ptr, payload_len) };
95    let ns_table_bytes = unsafe { std::slice::from_raw_parts(ns_table_ptr, ns_table_len) };
96
97    let ns_table = NsTable::from_bytes_unchecked(ns_table_bytes);
98
99    let payload = Payload::from_bytes(payload_bytes, &ns_table);
100    if let Err(err) = ns_table.validate(&payload.byte_len()) {
101        return DecodingResult::err(&format!("Invalid namespace table: {:?}", err));
102    }
103
104    let transactions: Vec<Transaction> = payload
105        .transactions(&ns_table)
106        .map(|tx| {
107            let namespace = tx.namespace().into();
108
109            let tx_payload = tx.into_payload();
110            let payload_ptr = tx_payload.as_ptr();
111            let payload_len = tx_payload.len();
112            let payload_cap = tx_payload.capacity();
113
114            // Leak the payload to prevent it from being freed
115            std::mem::forget(tx_payload);
116
117            Transaction {
118                namespace,
119                payload_ptr,
120                payload_len,
121                payload_cap,
122            }
123        })
124        .collect();
125
126    DecodingResult::success(transactions)
127}

129/// Free a TransactionVecResult that was allocated via FFI.
130/// This function will free:
131/// - The error string if present
132/// - The transactions array
133/// - Each transaction's payload within the array
134///
135/// Caller is responsible for ensuring this is called only on TransactionVecResult allocated
136/// by this library.
137///
138/// # Safety
139/// Caller promised this was allocated by us,
140#[no_mangle]
141pub unsafe extern "C" fn free_transaction_vec_result(result: DecodingResult) {
142    // Free the error string if present
143    if !result.error.is_null() {
144        let _ = CString::from_raw(result.error);
145    }
146
147    // Free the transactions array and payloads if present
148    if !result.transactions.is_null() {
149        // First free each transaction's payload
150        let transactions = Vec::from_raw_parts(
151            result.transactions,
152            result.transactions_len,
153            result.transactions_cap,
154        );
155
156        for transaction in transactions.iter() {
157            if !transaction.payload_ptr.is_null() {
158                let _ = Vec::from_raw_parts(
159                    transaction.payload_ptr as *mut u8,
160                    transaction.payload_len,
161                    transaction.payload_cap,
162                );
163            }
164        }
165
166        // The transactions vector itself is dropped when it goes out of scope
167    }
