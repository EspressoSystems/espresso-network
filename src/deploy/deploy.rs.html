22/// Deploy contracts needed to run the sequencer.
23///
24/// This script deploys contracts needed to run the sequencer to an L1. It outputs a .env file
25/// containing the addresses of the deployed contracts.
26///
27/// This script can also be used to do incremental deployments. The only contract addresses
28/// needed to configure the sequencer network are ESPRESSO_SEQUENCER_FEE_CONTRACT_PROXY_ADDRESS,
29/// ESPRESSO_SEQUENCER_LIGHT_CLIENT_PROXY_ADDRESS and ESPRESSO_SEQUENCER_STAKE_TABLE_ADDRESS.
30/// These contracts, however, have dependencies, and a full deployment involves several
31/// contracts. Some of these contracts, especially libraries may already have been deployed, or
32/// perhaps one of the top-level contracts has been deployed and we only need to deploy the other
33/// one.
34///
35/// It is possible to pass in the addresses of already deployed contracts, in which case those
36/// addresses will be used in place of deploying a new contract wherever that contract is required
37/// in the deployment process. The generated .env file will include all the addresses passed in as
38/// well as those newly deployed.

256#[tokio::main]
257async fn main() -> anyhow::Result<()> {
258    let opt = Options::parse();
259
260    opt.logging.init();
261
262    let mut contracts = Contracts::from(opt.contracts);
263    let provider = if opt.ledger {
264        let signer = connect_ledger(opt.account_index as usize).await?;
265        tracing::info!("Using ledger for signing, watch ledger device for prompts.");
266        build_provider_ledger(signer, opt.rpc_url.clone(), Some(opt.l1_polling_interval))
267    } else {
268        build_provider(
269            opt.mnemonic
270                .expect("Mnemonic provided when not using ledger"),
271            opt.account_index,
272            opt.rpc_url.clone(),
273            Some(opt.l1_polling_interval),
274        )
275    };
276
277    let account = provider.default_signer_address();
278    if let Some(command) = &opt.command {
279        match command {
280            Command::Account => {
281                println!("{account}");
282                return Ok(());
283            },
284            Command::Balance => {
285                let balance = provider.get_balance(account).await?;
286                println!("{account}: {} Eth", format_ether(balance));
287                return Ok(());
288            },
289            Command::VerifyNodeJsFiles => {
290                verify_node_js_files().await?;
291                return Ok(());
292            },
293        };
294    };
295
296    // No subcommand specified. Deploy contracts.
297
298    let balance = provider.get_balance(account).await?;
299    tracing::info!(
300        "Using deployer account {account} with balance: {}",
301        format_ether(balance),
302    );
303    if balance.is_zero() {
304        anyhow::bail!(
305            "account_index {}, address={account} has no balance. A funded account is required.",
306            opt.account_index
307        );
308    }
309
310    // First use builder to build constructor input arguments
311    let mut args_builder = DeployerArgsBuilder::default();
312    args_builder
313        .deployer(provider)
314        .mock_light_client(opt.use_mock)
315        .use_multisig(opt.use_multisig)
316        .dry_run(opt.dry_run)
317        .rpc_url(opt.rpc_url.to_string());
318    if let Some(multisig) = opt.multisig_address {
319        args_builder.multisig(multisig);
320    }
321    if let Some(multisig_pauser) = opt.multisig_pauser_address {
322        args_builder.multisig_pauser(multisig_pauser);
323    }
324    if let Some(token_recipient) = opt.initial_token_grant_recipient {
325        args_builder.token_recipient(token_recipient);
326    }
327
328    if let Some(blocks_per_epoch) = opt.blocks_per_epoch {
329        args_builder.blocks_per_epoch(blocks_per_epoch);
330    }
331    if let Some(epoch_start_block) = opt.epoch_start_block {
332        args_builder.epoch_start_block(epoch_start_block);
333    }
334
335    if opt.deploy_light_client_v1 {
336        let (genesis_state, genesis_stake) = if opt.mock_espresso_live_network {
337            light_client_genesis_from_stake_table(&Default::default(), DEFAULT_STAKE_TABLE_CAPACITY)
338                .unwrap()
339        } else {
340            light_client_genesis(&opt.sequencer_url, opt.stake_table_capacity).await?
341        };
342        args_builder
343            .genesis_lc_state(genesis_state)
344            .genesis_st_state(genesis_stake);
345        if let Some(prover) = opt.permissioned_prover {
346            args_builder.permissioned_prover(prover);
347        }
348    }
349    if opt.upgrade_light_client_v2 {
350        let (blocks_per_epoch, epoch_start_block) =
351            if (opt.dry_run && opt.use_multisig) || opt.mock_espresso_live_network {
352                (10, 22)
353            } else {
354                // fetch epoch length from HotShot config
355                // Request the configuration until it is successful
356                loop {
357                    match surf_disco::Client::<ServerError, StaticVersion<0, 1>>::new(
358                        opt.sequencer_url.clone(),
359                    )
360                    .get::<PublicNetworkConfig>("config/hotshot")
361                    .send()
362                    .await
363                    {
364                        Ok(resp) => {
365                            let config = resp.hotshot_config();
366                            break (config.blocks_per_epoch(), config.epoch_start_block());
367                        },
368                        Err(e) => {
369                            tracing::error!("Failed to fetch the network config: {e}");
370                            sleep(Duration::from_secs(5));
371                        },
372                    }
373                }
374            };
375        args_builder.blocks_per_epoch(blocks_per_epoch);
376        args_builder.epoch_start_block(epoch_start_block);
377    }
378
379    if opt.deploy_stake_table {
380        if let Some(escrow_period) = opt.exit_escrow_period {
381            args_builder.exit_escrow_period(U256::from(escrow_period.as_secs()));
382        }
383    }
384    if opt.deploy_timelock {
385        let timelock_admin = opt
386            .timelock_admin
387            .expect("Must provide --timelock-admin when deploying timelock");
388        args_builder.timelock_admin(timelock_admin);
389        let timelock_delay = opt
390            .timelock_delay
391            .expect("Must provide --timelock-delay when deploying timelock");
392        args_builder.timelock_delay(U256::from(timelock_delay));
393        let timelock_executors = opt
394            .timelock_executors
395            .expect("Must provide --timelock-executors when deploying timelock");
396        args_builder.timelock_executors(timelock_executors.into_iter().collect());
397        let timelock_proposers = opt
398            .timelock_proposers
399            .expect("Must provide --timelock-proposers when deploying timelock");
400        args_builder.timelock_proposers(timelock_proposers.into_iter().collect());
401    }
402
403    // then deploy specified contracts
404    let args = args_builder.build()?;
405    if opt.deploy_fee {
406        args.deploy(&mut contracts, Contract::FeeContractProxy)
407            .await?;
408    }
409    if opt.deploy_esp_token {
410        args.deploy(&mut contracts, Contract::EspTokenProxy).await?;
411    }
412    if opt.deploy_light_client_v1 {
413        args.deploy(&mut contracts, Contract::LightClientProxy)
414            .await?;
415    }
416    if opt.upgrade_light_client_v2 {
417        args.deploy(&mut contracts, Contract::LightClientV2).await?;
418    }
419    if opt.deploy_stake_table {
420        args.deploy(&mut contracts, Contract::StakeTableProxy)
421            .await?;
422    }
423    if opt.upgrade_stake_table_v2 {
424        args.deploy(&mut contracts, Contract::StakeTableV2).await?;
425    }
426    if opt.deploy_timelock {
427        args.deploy(&mut contracts, Contract::Timelock).await?;
428    }
429
430    // finally print out or persist deployed addresses
431    if let Some(out) = &opt.out {
432        let file = File::options()
433            .create(true)
434            .truncate(true)
435            .write(true)
436            .open(out)?;
437        contracts.write(file)?;
438    } else {
439        contracts.write(stdout())?;
440    }
441
442    Ok(())
443}
