73#[async_trait]
74impl<Types> Fetchable<Types> for LeafQueryData<Types>
75where
76    Types: NodeType,
77    Header<Types>: QueryableHeader<Types>,
78    Payload<Types>: QueryablePayload<Types>,
79{
80    type Request = LeafId<Types>;
81
82    fn satisfies(&self, req: Self::Request) -> bool {
83        match req {
84            LeafId::Number(n) => self.height() == n as u64,
85            LeafId::Hash(h) => self.hash() == h,
86        }
87    }
88
89    async fn passive_fetch(
90        notifiers: &Notifiers<Types>,
91        req: Self::Request,
92    ) -> BoxFuture<'static, Option<Self>> {
93        notifiers
94            .leaf
95            .wait_for(move |leaf| leaf.satisfies(req))
96            .await
97            .into_future()
98            .boxed()
99    }
100
101    async fn active_fetch<S, P>(
102        tx: &mut impl AvailabilityStorage<Types>,
103        fetcher: Arc<Fetcher<Types, S, P>>,
104        req: Self::Request,
105    ) -> anyhow::Result<()>
106    where
107        S: VersionedDataSource + 'static,
108        for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
109        for<'a> S::ReadOnly<'a>:
110            AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
111        P: AvailabilityProvider<Types>,
112    {
113        fetch_leaf_with_callbacks(tx, fetcher, req, None).await
114    }
115
116    async fn load<S>(storage: &mut S, req: Self::Request) -> QueryResult<Self>
117    where
118        S: AvailabilityStorage<Types>,
119    {
120        storage.get_leaf(req).await
121    }

124pub(super) async fn fetch_leaf_with_callbacks<Types, S, P, I>(
125    tx: &mut impl AvailabilityStorage<Types>,
126    fetcher: Arc<Fetcher<Types, S, P>>,
127    req: LeafId<Types>,
128    callbacks: I,
129) -> anyhow::Result<()>
130where
131    Types: NodeType,
132    Header<Types>: QueryableHeader<Types>,
133    Payload<Types>: QueryablePayload<Types>,
134    S: VersionedDataSource + 'static,
135    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
136    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
137    P: AvailabilityProvider<Types>,
138    I: IntoIterator<Item = LeafCallback<Types, S, P>> + Send + 'static,
139    I::IntoIter: Send,
140{
141    match req {
142        LeafId::Number(n) => {
143            // We need the next leaf in the chain so we can figure out what hash we expect for this
144            // leaf, so we can fetch it securely from an untrusted provider.
145            let next = (n + 1) as u64;
146            let next = match tx.first_available_leaf(next).await {
147                Ok(leaf) if leaf.height() == next => leaf,
148                Ok(leaf) => {
149                    // If we don't have the immediate successor leaf, but we have some later leaf,
150                    // then we can't trigger this exact fetch, but we can fetch the (apparently)
151                    // missing parent of the leaf we do have, which will trigger a chain of fetches
152                    // that eventually reaches all the way back to the desired leaf.
153                    tracing::debug!(
154                        n,
155                        fetching = leaf.height() - 1,
156                        "do not have necessary leaf; trigger fetch of a later leaf"
157                    );
158
159                    let mut callbacks = vec![LeafCallback::Leaf {
160                        fetcher: fetcher.clone(),
161                    }];
162
163                    if !fetcher.leaf_only {
164                        callbacks.push(
165                            HeaderCallback::Payload {
166                                fetcher: fetcher.clone(),
167                            }
168                            .into(),
169                        );
170                        callbacks.push(
171                            HeaderCallback::VidCommon {
172                                fetcher: fetcher.clone(),
173                            }
174                            .into(),
175                        );
176                    }
177
178                    let parent_qc = leaf.leaf().justify_qc();
179                    fetcher.leaf_fetcher.clone().spawn_fetch(
180                        request::LeafRequest::new(
181                            leaf.height() - 1,
182                            leaf.leaf().parent_commitment(),
183                            parent_qc.view_number(),
184                            parent_qc.data,
185                        ),
186                        fetcher.provider.clone(),
187                        // After getting the leaf, grab the other data as well; that will be missing
188                        // whenever the leaf was.
189                        callbacks,
190                    );
191                    return Ok(());
192                },
193                Err(QueryError::Missing | QueryError::NotFound) => {
194                    // We successfully queried the database, but the next leaf wasn't there. We
195                    // know for sure that based on the current state of the DB, we cannot fetch this
196                    // leaf.
197                    tracing::debug!(n, "not fetching leaf with unknown successor");
198                    return Ok(());
199                },
200                Err(QueryError::Error { message }) => {
201                    // An error occurred while querying the database. We don't know if we need to
202                    // fetch the leaf or not. Return an error so we can try again.
203                    bail!("failed to fetch successor for leaf {n}: {message}");
204                },
205            };
206
207            let fetcher = fetcher.clone();
208            let parent_qc = next.leaf().justify_qc();
209            fetcher.leaf_fetcher.clone().spawn_fetch(
210                request::LeafRequest::new(
211                    n as u64,
212                    next.leaf().parent_commitment(),
213                    parent_qc.view_number(),
214                    parent_qc.data,
215                ),
216                fetcher.provider.clone(),
217                once(LeafCallback::Leaf { fetcher }).chain(callbacks),
218            );
219        },
220        LeafId::Hash(h) => {
221            // We don't actively fetch leaves when requested by hash, because we have no way of
222            // knowing whether a leaf with such a hash actually exists, and we don't want to bother
223            // peers with requests for non-existent leaves.
224            tracing::debug!("not fetching unknown leaf {h}");
225        },
226    }
227
228    Ok(())
229}

231/// Trigger a fetch of the parent of the given `leaf`, if it is missing.
232///
233/// Leaves have a unique constraint among fetchable objects: we cannot fetch a given leaf at height
234/// `h` unless we have its child at height `h + 1`. This is because the child, through its
235/// `parent_commitment`, tells us what the hash of the parent should be, which lets us authenticate
236/// it when fetching from an untrusted provider. Thus, requests for leaf `h` might block if `h + 1`
237/// is not available. To ensure all these requests are eventually unblocked, and all leaves are
238/// eventually fetched, we call this function whenever we receive leaf `h + 1` to check if we need
239/// to then fetch leaf `h`.
240pub(super) fn trigger_fetch_for_parent<Types, S, P>(
241    fetcher: &Arc<Fetcher<Types, S, P>>,
242    leaf: &LeafQueryData<Types>,
243) where
244    Types: NodeType,
245    Header<Types>: QueryableHeader<Types>,
246    Payload<Types>: QueryablePayload<Types>,
247    S: VersionedDataSource + 'static,
248    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
249    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
250    P: AvailabilityProvider<Types>,
251{
252    let height = leaf.height();
253    let parent = leaf.leaf().parent_commitment();
254    let parent_qc = leaf.leaf().justify_qc();
255    let parent_qc_view = parent_qc.view_number();
256    let parent_qc_data = parent_qc.data;
257
258    // Check that there is a parent to fetch.
259    if height == 0 {
260        return;
261    }
262
263    // Spawn an async task; we're triggering a fire-and-forget fetch of a leaf that might now be
264    // available; we don't need to block the caller on this.
265    let fetcher = fetcher.clone();
266    let span = tracing::info_span!("fetch parent leaf", height, %parent, %parent_qc_view);
267    spawn(
268        async move {
269            // Check if we already have the parent.
270            match fetcher.storage.read().await {
271                Ok(mut tx) => {
272                    // Don't bother fetching a pruned leaf.
273                    if let Ok(pruned_height) = tx.load_pruned_height().await
274                        && pruned_height.is_some_and(|ph| height <= ph)
275                    {
276                        tracing::info!(height, ?pruned_height, "not fetching pruned parent leaf");
277                        return;
278                    }
279
280                    if tx.get_leaf(((height - 1) as usize).into()).await.is_ok() {
281                        return;
282                    }
283                },
284                Err(err) => {
285                    // If we can't open a transaction, we can't be sure that we already have the
286                    // parent, so we fall through to fetching it just to be safe.
287                    tracing::warn!(
288                        height,
289                        %parent,
290                        "error opening transaction to check for parent leaf: {err:#}",
291                    );
292                },
293            }
294
295            tracing::info!(height, %parent, "received new leaf; fetching missing parent");
296            fetcher.leaf_fetcher.clone().spawn_fetch(
297                request::LeafRequest::new(height - 1, parent, parent_qc_view, parent_qc_data),
298                fetcher.provider.clone(),
299                // After getting the leaf, grab the other data as well; that will be missing
300                // whenever the leaf was.
301                [
302                    LeafCallback::Leaf {
303                        fetcher: fetcher.clone(),
304                    },
305                    HeaderCallback::Payload {
306                        fetcher: fetcher.clone(),
307                    }
308                    .into(),
309                    HeaderCallback::VidCommon {
310                        fetcher: fetcher.clone(),
311                    }
312                    .into(),
313                ],
314            );
315        }
316        .instrument(span),
317    );
318}

359#[derive(Derivative, From)]
360#[derivative(Debug(bound = ""))]
361pub(super) enum LeafCallback<Types: NodeType, S, P> {
362    /// Callback when fetching the leaf for its own sake.
363    #[from(ignore)]
364    Leaf {
365        #[derivative(Debug = "ignore")]
366        fetcher: Arc<Fetcher<Types, S, P>>,
367    },
368    /// Callback when fetching the leaf in order to then look up something else.
369    Continuation {
370        callback: HeaderCallback<Types, S, P>,
371    },
372}
373
374impl<Types: NodeType, S, P> PartialEq for LeafCallback<Types, S, P> {
375    fn eq(&self, other: &Self) -> bool {
376        self.cmp(other).is_eq()
377    }
378}
379
380impl<Types: NodeType, S, P> Eq for LeafCallback<Types, S, P> {}
381
382impl<Types: NodeType, S, P> Ord for LeafCallback<Types, S, P> {
383    fn cmp(&self, other: &Self) -> Ordering {
384        match (self, other) {
385            // Store leaves in the database before storing derived objects.
386            (Self::Leaf { .. }, Self::Continuation { .. }) => Ordering::Less,
387            (Self::Continuation { .. }, Self::Leaf { .. }) => Ordering::Greater,
388
389            (Self::Continuation { callback: cb1 }, Self::Continuation { callback: cb2 }) => {
390                cb1.cmp(cb2)
391            },
392            _ => Ordering::Equal,
393        }
394    }
395}
396
397impl<Types: NodeType, S, P> PartialOrd for LeafCallback<Types, S, P> {
398    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
399        Some(self.cmp(other))
400    }
401}

403impl<Types: NodeType, S, P> Callback<LeafQueryData<Types>> for LeafCallback<Types, S, P>
404where
405    Header<Types>: QueryableHeader<Types>,
406    Payload<Types>: QueryablePayload<Types>,
407    S: VersionedDataSource + 'static,
408    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
409    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
410    P: AvailabilityProvider<Types>,
411{
412    async fn run(self, leaf: LeafQueryData<Types>) {
413        match self {
414            Self::Leaf { fetcher } => {
415                tracing::info!("fetched leaf {}", leaf.height());
416                // Trigger a fetch of the parent leaf, if we don't already have it.
417                trigger_fetch_for_parent(&fetcher, &leaf);
418                fetcher.store_and_notify(&leaf).await;
419            },
420            Self::Continuation { callback } => callback.run(leaf.leaf.block_header().clone()),
421        }
422    }
423}

425impl<Types: NodeType, S, P> Callback<NonEmptyRange<LeafQueryData<Types>>>
426    for LeafCallback<Types, S, P>
427where
428    Header<Types>: QueryableHeader<Types>,
429    Payload<Types>: QueryablePayload<Types>,
430    S: VersionedDataSource + 'static,
431    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
432    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
433    P: AvailabilityProvider<Types>,
434{
435    async fn run(self, leaves: NonEmptyRange<LeafQueryData<Types>>) {
436        match self {
437            Self::Leaf { fetcher } => {
438                tracing::info!("fetched leaf range {}..{}", leaves.start(), leaves.end());
439                fetcher.store_and_notify(&leaves).await;
440
441                // Unlike in the singular leaf version of this callback, we do not call
442                // `trigger_fetch_for_parent` to start a potential chain reaction for a
443                // contiguous range of leaves. This is because we have already just fetched a
444                // contiguous range, and if we are currently bulk fetching, it is more efficient to
445                // continue bulk fetching, rather than kick of a chain reaction of individual
446                // fetches, which will end up fetching the same data, slower.
447            },
448            Self::Continuation { callback } => callback.run_range(leaves.as_ref_cloned()),
449        }
450    }
451}

453/// A request for a semi-open range of height-indexed objects.
454#[derive(Clone, Copy, Debug)]
455pub struct RangeRequest {
456    pub start: u64,
457    pub end: u64,
458}
459
460impl IntoIterator for RangeRequest {
461    type Item = u64;
462    type IntoIter = Range<u64>;
463
464    fn into_iter(self) -> Self::IntoIter {
465        self.start..self.end
466    }
467}
468
469impl FetchRequest for RangeRequest {
470    fn might_exist(self, heights: Heights) -> bool {
471        heights.pruned_height.is_none_or(|h| h < self.start) && self.end < heights.height
472    }
473}
474
475impl RangeRequest {
476    pub fn is_satisfied(&self, range: &NonEmptyRange<impl HeightIndexed>) -> bool {
477        range.start() == self.start && range.end() == self.end
478    }
479
480    pub fn len(&self) -> usize {
481        (self.end - self.start) as usize
482    }
483}

485#[async_trait]
486impl<Types> Fetchable<Types> for NonEmptyRange<LeafQueryData<Types>>
487where
488    Types: NodeType,
489    Header<Types>: QueryableHeader<Types>,
490    Payload<Types>: QueryablePayload<Types>,
491{
492    type Request = RangeRequest;
493
494    fn satisfies(&self, req: Self::Request) -> bool {
495        req.is_satisfied(self)
496    }
497
498    async fn passive_fetch(
499        notifiers: &Notifiers<Types>,
500        req: Self::Request,
501    ) -> BoxFuture<'static, Option<Self>> {
502        let waits = join_all(req.into_iter().map(|i| {
503            notifiers
504                .leaf
505                .wait_for(move |leaf| leaf.satisfies(LeafId::Number(i as usize)))
506        }))
507        .await;
508
509        join_all(waits.into_iter().map(|wait| wait.into_future()))
510            .map(|options| NonEmptyRange::new(options.into_iter().flatten()).ok())
511            .boxed()
512    }
513
514    async fn active_fetch<S, P>(
515        tx: &mut impl AvailabilityStorage<Types>,
516        fetcher: Arc<Fetcher<Types, S, P>>,
517        req: Self::Request,
518    ) -> anyhow::Result<()>
519    where
520        S: VersionedDataSource + 'static,
521        for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
522        for<'a> S::ReadOnly<'a>:
523            AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
524        P: AvailabilityProvider<Types>,
525    {
526        fetch_leaf_range_with_callbacks(tx, fetcher, req, None).await
527    }
528
529    async fn load<S>(storage: &mut S, req: Self::Request) -> QueryResult<Self>
530    where
531        S: AvailabilityStorage<Types>,
532    {
533        let leaves = storage
534            .get_leaf_range((req.start as usize)..(req.end as usize))
535            .await?
536            .into_iter()
537            .collect::<QueryResult<Vec<_>>>()?;
538        if leaves.len() != req.len() {
539            tracing::debug!(
540                ?req,
541                len = leaves.len(),
542                "database returned partial result, unable to load full range"
543            );
544            return Err(QueryError::Missing);
545        }
546        NonEmptyRange::new(leaves).map_err(|err| QueryError::Error {
547            message: format!("expected contiguous range, but: {err:#}"),
548        })
549    }
550}

575pub(super) async fn fetch_leaf_range_with_callbacks<Types, S, P, I>(
576    tx: &mut impl AvailabilityStorage<Types>,
577    fetcher: Arc<Fetcher<Types, S, P>>,
578    req: RangeRequest,
579    callbacks: I,
580) -> anyhow::Result<()>
581where
582    Types: NodeType,
583    Header<Types>: QueryableHeader<Types>,
584    Payload<Types>: QueryablePayload<Types>,
585    S: VersionedDataSource + 'static,
586    for<'a> S::Transaction<'a>: UpdateAvailabilityStorage<Types>,
587    for<'a> S::ReadOnly<'a>: AvailabilityStorage<Types> + NodeStorage<Types> + PrunedHeightStorage,
588    P: AvailabilityProvider<Types>,
589    I: IntoIterator<Item = LeafCallback<Types, S, P>> + Send + 'static,
590    I::IntoIter: Send,
591{
592    // We need the next leaf after the chain so we can figure out what hash we expect for the last
593    // leaf in the chain, so we can fetch it securely from an untrusted provider.
594    let next = match tx.first_available_leaf(req.end).await {
595        Ok(leaf) if leaf.height() == req.end => leaf,
596        Ok(leaf) => {
597            // If we don't have the immediate successor leaf, but we have some later leaf,
598            // then we can't trigger this exact fetch, but we can fetch the (apparently)
599            // missing parent of the leaf we do have, which will trigger a chain of fetches
600            // that eventually reaches all the way back to the desired leaf.
601            tracing::debug!(
602                req.end,
603                fetching = leaf.height() - 1,
604                "do not have necessary leaf; trigger fetch of a later leaf"
605            );
606
607            let mut callbacks = vec![LeafCallback::Leaf {
608                fetcher: fetcher.clone(),
609            }];
610
611            if !fetcher.leaf_only {
612                callbacks.push(
613                    HeaderCallback::Payload {
614                        fetcher: fetcher.clone(),
615                    }
616                    .into(),
617                );
618                callbacks.push(
619                    HeaderCallback::VidCommon {
620                        fetcher: fetcher.clone(),
621                    }
622                    .into(),
623                );
624            }
625
626            let parent_qc = leaf.leaf().justify_qc();
627            fetcher.leaf_fetcher.clone().spawn_fetch(
628                request::LeafRequest::new(
629                    leaf.height() - 1,
630                    leaf.leaf().parent_commitment(),
631                    parent_qc.view_number(),
632                    parent_qc.data,
633                ),
634                fetcher.provider.clone(),
635                // After getting the leaf, grab the other data as well; that will be missing
636                // whenever the leaf was.
637                callbacks,
638            );
639            return Ok(());
640        },
641        Err(QueryError::Missing | QueryError::NotFound) => {
642            // We successfully queried the database, but the next leaf wasn't there. We know for
643            // sure that based on the current state of the DB, we cannot fetch this leaf.
644            tracing::debug!(req.end, "not fetching leaf chain with unknown successor");
645            return Ok(());
646        },
647        Err(QueryError::Error { message }) => {
648            // An error occurred while querying the database. We don't know if we need to fetch the
649            // leaf or not. Return an error so we can try again.
650            bail!(
651                "failed to fetch successor for leaf chain {}: {message}",
652                req.end
653            );
654        },
655    };
656
657    let fetcher = fetcher.clone();
658    fetcher.leaf_range_fetcher.clone().spawn_fetch(
659        LeafRangeRequest {
660            start: req.start,
661            end: req.end,
662            last_leaf: next.leaf().parent_commitment(),
663            last_qc_view: next.leaf().justify_qc().view_number(),
664            last_qc_data: next.leaf().justify_qc().data,
665        },
666        fetcher.provider.clone(),
667        once(LeafCallback::Leaf { fetcher }).chain(callbacks),
668    );
669
670    Ok(())
