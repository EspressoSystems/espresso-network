10impl AvidMTxProof {
11    pub fn new(
12        index: &Index,
13        payload: &Payload,
14        common: &AvidMCommon,
15    ) -> Option<(Transaction, Self)> {
16        let ns_index = &index.ns_index;
17        let tx_index = &TxIndex(index.position as usize);
18
19        let payload_byte_len = payload.byte_len();
20        if !payload.ns_table().in_bounds(ns_index) {
21            tracing::warn!("ns_index {:?} out of bounds", ns_index);
22            return None; // error: ns index out of bounds
23        }
24        // check tx index below
25
26        let ns_range = payload.ns_table().ns_range(ns_index, &payload_byte_len);
27        let ns_byte_len = ns_range.byte_len();
28        let ns_payload = payload.read_ns_payload(&ns_range);
29        let ns_proof = AvidMNsProof::new(payload, ns_index, common)?;
30
31        // Read the tx table len from this namespace's tx table and compute a
32        // proof of correctness.
33        let num_txs_range = NumTxsRange::new(&ns_byte_len);
34        let payload_num_txs = ns_payload.read(&num_txs_range);
35
36        // Check tx index.
37        //
38        // TODO the next line of code (and other code) could be easier to read
39        // if we make a helpers that repeat computation we've already done.
40        if !NumTxs::new(&payload_num_txs, &ns_byte_len).in_bounds(tx_index) {
41            return None; // error: tx index out of bounds
42        }
43
44        // Read the tx table entries for this tx and compute a proof of
45        // correctness.
46        let tx_table_entries_range = TxTableEntriesRange::new(tx_index);
47        let payload_tx_table_entries = ns_payload.read(&tx_table_entries_range);
48
49        // Read the tx payload and compute a proof of correctness.
50        let tx_payload_range =
51            TxPayloadRange::new(&payload_num_txs, &payload_tx_table_entries, &ns_byte_len);
52
53        let tx = {
54            let ns_id = payload.ns_table().read_ns_id_unchecked(ns_index);
55            let tx_payload = ns_payload
56                .read(&tx_payload_range)
57                .to_payload_bytes()
58                .to_vec();
59            Transaction::new(ns_id, tx_payload)
60        };
61
62        Some((
63            tx,
64            AvidMTxProof {
65                tx_index: tx_index.clone(),
66                ns_proof,
67            },
68        ))
69    }

71    pub fn verify(
72        &self,
73        ns_table: &NsTable,
74        tx: &Transaction,
75        commit: &VidCommitment,
76        common: &VidCommon,
77    ) -> Option<bool> {
78        let VidCommon::V1(common) = common else {
79            tracing::info!("VID version mismatch");
80            return None;
81        };
82
83        let (txs, _) = self.ns_proof.verify(ns_table, commit, common)?;
84        if self.tx_index.0 > txs.len() {
85            return Some(false);
86        }
87        Some(&txs[self.tx_index.0] == tx)
88    }
