43// TODO: remove this once we no longer have a conflict due to `alloy` and `ark` `rand` version.
44trait RandomCompat: Sized {
45    fn random_with(rng: &mut impl RngCore) -> Self;
46}
47
48impl RandomCompat for U256 {
49    fn random_with(rng: &mut impl RngCore) -> Self {
50        let mut bytes = [0u8; 32];
51        rng.fill(&mut bytes);
52        U256::from_be_bytes(bytes)
53    }
54}
55
56impl RandomCompat for Address {
57    fn random_with(rng: &mut impl RngCore) -> Self {
58        let mut bytes = [0u8; 20];
59        rng.fill(&mut bytes);
60        Address::from(bytes)
61    }
62}

66struct Cli {

82    EvalDataGen,

84    TranscriptAppendMsg,
85    /// Get jf_plonk::Transcript::append_challenge()
86    TranscriptAppendField,
87    /// Get jf_plonk::Transcript::append_commitment()
88    TranscriptAppendGroup,
89    /// Get jf_plonk::Transcript::get_and_append_challenge()
90    TranscriptGetChal,
91    /// Get jf_plonk::Transcript::append_vk_and_pub_input()
92    TranscriptAppendVkAndPi,
93    /// Get jf_plonk::Transcript::append_proof_evaluations()
94    TranscriptAppendProofEvals,
95    /// Return the Plonk Verifier related constants
96    PlonkConstants,
97    /// Get jf_plonk::Verifier::compute_challenges()
98    PlonkComputeChal,
99    /// Get jf_plonk::Verifier::batch_verify()
100    PlonkVerify,

102    DummyProof,
103    /// Test only logic
104    TestOnly,
105    /// Generate Client Wallet
106    GenClientWallet,
107    /// Generate internal hash values for the BLS signature scheme
108    GenBLSHashes,

123    /// Generate reward claim test fixtures with configurable accounts and seeds
124    GenRewardFixture,

674        Action::EvolveRewardState => {
675            if cli.args.len() != 6 {
676                panic!(
677                    "Should provide arg1=currentState, arg2=seed, arg3=numAccountsToAdd, \
678                     arg4=numAccountsToUpdate, arg5=maxRewardIncrement, arg6=numFixtures"
679                );
680            }
681
682            let state_vec =
683                <Vec<(Address, U256)>>::abi_decode(&hex::decode(&cli.args[0]).unwrap()).unwrap();
684            let seed = cli.args[1].parse::<u64>().unwrap();
685            let num_add = cli.args[2].parse::<usize>().unwrap();
686            let num_update = cli.args[3].parse::<usize>().unwrap();
687            let max_reward_increment = cli.args[4].parse::<U256>().unwrap();
688            let num_fixtures = cli.args[5].parse::<usize>().unwrap();
689
690            let mut rng = StdRng::seed_from_u64(seed);
691            let mut state: HashMap<Address, U256> = state_vec.into_iter().collect();
692
693            // Randomly select existing accounts to update
694            let existing_accounts: Vec<Address> = state.keys().copied().collect();
695            let num_to_update = num_update.min(existing_accounts.len());
696            let accounts_to_update: Vec<Address> = existing_accounts
697                .choose_multiple(&mut rng, num_to_update)
698                .copied()
699                .collect();
700
701            // Update existing accounts
702            for account in accounts_to_update {
703                let amount = if max_reward_increment > U256::ZERO {
704                    (U256::random_with(&mut rng) % max_reward_increment) + U256::from(1)
705                } else {
706                    U256::from(1)
707                };
708                *state.get_mut(&account).unwrap() += amount;
709            }
710
711            // Add new accounts
712            for _ in 0..num_add {
713                let account = Address::random_with(&mut rng);
714                let amount = if max_reward_increment > U256::ZERO {
715                    (U256::random_with(&mut rng) % max_reward_increment) + U256::from(1)
716                } else {
717                    U256::from(1)
718                };
719                state.insert(account, amount);
720            }
721
722            // Build tree from final state
723            let mut tree = RewardMerkleTreeV2::new(REWARD_MERKLE_TREE_V2_HEIGHT);
724            for (account, lifetime_rewards) in &state {
725                let reward_account = RewardAccountV2::from(*account);
726                tree.update(reward_account, RewardAmount(*lifetime_rewards))
727                    .unwrap();
728            }
729
730            let commitment = tree.commitment();
731            let reward_commitment: B256 = commitment.digest().into();
732            let auth_root_fields: [B256; 8] = [
733                reward_commitment,
734                B256::ZERO,
735                B256::ZERO,
736                B256::ZERO,
737                B256::ZERO,
738                B256::ZERO,
739                B256::ZERO,
740                B256::ZERO,
741            ];
742            let auth_root = alloy::primitives::keccak256(auth_root_fields.abi_encode());
743            let auth_root_u256: U256 = auth_root.into();
744
745            // Generate fixtures for a random subset of accounts
746            let all_accounts: Vec<Address> = state.keys().copied().collect();
747            let num_fixtures_bounded = num_fixtures.min(all_accounts.len());
748            let selected_accounts: Vec<Address> = all_accounts
749                .choose_multiple(&mut rng, num_fixtures_bounded)
750                .copied()
751                .collect();
752
753            let fixtures: Vec<(Address, U256, Bytes)> = selected_accounts
754                .iter()
755                .map(|account| {
756                    let proof = RewardAccountProofV2::prove(&tree, *account).unwrap();
757                    let query_data: RewardAccountQueryDataV2 = proof.into();
758                    let claim_input = query_data.to_reward_claim_input().unwrap();
759                    (
760                        *account,
761                        claim_input.lifetime_rewards,
762                        claim_input.auth_data.into(),
763                    )
764                })
765                .collect();
766
767            let state_vec: Vec<(Address, U256)> = state.into_iter().collect();
768            let res = (auth_root_u256, state_vec, fixtures);
769            println!("{}", res.abi_encode_params().encode_hex());
770        },
771    };
