100/// Mainnet libp2p protocol identifiers. The snapshot tests below lock these down so a
101/// change that would partition mainnet (e.g. a stray `protocol_id_prefix` call) is caught.
102/// `None` for gossipsub means "do not call `protocol_id_prefix`" — libp2p's defaults
103/// (`/meshsub/1.1.0` and `/meshsub/1.0.0`) are then used.
104pub(crate) fn mainnet_gossipsub_prefix() -> Option<&'static str> {
105    None

111    StreamProtocol::new("/HotShot/direct_message/1.0")

114    "HotShot/identify/1.0"

117/// Resolve the gossipsub `protocol_id_prefix` for the given network discriminator.
118/// `None` returns the mainnet value (the libp2p default).
119pub(crate) fn gossipsub_prefix(discriminator: Option<U256>) -> Option<String> {
120    match discriminator {
121        None => mainnet_gossipsub_prefix().map(String::from),
122        Some(d) => Some(format!("/HotShot/gossipsub/1.0/{d:#x}")),
123    }
124}

126/// Resolve the kademlia stream protocol for the given network discriminator.
127pub(crate) fn kad_protocol(discriminator: Option<U256>) -> Result<StreamProtocol, NetworkError> {
128    match discriminator {
129        None => Ok(mainnet_kad_protocol()),
130        Some(d) => StreamProtocol::try_from_owned(format!("/ipfs/kad/1.0.0/{d:#x}"))
131            .map_err(|err| NetworkError::ConfigError(format!("invalid kademlia protocol: {err}"))),
132    }
133}

135/// Resolve the identify protocol string for the given network discriminator.
136pub(crate) fn identify_protocol(discriminator: Option<U256>) -> String {
137    match discriminator {
138        None => mainnet_identify_protocol().to_string(),
139        Some(d) => format!("HotShot/identify/1.0/{d:#x}"),
140    }
141}

143/// Resolve the direct-message stream protocol for the given network discriminator.
144pub(crate) fn direct_message_protocol(
145    discriminator: Option<U256>,
146) -> Result<StreamProtocol, NetworkError> {
147    match discriminator {
148        None => Ok(mainnet_direct_message_protocol()),
149        Some(d) => StreamProtocol::try_from_owned(format!("/HotShot/direct_message/1.0/{d:#x}"))

151                NetworkError::ConfigError(format!("invalid direct_message protocol: {err}"))

156fn resolve_put_quorum(
157    quorum_override: Option<NonZeroUsize>,
158    replication_factor: NonZeroUsize,
159) -> NonZeroUsize {
160    quorum_override.unwrap_or_else(|| {
161        NonZeroUsize::new(replication_factor.get() / 2)
162            .expect("replication factor should be bigger than 0")
163    })
164}

203    /// starts the swarm listening on `listen_addr`
204    /// and optionally dials into peer `known_peer`
205    /// returns the address the swarm is listening upon
206    #[instrument(skip(self))]
207    pub async fn start_listen(
208        &mut self,
209        listen_addr: Multiaddr,
210    ) -> Result<Multiaddr, NetworkError> {
211        self.listener_id = Some(self.swarm.listen_on(listen_addr).map_err(|err| {
212            NetworkError::ListenError(format!("failed to listen for Libp2p: {err}"))
213        })?);
214        let addr = loop {
215            if let Some(SwarmEvent::NewListenAddr { address, .. }) = self.swarm.next().await {
216                break address;
217            }
218        };
219        info!("Libp2p listening on {addr:?}");
220        Ok(addr)
221    }

223    /// initialize the DHT with known peers
224    /// add the peers to kademlia and then
225    /// the `spawn_listeners` function
226    /// will start connecting to peers
227    #[instrument(skip(self))]
228    pub fn add_known_peers(&mut self, known_peers: &[(PeerId, Multiaddr)]) {
229        debug!("Adding {} known peers", known_peers.len());
230        let behaviour = self.swarm.behaviour_mut();
231        let mut bs_nodes = HashMap::<PeerId, HashSet<Multiaddr>>::new();
232        let mut shuffled = known_peers.iter().collect::<Vec<_>>();
233        shuffled.shuffle(&mut thread_rng());
234        for (peer_id, addr) in shuffled {
235            if *peer_id != self.peer_id {
236                behaviour.dht.add_address(peer_id, addr.clone());
237                behaviour.autonat.add_server(*peer_id, Some(addr.clone()));
238                bs_nodes.insert(*peer_id, iter::once(addr.clone()).collect());
239            }
240        }
241    }

243    /// Creates a new `Network` with the given settings.
244    ///
245    /// Currently:
246    ///   * Generates a random key pair and associated [`PeerId`]
247    ///   * Launches a hopefully production ready transport: QUIC v1 (RFC 9000) + DNS
248    ///   * Generates a connection to the "broadcast" topic
249    ///   * Creates a swarm to manage peers and events
250    ///
251    /// # Errors
252    /// - If we fail to generate the transport or any of the behaviours
253    ///
254    /// # Panics
255    /// If 5 < 0
256    #[allow(clippy::too_many_lines)]
257    pub async fn new(
258        config: NetworkNodeConfig,
259        dht_persistent_storage: D,
260        consensus_key_to_pid_map: Arc<Mutex<BiMap<T::SignatureKey, PeerId>>>,
261    ) -> Result<Self, NetworkError> {
262        // Generate a random `KeyPair` if one is not specified
263        let keypair = config
264            .keypair
265            .clone()
266            .unwrap_or_else(Keypair::generate_ed25519);
267
268        // Get the `PeerId` from the `KeyPair`
269        let peer_id = PeerId::from(keypair.public());
270
271        // Generate the transport from the keypair and auth message
272        let transport: BoxedTransport = gen_transport::<T>(
273            keypair.clone(),
274            config.auth_message.clone(),
275            Arc::clone(&consensus_key_to_pid_map),
276        )
277        .await?;
278
279        // Calculate the record republication interval
280        let kademlia_record_republication_interval = config
281            .republication_interval
282            .unwrap_or(Duration::from_secs(KAD_DEFAULT_REPUB_INTERVAL_SEC));
283
284        // Calculate the Kademlia record TTL
285        let kademlia_ttl = config
286            .ttl
287            .unwrap_or(16 * kademlia_record_republication_interval);
288
289        // Generate the swarm
290        let mut swarm: Swarm<NetworkDef<T::SignatureKey, D>> = {
291            // Use the `Blake3` hash of the message's contents as the ID
292            let message_id_fn = |message: &GossipsubMessage| {
293                let hash = blake3::hash(&message.data);
294                MessageId::from(hash.as_bytes().to_vec())
295            };
296
297            // Derive a `Gossipsub` config from our gossip config
298            let mut gossipsub_builder = GossipsubConfigBuilder::default();
299            if let Some(prefix) = gossipsub_prefix(config.network_discriminator) {
300                gossipsub_builder.protocol_id_prefix(prefix);
301            }
302            let gossipsub_config = gossipsub_builder
303                .message_id_fn(message_id_fn) // Use the (blake3) hash of a message as its ID
304                .validation_mode(ValidationMode::Strict) // Force all messages to have valid signatures
305                .heartbeat_interval(config.gossip_config.heartbeat_interval) // Time between gossip heartbeats
306                .history_gossip(config.gossip_config.history_gossip) // Number of heartbeats to gossip about
307                .history_length(config.gossip_config.history_length) // Number of heartbeats to remember the full message for
308                .mesh_n(config.gossip_config.mesh_n) // Target number of mesh peers
309                .mesh_n_high(config.gossip_config.mesh_n_high) // Upper limit of mesh peers
310                .mesh_n_low(config.gossip_config.mesh_n_low) // Lower limit of mesh peers
311                .mesh_outbound_min(config.gossip_config.mesh_outbound_min) // Minimum number of outbound peers in mesh
312                .max_transmit_size(config.gossip_config.max_transmit_size) // Maximum size of a message
313                .max_ihave_length(config.gossip_config.max_ihave_length) // Maximum number of messages to include in an IHAVE message
314                .max_ihave_messages(config.gossip_config.max_ihave_messages) // Maximum number of IHAVE messages to accept from a peer within a heartbeat
315                .published_message_ids_cache_time(
316                    config.gossip_config.published_message_ids_cache_time,
317                ) // Cache duration for published message IDs
318                .iwant_followup_time(config.gossip_config.iwant_followup_time) // Time to wait for a message requested through IWANT following an IHAVE advertisement
319                .max_messages_per_rpc(config.gossip_config.max_messages_per_rpc) // The maximum number of messages we will process in a given RPC
320                .gossip_retransimission(config.gossip_config.gossip_retransmission) // Controls how many times we will allow a peer to request the same message id through IWANT gossip before we start ignoring them.
321                .flood_publish(config.gossip_config.flood_publish) // If enabled newly created messages will always be sent to all peers that are subscribed to the topic and have a good enough score.
322                .duplicate_cache_time(config.gossip_config.duplicate_cache_time) // The time period that messages are stored in the cache
323                .fanout_ttl(config.gossip_config.fanout_ttl) // Time to live for fanout peers
324                .heartbeat_initial_delay(config.gossip_config.heartbeat_initial_delay) // Initial delay in each heartbeat
325                .gossip_factor(config.gossip_config.gossip_factor) // Affects how many peers we will emit gossip to at each heartbeat
326                .gossip_lazy(config.gossip_config.gossip_lazy) // Minimum number of peers to emit gossip to during a heartbeat
327                .build()
328                .map_err(|err| {
329                    NetworkError::ConfigError(format!("error building gossipsub config: {err:?}"))
330                })?;
331
332            // - Build a gossipsub network behavior
333            let gossipsub: Gossipsub = Gossipsub::new(
334                MessageAuthenticity::Signed(keypair.clone()),
335                gossipsub_config,
336            )
337            .map_err(|err| {
338                NetworkError::ConfigError(format!("error building gossipsub behaviour: {err:?}"))
339            })?;
340
341            //   Build a identify network behavior needed for own
342            //   node connection information
343            //   E.g. this will answer the question: how are other nodes
344            //   seeing the peer from behind a NAT
345            let identify_cfg = IdentifyConfig::new(
346                identify_protocol(config.network_discriminator),
347                keypair.public(),
348            );
349            let identify = IdentifyBehaviour::new(identify_cfg);
350
351            // - Build DHT needed for peer discovery
352            let mut kconfig = Config::new(kad_protocol(config.network_discriminator)?);
353            kconfig
354                .set_parallelism(NonZeroUsize::new(5).unwrap())
355                .set_provider_publication_interval(Some(kademlia_record_republication_interval))
356                .set_publication_interval(Some(kademlia_record_republication_interval))
357                .set_record_ttl(Some(kademlia_ttl));
358
359            // allowing panic here because something is very wrong if this fails
360            #[allow(clippy::panic)]
361            if let Some(factor) = config.replication_factor {
362                kconfig.set_replication_factor(factor);
363            } else {
364                panic!("Replication factor not set");
365            }
366
367            // Create the DHT behaviour with the given persistent storage
368            let mut kadem = Behaviour::with_config(
369                peer_id,
370                PersistentStore::new(
371                    ValidatedStore::new(MemoryStore::new(peer_id)),
372                    dht_persistent_storage,
373                    5,
374                )
375                .await,
376                kconfig,
377            );
378            kadem.set_mode(Some(Mode::Server));
379
380            let rrconfig = Libp2pRequestResponseConfig::default();
381
382            // Create a new `cbor` codec with the given request and response sizes
383            let cbor = Cbor::new(
384                config.request_response_config.request_size_maximum,
385                config.request_response_config.response_size_maximum,
386            );
387
388            let direct_message: super::cbor::Behaviour<Vec<u8>, Vec<u8>> =
389                RequestResponse::with_codec(
390                    cbor,
391                    [(
392                        direct_message_protocol(config.network_discriminator)?,
393                        ProtocolSupport::Full,
394                    )],
395                    rrconfig.clone(),
396                );
397
398            let autonat_config = autonat::Config {
399                only_global_ips: false,
400                ..Default::default()
401            };
402
403            let network = NetworkDef::new(
404                gossipsub,
405                kadem,
406                identify,
407                direct_message,
408                autonat::Behaviour::new(peer_id, autonat_config),
409            );
410
411            // build swarm
412            let swarm = SwarmBuilder::with_existing_identity(keypair.clone());
413            let swarm = swarm.with_tokio();
414
415            swarm
416                .with_other_transport(|_| transport)
417                .unwrap()
418                .with_behaviour(|_| network)
419                .unwrap()
420                .with_swarm_config(|cfg| {
421                    cfg.with_idle_connection_timeout(Duration::from_secs(10))
422                        .with_substream_upgrade_protocol_override(V1Lazy)
423                })
424                .build()
425        };
426        for (peer, addr) in &config.to_connect_addrs {
427            if peer != swarm.local_peer_id() {
428                swarm.behaviour_mut().add_address(peer, addr.clone());
429                swarm.add_peer_address(*peer, addr.clone());
430            }
431        }
432
433        for addr in &config.announce_addresses {
434            info!("Adding announce address {addr}");
435            swarm.add_external_address(addr.clone());
436        }
437
438        Ok(Self {
439            peer_id,
440            swarm,
441            kademlia_record_ttl: kademlia_ttl,
442            consensus_key_to_pid_map,
443            listener_id: None,
444            direct_message_state: DMBehaviour::default(),
445            dht_handler: DHTBehaviour::new(
446                peer_id,
447                config
448                    .replication_factor
449                    .unwrap_or(NonZeroUsize::new(4).unwrap()),
450            ),
451            resend_tx: None,
452            autonat_private_logged: false,
453            dht_put_quorum: config.dht_put_quorum,
454        })
455    }

457    /// Publish a key/value to the record store.
458    ///
459    /// # Panics
460    /// If the default replication factor is `None`
461    pub fn put_record(&mut self, mut query: KadPutQuery) {
462        // Create the new record
463        let mut record = Record::new(query.key.clone(), query.value.clone());
464
465        // Set the record's expiration time to the proper time
466        record.expires = Some(Instant::now() + self.kademlia_record_ttl);
467
468        let quorum = resolve_put_quorum(self.dht_put_quorum, self.dht_handler.replication_factor());
469        match self
470            .swarm
471            .behaviour_mut()
472            .dht
473            .put_record(record, libp2p::kad::Quorum::N(quorum))
474        {
475            Err(e) => {
476                // failed try again later
477                query.progress = DHTProgress::NotStarted;
478                query.backoff.start_next(false);
479                error!("Error publishing to DHT: {e:?} for peer {:?}", self.peer_id);
480            },
481            Ok(qid) => {
482                debug!("Published record to DHT with qid {qid:?}");
483                let query = KadPutQuery {
484                    progress: DHTProgress::InProgress(qid),
485                    ..query
486                };
487                self.dht_handler.put_record(qid, query);
488            },
489        }
490    }

492    /// event handler for client events
493    /// currently supported actions include
494    /// - shutting down the swarm
495    /// - gossipping a message to known peers on the `global` topic
496    /// - returning the id of the current peer
497    /// - subscribing to a topic
498    /// - unsubscribing from a toipc
499    /// - direct messaging a peer
500    #[instrument(skip(self))]
501    async fn handle_client_requests(
502        &mut self,
503        msg: Option<ClientRequest>,
504    ) -> Result<bool, NetworkError> {
505        let behaviour = self.swarm.behaviour_mut();
506        match msg {
507            Some(msg) => {
508                match msg {
509                    ClientRequest::BeginBootstrap => {
510                        debug!("Beginning Libp2p bootstrap");
511                        let _ = self.swarm.behaviour_mut().dht.bootstrap();
512                    },
513                    ClientRequest::LookupPeer(pid, chan) => {
514                        let id = self.swarm.behaviour_mut().dht.get_closest_peers(pid);
515                        self.dht_handler
516                            .in_progress_get_closest_peers
517                            .insert(id, chan);
518                    },
519                    ClientRequest::GetRoutingTable(chan) => {
520                        self.dht_handler
521                            .print_routing_table(&mut self.swarm.behaviour_mut().dht);
522                        if chan.send(()).is_err() {
523                            warn!("Tried to notify client but client not tracking anymore");
524                        }
525                    },
526                    ClientRequest::PutDHT { key, value, notify } => {
527                        let query = KadPutQuery {
528                            progress: DHTProgress::NotStarted,
529                            notify,
530                            key,
531                            value,
532                            backoff: ExponentialBackoff::default(),
533                        };
534                        self.put_record(query);
535                    },
536                    ClientRequest::GetConnectedPeerNum(s) => {
537                        if s.send(self.num_connected()).is_err() {
538                            error!("error sending peer number to client");
539                        }
540                    },
541                    ClientRequest::GetConnectedPeers(s) => {
542                        if s.send(self.connected_pids()).is_err() {
543                            error!("error sending peer set to client");
544                        }
545                    },
546                    ClientRequest::GetDHT {
547                        key,
548                        notify,
549                        retry_count,
550                    } => {
551                        self.dht_handler.get_record(
552                            key,
553                            notify,
554                            ExponentialBackoff::default(),
555                            retry_count,
556                            &mut self.swarm.behaviour_mut().dht,
557                        );
558                    },
559                    ClientRequest::IgnorePeers(_peers) => {
560                        // NOTE used by test with conductor only
561                    },
562                    ClientRequest::Shutdown => {
563                        if let Some(listener_id) = self.listener_id {
564                            self.swarm.remove_listener(listener_id);
565                        }
566
567                        return Ok(true);
568                    },
569                    ClientRequest::GossipMsg(topic, contents) => {
570                        behaviour.publish_gossip(Topic::new(topic.clone()), contents.clone());
571                    },
572                    ClientRequest::Subscribe(t, chan) => {
573                        behaviour.subscribe_gossip(&t);
574                        if let Some(chan) = chan
575                            && chan.send(()).is_err()
576                        {
577                            error!("finished subscribing but response channel dropped");
578                        }
579                    },
580                    ClientRequest::Unsubscribe(t, chan) => {
581                        behaviour.unsubscribe_gossip(&t);
582                        if let Some(chan) = chan
583                            && chan.send(()).is_err()
584                        {
585                            error!("finished unsubscribing but response channel dropped");
586                        }
587                    },
588                    ClientRequest::DirectRequest {
589                        pid,
590                        contents,
591                        retry_count,
592                    } => {
593                        debug!("Sending direct request to {pid:?}");
594                        let id = behaviour.add_direct_request(pid, contents.clone());
595                        let req = DMRequest {
596                            peer_id: pid,
597                            data: contents,
598                            backoff: ExponentialBackoff::default(),
599                            retry_count,
600                        };
601                        self.direct_message_state.add_direct_request(req, id);
602                    },
603                    ClientRequest::DirectResponse(chan, msg) => {
604                        behaviour.add_direct_response(chan, msg);
605                    },
606                    ClientRequest::AddKnownPeers(peers) => {
607                        self.add_known_peers(&peers);
608                    },
609                    ClientRequest::Prune(pid) => {
610                        if self.swarm.disconnect_peer_id(pid).is_err() {
611                            warn!("Could not disconnect from {pid:?}");
612                        }
613                    },
614                }
615            },
616            None => {
617                error!("Error receiving msg in main behaviour loop: channel closed");
618            },
619        }
620        Ok(false)
621    }

623    /// event handler for events emitted from the swarm
624    #[allow(clippy::type_complexity)]
625    #[instrument(skip(self))]
626    async fn handle_swarm_events(
627        &mut self,
628        event: SwarmEvent<NetworkEventInternal>,
629        send_to_client: &UnboundedSender<NetworkEvent>,
630    ) -> Result<(), NetworkError> {
631        // Make the match cleaner
632        debug!("Swarm event observed {:?}", event);
633
634        #[allow(deprecated)]
635        match event {
636            SwarmEvent::ConnectionEstablished {
637                connection_id: _,
638                peer_id,
639                endpoint,
640                num_established,
641                concurrent_dial_errors,
642                established_in: _established_in,
643            } => {
644                if num_established > ESTABLISHED_LIMIT {
645                    error!(
646                        "Num concurrent connections to a single peer exceeding \
647                         {ESTABLISHED_LIMIT:?} at {num_established:?}!"
648                    );
649                } else {
650                    debug!(
651                        "Connection established with {peer_id:?} at {endpoint:?} with \
652                         {concurrent_dial_errors:?} concurrent dial errors"
653                    );
654                }
655
656                // Send the number of connected peers to the client
657                send_to_client
658                    .send(NetworkEvent::ConnectedPeersUpdate(self.num_connected()))
659                    .map_err(|err| NetworkError::ChannelSendError(err.to_string()))?;
660            },
661            SwarmEvent::ConnectionClosed {
662                connection_id: _,
663                peer_id,
664                endpoint,
665                num_established,
666                cause,
667            } => {
668                if num_established > ESTABLISHED_LIMIT_UNWR {
669                    error!(
670                        "Num concurrent connections to a single peer exceeding \
671                         {ESTABLISHED_LIMIT:?} at {num_established:?}!"
672                    );
673                } else {
674                    debug!("Connection closed with {peer_id:?} at {endpoint:?} due to {cause:?}");
675                }
676
677                // If we are no longer connected to the peer, remove the consensus key from the map
678                if num_established == 0 {
679                    self.consensus_key_to_pid_map
680                        .lock()
681                        .remove_by_right(&peer_id);
682                }
683
684                // Send the number of connected peers to the client
685                send_to_client
686                    .send(NetworkEvent::ConnectedPeersUpdate(self.num_connected()))
687                    .map_err(|err| NetworkError::ChannelSendError(err.to_string()))?;
688            },
689            SwarmEvent::Dialing {
690                peer_id,
691                connection_id: _,
692            } => {
693                debug!("Attempting to dial {peer_id:?}");
694            },
695            SwarmEvent::ListenerClosed {
696                listener_id: _,
697                addresses: _,
698                reason: _,
699            }
700            | SwarmEvent::NewListenAddr {
701                listener_id: _,
702                address: _,
703            }
704            | SwarmEvent::ExpiredListenAddr {
705                listener_id: _,
706                address: _,
707            }
708            | SwarmEvent::NewExternalAddrCandidate { .. }
709            | SwarmEvent::ExternalAddrExpired { .. }
710            | SwarmEvent::IncomingConnection {
711                connection_id: _,
712                local_addr: _,
713                send_back_addr: _,
714            } => {},
715            SwarmEvent::Behaviour(b) => {
716                let maybe_event = match b {
717                    NetworkEventInternal::DHTEvent(e) => self
718                        .dht_handler
719                        .dht_handle_event(e, self.swarm.behaviour_mut().dht.store_mut()),
720                    NetworkEventInternal::IdentifyEvent(e) => {
721                        // NOTE feed identified peers into kademlia's routing table for peer discovery.
722                        if let IdentifyEvent::Received {
723                            peer_id,
724                            info:
725                                IdentifyInfo {
726                                    listen_addrs,
727                                    protocols: _,
728                                    public_key: _,
729                                    protocol_version: _,
730                                    agent_version: _,
731                                    observed_addr: _,
732                                    signed_peer_record: _,
733                                },
734                            connection_id: _,
735                        } = *e
736                        {
737                            let behaviour = self.swarm.behaviour_mut();
738
739                            // into hashset to delete duplicates (I checked: there are duplicates)
740                            for addr in listen_addrs.iter().collect::<HashSet<_>>() {
741                                behaviour.dht.add_address(&peer_id, addr.clone());
742                            }
743                        }
744                        None
745                    },
746                    NetworkEventInternal::GossipEvent(e) => match *e {
747                        GossipEvent::Message {
748                            propagation_source: _peer_id,
749                            message_id: _id,
750                            message,
751                        } => Some(NetworkEvent::GossipMsg(message.data)),
752                        GossipEvent::Subscribed { peer_id, topic } => {
753                            debug!("Peer {peer_id:?} subscribed to topic {topic:?}");
754                            None
755                        },
756                        GossipEvent::Unsubscribed { peer_id, topic } => {
757                            debug!("Peer {peer_id:?} unsubscribed from topic {topic:?}");
758                            None
759                        },
760                        GossipEvent::GossipsubNotSupported { peer_id } => {
761                            LogEvent::GossipsubNotSupported.record();
762                            debug!("Peer {peer_id:?} does not support gossipsub");
763                            None
764                        },
765                        GossipEvent::SlowPeer {
766                            peer_id,
767                            failed_messages: _,
768                        } => {
769                            LogEvent::GossipsubSlowPeer.record();
770                            debug!("Peer {peer_id:?} is slow");
771                            None
772                        },
773                    },
774                    NetworkEventInternal::DMEvent(e) => self
775                        .direct_message_state
776                        .handle_dm_event(e, self.resend_tx.clone()),
777                    NetworkEventInternal::AutonatEvent(e) => {
778                        match e {
779                            autonat::Event::InboundProbe(_) => {},
780                            autonat::Event::OutboundProbe(e) => match e {
781                                autonat::OutboundProbeEvent::Request { .. }
782                                | autonat::OutboundProbeEvent::Response { .. } => {},
783                                autonat::OutboundProbeEvent::Error {
784                                    probe_id: _,
785                                    peer,
786                                    error,
787                                } => {
788                                    debug!("AutoNAT outbound probe to {peer:?} failed: {error:?}");
789                                },
790                            },
791                            autonat::Event::StatusChanged { old, new } => match &new {
792                                autonat::NatStatus::Public(addr) => {
793                                    info!(
794                                        "AutoNAT: this node is publicly reachable at {addr} (was \
795                                         {old:?})"
796                                    );
797                                    self.autonat_private_logged = false;
798                                },
799                                autonat::NatStatus::Private => {
800                                    warn!(
801                                        "AutoNAT: probe reports this node may not be publicly \
802                                         reachable (was {old:?}). Treating as transient until \
803                                         confirmed by repeated probes."
804                                    );
805                                },
806                                autonat::NatStatus::Unknown => {
807                                    debug!("AutoNAT status: {old:?} -> Unknown");
808                                    self.autonat_private_logged = false;
809                                },
810                            },
811                        };
812                        let autonat = &self.swarm.behaviour().autonat;
813                        if matches!(autonat.nat_status(), autonat::NatStatus::Private)
814                            && autonat.confidence() >= AUTONAT_CONFIDENCE_MAX
815                            && !self.autonat_private_logged
816                        {
817                            error!(
818                                "AutoNAT: this node is NOT publicly reachable (confirmed by \
819                                 repeated probes). Peers cannot direct-message us, so leader \
820                                 views may fail and we may accumulate missed slots. Verify \
821                                 --libp2p-advertise-address (env \
822                                 ESPRESSO_NODE_LIBP2P_ADVERTISE_ADDRESS) is set a publicly \
823                                 reachable host:port, and ensure inbound UDP at that port is open \
824                                 from the public internet (firewall/NAT/security group)."
825                            );
826                            self.autonat_private_logged = true;
827                        }
828                        None
829                    },
830                };
831
832                if let Some(event) = maybe_event {
833                    // forward messages directly to Client
834                    send_to_client
835                        .send(event)
836                        .map_err(|err| NetworkError::ChannelSendError(err.to_string()))?;
837                }
838            },
839            SwarmEvent::OutgoingConnectionError {
840                connection_id: _,
841                peer_id,
842                error,
843            } => {
844                LogEvent::DialFailure.record();
845                debug!("Outgoing connection error to {peer_id:?}: {error:?}");
846            },
847            SwarmEvent::IncomingConnectionError {
848                connection_id: _,
849                local_addr: _,
850                send_back_addr: _,
851                error,
852                peer_id: _,
853            } => {
854                LogEvent::IncomingConnError.record();
855                debug!("Incoming connection error: {error:?}");
856            },
857            SwarmEvent::ListenerError {
858                listener_id: _,
859                error,
860            } => {
861                LogEvent::ListenerError.record();
862                debug!("Listener error: {error:?}");
863            },
864            SwarmEvent::ExternalAddrConfirmed { address } => {
865                let my_id = *self.swarm.local_peer_id();
866                self.swarm
867                    .behaviour_mut()
868                    .dht
869                    .add_address(&my_id, address.clone());
870            },
871            SwarmEvent::NewExternalAddrOfPeer { peer_id, address } => {
872                self.swarm
873                    .behaviour_mut()
874                    .dht
875                    .add_address(&peer_id, address.clone());
876            },
877            _ => {
878                debug!("Unhandled swarm event {event:?}");
879            },
880        }
881        Ok(())
882    }

884    /// Spawn a task to listen for requests on the returned channel
885    /// as well as any events produced by libp2p
886    ///
887    /// # Errors
888    /// - If we fail to create the channels or the bootstrap channel
889    pub fn spawn_listeners(
890        mut self,
891    ) -> Result<
892        (
893            UnboundedSender<ClientRequest>,
894            UnboundedReceiver<NetworkEvent>,
895            SwarmTaskHandle,
896        ),
897        NetworkError,
898    > {
899        let (s_input, mut s_output) = unbounded_channel::<ClientRequest>();
900        let (r_input, r_output) = unbounded_channel::<NetworkEvent>();
901        let (mut bootstrap_tx, bootstrap_rx) = mpsc::channel(100);
902        self.resend_tx = Some(s_input.clone());
903        self.dht_handler.set_bootstrap_sender(bootstrap_tx.clone());
904
905        DHTBootstrapTask::run(bootstrap_rx, s_input.clone());
906        // Keep the task's `JoinHandle` so callers can await the swarm loop
907        // ending on shutdown, ensuring the listening socket is released.
908        let task = spawn(
909            async move {
910                loop {
911                    select! {
912                        event = self.swarm.next() => {
913                            debug!("peerid {:?}\t\thandling maybe event {:?}", self.peer_id, event);
914                            if let Some(event) = event {
915                                debug!("peerid {:?}\t\thandling event {:?}", self.peer_id, event);
916                                self.handle_swarm_events(event, &r_input).await?;
917                            }
918                        },
919                        msg = s_output.recv() => {
920                            debug!("peerid {:?}\t\thandling msg {:?}", self.peer_id, msg);
921                            let shutdown = self.handle_client_requests(msg).await?;
922                            if shutdown {
923                                let _ = bootstrap_tx.send(InputEvent::ShutdownBootstrap).await;
924                                break
925                            }
926                        }
927                    }
928                }
929                drop(self.swarm);
930                Ok::<(), NetworkError>(())
931            }
932            .instrument(info_span!("Libp2p NetworkBehaviour Handler")),
933        );
934        Ok((s_input, r_output, task))
935    }
