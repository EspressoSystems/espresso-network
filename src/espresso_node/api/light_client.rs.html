31async fn get_leaf_proof_with_qc_chain<State>(
32    state: &State,
33    requested_leaf: LeafQueryData<SeqTypes>,
34    fetch_timeout: Duration,
35) -> Result<LeafProof, Error>
36where
37    State: AvailabilityDataSource<SeqTypes> + VersionedDataSource,
38    for<'a> State::ReadOnly<'a>: NodeStorage<SeqTypes>,
39{
40    let requested = requested_leaf.height() as usize;
41    // Grab the endpoint and the final QC chain in the same transaction, to ensure that
42    // the QC chain actually corresponds to the endpoint block (and is not subject to
43    // concurrent updates).
44    let mut tx = state.read().await.map_err(internal)?;
45    let latest_height = NodeStorage::block_height(&mut tx).await.map_err(internal)?;
46    let qc_chain = tx.latest_qc_chain().await.map_err(internal)?;
47    drop(tx);
48
49    let mut leaves = state.get_leaf_range(requested + 1..latest_height).await;
50    let mut proof = LeafProof::default();
51    proof.push(requested_leaf);
52
53    while let Some(leaf) = leaves.next().await {
54        let leaf = leaf
55            .with_timeout(fetch_timeout)
56            .await
57            .ok_or_else(|| not_found("missing leaves"))?;
58
59        if proof.push(leaf) {
60            return Ok(proof);
61        }
62    }
63
64    // We reached the end of the range of interest without encountering a 3-chain. Thus, if the last
65    // leaf in the chain is not already assumed finalized by the client, we must prove it finalized
66    // by appending two more QCs.
67    let Some([committing_qc, deciding_qc]) = qc_chain else {
68        return Err(not_found("missing QC 2-chain to prove finality"));
69    };
70    proof.add_qc_chain(Arc::new(committing_qc), Arc::new(deciding_qc));
71
72    Ok(proof)
73}

75/// Build a leaf proof for the new protocol using certificate2 finality.
76///
77/// Certificate2 directly commits the leaf at `cert2.data.block_number`. The indirect commit rule then
78/// commits all of that leaf's uncommitted ancestors. If cert2 directly commits the requested leaf,
79/// the proof contains only that leaf plus cert2. Otherwise, the proof includes the chain from the
80/// requested leaf through the directly committed descendant; the verifier checks parent commitments
81/// to prove the requested leaf is on that finalized chain.
82async fn get_leaf_proof_with_cert2<State>(
83    state: &State,
84    requested_leaf: LeafQueryData<SeqTypes>,
85    fetch_timeout: Duration,
86) -> Result<LeafProof, Error>
87where
88    State: AvailabilityDataSource<SeqTypes> + VersionedDataSource,
89    for<'a> State::ReadOnly<'a>: NodeStorage<SeqTypes>,
90{
91    let requested_height = requested_leaf.height();
92    let cert2 = state
93        .read()
94        .await
95        .map_err(internal)?
96        .load_earliest_cert2(requested_height)
97        .await
98        .map_err(internal)?
99        .ok_or_else(|| not_found("no cert2 finality proof available"))?;
100
101    let cert2_height = cert2.data.block_number;
102    if cert2_height < requested_height {
103        return Err(not_found(
104            "cert2 finality proof is older than requested leaf",
105        ));
106    }
107
108    let mut proof = LeafProof::default();
109    let requested_leaf_qc = requested_leaf.qc().clone();
110
111    if cert2_height == requested_height {
112        if requested_leaf.leaf().commit() != cert2.data.leaf_commit {
113            return Err(internal("stored cert2 does not finalize the expected leaf"));
114        }
115        proof.push(requested_leaf);
116        proof.add_certificate(Arc::new(cert2), requested_leaf_qc);
117        return Ok(proof);
118    }
119
120    proof.push(requested_leaf);
121    let mut leaves = state
122        .get_leaf_range(requested_height as usize + 1..cert2_height as usize + 1)
123        .await;
124    // Extend the proof chain until we reach the leaf directly committed by cert2. Once that leaf is
125    // present and matches cert2's commitment, the requested leaf is finalized by the indirect
126    // commit rule.
127    while let Some(leaf) = leaves.next().await {
128        let leaf = leaf
129            .with_timeout(fetch_timeout)
130            .await
131            .ok_or_else(|| not_found("missing leaves"))?;
132
133        if leaf.height() == cert2_height {
134            if leaf.leaf().commit() != cert2.data.leaf_commit {
135                return Err(internal("stored cert2 does not finalize the expected leaf"));
136            }
137            proof.push(leaf);
138            proof.add_certificate(Arc::new(cert2), requested_leaf_qc);
139            return Ok(proof);
140        }
141        proof.push(leaf);
142    }
143
144    Err(not_found("missing cert2 leaf"))
145}

147async fn get_leaf_proof_with_finalized_assumption<State>(
148    state: &State,
149    requested_leaf: LeafQueryData<SeqTypes>,
150    finalized: usize,
151    fetch_timeout: Duration,
152) -> Result<LeafProof, Error>
153where
154    State: AvailabilityDataSource<SeqTypes>,
155{
156    let requested = requested_leaf.height() as usize;
157    // If we have a known-finalized block, we will not need a final 2-chain of QCs to prove
158    // the last leaf in the result finalized, since we will either terminate with a 3-chain
159    // of leaves or at the `finalized` leaf.
160    if finalized <= requested {
161        return Err(Error::Custom {
162            message: format!(
163                "finalized leaf height ({finalized}) must be greater than requested ({requested})"
164            ),
165            status: StatusCode::BAD_REQUEST,
166        });
167    }
168
169    let mut leaves = state.get_leaf_range(requested + 1..finalized).await;
170    let mut proof = LeafProof::default();
171    proof.push(requested_leaf);
172
173    while let Some(leaf) = leaves.next().await {
174        let leaf = leaf
175            .with_timeout(fetch_timeout)
176            .await
177            .ok_or_else(|| not_found("missing leaves"))?;
178
179        if proof.push(leaf) {
180            return Ok(proof);
181        }
182    }
183
184    Ok(proof)
185}

187async fn get_header_proof<State>(
188    state: &State,
189    root: u64,
190    requested: BlockId<SeqTypes>,
191    fetch_timeout: Duration,
192) -> Result<HeaderProof, Error>
193where
194    State: AvailabilityDataSource<SeqTypes>
195        + MerklizedStateDataSource<SeqTypes, BlockMerkleTree, { BlockMerkleTree::ARITY }>
196        + VersionedDataSource,
197{
198    let header = state
199        .get_header(requested)
200        .await
201        .with_timeout(fetch_timeout)
202        .await
203        .ok_or_else(|| not_found(format!("unknown header {requested}")))?;
204    if header.height() >= root {
205        return Err(Error::Custom {
206            message: format!(
207                "height ({}) must be less than root ({root})",
208                header.height()
209            ),
210            status: StatusCode::BAD_REQUEST,
211        });
212    }
213    let path = MerklizedStateDataSource::<SeqTypes, BlockMerkleTree, _>::get_path(
214        state,
215        Snapshot::Index(root),
216        header.height(),
217    )
218    .await
219    .map_err(|source| Error::MerklizedState {
220        source: source.into(),
221    })?;
222
223    Ok(HeaderProof::new(header, path))
224}

226async fn get_namespace_proof_range<State>(
227    state: &State,
228    start: usize,
229    end: usize,
230    namespace: u64,
231    fetch_timeout: Duration,
232    large_object_range_limit: usize,
233) -> Result<Vec<NamespaceProof>, Error>
234where
235    State: AvailabilityDataSource<SeqTypes>,
236{
237    if end <= start {
238        return Err(Error::Custom {
239            message: format!("requested empty interval [{start}, {end})"),
240            status: StatusCode::BAD_REQUEST,
241        });
242    }
243    if end - start > large_object_range_limit {
244        return Err(Error::Custom {
245            message: format!(
246                "requested range [{start}, {end}) exceeds maximum size {large_object_range_limit}"
247            ),
248            status: StatusCode::BAD_REQUEST,
249        });
250    }
251
252    let fetch_headers = async move {
253        state
254            .get_header_range(start..end)
255            .await
256            .enumerate()
257            .then(|(i, fetch)| async move {
258                fetch
259                    .with_timeout(fetch_timeout)
260                    .await
261                    .ok_or_else(|| Error::Custom {
262                        message: format!("missing header {}", start + i),
263                        status: StatusCode::NOT_FOUND,
264                    })
265            })
266            .try_collect::<Vec<_>>()
267            .await
268    };
269    let fetch_payloads = async move {
270        state
271            .get_payload_range(start..end)
272            .await
273            .enumerate()
274            .then(|(i, fetch)| async move {
275                fetch
276                    .with_timeout(fetch_timeout)
277                    .await
278                    .ok_or_else(|| Error::Custom {
279                        message: format!("missing payload {}", start + i),
280                        status: StatusCode::NOT_FOUND,
281                    })
282            })
283            .try_collect::<Vec<_>>()
284            .await
285    };
286    let fetch_vid_commons = async move {
287        state
288            .get_vid_common_range(start..end)
289            .await
290            .enumerate()
291            .then(|(i, fetch)| async move {
292                fetch
293                    .with_timeout(fetch_timeout)
294                    .await
295                    .ok_or_else(|| Error::Custom {
296                        message: format!("missing VID common {}", start + i),
297                        status: StatusCode::NOT_FOUND,
298                    })
299            })
300            .try_collect::<Vec<_>>()
301            .await
302    };
303    let (headers, (payloads, vid_commons)) =
304        try_join(fetch_headers, try_join(fetch_payloads, fetch_vid_commons)).await?;
305
306    izip!(headers, payloads, vid_commons)
307        .map(|(header, payload, vid_common)| {
308            let Some(ns_index) = header.ns_table().find_ns_id(&namespace.into()) else {
309                return Ok(NamespaceProof::not_present());
310            };
311            let ns_proof = NsProof::new(payload.data(), &ns_index, vid_common.common())
312                .ok_or_else(|| Error::Custom {
313                    message: "failed to construct namespace proof".into(),
314                    status: StatusCode::INTERNAL_SERVER_ERROR,
315                })?;
316            Ok(NamespaceProof::new(ns_proof, vid_common.common().clone()))
317        })
318        .collect()
319}

321#[derive(Debug)]
322pub(super) struct Options {
323    /// Timeout for failing requests due to missing data.
324    ///
325    /// If data needed to respond to a request is missing, it can (in some cases) be fetched from an
326    /// external provider. This parameter controls how long the request handler will wait for
327    /// missing data to be fetched before giving up and failing the request.
328    pub fetch_timeout: Duration,
329
330    /// The maximum number of large objects which can be loaded in a single range query.
331    ///
332    /// Large objects include anything that _might_ contain a full payload or an object proportional
333    /// in size to a payload. Note that this limit applies to the entire class of objects: we do not
334    /// check the size of objects while loading to determine which limit to apply. If an object
335    /// belongs to a class which might contain a large payload, the large object limit always
336    /// applies.
337    pub large_object_range_limit: usize,
338}
339
340impl Default for Options {
341    fn default() -> Self {
342        Self {
343            fetch_timeout: Duration::from_millis(500),
344            large_object_range_limit: availability::Options::default().large_object_range_limit,
345        }
346    }
347}

349pub(super) fn define_api<S, ApiVer: StaticVersionType + 'static>(
350    opt: Options,
351    api_ver: semver::Version,
352) -> Result<Api<S, Error, ApiVer>>
353where
354    S: ReadState + Send + Sync + 'static,
355    S::State: AvailabilityDataSource<SeqTypes>
356        + MerklizedStateDataSource<SeqTypes, BlockMerkleTree, { BlockMerkleTree::ARITY }>
357        + NodeStateDataSource
358        + StakeTableDataSource<SeqTypes>
359        + VersionedDataSource,
360    for<'a> <S::State as VersionedDataSource>::ReadOnly<'a>: NodeStorage<SeqTypes>,
361{
362    let toml = toml::from_str::<toml::Value>(include_str!("../../api/light-client.toml"))?;
363    let mut api = Api::<S, Error, ApiVer>::new(toml)?;
364    api.with_version(api_ver);
365
366    let Options {
367        fetch_timeout,
368        large_object_range_limit,
369    } = opt;
370
371    api.get("leaf", move |req, state| {
372        async move {
373            let requested_leaf = leaf_from_req(&req, state, fetch_timeout).await?;
374            let finalized = req
375                .opt_integer_param("finalized")
376                .map_err(bad_param("finalized"))?;
377
378            if let Some(finalized) = finalized {
379                get_leaf_proof_with_finalized_assumption(
380                    state,
381                    requested_leaf,
382                    finalized,
383                    fetch_timeout,
384                )
385                .await
386            } else if requested_leaf.header().version() >= NEW_PROTOCOL_VERSION {
387                get_leaf_proof_with_cert2(state, requested_leaf, fetch_timeout).await
388            } else {
389                get_leaf_proof_with_qc_chain(state, requested_leaf, fetch_timeout).await
390            }
391        }
392        .boxed()
393    })?
394    .get("header", move |req, state| {
395        async move {
396            let root = req.integer_param("root").map_err(bad_param("root"))?;
397            let requested = block_id_from_req(&req)?;
398            get_header_proof(state, root, requested, fetch_timeout).await
399        }
400        .boxed()
401    })?
402    .get("stake_table", move |req, state| {
403        async move {
404            let epoch: u64 = req.integer_param("epoch").map_err(bad_param("epoch"))?;
405
406            let node_state = state.node_state().await;
407            let epoch_height = node_state.epoch_height.ok_or_else(|| Error::Custom {
408                message: "epoch state not set".into(),
409                status: StatusCode::INTERNAL_SERVER_ERROR,
410            })?;
411            let first_epoch = epoch_from_block_number(node_state.epoch_start_block, epoch_height);
412
413            if epoch < first_epoch + 2 {
414                return Err(Error::Custom {
415                    message: format!("epoch must be at least {}", first_epoch + 2),
416                    status: StatusCode::BAD_REQUEST,
417                });
418            }
419
420            // Find the range of L1 block containing events for this epoch. This is determined by
421            // the `l1_finalized` field of the epoch root (from two epochs prior) and the previous
422            // epoch's epoch root.
423            let epoch_root_height = root_block_in_epoch(epoch - 2, epoch_height) as usize;
424            let epoch_root = state
425                .get_header(epoch_root_height)
426                .await
427                .with_timeout(fetch_timeout)
428                .await
429                .ok_or_else(|| {
430                    not_found(format!("missing epoch root header {epoch_root_height}"))
431                })?;
432            let to_l1_block = epoch_root
433                .l1_finalized()
434                .ok_or_else(|| Error::Custom {
435                    message: "epoch root header is missing L1 finalized block".into(),
436                    status: StatusCode::INTERNAL_SERVER_ERROR,
437                })?
438                .number();
439
440            let from_l1_block = if epoch >= first_epoch + 3 {
441                let prev_epoch_root_height = root_block_in_epoch(epoch - 3, epoch_height) as usize;
442                let prev_epoch_root = state
443                    .get_header(prev_epoch_root_height)
444                    .await
445                    .with_timeout(fetch_timeout)
446                    .await
447                    .ok_or_else(|| {
448                        not_found(format!(
449                            "missing previous epoch root header {prev_epoch_root_height}"
450                        ))
451                    })?;
452                prev_epoch_root
453                    .l1_finalized()
454                    .ok_or_else(|| Error::Custom {
455                        message: "previous epoch root header is missing L1 finalized block".into(),
456                        status: StatusCode::INTERNAL_SERVER_ERROR,
457                    })?
458                    .number()
459                    + 1
460            } else {
461                0
462            };
463
464            state
465                .stake_table_events(from_l1_block, to_l1_block)
466                .await
467                .map_err(|err| Error::Custom {
468                    message: format!("failed to load stake table events: {err:#}"),
469                    status: StatusCode::INTERNAL_SERVER_ERROR,
470                })
471        }
472        .boxed()
473    })?
474    .get("payload", move |req, state| {
475        async move {
476            let height: usize = req.integer_param("height").map_err(bad_param("height"))?;
477            let fetch_payload = async move {
478                state
479                    .get_payload(height)
480                    .await
481                    .with_timeout(fetch_timeout)
482                    .await
483                    .ok_or_else(|| Error::Custom {
484                        message: format!("missing payload {height}"),
485                        status: StatusCode::NOT_FOUND,
486                    })
487            };
488            let fetch_vid_common = async move {
489                state
490                    .get_vid_common(height)
491                    .await
492                    .with_timeout(fetch_timeout)
493                    .await
494                    .ok_or_else(|| Error::Custom {
495                        message: format!("missing VID common {height}"),
496                        status: StatusCode::NOT_FOUND,
497                    })
498            };
499            let (payload, vid_common) = try_join(fetch_payload, fetch_vid_common).await?;
500            Ok(PayloadProof::new(
501                payload.data().clone(),
502                vid_common.common().clone(),
503            ))
504        }
505        .boxed()
506    })?
507    .get("payload_range", move |req, state| {
508        async move {
509            let start: usize = req.integer_param("start").map_err(bad_param("start"))?;
510            let end: usize = req.integer_param("end").map_err(bad_param("end"))?;
511            let fetch_payloads = async move {
512                state.get_payload_range(start..end).await.enumerate().then(
513                    move |(i, fetch)| async move {
514                        fetch
515                            .with_timeout(fetch_timeout)
516                            .await
517                            .ok_or_else(|| Error::Custom {
518                                message: format!("missing payload {}", start + i),
519                                status: StatusCode::NOT_FOUND,
520                            })
521                    },
522                )
523            };
524            let fetch_vid_commons = async move {
525                state
526                    .get_vid_common_range(start..end)
527                    .await
528                    .enumerate()
529                    .then(move |(i, fetch)| async move {
530                        fetch
531                            .with_timeout(fetch_timeout)
532                            .await
533                            .ok_or_else(|| Error::Custom {
534                                message: format!("missing VID common {}", start + i),
535                                status: StatusCode::NOT_FOUND,
536                            })
537                    })
538            };
539            let (payloads, vid_commons) = join(fetch_payloads, fetch_vid_commons).await;
540            payloads
541                .zip(vid_commons)
542                .map(|(payload, vid_common)| {
543                    Ok(PayloadProof::new(
544                        payload?.data().clone(),
545                        vid_common?.common().clone(),
546                    ))
547                })
548                .try_collect::<Vec<_>>()
549                .await
550        }
551        .boxed()
552    })?
553    .get("namespace", move |req, state| {
554        async move {
555            let height = req.integer_param("height").map_err(bad_param("height"))?;
556            let namespace = req
557                .integer_param("namespace")
558                .map_err(bad_param("namespace"))?;
559            let mut proofs = get_namespace_proof_range(
560                state,
561                height,
562                height + 1,
563                namespace,
564                fetch_timeout,
565                large_object_range_limit,
566            )
567            .await?;
568            if proofs.len() != 1 {
569                tracing::error!(
570                    height,
571                    namespace,
572                    ?proofs,
573                    "get_namespace_proof_range should have returned exactly one proof"
574                );
575                return Err(Error::Custom {
576                    message: "internal consistency error".into(),
577                    status: StatusCode::INTERNAL_SERVER_ERROR,
578                });
579            }
580            Ok(proofs.remove(0))
581        }
582        .boxed()
583    })?
584    .get("namespace_range", move |req, state| {
585        async move {
586            let start = req.integer_param("start").map_err(bad_param("start"))?;
587            let end = req.integer_param("end").map_err(bad_param("end"))?;
588            let namespace = req
589                .integer_param("namespace")
590                .map_err(bad_param("namespace"))?;
591            get_namespace_proof_range(
592                state,
593                start,
594                end,
595                namespace,
596                fetch_timeout,
597                large_object_range_limit,
598            )
599            .await
600        }
601        .boxed()
602    })?;
603
604    Ok(api)
605}

607async fn leaf_from_req<S>(
608    req: &RequestParams,
609    state: &S,
610    fetch_timeout: Duration,
611) -> Result<LeafQueryData<SeqTypes>, Error>
612where
613    S: AvailabilityDataSource<SeqTypes>,
614{
615    let requested = if let Some(height) = req
616        .opt_integer_param::<_, usize>("height")
617        .map_err(bad_param("height"))?
618    {
619        LeafId::Number(height)
620    } else if let Some(hash) = req.opt_blob_param("hash").map_err(bad_param("hash"))? {
621        LeafId::Hash(hash)
622    } else if let Some(hash) = req
623        .opt_blob_param("block-hash")
624        .map_err(bad_param("block-hash"))?
625    {
626        let header = state
627            .get_header(BlockId::Hash(hash))
628            .await
629            .with_timeout(fetch_timeout)
630            .await
631            .ok_or_else(|| not_found(format!("unknown block hash {hash}")))?;
632        LeafId::Number(header.height() as usize)
633    } else if let Some(hash) = req
634        .opt_blob_param("payload-hash")
635        .map_err(bad_param("payload-hash"))?
636    {
637        let header = state
638            .get_header(BlockId::PayloadHash(hash))
639            .await
640            .with_timeout(fetch_timeout)
641            .await
642            .ok_or_else(|| not_found(format!("unknown payload hash {hash}")))?;
643        LeafId::Number(header.height() as usize)
644    } else {
645        return Err(Error::Custom {
646            message: "missing parameter: requested leaf must be identified by height, hash, block \
647                      hash, or payload hash"
648                .into(),
649            status: StatusCode::BAD_REQUEST,
650        });
651    };
652
653    state
654        .get_leaf(requested)
655        .await
656        .with_timeout(fetch_timeout)
657        .await
658        .ok_or_else(|| not_found(format!("unknown leaf {requested}")))
659}

661fn block_id_from_req(req: &RequestParams) -> Result<BlockId<SeqTypes>, Error> {
662    if let Some(height) = req
663        .opt_integer_param("height")
664        .map_err(bad_param("height"))?
665    {
666        Ok(BlockId::Number(height))
667    } else if let Some(hash) = req.opt_blob_param("hash").map_err(bad_param("hash"))? {
668        Ok(BlockId::Hash(hash))
669    } else if let Some(hash) = req
670        .opt_blob_param("payload-hash")
671        .map_err(bad_param("payload-hash"))?
672    {
673        Ok(BlockId::PayloadHash(hash))
674    } else {
675        Err(Error::Custom {
676            message: "missing parameter: requested header must be identified by height, hash, or \
677                      payload hash"
678                .into(),
679            status: StatusCode::BAD_REQUEST,
680        })

684fn bad_param<E>(name: &'static str) -> impl FnOnce(E) -> Error
685where
686    E: Display,
687{
688    move |err| Error::Custom {
689        message: format!("{name}: {err:#}"),
690        status: StatusCode::BAD_REQUEST,
691    }
692}

694fn internal(err: impl Display) -> Error {
695    Error::Custom {
696        message: err.to_string(),
697        status: StatusCode::INTERNAL_SERVER_ERROR,
698    }

702    Error::Custom {
703        message: msg.into(),
704        status: StatusCode::NOT_FOUND,

1036    async fn test_header_proof() {
1037        let storage = <DataSource as TestableSequencerDataSource>::create_storage().await;
1038        let ds = DataSource::create(
1039            DataSource::persistence_options(&storage),
1040            Default::default(),
1041            false,
1042        )

1044        .unwrap();
1045
1046        // Construct a chain of leaves, plus the corresponding block Merkle tree at each leaf.
1047        let leaves = leaf_chain(0..=2, EPOCH_VERSION).await;
1048        let mts = leaves
1049            .iter()
1050            .scan(
1051                BlockMerkleTree::new(BLOCK_MERKLE_TREE_HEIGHT),
1052                |mt, leaf| {
1053                    assert_eq!(mt.commitment(), leaf.header().block_merkle_tree_root());
1054                    let item = mt.clone();
1055                    mt.push(leaf.block_hash()).unwrap();
1056                    Some(item)
1057                },
1058            )
1059            .collect::<Vec<_>>();
1060
1061        // Save all those objects in the DB.
1062        {
1063            let mut tx = ds.write().await.unwrap();
1064            for (leaf, mt) in leaves.iter().zip(&mts) {
1065                tx.insert_leaf(leaf).await.unwrap();
1066
1067                if leaf.height() > 0 {
1068                    let merkle_path = mt.lookup(leaf.height() - 1).expect_ok().unwrap().1;
1069                    UpdateStateData::<SeqTypes, BlockMerkleTree, _>::insert_merkle_nodes(
1070                        &mut tx,
1071                        merkle_path,
1072                        ToTraversalPath::<{ BlockMerkleTree::ARITY }>::to_traversal_path(
1073                            &(leaf.height() - 1),
1074                            BLOCK_MERKLE_TREE_HEIGHT,
1075                        ),
1076                        leaf.height(),
1077                    )
1078                    .await
1079                    .unwrap();
1080                    UpdateStateData::<SeqTypes, BlockMerkleTree, _>::set_last_state_height(
1081                        &mut tx,
1082                        leaf.height() as usize,
1083                    )
1084                    .await
1085                    .unwrap();
1086                }
1087            }
1088            tx.commit().await.unwrap();
1089        }
1090
1091        // Test happy path.
1092        for (root, mt) in mts.iter().enumerate().skip(1) {
1093            for (height, leaf) in leaves.iter().enumerate().take(root) {
1094                tracing::info!(root, height, "test happy path");
1095                let proof =
1096                    get_header_proof(&ds, root as u64, BlockId::Number(height), Duration::MAX)
1097                        .await
1098                        .unwrap();
1099                assert_eq!(proof.verify_ref(mt.commitment()).unwrap(), leaf.header());
1100            }
1101        }
1102
1103        // Test unknown leaf.
1104        let err = get_header_proof(&ds, 5, BlockId::Number(4), Duration::from_secs(1))
1105            .await
1106            .unwrap_err();
1107        assert_eq!(err.status(), StatusCode::NOT_FOUND);
1108

1112            .unwrap_err();
