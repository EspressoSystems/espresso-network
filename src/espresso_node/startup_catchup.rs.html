1//! Startup stake-table catchup for new-protocol nodes.
2//!
3//! Cliquenet only connects to validators in the current epoch's stake table
4//! window (`N-1`, `N`, `N+1`). On a fresh-join or cold-restart node, no
5//! consensus messages can be received until those stake tables are populated,
6//! so the existing reactive catchup (triggered by an unknown-epoch proposal)
7//! never fires.
8//!
9//! [`bootstrap_epoch_window`] drives the existing catchup machinery
10//! synchronously at startup: it walks forward one epoch at a time from the
11//! highest already-known epoch (loaded from persistence by `reload_stake`)
12//! until peers can no longer serve the next epoch root leaf — which is the
13//! point at which the live network currently is.
14
15use std::time::Duration;
16
17use anyhow::{Context, ensure};
18use espresso_types::SeqTypes;
19use hotshot_types::{
20    data::EpochNumber, epoch_membership::EpochMembershipCoordinator, traits::election::Membership,
21};
22
23/// Walk forward from the highest already-known epoch until peers can no
24/// longer serve the next epoch root leaf, populating the membership with
25/// stake tables for every epoch up through `N+1` (where `N` is the current
26/// epoch). Returns `N`.
27///
28/// Preconditions: `reload_stake` should have run before this — it populates
29/// the membership from local persistence so the walk skips epochs we
30/// already know.
31pub async fn bootstrap_epoch_window(
32    coordinator: &EpochMembershipCoordinator<SeqTypes>,
33    epoch_height: u64,
34    step_timeout: Duration,
35) -> anyhow::Result<EpochNumber> {
36    if epoch_height == 0 {
37        // Pre-epoch chain: epochs aren't enabled yet, the non-epoch
38        // committee path is what gets used.
39        return Ok(EpochNumber::genesis());
40    }
41
42    let membership = coordinator.membership();
43    let first_epoch = membership
44        .read()
45        .await
46        .first_epoch()
47        .context("first_epoch not seeded; genesis stake table missing")?;
48
49    // Find the highest contiguous pair `(H, H-1)` already in memory. Both
50    // are needed as the starting point of the forward walk: `add_epoch_root`
51    // for epoch `K+2` requires the stake table at `K`, so to derive both
52    // `H+1` (needs `H-1`) and `H+2` (needs `H`) we need `H` and `H-1`
53    // present. If only `H` is present (e.g. `set_first_epoch` ran without a
54    // matching reload, or persistence has gaps near the tip), the walk's
55    // first iteration would otherwise fall into a deep walk-back that may
56    // be unfillable from peers and would silently terminate the bootstrap
57    // at a stale epoch.
58    //
59    // `set_first_epoch` always seeds `first_epoch` and `first_epoch + 1`,
60    // so the scan terminates at worst at `first_epoch + 1`.
61    let mut highest = {
62        let m = membership.read().await;
63        let initial = m.highest_known_epoch().unwrap_or(first_epoch + 1);
64        let mut h = initial;
65        while h > first_epoch + 1 && !(m.has_stake_table(h) && m.has_stake_table(h - 1)) {
66            h = h - 1;
67        }
68        h
69    };
70
71    tracing::info!(
72        %first_epoch,
73        starting_from = %highest,
74        "bootstrap_epoch_window: walking forward",
75    );
76
77    // Walk forward; each successful iteration drives `add_epoch_root` via
78    // the existing catchup machinery, persisting the new stake table. The
79    // walk terminates when the catchup chain returns an error
80    // (`Ok(Err(_))`) or a single step exceeds the hang bound (`Err(_)`).
81    loop {
82        let target = highest + 1;
83        let result =
84            tokio::time::timeout(step_timeout, coordinator.wait_for_stake_table(target)).await;
85        match result {
86            Ok(Ok(_)) => {
87                tracing::info!(%target, "bootstrap_epoch_window: derived stake table");
88                highest = target;
89            },
90            Ok(Err(err)) => {
91                tracing::info!(
92                    %target,
93                    %err,
94                    "bootstrap_epoch_window: catchup failed; treating as live tip",
95                );
96                break;
97            },
98            Err(_) => {
99                tracing::info!(
100                    %target,
101                    timeout_secs = step_timeout.as_secs(),
102                    "bootstrap_epoch_window: catchup timed out; treating as live tip",
103                );
104                break;
105            },
106        }
107    }
108
109    // `highest` corresponds to N+1 (the leaf at root_block_in_epoch(N-1) is
110    // the last finalized one peers can serve). So current epoch N = highest - 1.
111    let current = EpochNumber::new(highest.saturating_sub(1));
112
113    let m = membership.read().await;
114    ensure!(
115        m.has_stake_table(current),
116        "missing stake table for current epoch {current} after bootstrap"
117    );
118    ensure!(
119        m.has_stake_table(highest),
120        "missing stake table for next epoch {highest} after bootstrap"
121    );
122
123    tracing::info!(%current, "bootstrap_epoch_window: complete");
124    Ok(current)
125}
