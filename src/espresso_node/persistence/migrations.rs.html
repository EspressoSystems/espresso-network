9pub struct BackfillHash;
10
11#[async_trait]
12impl DataBackfill for BackfillHash {
13    fn name(&self) -> &'static str {
14        "hash_bigint_backfill_hash"
15    }
16
17    async fn run_batch(
18        &self,
19        tx: &mut Transaction<Write>,
20        // Reused as last-seen id (keyset cursor), not a row count.
21        // Initial value 0 is safe because auto-increment ids start at 1.
22        offset: u64,
23    ) -> anyhow::Result<Option<u64>> {
24        let rows: Vec<(i32, Vec<u8>)> =
25            sqlx::query_as("SELECT id, value FROM hash WHERE id > $1 ORDER BY id LIMIT $2")
26                .bind(offset as i64)
27                .bind(self.batch_size() as i64)
28                .fetch_all(tx.as_mut())
29                .await?;
30
31        if rows.is_empty() {
32            return Ok(None);
33        }
34        let n = rows.len();
35        let last_id = rows.last().expect("non-empty").0 as u64;
36
37        let (ids, values): (Vec<i64>, Vec<Vec<u8>>) = rows
38            .into_iter()
39            .map(|(id, value)| (id as i64, value))
40            .unzip();
41
42        sqlx::query(
43            "INSERT INTO hash_bigint (id, value)
44             SELECT * FROM UNNEST($1::bigint[], $2::bytea[])
45             ON CONFLICT DO NOTHING",
46        )
47        .bind(&ids)
48        .bind(&values)
49        .execute(tx.as_mut())
50        .await?;
51
52        if n < self.batch_size() {
53            Ok(None)
54        } else {
55            Ok(Some(last_id))
56        }
57    }
58}

60macro_rules! merkle_tree_backfill {
61    ($struct_name:ident, $migration_name:literal, $legacy_table:literal, $new_table:literal) => {
62        pub struct $struct_name;
63
64        #[async_trait]
65        impl DataBackfill for $struct_name {
66            fn name(&self) -> &'static str {
67                $migration_name
68            }
69
70            fn requires(&self) -> &'static [&'static str] {
71                &["hash_bigint_backfill_hash"]
72            }
73
74            async fn run_batch(
75                &self,
76                tx: &mut Transaction<Write>,
77                // Cursor: the start of the `created` (block height) range for this batch.
78                // Each batch covers [offset, offset + batch_size), using the index on `created`.
79                offset: u64,
80            ) -> anyhow::Result<Option<u64>> {
81                let batch_size = self.batch_size() as i64;
82
83                // Check if any rows remain at or beyond the current offset. Checking only the
84                // current window would cause early termination if block heights have gaps larger
85                // than batch_size; checking the open-ended tail means gaps are just a few fast
86                // no-op iterations.
87                let any: Option<(i64,)> = sqlx::query_as(concat!(
88                    "SELECT created FROM ",
89                    $legacy_table,
90                    " WHERE created >= $1 LIMIT 1"
91                ))
92                .bind(offset as i64)
93                .fetch_optional(tx.as_mut())
94                .await?;
95                if any.is_none() {
96                    return Ok(None);
97                }
98
99                // Move rows from legacy into the new _bigint table, translating both
100                // `hash_id` and every element of `children` from legacy hash ids into
101                // hash_bigint ids by joining on the hash `value`. This removes any
102                // dependency on `BackfillHash` having preserved the original ids.
103                sqlx::query(concat!(
104                    "INSERT INTO ",
105                    $new_table,
106                    " (path, created, hash_id, children, children_bitvec, idx, entry)
107                     SELECT
108                         legacy_merkle_node.path,
109                         legacy_merkle_node.created,
110                         new_hash.id,
111                         CASE WHEN legacy_merkle_node.children IS NULL THEN NULL
112                              ELSE COALESCE((
113                                  SELECT jsonb_agg(child_new_hash.id ORDER BY \
114                     child_position)
115                                  FROM \
116                     jsonb_array_elements_text(legacy_merkle_node.children)
117                                       WITH ORDINALITY AS child_elem(legacy_hash_id, \
118                     child_position)
119                                  JOIN hash AS child_legacy_hash
120                                    ON child_legacy_hash.id = \
121                     child_elem.legacy_hash_id::INT
122                                  JOIN hash_bigint AS child_new_hash
123                                    ON child_new_hash.value = child_legacy_hash.value
124                              ), '[]'::jsonb)
125                         END,
126                         legacy_merkle_node.children_bitvec,
127                         legacy_merkle_node.idx,
128                         legacy_merkle_node.entry
129                     FROM ",
130                    $legacy_table,
131                    " AS legacy_merkle_node
132                     JOIN hash AS legacy_hash ON legacy_hash.id = \
133                     legacy_merkle_node.hash_id
134                     JOIN hash_bigint AS new_hash ON new_hash.value = legacy_hash.value
135                     WHERE legacy_merkle_node.created >= $1 AND legacy_merkle_node.created \
136                     < $2
137                     ON CONFLICT (path, created) DO NOTHING"
138                ))
139                .bind(offset as i64)
140                .bind(offset as i64 + batch_size)
141                .execute(tx.as_mut())
142                .await?;
143
144                // Delete moved rows from the legacy table in the same transaction so that
145                // storage stays roughly flat during the migration (move, not copy).
146                sqlx::query(concat!(
147                    "DELETE FROM ",
148                    $legacy_table,
149                    " WHERE created >= $1 AND created < $2"
150                ))
151                .bind(offset as i64)
152                .bind(offset as i64 + batch_size)
153                .execute(tx.as_mut())
154                .await?;
155
156                Ok(Some(offset + self.batch_size() as u64))
157            }
158        }
159    };
160}

175pub struct CleanupLegacyHashTable;
176
177#[async_trait]
178impl DataBackfill for CleanupLegacyHashTable {
179    fn name(&self) -> &'static str {
180        "hash_bigint_cleanup_legacy_hash"
181    }
182
183    fn requires(&self) -> &'static [&'static str] {
184        &[
185            "hash_bigint_backfill_fee_merkle_tree",
186            "hash_bigint_backfill_block_merkle_tree",
187        ]
188    }
189
190    async fn run_batch(
191        &self,
192        tx: &mut Transaction<Write>,
193        // Keyset cursor: last deleted id (0 on first batch).
194        offset: u64,
195    ) -> anyhow::Result<Option<u64>> {
196        // Both merkle tree tables are now empty so there are no FK references to hash.id.
197        let deleted: Vec<(i64,)> = sqlx::query_as(
198            "DELETE FROM hash WHERE id IN (
199                SELECT id FROM hash WHERE id > $1 ORDER BY id LIMIT $2
200             ) RETURNING id",
201        )
202        .bind(offset as i64)
203        .bind(self.batch_size() as i64)
204        .fetch_all(tx.as_mut())
205        .await?;
206
207        Ok(deleted.last().map(|(id,)| *id as u64))
208    }
209}

244    async fn run_to_completion(
245        backfill: &dyn DataBackfill,
246        storage: &SqlStorage,
247    ) -> anyhow::Result<()> {
248        let mut offset = 0u64;
249        loop {
250            let mut tx = storage.write().await?;
251            let next = backfill.run_batch(&mut tx, offset).await?;
252            tx.commit().await?;
253            match next {
254                Some(o) => offset = o,
255                None => return Ok(()),
256            }
257        }
258    }

262        tree: &FeeMerkleTree,
263        accounts: &[FeeAccount],

282        UpdateStateData::<SeqTypes, FeeMerkleTree, { FeeMerkleTree::ARITY }>::insert_merkle_nodes_batch(
283            tx,
284            proofs,
285            block_height,
286        )
287        .await
288        .expect("insert_merkle_nodes_batch");
289    }

329    /// Regression test: reads during the backfill window must return the latest
330    /// version of a node, not a stale one.
331    ///
332    /// Each Merkle node row is keyed `(path, created)` where `created` is the block
333    /// height at which the node changed, so a single path accumulates one row per
334    /// such height. The backfill moves rows by ascending `created`, so mid-migration
335    /// a path's history is split: an older `created` lives in `*_bigint` while a
336    /// newer `created` still lives in the legacy table.
337    ///
338    /// `get_path`'s legacy fallback keys on whether the path exists in `*_bigint` at
339    /// all, not on `created`. With any older version present in `*_bigint` the path
340    /// counts as "found", legacy is never consulted, and the snapshot read returns
341    /// the stale older version. The reconstructed commitment then fails to match the
342    /// header root and `get_path` errors (or, absent that check, returns wrong data).
343    #[test_log::test(tokio::test(flavor = "multi_thread"))]
344    async fn read_during_backfill_returns_latest_version() {
345        let db = TmpDb::init().await;
346        let opt = tmp_options(&db);
347        let cfg = Config::try_from(&opt).expect("config");
348        let storage = SqlStorage::connect(cfg, StorageConnectionType::Query)
349            .await
350            .expect("connect");
351
352        let account = FeeAccount::from(Address::repeat_byte(0x42));
353        let mut tree = FeeMerkleTree::new(FEE_MERKLE_TREE_HEIGHT);
354
355        // Height 1: account = 100. Live write goes to the *_bigint tables.
356        tree.update(account, FeeAmount::from(100u64)).unwrap();
357        let proof_v1 = membership_proof(&tree, &account);
358        let commit_v1 = tree.commitment();
359        let mut tx = storage.write().await.unwrap();
360        write_fee_merkle_proofs(&mut tx, &tree, &[account], 1).await;
361        tx.commit().await.unwrap();
362
363        // Height 2: account = 200. Also written to the *_bigint tables.
364        tree.update(account, FeeAmount::from(200u64)).unwrap();
365        let proof_v2 = membership_proof(&tree, &account);
366        let commit_v2 = tree.commitment();
367        let mut tx = storage.write().await.unwrap();
368        write_fee_merkle_proofs(&mut tx, &tree, &[account], 2).await;
369        tx.commit().await.unwrap();
370
371        // End-state of the mid-migration split, constructed in reverse: both
372        // writes went to *_bigint above, now move the height-2 row across to
373        // legacy. The natural flow is legacy → *_bigint by ascending `created`,
374        // but the resulting (older in *_bigint, newer in legacy) shape is the
375        // same. Also seed legacy `hash` so the legacy table's hash_id FK resolves.
376        let mut tx = storage.write().await.unwrap();
377        sqlx::query(
378            "INSERT INTO hash (id, value) SELECT id::INT, value FROM hash_bigint ON CONFLICT DO \
379             NOTHING",
380        )
381        .execute(tx.as_mut())
382        .await
383        .unwrap();
384        sqlx::query(
385            "INSERT INTO fee_merkle_tree (path, created, hash_id, children, children_bitvec, idx, \
386             entry) SELECT path, created, hash_id::INT, children, children_bitvec::BIT(256), idx, \
387             entry FROM fee_merkle_tree_bigint WHERE created = 2",
388        )
389        .execute(tx.as_mut())
390        .await
391        .unwrap();
392        sqlx::query("DELETE FROM fee_merkle_tree_bigint WHERE created = 2")
393            .execute(tx.as_mut())
394            .await
395            .unwrap();
396        tx.commit().await.unwrap();
397
398        // Seed headers carrying each height's fee root, and mark height 2 decided.
399        let mut tx = storage.write().await.unwrap();
400        insert_fee_header(&mut tx, 1, commit_v1).await;
401        insert_fee_header(&mut tx, 2, commit_v2).await;
402        UpdateStateData::<SeqTypes, FeeMerkleTree, { FeeMerkleTree::ARITY }>::set_last_state_height(
403            &mut tx, 2,
404        )
405        .await
406        .unwrap();
407        tx.commit().await.unwrap();
408
409        // Control: height 1 is served from the *_bigint row and is correct.
410        let mut tx = storage.read().await.unwrap();
411        let got_v1 = tx
412            .get_path(
413                Snapshot::<SeqTypes, FeeMerkleTree, { FeeMerkleTree::ARITY }>::Index(1),
414                account,
415            )
416            .await
417            .expect("get_path at height 1");
418        assert_eq!(got_v1, proof_v1, "height 1 should return the V1 proof");
419
420        // Bug: height 2's latest node lives in legacy, but *_bigint still holds the
421        // height-1 version of the same path, so the fallback returns it stale.
422        let mut tx = storage.read().await.unwrap();
423        let got_v2 = tx
424            .get_path(
425                Snapshot::<SeqTypes, FeeMerkleTree, { FeeMerkleTree::ARITY }>::Index(2),
426                account,
427            )
428            .await
429            .expect("get_path at height 2 (latest version is in the legacy table)");
430        assert_eq!(
431            got_v2, proof_v2,
432            "height 2 must return the latest (V2) proof, not the stale V1 row from *_bigint"
433        );
434    }

436    /// Regression test for the FK race between `BackfillHash` and live writes
437    /// to `hash_bigint`.
438    ///
439    /// V1501 seeds the `hash_bigint(id)` sequence above `MAX(hash.id)` so new
440    /// auto-ids cannot collide with legacy ids — but nothing protects the
441    /// `value` UNIQUE constraint. Whenever a post-migration write inserts a
442    /// value that also lives in legacy `hash` (the common case: empty-subtree
443    /// hashes and unchanged branch hashes are byte-identical across blocks),
444    /// the live row claims a new id, and the backfill's `INSERT (old_id, value)
445    /// ON CONFLICT DO NOTHING` is silently dropped. The Merkle tree backfill
446    /// then copies the legacy `hash_id` verbatim and the FK to `hash_bigint(id)`
447    /// fires because that id was never inserted.
448    ///
449    /// This test exercises the real `UpdateStateData::insert_merkle_nodes_batch`
450    /// path so the shared-hash overlap arises from realistic Merkle proofs.
451    #[test_log::test(tokio::test(flavor = "multi_thread"))]
452    async fn backfill_preserves_fk_when_live_write_shares_hash_value() {
453        let db = TmpDb::init().await;
454        let opt = tmp_options(&db);
455        let cfg = Config::try_from(&opt).expect("config");
456        let storage = SqlStorage::connect(cfg, StorageConnectionType::Query)
457            .await
458            .expect("connect");
459
460        let mut tree = FeeMerkleTree::new(FEE_MERKLE_TREE_HEIGHT);
461        let account = FeeAccount::from(Address::repeat_byte(0x42));
462        tree.update(account, FeeAmount::from(100_u64)).unwrap();
463
464        // Write a real Merkle proof for `account` into the *new* _bigint tables.
465        let mut tx = storage.write().await.unwrap();
466        write_fee_merkle_proofs(&mut tx, &tree, &[account], 1).await;
467        tx.commit().await.unwrap();
468
469        // Move every row from the _bigint tables back into the legacy tables to
470        // simulate a database that was populated before V1501 ran. Then reset
471        // the hash_bigint sequence the way V1501 itself does.
472        let mut tx = storage.write().await.unwrap();
473        sqlx::query("INSERT INTO hash (id, value) SELECT id::INT, value FROM hash_bigint")
474            .execute(tx.as_mut())
475            .await
476            .unwrap();
477        sqlx::query(
478            "INSERT INTO fee_merkle_tree (path, created, hash_id, children, children_bitvec, idx, \
479             entry) SELECT path, created, hash_id::INT, children, children_bitvec::BIT(256), idx, \
480             entry FROM fee_merkle_tree_bigint",
481        )
482        .execute(tx.as_mut())
483        .await
484        .unwrap();
485        sqlx::query("TRUNCATE fee_merkle_tree_bigint, block_merkle_tree_bigint, hash_bigint")
486            .execute(tx.as_mut())
487            .await
488            .unwrap();
489        sqlx::query(
490            "SELECT setval(pg_get_serial_sequence('hash_bigint', 'id'), GREATEST(COALESCE((SELECT \
491             MAX(id) FROM hash), 1), 1))",
492        )
493        .execute(tx.as_mut())
494        .await
495        .unwrap();
496        tx.commit().await.unwrap();
497
498        // Live post-V1501 write at a new block height. The proof for the same
499        // account shares almost every hash value with the legacy proof above,
500        // so the live `batch_insert_hashes` calls collide on `value` with every
501        // row BackfillHash is about to copy.
502        let mut tx = storage.write().await.unwrap();
503        write_fee_merkle_proofs(&mut tx, &tree, &[account], 2).await;
504        tx.commit().await.unwrap();
505
506        // Drive backfills directly so a failure surfaces immediately rather
507        // than entering the registry's 5-minute retry loop.
508        run_to_completion(&BackfillHash, &storage)
509            .await
510            .expect("BackfillHash failed");
511        run_to_completion(&BackfillFeeMerkleTree, &storage)
512            .await
513            .expect("BackfillFeeMerkleTree failed (FK violation from dropped hash row)");
514
515        let mut tx = storage.read().await.unwrap();
516        let (n_legacy,): (i64,) = sqlx::query_as("SELECT COUNT(*) FROM fee_merkle_tree")
517            .fetch_one(tx.as_mut())
518            .await
519            .unwrap();
520        assert_eq!(n_legacy, 0, "legacy fee_merkle_tree rows were not migrated");
521
522        let (n_heights,): (i64,) =
523            sqlx::query_as("SELECT COUNT(DISTINCT created) FROM fee_merkle_tree_bigint")
524                .fetch_one(tx.as_mut())
525                .await
526                .unwrap();
527        assert_eq!(n_heights, 2, "expected rows at both heights");
528
529        let (n_orphans,): (i64,) = sqlx::query_as(
530            "SELECT COUNT(*) FROM fee_merkle_tree_bigint fmt LEFT JOIN hash_bigint hb ON hb.id = \
531             fmt.hash_id WHERE hb.id IS NULL",
532        )
533        .fetch_one(tx.as_mut())
534        .await
535        .unwrap();
536        assert_eq!(n_orphans, 0, "fee_merkle_tree_bigint has dangling hash_id");
537
538        let (n_orphan_children,): (i64,) = sqlx::query_as(
539            "SELECT COUNT(*) FROM ( SELECT child_id FROM fee_merkle_tree_bigint fmt, \
540             jsonb_array_elements_text(fmt.children) AS arr(child_id) WHERE fmt.children IS NOT \
541             NULL ) c LEFT JOIN hash_bigint hb ON hb.id = c.child_id::BIGINT WHERE hb.id IS NULL",
542        )
543        .fetch_one(tx.as_mut())
544        .await
545        .unwrap();
546        assert_eq!(
547            n_orphan_children, 0,
548            "fee_merkle_tree_bigint.children has dangling hash_id"
549        );
550    }
