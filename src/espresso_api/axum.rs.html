34/// API error response
35#[derive(Debug, Serialize, schemars::JsonSchema)]
36struct ErrorResponse {
37    error: String,
38}
39
40impl IntoResponse for ApiError {
41    fn into_response(self) -> Response {
42        let status = match &self {
43            ApiError::BadRequest(_) => StatusCode::BAD_REQUEST,
44            ApiError::NotFound(_) => StatusCode::NOT_FOUND,
45            ApiError::Internal(_) => StatusCode::INTERNAL_SERVER_ERROR,
46        };
47
48        let body = Json(ErrorResponse {
49            error: self.to_string(),
50        });
51
52        (status, body).into_response()
53    }
54}

56/// Classify an `anyhow::Error` from an availability handler into the appropriate `ApiError`
57/// variant. Errors produced via [`AvailabilityError`] in the state implementation carry semantic
58/// meaning; everything else falls back to a 500 Internal Server Error.
59fn classify_availability_error(err: anyhow::Error) -> ApiError {
60    let is_not_found = err
61        .downcast_ref::<AvailabilityError>()
62        .map(|e| matches!(e, AvailabilityError::NotFound(_)));
63    match is_not_found {
64        Some(true) => ApiError::NotFound(err),
65        Some(false) => ApiError::BadRequest(err),
66        None => ApiError::Internal(err),
67    }
68}
69
70impl OperationOutput for ApiError {
71    type Inner = Self;
72}

84/// Middleware to rewrite root paths to /v2 paths
85///
86/// Requests to `/rewards/...` get rewritten to `/v2/rewards/...`
87/// Paths already prefixed with `/v2` are left unchanged
88///
89/// Note: This middleware is only applied to the v2 router, so v1 routes never pass through it
90async fn rewrite_root_to_v2(mut req: Request, next: Next) -> Response {
91    let uri = req.uri().clone();
92    let path = uri.path();
93
94    // Only rewrite unversioned paths (not starting with /v2)
95    if !path.starts_with("/v2") && path != "/" {
96        let new_path = format!("/v2{}", path);
97        let pq = if let Some(q) = uri.query() {
98            format!("{}?{}", new_path, q)
99        } else {
100            new_path
101        };
102        if let Ok(new_uri) = Uri::builder().path_and_query(pq).build() {
103            *req.uri_mut() = new_uri;
104        }
105    }
106
107    next.run(req).await
108}

115struct SendQuery<T>(T);
116
117impl<T, S> axum::extract::FromRequestParts<S> for SendQuery<T>
118where
119    T: serde::de::DeserializeOwned + Send,
120    S: Send + Sync,
121{
122    type Rejection = axum::extract::rejection::QueryRejection;
123
124    async fn from_request_parts(
125        parts: &mut axum::http::request::Parts,
126        state: &S,
127    ) -> Result<Self, Self::Rejection> {
128        axum::extract::Query::<T>::from_request_parts(parts, state)
129            .await
130            .map(|axum::extract::Query(inner)| SendQuery(inner))
131    }
132}

134impl<T: schemars::JsonSchema> aide::operation::OperationInput for SendQuery<T> {
135    fn operation_input(
136        ctx: &mut aide::generate::GenContext,
137        operation: &mut aide::openapi::Operation,
138    ) {
139        let schema = ctx.schema.subschema_for::<T>();
140        let params = aide::operation::parameters_from_schema(
141            ctx,
142            schema,
143            aide::operation::ParamLocation::Query,
144        );
145        aide::operation::add_parameters(ctx, operation, params);
146    }
147}

149async fn drive_ws_stream<T: Serialize>(
150    mut socket: axum::extract::ws::WebSocket,
151    stream: BoxStream<'static, T>,
152) {
153    use futures::StreamExt as _;
154    futures::pin_mut!(stream);
155    while let Some(item) = stream.next().await {
156        let Ok(json) = serde_json::to_string(&item) else {
157            break;
158        };
159        if socket
160            .send(axum::extract::ws::Message::Text(json.into()))
161            .await
162            .is_err()
163        {
164            break;
165        }
166    }
167}

169/// Create a combined router serving both v1 and v2 APIs
170pub fn create_combined_router<S>(state: S) -> Router
171where
172    S: v1::RewardApi
173        + v1::AvailabilityApi
174        + v1::HotShotAvailabilityApi
175        + v1::BlockStateApi
176        + v1::FeeStateApi
177        + v2::RewardApi
178        + v2::DataApi
179        + v2::ConsensusApi
180        + Clone
181        + Send
182        + Sync
183        + 'static,
184{
185    let router_v1 = create_router_v1(state.clone());
186    let router_v2 = create_router_v2(state).layer(middleware::from_fn(rewrite_root_to_v2));
187
188    router_v2.merge(router_v1).route("/", get(redirect_to_docs))
189}

926        };

931    >| async move {

978            }),
979        )
980        .api_route(
981            routes::v2::INCORRECT_ENCODING_PROOF_ROUTE.http,
982            get_with(get_incorrect_encoding_proof, |op| {
983                op.description(routes::v2::INCORRECT_ENCODING_PROOF_ROUTE.description)
984                    .tag(routes::v2::INCORRECT_ENCODING_PROOF_ROUTE.tag)
985            }),
986        )
987        .api_route(
988            routes::v2::STATE_CERTIFICATE_ROUTE.http,
989            get_with(get_state_certificate, |op| {
990                op.description(routes::v2::STATE_CERTIFICATE_ROUTE.description)
991                    .tag(routes::v2::STATE_CERTIFICATE_ROUTE.tag)
992            }),
993        )
994        .api_route(
995            routes::v2::STAKE_TABLE_ROUTE.http,
996            get_with(get_stake_table, |op| {
997                op.description(routes::v2::STAKE_TABLE_ROUTE.description)
998                    .tag(routes::v2::STAKE_TABLE_ROUTE.tag)
999            }),
1000        )
1001        .finish_api(&mut api);
1002
1003    // Transform examples (array) to example (singular) for OpenAPI 3.0/Swagger compatibility
1004    if let Some(ref mut components) = api.components {
1005        let mut transform = schemars::transform::SetSingleExample::default();
1006        for schema in components.schemas.values_mut() {
1007            transform.transform(&mut schema.json_schema);
1008        }
1009    }
1010
1011    // Also transform path parameter schemas
1012    if let Some(ref mut paths) = api.paths {
1013        let mut transform = schemars::transform::SetSingleExample::default();
1014        for path_item_ref in paths.paths.values_mut() {
1015            if let aide::openapi::ReferenceOr::Item(path_item) = path_item_ref {
1016                for operation in [
1017                    &mut path_item.get,
1018                    &mut path_item.post,
1019                    &mut path_item.put,
1020                    &mut path_item.delete,
1021                    &mut path_item.patch,
1022                ]
1023                .into_iter()
1024                .flatten()
1025                {
1026                    for param in &mut operation.parameters {
1027                        if let aide::openapi::ReferenceOr::Item(param_item) = param {
1028                            let parameter_data = match param_item {
1029                                aide::openapi::Parameter::Query { parameter_data, .. } => {
1030                                    parameter_data
1031                                },
1032                                aide::openapi::Parameter::Header { parameter_data, .. } => {
1033                                    parameter_data
1034                                },
1035                                aide::openapi::Parameter::Path { parameter_data, .. } => {
1036                                    parameter_data
1037                                },
1038                                aide::openapi::Parameter::Cookie { parameter_data, .. } => {
1039                                    parameter_data
1040                                },
1041                            };
1042                            if let aide::openapi::ParameterSchemaOrContent::Schema(ref mut schema) =
1043                                parameter_data.format
1044                            {
1045                                transform.transform(&mut schema.json_schema);
1046                            }
1047                        }
1048                    }
1049                }
1050            }
1051        }
1052    }
1053
1054    router
1055        .route(routes::v2::OPENAPI_SPEC_ROUTE, get(serve_openapi_spec))
1056        .route(routes::v2::SWAGGER_ROUTE, get(serve_swagger_ui))
1057        .route("/v2/", get(serve_swagger_ui))
1058        .route(
1059            routes::v2::SCALAR_ROUTE,
1060            get(Scalar::new(routes::v2::OPENAPI_SPEC_ROUTE)
1061                .with_title("Espresso Node API v2")
1062                .axum_handler()),
1063        )
1064        .route(
1065            routes::v2::REDOC_ROUTE,
1066            get(Redoc::new(routes::v2::OPENAPI_SPEC_ROUTE)
1067                .with_title("Espresso Node API v2")
1068                .axum_handler()),
1069        )
1070        .layer(Extension(api))
1071        .with_state(state)
