30/// API error response
31#[derive(Debug, Serialize, schemars::JsonSchema)]
32struct ErrorResponse {
33    error: String,
34}
35
36impl IntoResponse for ApiError {
37    fn into_response(self) -> Response {
38        let status = match self {
39            ApiError::BadRequest(_) => StatusCode::BAD_REQUEST,
40            ApiError::Internal(_) => StatusCode::INTERNAL_SERVER_ERROR,
41        };
42
43        let body = Json(ErrorResponse {
44            error: self.to_string(),
45        });
46
47        (status, body).into_response()
48    }
49}
50
51impl OperationOutput for ApiError {
52    type Inner = Self;
53}

65/// Middleware to rewrite root paths to /v2 paths
66///
67/// Requests to `/rewards/...` get rewritten to `/v2/rewards/...`
68/// Paths already prefixed with `/v2` are left unchanged
69///
70/// Note: This middleware is only applied to the v2 router, so v1 routes never pass through it
71async fn rewrite_root_to_v2(mut req: Request, next: Next) -> Response {
72    let uri = req.uri().clone();
73    let path = uri.path();
74
75    // Only rewrite unversioned paths (not starting with /v2)
76    if !path.starts_with("/v2") && path != "/" {
77        let new_path = format!("/v2{}", path);
78        let pq = if let Some(q) = uri.query() {
79            format!("{}?{}", new_path, q)
80        } else {
81            new_path
82        };
83        if let Ok(new_uri) = Uri::builder().path_and_query(pq).build() {
84            *req.uri_mut() = new_uri;
85        }
86    }
87
88    next.run(req).await
89}

96struct SendQuery<T>(T);
97
98impl<T, S> axum::extract::FromRequestParts<S> for SendQuery<T>
99where
100    T: serde::de::DeserializeOwned + Send,
101    S: Send + Sync,
102{
103    type Rejection = axum::extract::rejection::QueryRejection;
104
105    async fn from_request_parts(
106        parts: &mut axum::http::request::Parts,
107        state: &S,
108    ) -> Result<Self, Self::Rejection> {
109        axum::extract::Query::<T>::from_request_parts(parts, state)
110            .await
111            .map(|axum::extract::Query(inner)| SendQuery(inner))
112    }
113}
114
115impl<T: schemars::JsonSchema> aide::operation::OperationInput for SendQuery<T> {
116    fn operation_input(
117        ctx: &mut aide::generate::GenContext,
118        operation: &mut aide::openapi::Operation,
119    ) {
120        let schema = ctx.schema.subschema_for::<T>();
121        let params = aide::operation::parameters_from_schema(
122            ctx,
123            schema,
124            aide::operation::ParamLocation::Query,
125        );
126        aide::operation::add_parameters(ctx, operation, params);
127    }
128}

130/// Create a combined router serving both v1 and v2 APIs
131pub fn create_combined_router<S>(state: S) -> Router
132where
133    S: v1::RewardApi
134        + v1::AvailabilityApi
135        + v2::RewardApi
136        + v2::DataApi
137        + v2::ConsensusApi
138        + Clone
139        + Send
140        + Sync
141        + 'static,
142{
143    let router_v1 = create_router_v1(state.clone());
144    let router_v2 = create_router_v2(state).layer(middleware::from_fn(rewrite_root_to_v2));
145
146    router_v2.merge(router_v1).route("/", get(redirect_to_docs))
147}

391                .map(Json)

395        |State(state): State<S>, SendQuery(request): SendQuery<GetStakeTableRequest>| async move {

397        };
398
399    let get_namespace_proof =
400        |State(state): State<S>, SendQuery(query): SendQuery<GetNamespaceProofRequest>| async move {

402        };
403
404    let get_incorrect_encoding_proof = |State(state): State<S>,
405                                        SendQuery(query): SendQuery<

407    >| async move {
408        handlers::get_incorrect_encoding_proof(&state, query)

410            .map(Json)

434        )

436            routes::v2::REWARD_BALANCES_ROUTE.http,
437            get_with(get_reward_balances, |op| {

447            }),
448        )
449        .api_route(
450            routes::v2::NAMESPACE_PROOF_ROUTE.http,
451            get_with(get_namespace_proof, |op| {
452                op.description(routes::v2::NAMESPACE_PROOF_ROUTE.description)
453                    .tag(routes::v2::NAMESPACE_PROOF_ROUTE.tag)
454            }),
455        )
456        .api_route(
457            routes::v2::INCORRECT_ENCODING_PROOF_ROUTE.http,
458            get_with(get_incorrect_encoding_proof, |op| {
459                op.description(routes::v2::INCORRECT_ENCODING_PROOF_ROUTE.description)
460                    .tag(routes::v2::INCORRECT_ENCODING_PROOF_ROUTE.tag)
461            }),
462        )
463        .api_route(
464            routes::v2::STATE_CERTIFICATE_ROUTE.http,
465            get_with(get_state_certificate, |op| {
466                op.description(routes::v2::STATE_CERTIFICATE_ROUTE.description)
467                    .tag(routes::v2::STATE_CERTIFICATE_ROUTE.tag)
468            }),
469        )
470        .api_route(
471            routes::v2::STAKE_TABLE_ROUTE.http,
472            get_with(get_stake_table, |op| {
473                op.description(routes::v2::STAKE_TABLE_ROUTE.description)
474                    .tag(routes::v2::STAKE_TABLE_ROUTE.tag)
475            }),
476        )
477        .finish_api(&mut api);
478
479    // Transform examples (array) to example (singular) for OpenAPI 3.0/Swagger compatibility
480    if let Some(ref mut components) = api.components {
481        let mut transform = schemars::transform::SetSingleExample::default();
482        for schema in components.schemas.values_mut() {
483            transform.transform(&mut schema.json_schema);
484        }
485    }
486
487    // Also transform path parameter schemas
488    if let Some(ref mut paths) = api.paths {
489        let mut transform = schemars::transform::SetSingleExample::default();
490        for path_item_ref in paths.paths.values_mut() {
491            if let aide::openapi::ReferenceOr::Item(path_item) = path_item_ref {
492                for operation in [
493                    &mut path_item.get,
494                    &mut path_item.post,
495                    &mut path_item.put,
496                    &mut path_item.delete,
497                    &mut path_item.patch,
498                ]
499                .into_iter()
500                .flatten()
501                {
502                    for param in &mut operation.parameters {
503                        if let aide::openapi::ReferenceOr::Item(param_item) = param {
504                            let parameter_data = match param_item {
505                                aide::openapi::Parameter::Query { parameter_data, .. } => {
506                                    parameter_data
507                                },
508                                aide::openapi::Parameter::Header { parameter_data, .. } => {
509                                    parameter_data
510                                },
511                                aide::openapi::Parameter::Path { parameter_data, .. } => {
512                                    parameter_data
513                                },
514                                aide::openapi::Parameter::Cookie { parameter_data, .. } => {
515                                    parameter_data
516                                },
517                            };
518                            if let aide::openapi::ParameterSchemaOrContent::Schema(ref mut schema) =
519                                parameter_data.format
520                            {
521                                transform.transform(&mut schema.json_schema);
522                            }
523                        }
524                    }
525                }
526            }
527        }
528    }
529
530    router
531        .route(routes::v2::OPENAPI_SPEC_ROUTE, get(serve_openapi_spec))
532        .route(routes::v2::SWAGGER_ROUTE, get(serve_swagger_ui))
533        .route("/v2/", get(serve_swagger_ui))
534        .route(
535            routes::v2::SCALAR_ROUTE,
536            get(Scalar::new(routes::v2::OPENAPI_SPEC_ROUTE)
537                .with_title("Espresso Node API v2")
538                .axum_handler()),
539        )
540        .route(
541            routes::v2::REDOC_ROUTE,
542            get(Redoc::new(routes::v2::OPENAPI_SPEC_ROUTE)
543                .with_title("Espresso Node API v2")
544                .axum_handler()),
545        )
546        .layer(Extension(api))
547        .with_state(state)
548}
