57    /// Storage path for persistent data.
58    #[clap(long, env = "ESPRESSO_SEQUENCER_STORAGE_PATH")]
59    path: PathBuf,
60
61    /// Number of views to retain in consensus storage before data that hasn't been archived is
62    /// garbage collected.
63    ///
64    /// The longer this is, the more certain that all data will eventually be archived, even if
65    /// there are temporary problems with archive storage or partially missing data. This can be set
66    /// very large, as most data is garbage collected as soon as it is finalized by consensus. This
67    /// setting only applies to views which never get decided (ie forks in consensus) and views for
68    /// which this node is partially offline. These should be exceptionally rare.

71    /// view time of 2s.
72    #[clap(
73        long,
74        env = "ESPRESSO_SEQUENCER_CONSENSUS_VIEW_RETENTION",
75        default_value = "130000"
76    )]
77    pub(crate) consensus_view_retention: u64,
78}

263    /// Overwrite a file if a condition is met.
264    ///
265    /// The file at `path`, if it exists, is opened in read mode and passed to `pred`. If `pred`
266    /// returns `true`, or if there was no existing file, then `write` is called to update the
267    /// contents of the file. `write` receives a truncated file open in write mode and sets the
268    /// contents of the file.
269    ///
270    /// The final replacement of the original file is atomic; that is, `path` will be modified only
271    /// if the entire update succeeds.
272    fn replace(
273        &mut self,
274        path: &Path,
275        pred: impl FnOnce(File) -> anyhow::Result<bool>,
276        write: impl FnOnce(File) -> anyhow::Result<()>,
277    ) -> anyhow::Result<()> {
278        if path.is_file() {
279            // If there is an existing file, check if it is suitable to replace. Note that this
280            // check is not atomic with respect to the subsequent write at the file system level,
281            // but this object is the only one which writes to this file, and we have a mutable
282            // reference, so this should be safe.
283            if !pred(File::open(path)?)? {
284                // If we are not overwriting the file, we are done and consider the whole operation
285                // successful.
286                return Ok(());
287            }
288        }
289
290        // Either there is no existing file or we have decided to overwrite the file. Write the new
291        // contents into a temporary file so we can update `path` atomically using `rename`.
292        let mut swap_path = path.to_owned();
293        swap_path.set_extension("swp");
294        let swap = OpenOptions::new()
295            .write(true)
296            .truncate(true)
297            .create(true)
298            .open(&swap_path)?;
299        write(swap)?;
300
301        // Now we can replace the original file.
302        fs::rename(swap_path, path)?;
303
304        Ok(())
305    }

340    fn prune_files(
341        &mut self,
342        dir_path: PathBuf,
343        prune_view: ViewNumber,
344        keep_decided_view: Option<ViewNumber>,
345        prune_intervals: &[RangeInclusive<ViewNumber>],
346    ) -> anyhow::Result<()> {
347        if !dir_path.is_dir() {
348            return Ok(());
349        }
350
351        for (file_view, path) in view_files(dir_path)? {
352            // If the view is the anchor view, keep it no matter what.
353            if let Some(decided_view) = keep_decided_view {
354                if decided_view == file_view {
355                    continue;
356                }
357            }
358            // Otherwise, delete it if it is time to prune this view _or_ if the given intervals,
359            // which we've already successfully processed, contain the view; in this case we simply
360            // don't need it anymore.
361            if file_view < prune_view || prune_intervals.iter().any(|i| i.contains(&file_view)) {
362                fs::remove_file(&path)?;
363            }
364        }
365
366        Ok(())
367    }

369    fn parse_decided_leaf(
370        &self,
371        bytes: &[u8],
372    ) -> anyhow::Result<(Leaf2, CertificatePair<SeqTypes>)> {
373        // Old versions of the software did not store the next epoch QC. Without knowing which
374        // version this file was created with, we can simply try parsing both ways and then
375        // reconstruct a certificate pair with or without the next epoch QC.
376        match bincode::deserialize(bytes) {
377            Ok((leaf, cert)) => Ok((leaf, cert)),
378            Err(err) => {
379                tracing::warn!(
380                    "error parsing decided leaf, maybe file was created without next epoch QC? \
381                     {err}"
382                );
383                let (leaf, qc) =
384                    bincode::deserialize::<(Leaf2, QuorumCertificate2<SeqTypes>)>(bytes)
385                        .context("parsing decided leaf")?;
386                Ok((leaf, CertificatePair::non_epoch_change(qc)))
387            },
388        }

391    /// Generate events based on persisted decided leaves.
392    ///
393    /// Returns a list of closed intervals of views which can be safely deleted, as all leaves
394    /// within these view ranges have been processed by the event consumer.
395    async fn generate_decide_events(
396        &mut self,
397        view: ViewNumber,
398        deciding_qc: Option<Arc<CertificatePair<SeqTypes>>>,
399        consumer: &impl EventConsumer,
400    ) -> anyhow::Result<Vec<RangeInclusive<ViewNumber>>> {
401        // Generate a decide event for each leaf, to be processed by the event consumer. We make a
402        // separate event for each leaf because it is possible we have non-consecutive leaves in our
403        // storage, which would not be valid as a single decide with a single leaf chain.
404        let mut leaves = BTreeMap::new();
405        for (v, path) in view_files(self.decided_leaf2_path())? {
406            if v > view {
407                continue;
408            }
409
410            let bytes =
411                fs::read(&path).context(format!("reading decided leaf {}", path.display()))?;
412            let (mut leaf, cert) = self.parse_decided_leaf(&bytes)?;
413
414            // Include the VID share if available.
415            let vid_share = self.load_vid_share(v)?.map(|proposal| proposal.data);
416            if vid_share.is_none() {
417                tracing::debug!(?v, "VID share not available at decide");
418            }
419
420            // Move the state cert to the finalized dir if it exists.
421            let state_cert = self.store_finalized_state_cert(v)?;
422
423            // Fill in the full block payload using the DA proposals we had persisted.
424            if let Some(proposal) = self.load_da_proposal(v)? {
425                let payload = Payload::from_bytes(
426                    &proposal.data.encoded_transactions,
427                    &proposal.data.metadata,
428                );
429                leaf.fill_block_payload_unchecked(payload);
430            } else {
431                tracing::debug!(?v, "DA proposal not available at decide");
432            }
433
434            let info = LeafInfo {
435                leaf,
436                vid_share,
437                state_cert,
438                // Note: the following fields are not used in Decide event processing, and should be
439                // removed. For now, we just default them.
440                state: Default::default(),
441                delta: Default::default(),
442            };
443
444            leaves.insert(v, (info, cert));
445        }
446
447        // The invariant is that the oldest existing leaf in the `anchor_leaf` table -- if there is
448        // one -- was always included in the _previous_ decide event...but not removed from the
449        // database, because we always persist the most recent anchor leaf.
450        if let Some((oldest_view, _)) = leaves.first_key_value() {
451            // The only exception is when the oldest leaf is the genesis leaf; then there was no
452            // previous decide event.
453            if *oldest_view > ViewNumber::genesis() {
454                leaves.pop_first();
455            }
456        }
457
458        let mut intervals = vec![];
459        let mut current_interval = None;
460        for (view, (leaf, cert)) in leaves {
461            let height = leaf.leaf.block_header().block_number();
462
463            {
464                // Insert the deciding QC at the appropriate position, with the last decide event in the
465                // chain.
466                let deciding_qc = if let Some(deciding_qc) = &deciding_qc {
467                    (deciding_qc.view_number() == cert.view_number() + 1)
468                        .then_some(deciding_qc.clone())
469                } else {
470                    None
471                };
472
473                consumer
474                    .handle_event(&Event {
475                        view_number: view,
476                        event: EventType::Decide {
477                            committing_qc: Arc::new(cert),
478                            deciding_qc,
479                            leaf_chain: Arc::new(vec![leaf]),
480                            block_size: None,
481                        },
482                    })
483                    .await?;
484            }
485            if let Some((start, end, current_height)) = current_interval.as_mut() {
486                if height == *current_height + 1 {
487                    // If we have a chain of consecutive leaves, extend the current interval of
488                    // views which are safe to delete.
489                    *current_height += 1;
490                    *end = view;
491                } else {
492                    // Otherwise, end the current interval and start a new one.
493                    intervals.push(*start..=*end);
494                    current_interval = Some((view, view, height));
495                }
496            } else {
497                // Start a new interval.
498                current_interval = Some((view, view, height));
499            }
500        }
501        if let Some((start, end, _)) = current_interval {
502            intervals.push(start..=end);
503        }
504
505        Ok(intervals)
506    }

545    fn load_anchor_leaf(&self) -> anyhow::Result<Option<(Leaf2, QuorumCertificate2<SeqTypes>)>> {
546        tracing::info!("Checking `Leaf2` to load the anchor leaf.");
547        if self.decided_leaf2_path().is_dir() {
548            let mut anchor: Option<(Leaf2, QuorumCertificate2<SeqTypes>)> = None;
549
550            // Return the latest decided leaf.
551            for (_, path) in view_files(self.decided_leaf2_path())? {
552                let bytes =
553                    fs::read(&path).context(format!("reading decided leaf {}", path.display()))?;
554                let (leaf, cert) = self.parse_decided_leaf(&bytes)?;
555                if let Some((anchor_leaf, _)) = &anchor {
556                    if leaf.view_number() > anchor_leaf.view_number() {
557                        anchor = Some((leaf, cert.qc().clone()));
558                    }
559                } else {
560                    anchor = Some((leaf, cert.qc().clone()));
561                }
562            }
563
564            return Ok(anchor);
565        }
566
567        tracing::warn!(
568            "Failed to find an anchor leaf in `Leaf2` storage. Checking legacy `Leaf` storage. \
569             This is very likely to fail."
570        );
571        if self.legacy_anchor_leaf_path().is_file() {
572            // We may have an old version of storage, where there is just a single file for the
573            // anchor leaf. Read it and return the contents.
574            let mut file = BufReader::new(File::open(self.legacy_anchor_leaf_path())?);
575
576            // The first 8 bytes just contain the height of the leaf. We can skip this.
577            file.seek(SeekFrom::Start(8)).context("seek")?;
578            let bytes = file
579                .bytes()
580                .collect::<Result<Vec<_>, _>>()
581                .context("read")?;
582            return Ok(Some(bincode::deserialize(&bytes).context("deserialize")?));
583        }
584
585        Ok(None)
586    }

589        &mut self,
590        view: ViewNumber,
591    ) -> anyhow::Result<Option<LightClientStateUpdateCertificateV2<SeqTypes>>> {
592        let dir_path = self.state_cert_dir_path();
593        let file_path = dir_path.join(view.u64().to_string()).with_extension("txt");
594
595        if !file_path.exists() {
596            return Ok(None);
597        }
598
599        let bytes = fs::read(&file_path)?;
600
601        let state_cert: LightClientStateUpdateCertificateV2<SeqTypes> =
602            bincode::deserialize(&bytes).or_else(|err_v2| {
603                tracing::info!(
604                    error = %err_v2,
605                    path = %file_path.display(),
606                    "Failed to deserialize state certificate, attempting with v1"
607                );
608
609                bincode::deserialize::<LightClientStateUpdateCertificateV1<SeqTypes>>(&bytes)
610                    .map(Into::into)
611                    .with_context(|| {
612                        format!(
613                            "Failed to deserialize with both v2 and v1 from file '{}'. error: \
614                             {err_v2}",
615                            file_path.display()
616                        )
617                    })
618            })?;
619
620        let epoch = state_cert.epoch.u64();
621        let finalized_dir_path = self.finalized_state_cert_dir_path();
622        fs::create_dir_all(&finalized_dir_path).context("creating finalized state cert dir")?;
623
624        let finalized_file_path = finalized_dir_path
625            .join(epoch.to_string())
626            .with_extension("txt");
627
628        self.replace(
629            &finalized_file_path,
630            |_| Ok(true),
631            |mut file| {
632                file.write_all(&bytes)?;
633                Ok(())
634            },
635        )
636        .context(format!(
637            "finalizing light client state update certificate file for epoch {epoch:?}"
638        ))?;
639
640        Ok(Some(state_cert))
641    }

694    async fn append_decided_leaves(
695        &self,
696        view: ViewNumber,
697        leaf_chain: impl IntoIterator<Item = (&LeafInfo<SeqTypes>, CertificatePair<SeqTypes>)> + Send,
698        deciding_qc: Option<Arc<CertificatePair<SeqTypes>>>,
699        consumer: &impl EventConsumer,
700    ) -> anyhow::Result<()> {
701        let mut inner = self.inner.write().await;
702        let path = inner.decided_leaf2_path();
703
704        // Ensure the anchor leaf directory exists.
705        fs::create_dir_all(&path).context("creating anchor leaf directory")?;
706
707        // Earlier versions stored only a single decided leaf in a regular file. If our storage is
708        // still on this version, migrate to a directory structure storing (possibly) many leaves.
709        let legacy_path = inner.legacy_anchor_leaf_path();
710        if !path.is_dir() && legacy_path.is_file() {
711            tracing::info!("migrating to multi-leaf storage");
712
713            // Move the existing data into the new directory.
714            let (leaf, qc) = inner
715                .load_anchor_leaf()?
716                .context("anchor leaf file exists but unable to load contents")?;
717            let view = leaf.view_number().u64();
718            let bytes = bincode::serialize(&(leaf, qc))?;
719            let new_file = path.join(view.to_string()).with_extension("txt");
720            inner
721                .replace(
722                    &new_file,
723                    |_| Ok(true),
724                    |mut file| {
725                        file.write_all(&bytes)?;
726                        Ok(())
727                    },
728                )
729                .context(format!("writing anchor leaf file {view}"))?;
730
731            // Now we can remove the old file.
732            fs::remove_file(&legacy_path).context("removing legacy anchor leaf file")?;
733        }
734
735        for (info, cert) in leaf_chain {
736            let view = info.leaf.view_number().u64();
737            let file_path = path.join(view.to_string()).with_extension("txt");
738            inner.replace(
739                &file_path,
740                |_| {
741                    // Don't overwrite an existing leaf, but warn about it as this is likely not
742                    // intended behavior from HotShot.
743                    tracing::warn!(view, "duplicate decided leaf");
744                    Ok(false)
745                },
746                |mut file| {
747                    let bytes = bincode::serialize(&(&info.leaf, cert))?;
748                    file.write_all(&bytes)?;
749                    Ok(())
750                },
751            )?;
752        }
753
754        match inner
755            .generate_decide_events(view, deciding_qc, consumer)
756            .await
757        {
758            Err(err) => {
759                // Event processing failure is not an error, since by this point we have at least
760                // managed to persist the decided leaves successfully, and the event processing will
761                // just run again at the next decide.
762                tracing::warn!(?view, "event processing failed: {err:#}");
763            },
764            Ok(intervals) => {
765                if let Err(err) = inner.collect_garbage(view, &intervals) {
766                    // Similarly, garbage collection is not an error. We have done everything we
767                    // strictly needed to do, and GC will run again at the next decide. Log the
768                    // error but do not return it.
769                    tracing::warn!(?view, "GC failed: {err:#}");
770                }
771            },
772        }
773
774        Ok(())
775    }

979        &self,
980    ) -> anyhow::Result<BTreeMap<ViewNumber, Proposal<SeqTypes, QuorumProposalWrapper<SeqTypes>>>>
981    {
982        let inner = self.inner.read().await;
983
984        // First, get the proposal directory.
985        let dir_path = inner.quorum_proposals2_dir_path();
986        if !dir_path.is_dir() {
987            return Ok(Default::default());
988        }
989
990        // Read quorum proposals from every data file in this directory.
991        let mut map = BTreeMap::new();
992        for (view, path) in view_files(&dir_path)? {
993            let proposal_bytes = fs::read(path)?;
994            let Some(proposal) = bincode::deserialize::<
995                Proposal<SeqTypes, QuorumProposalWrapper<SeqTypes>>,
996            >(&proposal_bytes)
997            .or_else(|error| {
998                bincode::deserialize::<Proposal<SeqTypes, QuorumProposalWrapperLegacy<SeqTypes>>>(
999                    &proposal_bytes,
1000                )
1001                .map(convert_proposal)
1002                .inspect_err(|err_v3| {
1003                    // At this point, if the file contents are invalid, it is most likely an
1004                    // error rather than a miscellaneous file somehow ending up in the
1005                    // directory. However, we continue on, because it is better to collect as
1006                    // many proposals as we can rather than letting one bad proposal cause the
1007                    // entire operation to fail, and it is still possible that this was just
1008                    // some unintended file whose name happened to match the naming convention.
1009
1010                    tracing::warn!(
1011                        ?view,
1012                        %error,
1013                        error_v3 = %err_v3,
1014                        "ignoring malformed quorum proposal file"
1015                    );
1016                })
1017            })
1018            .ok() else {
1019                continue;
1020            };
1021
1022            let proposal2 = convert_proposal(proposal);
1023
1024            // Push to the map and we're done.
1025            map.insert(view, proposal2);
1026        }
1027
1028        Ok(map)
1029    }

2028    /// - `Vec<(EventKey, StakeTableEvent)>` - A list of events, where each entry is a tuple of the event key
2029    /// event key is (l1 block number, log index)

2034        to_l1_block: u64,
2035    ) -> anyhow::Result<(
2036        Option<EventsPersistenceRead>,
2037        Vec<(EventKey, StakeTableEvent)>,

2040        let dir_path = inner.stake_table_dir_path();
2041        let events_dir = dir_path.join("events");
2042
2043        // check if we have any events in storage
2044        // we can do this by checking last l1 finalized block for which we processed events
2045        let last_l1_finalized_path = events_dir.join("last_l1_finalized").with_extension("bin");
2046
2047        if !last_l1_finalized_path.exists() || !events_dir.exists() {
2048            return Ok((None, Vec::new()));
2049        }

2051        let mut events = Vec::new();
2052
2053        let bytes = fs::read(&last_l1_finalized_path)

2057            .as_slice()
2058            .read_exact(&mut buf[..8])
2059            .with_context(|| {
2060                format!("Failed to read 8 bytes from file at path: {last_l1_finalized_path:?}")

2065        // Determine the L1 block for querying events.
2066        // If the last stored L1 block is greater than the requested block, limit the query to the requested block.
2067        // Otherwise, query up to the last stored block.
2068        let query_l1_block = if last_processed_l1_block > to_l1_block {
2069            to_l1_block
2070        } else {

2072        };

2075            let entry = entry?;

2104                continue;
2105            }
2106

2108                File::open(&path).context(format!("Failed to open event file. path={path:?}"))?;

2115        }
2116
2117        events.sort_by_key(|(key, _)| *key);
2118
2119        if query_l1_block == to_l1_block {
2120            Ok((Some(EventsPersistenceRead::Complete), events))
2121        } else {
2122            Ok((
2123                Some(EventsPersistenceRead::UntilL1Block(query_l1_block)),

2125            ))
2126        }
2127    }
